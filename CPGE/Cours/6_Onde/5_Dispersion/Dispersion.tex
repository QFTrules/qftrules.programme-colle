\section{Relation de dispersion}

\subsection{Relation de dispersion}
 Considérons une OPPH$^{\star}$ (OPPH généralisée) scalaire de la forme $\udl{s}(x,t)=s_{\rm{m}}\e^{i(\omega t-\udl{k}x)}$ avec $\udl{k}$ le vecteur d'onde complexe. L’équation d’onde impose une \imp{relation de dispersion}, c’est-à-dire une relation entre $\udl{k}$ et $\omega$. Nous écrirons cette relation de dispersion sous la forme
% \begin{equation}
  \eqbox[]{
    \udl{k}(\omega)=k'(\omega)+ik''(\omega),
  }
% \end{equation}
où $k’=\re\,\udl{k}$ et $k''=\im\,\udl{k}$ sont réels.
En remplaçant dans l'OPPH$^{\star}$, et en prenant la partie réelle, il vient
\begin{equation}
   s(x,t)=\re \, \udl{s}(x,t) = s_{\rm{m}}\e^{k''x}\cos{(\omega t-k' x)}.
\end{equation}
 Si le milieu occupe l’espace $x>0$, on doit imposer $k'' \leq 0$ pour éviter toute divergence.

 \begin{definition}[Relation de dispersion]\label{def:}
  La \imp{relation de dispersion} est la relation $\udl{k}(\omega)$ imposée par l'équation d'onde entre le vecteur d'onde complexe $\udl{k}$ et la pulsation $\omega$.
  
 \end{definition}

 \subsection{Vitesse de phase et propagation} La grandeur $k'$ intervient dans la phase $\vphi = \omega t -k' x$ de l'onde et donc caractérise sa \imp{propagation}.
 % : la longueur d’onde vaut $\lambda = \frac{2\pi}{k’}$.
 Supposons $k'>0$ de sorte à ce que l’onde se propage vers les $x$ croissants. On peut mettre la partie oscillante de l’onde sous la forme
\begin{equation}
\label{eq:OPPH*_vitesse_phase}
s(x,t) 
% =s_{\rm{m}}\e^{k'' x}\cos\left[k'\left( \frac{\omega}{k'}t-x\right)\right] 
= s_{\rm{m}}\e^{k'' x}\cos\left[\omega\left(t-\textstyle\frac{x}{v_{\vphi}}\right)\right] = s_{\rm{m}}\e^{k'' x}f(t-\textstyle\frac{x}{v_{\vphi}}),
\end{equation}
avec $f(t)$ la fonction $t \mapsto \cos(\omega t) $ et $v_{\vphi} = \frac{\omega}{k'}$ la vitesse de phase. La fonction $f$ décrit une onde progressive, et caractérise donc la partie propagative de l'onde.
%Dans le cas général, on a la définition suivante.
Un plan d'onde d'une OPPH$^{\star}$ est  une surface équiphase : $\vphi = \omega t - k' x = \cte$, soit $x(t) = v_\vphi t - \cte$. Un plan d'onde se déplace donc à la vitesse de phase $v_{\vphi}$. Attention, la vitesse de phase décrit un effet purement géométrique :  elle ne correspond pas à la vitesse de propagation de l'information. 
\begin{definition}[Vitesse de phase]\label{def:}
 La vitesse de déplacement des plans d'onde d'une OPPH$^{\star}$  est donnée par  la  \imp{vitesse de phase} 
    $$
    v_{\vphi}(\omega) = \frac{\omega}{|k’(\omega)|},
    $$
    où  $k'(\omega) = \re\, \udl{k}(\omega)$. Le milieu est non dispersif si $v_{\vphi}(\omega) = \cte$  ; dispersif sinon.
\end{definition}
Deux situations peuvent se présenter.
\begin{liste}
  \item Si $k'(\omega) = \frac{\omega}{c}$ est linéaire, alors $v_{\vphi} = c$ est une constante (la célérité) : le milieu est \imp{non dispersif}.
  \item Si $k'(\omega)$ n’est pas linéaire, alors  $v_{\vphi}(\omega)$ n'est pas une constante : le milieu est \imp{dispersif}.
\end{liste}
% Une information a besoin d'être véhiculée par un paquet d'onde, construit par superposition d'OPPH. L'information physique se déplace alors à la vitesse de l'enveloppe du paquet d'onde, comme expliqué dans ce qui suit.

\begin{exemple}{Dispersion dans un plasma}
  Déterminer la vitesse de phase dans un plasma à haute fréquence $\omega > \omega_p$, de relation de dispersion
  \begin{equation}
    k(\omega) = \pm \frac{\sqrt{\omega^2-\omega_p^2}}{c}.
  \end{equation}
  \examplesol{
    Ici $k(\omega) = k'(\omega)$ est réel. La vitesse de phase vaut donc
    % \begin{equation}
      \eqbox{
        v_{\vphi}(\omega) = \frac{\omega}{|k’(\omega)|} = \frac{c}{\sqrt{1-(\omega/\omega_p)^2}}.
      }
    % \end{equation}
    Ce n'est pas une constante, donc un plasma est un milieu dispersif.
  }
\end{exemple}

\subsection{Distance d'amortissement et atténuation}  La partie imaginaire $k''$ caractérise l’\imp{atténuation} de l’onde.  Le milieu est absorbant ou réfléchissant si $k''<0$ ; transparent si $k''=0$. On peut mettre la partie non oscillante de l'onde sous la forme
\begin{equation}
\label{eq:OPPH*_vitesse_phase}
s(x,t)= s_{\rm{m}}\e^{k'' x}f(v_{\vphi} t-x) = s_{\rm{m}}\e^{-x/\delta}f(v_{\vphi} t-x) =  S(x)f(v_{\vphi} t-x),
\end{equation}
où $S(x) = s_{\rm{m}}\e^{-x/\delta}$ est l'enveloppe de l'onde et $\delta = -1/k''(\omega)$ une grandeur homogène à une distance. L'enveloppe décroît exponentiellement avec la profondeur dans le milieu.

\begin{definition}[Distance d'amortissement]\label{def:}
  L'atténuation d'une OPPH* est décrite par la \imp{distance d'amortissement}
    $$
    \delta(\omega) = \frac{1}{|k''(\omega)|},
    $$
    où $k''(\omega) = \im\, \udl{k}(\omega)$. Le milieu est absorbant ou réfléchissant si $\delta$ est finie ; transparent si $\delta=\infty$.
\end{definition}

\begin{exemple}{Atténuation dans un plasma}
  Déterminer la vitesse de phase dans un plasma à basse fréquence $\omega < \omega_p$, de relation de dispersion
  \begin{equation}
    k(\omega) = \pm i\frac{\sqrt{\omega_p^2-\omega^2}}{c}.
  \end{equation}
  \examplesol{
    Ici $k(\omega) = ik''(\omega)$ est imaginaire pur. La distance d'amortissement vaut donc
    % \begin{equation}
      \eqbox{
        \delta(\omega) = \frac{1}{|k''(\omega)|} = \frac{c}{\sqrt{\omega_p^2-\omega^2}}.
      }
    % \end{equation}
    Cette distance est finie, donc un plasma est un milieu absorbant ou réfléchissant. Pour savoir si l'onde est absorbée ou réfléchie, il faut mener une analyse énergétique. En l'occurence, le plasma réfléchit l'onde et n'absorbe pas.
  }
\end{exemple}


\section{Propagation d'un paquet d'onde}
Pour simplifier l'étude et se concentrer sur l'aspect propagatif, nous allons considérer un milieu dispersif mais non absorbant. Ainsi $\udl{k}(\omega) = k'(\omega) = k(\omega)$ est supposé réel. Les OPPH$^{\star}$ sont alors des OPPH.
% \parag{Caractère non physique d'une OPPH}
Un signal de la forme $s(x,t)=s_{\rm{m}}\cos(\omega t-kx)$ ne peut pas décrire un signal physique parce que son extension temporelle et spatiale sont infinies. Une OPPH transporte donc une énergie infinie.
  % \begin{equation}
  %   E \propto \int_{-\infty}^{\infty} s(x,t)^2 \d x = S_{\rm{m}}^2 \int_{-\infty}^{\infty} \cos(\omega t-k(\omega)x)^2 \d x  = + \infty.
  % \end{equation}
% \end{liste}
\begin{prop}\label{prop:}
  Une OPPH est un modèle de signal non physique, et ne peut pas  transporter de l'information.
\end{prop}

\subsection{Superposition de deux OPPH}  Pour transporter  l'information, il est indispensable de considérer une combinaison linéaire d'OPPH. L'information peut être par exemple encodée dans les variations d'amplitude de l'enveloppe (modulation d'amplitude) ou dans les fréquences explorées (modulation de fréquence). Pour commencer, étudions une superposition simple de deux OPPH de pulsations proches et de même amplitude,
\begin{equation}
\begin{split}
s(x,t) = s_1(x,t) + s_2(x,t) &= A\cos(\omega_1 t-k_1 x)+ A\cos(\omega_2 t-k_2 x),
\end{split}
\end{equation}
% soit
\eqbox[]{
  s(x,t)=2A\underbrace{\cos(\omega_0 t-k_0 x)}_{\text{porteuse}}\times\underbrace{\cos(\Delta\omega\,t-\Delta k\,x)}_{\text{enveloppe}}.
} 
Nous avons posé l'écart en pulsation $\Delta\omega=\frac{\omega_2-\omega_1}{2}$ et la pulsation moyenne $\omega_0=\frac{\omega_1+\omega_2}{2}$ (idem pour $k$). On observe un phénomène de battements, où une onde porteuse est modulée par une enveloppe (Fig.~\ref{fig:battements}).

\begin{figure}[H]
  \centering
  \includegraphics[width=0.8\textwidth]{battements.pdf}
  \caption{Exemple de phénomène de battements pour deux fréquences proches $f_1=\SI{10}{Hz}$ et $f_2=\SI{10,5}{Hz}$.}
  \label{fig:battements}
  \end{figure}
\begin{liste}
  \item La \imp{porteuse} varie rapidement dans le temps (pulsation $\omega_0$). Elle se déplace à la vitesse
  \begin{equation}
    \frac{\omega_0}{k_0} = v_{\varphi}(\omega_0),
  \end{equation}
  qui est la vitesse de phase d'une OPPH de pulsation $\omega_0$.
  \item L'\imp{enveloppe} varie lentement dans le temps (pulsation $\Delta \omega \ll \omega_0$). Elle se propage à la vitesse
  \begin{equation}
    \frac{\Delta \omega}{\Delta k} = v_g,
  \end{equation}
  appelée \imp{vitesse de groupe}.
\end{liste}
Si la relation de dispersion est linéaire, $\omega = c k$ et donc $ v_{\varphi} = v_g = c$. Mais en présence de dispersion, les vitesses de phase et de groupe sont différentes : l'enveloppe, qui encode l'information, ne se propage pas à la même vitesse que la porteuse. Généralisons ces résultats à une superposition d'une infinité d'onde.


\subsection{Paquet d'onde et vitesse de groupe} Une superposition d'OPPH complexes d'amplitudes $\tilde{s}_i$, de pulsations $\omega_i$, et de vecteurs d'onde $k_i=k(\omega_i)$ donnés par la relation de dispersion, s'écrit
\begin{equation}
  \udl{s}(x,t)  = \sum_i \tilde{s}_i \e^{i(\omega_i t-k_i x)}.
\end{equation}
Pour autoriser un continuum de pulsations on passe à une intégrale : c'est la transformation de \imp{Fourier}.
% On obtient ce qu'on appelle un paquet d'onde.
\begin{definition}[Paquet d'onde]\label{def:}
  On appelle \imp{paquet d'onde} la superposition continue d'une infinité d'OPPH, d'expression
  \begin{equation*}
  \label{eq:paquet_onde}
  \udl{s}(x,t)=\int_{-\infty}^{+\infty}\tilde{\udl{s}}(\omega)\e^{i(\omega t-k(\omega)x)}\d\omega,
\end{equation*}
où $k(\omega)$ et $\omega$ sont reliés par la relation de dispersion et $\tilde{\udl{s}}(\omega)$ est l'amplitude spectrale.
\end{definition}

\begin{figure}[H]
  \centering
  \begin{subfig}[label][0.45]
    \includegraphics[width=0.9\textwidth]{wave_packet_omega.pdf}
  \end{subfig}
  \begin{subfig}[label][0.45]
    \includegraphics[width=0.9\textwidth]{wave_packet.pdf}
  \end{subfig}
  \caption{Pquet d'onde. (a) Norme de l'amplitude spectrale $\tilde{s}(\omega)$ d'un signal typique en fonction de la pulsation $\omega$. Tout signal physique a une extension en pulsation finie $\Delta \omega$. (b) Signal temporelle associé $s(x,t)$ (en bleu). Les oscillations rapides (porteuse) sont déterminés par la pulsation moyenne $\omega_0$. L'enveloppe a une extension temporelle typique $\tau = 1/\Delta \omega$. La dépendance spatiale de $s(x,t)$ est déterminée par la relation de dispersion $k(\omega)$.}
  \label{fig:wave_packet}
\end{figure}
Un paquet d'onde physique possède une extension temporelle $\tau$ finie, et donc d'après les propriétés de la trannsformée de Fourier, une extension fréquentielle finie également
\eqbox[]{
\Delta\omega\sim\frac{1}{\tau}.
}
\begin{remarque}
Nous avions déjà rencontré la relation $\Delta\nu \times \tau_{\rm c}\sim 1$ en optique où elle était utile pour relier le temps de cohérence $\tau_c$ d'une source à son largeur en fréquence $\Delta \nu$.
\end{remarque}
 L'amplitude spectrale $\tilde{\udl{s}}(\omega)$ ne prend donc des valeurs significatives que sur un intervalle restreint de largeur $\Delta \omega$ centré autour d'une pulsation moyenne $\omega_0$.

% Ainsi
% \begin{equation}
% \udl{s}(x,t) \simeq \int_{\omega_0-\frac{\Delta\omega}{2}}^{\omega_0+\frac{\Delta\omega}{2}}\tilde{\udl{s}}(\omega)\e^{i(\omega t-k x)}\d\omega.
% %\e^{i(\omega_0 t-k_0 x)}\int_{\omega_0-\frac{\Delta\omega}{2}}^{\omega_0+\frac{\Delta\omega}{2}}\tilde{\udl{s}}(\omega)\e^{i((\omega-\omega_0) t-(k-k_0)x)}\d\omega.
% \end{equation}
Si de plus le milieu  est \imp{faiblement dispersif}, la norme du vecteur d'onde varie peu sur cet intervalle : on peut donc linéariser la relation de dispersion autour de $\omega_0$ :
  \begin{equation}
   k(\omega)\simeq k(\omega_0)+(\omega-\omega_0)\left.\frac{\d k}{\d\omega}\right|_{\omega=\omega_0} = k_0 + \frac{\omega-\omega_0}{v_g},
   \end{equation}
   avec $k_0=k(\omega_0)$ et  $v_{\rm{g}}=1/\left.\frac{\d k}{\d\omega}\right|_{\omega=\omega_0}=\left.\frac{\d \omega}{\d k}\right|_{\omega=\omega_0}$ la vitesse de groupe à la pulsation moyenne $\omega_0$.
   \begin{definition}[Vitesse de groupe]\label{def:}
    On appelle \imp{vitesse de groupe} d'une OPPH de relation de dispersion $\omega(k)$ la grandeur
    \begin{equation*}
    \label{eq:def_vg}
    v_{\rm{g}}(\omega) = \frac{\d \omega}{\d k}.
  \end{equation*}
  \end{definition}
Le paquet d'onde se réécrit alors
\begin{align}
\udl{s}(x,t)
& \simeq
\int_{ - \infty}^{ +\infty}
\tilde{\udl{s}}(\omega)
\e^{i[\omega t- x k_0 - \frac{x}{v_g}(\omega-\omega_0)]}\d \omega \\
& \simeq \e^{i(\omega_0 t-k_0 x)}\int_{ - \infty}^{ +\infty}\tilde{\udl{s}}(\omega)\e^{i(\omega-\omega_0)(t-\frac{x}{v_g})}\d\omega \\
& \simeq \e^{i(\omega_0 t-k_0 x)}\times\int_{ - \infty}^{ +\infty}\tilde{\udl{s}}(\omega + \omega_0)\e^{i\omega(t-\frac{x}{v_g})}\d\omega.
\end{align}
par changement de variable $\omega \mapsto \omega + \omega_0$. En définissant la fonction $\tilde{\udl{s}}(\omega + \omega_0) = \tilde{\udl{s}}_0(\omega)$, on peut écrire
\eqbox{
 \udl{s}(x,t)  \simeq \underbrace{\e^{i\omega_0(t- \frac{x}{v_{\vphi} })}}_{\text{porteuse}}\times\underbrace{\int_{ - \infty}^{ + \infty}\tilde{\udl{s}}_0(\omega)\e^{i\omega(t-\frac{x}{v_g})}\d\omega.}_{\text{enveloppe}}
}
Le paquet d'onde s'écrit comme le produit d'une \imp{porteuse} $f(t-\frac{x}{v_{\vphi}})$ et d'une \imp{enveloppe} $S(t -\frac{x}{v_g })$. La fonction $S(t)$ est la transformée de Fourier de l'amplitude spectrale  $\tilde{\udl{s}}_0(\omega)$ centrée autour de  $\omega_0$.
\begin{prop}\label{prop:}
  Un paquet d'onde se propageant dans un milieu \imp{faiblement dispersif} se comporte comme  une \imp{porteuse} de pulsation $\omega_0$ modulée par une \imp{enveloppe} de largeur spectrale $\Delta \omega$.
  \begin{liste}
    \item La porteuse est une onde progressive se déplacant à la vitesse de \imp{phase} $v_{\vphi}(\omega_0)=\frac{\omega_0}{k_0}$.
    \item L'enveloppe est une onde progressive se déplacant à la vitesse de \imp{groupe} $v_g(\omega_0) = \left.\dd{\omega}{k}\right|_{\omega=\omega_0} $.
  \end{liste}
\end{prop}

% On reconnaît sous l'intégrale une \important{enveloppe} qui se propage à la vitesse de groupe $v_g(\omega_0)$.

\begin{exemple}{Représentation spatiale d'un paquet d'onde}
  Nous avons représenté le paquet d'onde dans le domaine temporelle à l'aide des fonctions porteuse $f(t)$ et enveloppe $S(t)$.
  On considère ici un paquet d'onde caractérisé par la relation de dispersion $\omega(k)$ avec $k$ réel et d'amplitude spectrale $\tilde{\udl{a}}(k)$ en vecteur d'onde. On suppose que le milieu est faiblement dispersif autour d'un vecteur d'onde moyen $k_0$. Montrer que le paquet d'onde se met sous la forme
  \begin{equation}
    \udl{s}(x,t) \simeq g(x - v_{\vphi} t ) A(x - v_g t),
    % \int_{-\Delta k/2}^{+\Delta k/2} \tilde{a}(k_0 + k)\e^{i(k v_g t - k x)}\d k,
  \end{equation}
  où $g(x - v_{\vphi} t )$ est la porteuse et l'enveloppe $A(x - v_g t)$ est à écrire comme une intégrale sur $k$.
  \examplesol{
    On linéarise la relation de dispersion autour de $k_0$, soit 
    \begin{equation}
      \omega(k) \simeq \omega_0 + v_g(k-k_0),
    \end{equation}
    avec $\omega_0=\omega(k_0)$ la pulsation moyenne et $v_g=\dd{\omega}{k}|_{k=k_0}$ la vitesse de groupe. On peut alors écrire le paquet d'onde sous la forme
    \begin{equation}
      \begin{split}
      s(x,t) = \int_{-\infty}^{+\infty} \tilde{\udl{a}}(k)\e^{i(\omega(k) t - kx)}\d k & \simeq \int_{-\infty}^{+\infty} \tilde{\udl{a}}(k)\e^{i(\omega_0 t - k_0 x + (k -k_0)(v_g t - x))}\d k \\ 
      & \simeq \e^{ik_0(v_{\vphi} t - x)}\int_{-\infty}^{+\infty} \tilde{\udl{a}}(k)\e^{i(k - k_0)(v_g t - x)}\d k \\ 
      & \simeq \e^{ -ik_0(x - v_{\vphi} t)}\int_{-\infty}^{+\infty} \tilde{\udl{a}}(k_0 + k)\e^{ -ik(x -v_g t)}\d k.
      \end{split}
    \end{equation}
    On peut alors définir l'amplitude spectrale centrée $\tilde{\udl{a}}_0(k) = \tilde{\udl{a}}(k_0 + k)$ et mettre le paquet d'onde sous la forme attendue avec 
    $$
  \boxed{g(x) = \e^{-ik_0 x}} \quad \text{ et } \quad \boxed{A(x) = \int_{-\infty}^{+\infty} \tilde{\udl{a}}_0(k)\e^{-ikx}\d k.}
    $$
  
  }
\end{exemple}
