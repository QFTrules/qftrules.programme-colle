	\draw[fill, color = black,#1] (0.5,0.2) rectangle(1,-0.2) node[color=white,midway]{\tiny  \textbf{N}};
	\node[#1, xshift = -2.1mm, yshift = -2.5mm] at (0.5,0.2) {\tiny aimant};}

	scale=.7,

	]
