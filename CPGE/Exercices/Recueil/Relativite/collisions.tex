\begin{exo}[PC][1][TD]{Référentiel barycentrique de deux photons}
  \Notions{\item Collision élastique \task Photons \task Référentiel barycentrique}
  On considère la collision élastique entre deux photons, $\gamma + \gamma \rightarrow \gamma + \gamma$, de fréquences $\nu_1$ et $\nu_2$ dans le référentiel du laboratoire $\Rcal$, dont les directions font les angles $\theta_1$ et $\theta_2$ respectivement avec l'axe $(Ox)$.
  \begin{quest}
    \item	Rappeler l'expression de la vitesse d'entraînement $\Vec{\vboost}^*$ du référentiel barycentrique $\Rcal^{*}$ par rapport à $\Rcal$ en fonction de l'énergie totale du système $\Etot_{\rm tot}$ et de l'impulsion totale $\Vec{p}_{\rm tot}$ dans $\Rcal$.
    \sol{
      Voir cours,
      % \begin{equation}
      \eqbox{
        \Vec{\vboost}^* = \frac{\Vec{p}_{\rm tot}c^2}{\Etot_{\rm tot}}
        }
      % \end{equation}
    }
    \item Exprimer l'énergie totale $\Etot_{\rm tot}$ et l'impulsion totale $\Vec{p}_{\rm tot}$ dans $\Rcal$. On pourra introduire la constante de Planck $h$. En déduire que la vitesse d'entraînement adimensionnée a pour norme 
    \begin{equation}
      \bboost^* = \frac{\nu_1 \cos \theta_1+\nu_2 \cos \theta_2}{\nu_1+\nu_2}.
    \end{equation}

    \sol{
      On a d'après la relation de Planck-Einstein, 
      \eqbox{
      \mathcal{E}_{\rm tot}=\Etot_1 + \Etot_2 = h\left(\nu_1+\nu_2\right),
      }
      et d'après la relation de De Broglie, 
      \eqbox{
      \Vec{p}_{\rm tot}=\Vec{p}_1 + \Vec{p}_2  = \frac{h}{c}\left( \nu_1 \Vec{u}_1 + \nu_2 \Vec{u}_2 \right) =   \frac{h}{c}\left(\nu_1 \cos \theta_1+\nu_2 \cos \theta_2, \nu_1 \sin \theta_1+\nu_2 \sin \theta_2, 0\right),
      }
      où $\Vec{u}_1$ et $\Vec{u}_2$ sont les vecteurs unitaires dans les directions des photons 1 et 2 respectivement.
      Pour simplifier, on choisit l'axe des $x$ de sorte que le barycentre du système reste sur cet axe  dans $\Rcal$ au cours du mouvement, ce qui impose $\nu_1 \sin \theta_1+\nu_2 \sin \theta_2=0$. Le boost s'effectue dans ce cas suivant cet axe, et 
      \eqbox{
      \bboost^* = \frac{p_{{\rm tot},x} c}{\Etot_{\rm tot}} = \frac{\nu_1 \cos \theta_1+\nu_2 \cos \theta_2}{\nu_1+\nu_2}.
      } 
    }
    \item Que vaut cette vitesse lorsque les photons se déplacent dans la même direction? Lorsqu'ils se déplacent en sens opposés? Commenter.
    \sol{
      \begin{enumerate}
        \item 
        Si les photons ont même fréquence ($\nu_1=\nu_2$) et s'ils sont émis dans une même direction mais dans des sens opposés ($\theta_1=0$ et $\theta_2=\pi$), alors $v_e=0$. Si les fréquences diffèrent, on trouve 
        $$
        \bboost^* = \frac{\nu_1 - \nu_2}{\nu_1 + \nu_2},
        $$
        \item S'ils sont émis dans le même sens ($\theta_1=0$ et $\theta_2=0$), le résultat précédent donne $\bboost^*=1$, ce qui doit être exclu, car les expressions utilisées supposent que $\gamma_e$ soit fini; physiquement, aucun référentiel inertiel ne peut se déplacer à la vitesse $c$.
      \end{enumerate}
    }
  \end{quest}
  
\end{exo}

\begin{exo}[PC][2][TD]{Diffusion Compton}
  \Notions{\item Collision élastique \task Photons \task Conservation de la quadri-impulsion}
  L'effet Compton a été découvert par le physicien américain A. H. \textsc{Compton}, en 1923. En irradiant du graphite avec des rayons $X$ associés à la raie $\mathrm{K}_\alpha$ du molybdène, de longueur d'onde $\lambda = \SI{71,1}{pm}$, puis en analysant la lumière obtenu avec un spectromètre à réseau cristallin, dans une direction faisant un angle $\theta$ avec la lumière incidente, il observa deux raies spectrales :
  \begin{enumerate}
    \item Une première raie centrée sur la longueur d'onde incidente $\lambda$ ;
    \item Une seconde centrée sur une longueur d'onde $\lambda^{\prime}$ telle que 
    \begin{equation}\label{eq:compton}
      \lambda^{\prime}-\lambda \approx \num{2,4}\, (1 - \cos \theta) ~(\text{en pm}).
    \end{equation}
  \end{enumerate}
  Compton expliqua le résultat précédent en considérant la collision élastique entre un photon $\gamma$ sur un électron libre du graphite au repos :
  \begin{equation}
    \gamma+e^{-} \rightarrow \gamma+e^{-}.
  \end{equation}
  \begin{quest}
    \item	Rappeler la relation entre l'énergie $\Etot$  de l'impulsion $p$ d'un photon. Rappeler la relation entre l'énergie $\Etot_e$ et l'impulsion $p_e$ de l'électron diffusé, de masse $m_e$.
    \sol{
    La relation énergie-impulsion du photon s'écrit
    \eqbox{
    \Etot = pc,
    }
    le photon étant de masse nulle. Pour l'électron, on a
    \eqbox{
    \Etot_e^2 = p_e^2 c^2 + m_e^2 c^4.
    }
    }
    \item Écrire la conservation de l'énergie totale et de l'impulsion totale. En éliminant l'énergie et l'impulsion de l'électron, montrer que 
     \begin{equation}
      (\Etot -\Etot' + m_e c^2)^2 - (\Vec{p} - \Vec{p}')^2c^2 = m_e^2c^4 .  
    \end{equation}
    \sol{
    La conservation de l'énergie donne 
    $$
    m_e c^2 + \Etot = \Etot' + \Etot_e',
    $$
    l'électron étant au repos avant diffusion,
    tandis que la conservation de l'impulsion donne 
    $$
    \Vec{p} = \Vec{p}_e' + \Vec{p}'.
    $$
    La relation énergie-impulsion appliqué à l'électron après diffusion donne 
    \eqbox{
      m_e^2 c^4 = \Etot_e'^2 - p_e'^2 c^2 = (\Etot - \Etot' + m_e c^2)^2 - (\Vec{p} - \Vec{p}')^2 c^2.
      }
    }
    \item Simplifier et en déduire la relation  
    \begin{equation}
      \frac{1}{\mathcal{E}^{\prime}}-\frac{1}{{\mathcal{E}}}=\frac{1-\cos \theta}{m_e c^2}.
    \end{equation}
    Montrer que le facteur numérique de l'\Eq{compton} correspond à la longueur d'onde Compton  $\lambda_{\rm C} = \frac{h}{m_e c}$.
    \sol{
    En développant et simplifant le carré, on trouve
    % $$
    \begin{align}
      & (\Etot - \Etot' + m_e c^2)^2 - (\Vec{p} - \Vec{p}')^2 c^2 - m_e^2 c^4  \\  & = \Etot^2 - 2 \Etot \Etot' + \Etot'^2 + 2 m_e c^2 (\Etot - \Etot') + m_e^2 c^4 - (p^2 + p'^2 - 2 p p' \cos \theta) c^2 - m_e^2 c^4 \nonumber \\
      & = \left( \Etot^2 - p^2c^2 \right) + \left( \Etot'^2 - p'^2 c^2 \right) + 2 m_e c^2 (\Etot - \Etot') + 2 p p' c^2 \cos \theta - 2 \Etot \Etot' \nonumber \\
       & = 0 + 0 + 2 m_e c^2 (\Etot - \Etot') + 2 \Etot \Etot'( \cos \theta - 1) \nonumber \\
       & = 0 \nonumber
    \end{align}
    On en déduit
    $$
    m_e c^2 (\Etot - \Etot') = \Etot \Etot' (1 - \cos \theta) \Rar  \boxed{\frac{1}{\Etot'} - \frac{1}{\Etot} = \frac{1 - \cos \theta}{m_e c^2}.}
    $$
    On peut relier l'énergie à la longueur d'onde du photon par la relation $\Etot = h \nu = \frac{h c}{\lambda}$, d'où
    $$
    \lambda' - \lambda = \frac{h}{m_e c} (1 - \cos \theta) = \lambda_{\rm C} (1 - \cos \theta).
    $$
    La longueur d'onde Compton vaut
    \eqbox{
    \lambda_{\rm C} = \frac{h}{m_e c} \approx \SI{2.43}{pm},
    }
    en accord avec la valeur expérimentale.
    }
  \end{quest}
  

\end{exo}

On désigne par $\theta$ l'angle de diffusion que fait, avec la direction incidente, l’impulsion $\Vec{p}_{\!1}^{\prime}$ de l'électron diffusé $A_{1}^{\prime}$.
% , par $E_{1}$ l'énergie totale de $A_{1}$ et par $E_{1}^{\prime}$ celle de $A_{1}^{\prime}$.

  \item On note $p_{1}^{*}$ est la norme de l'impulsion de $A_{1}$ dans le référentiel barycentrique $\Rcal^{*}$ et $\vboost$ la vitesse de $\Rcal^{*}$ par rapport à $\Rcal$. Démontrer la relation
  \begin{equation}
    p_{1}^{*}=\gboost\left(p_{1}-\frac{\vboost E_{1}}{c^{2}}\right).
  \end{equation}
Exprimer $\bboost$ et $\gboost$ en fonction de $\gamma_1$. \AN ?

  % \item Établir en fonction de $\gamma_{1}$, l'expression du rapport $\beta_{C}$ de la vitesse du référentiel du centre de masse $\mathcal{R}^{*}$ du système sur la vitesse de la lumière dans le vide.
  
  % \item Même question pour le facteur relativiste $\gamma_{C}$ associé à $\mathcal{R}^{*}$
  
  % \item Calculer la vitesse du projectile par rapport à $\mathcal{R}^{*}$.
  
  \item Calculer l’énergie cinétique $\Ek_1^*$ de l’électron $A_1$ dans $\mathcal{R}^{*}$. Que devient cette énergie après la collision?
  
  \item Écrire la conservation de l’énergie et de l’impulsion dans $\mathcal{R}$.
  % , les équations caractéristiques de cette collision relatives à la quantité de mouvement et à l'énergie.
  
  \item Montrer que l'angle $\theta$ est relié aux énergies cinétiques par la relation :
  
% $$
\begin{equation}

  % $$

où $\alpha$ est un facteur à déterminer.

  \item En déduire $\Ek_1^{\prime}$ de $A_{1}^{\prime}$ en fonction de $\theta$ et de $\Ek_1$. \AN~pour $\theta=\pi / 3$ ?

  \item Quelle est la valeur maximale de $\Ek_1^{\prime}$ ? Ce résultat était-il prévisible?
  
\end{quest}

\begin{exo}[PC][2][TD]{L'électron libre}
  Montrer qu'un électron libre ne peut pas émettre de photon.
  \sol{
  Par l'absurde, considérons l'émission dans un photon $\gamma$ par un électron libre $e^-$, suivant la réaction :
  $$
  e^- \rightarrow e^- + \gamma .
  $$
  Soit $\Etot_1$ et $\Vec{p}_1$ l'énergie totale et l'impulsion de l'électron avant l'émission, $\Etot_2$ et $\Vec{p}_2$ celles de l'électron après l'émission, et $\Etot_\gamma$ et $\Vec{p}_\gamma$ celles du photon émis. Par conservation de l'énergie et de l'impulsion, on a :
  $$
  \Etot_1 = \Etot_2 + \Etot_\gamma , \quad \text { et } \quad \Vec{p}_1 = \Vec{p}_2 + \Vec{p}_\gamma.
  $$
  On rappelle que l'électron est de masse $m_e$, et le photon de masse nulle.
  D'après la relation impulsion-énergie pour l'électron, on a :
  \begin{align}
    m_e^2 c^4 & = \Etot_1^2 - p_1^2 c^2 \nonumber \\
              & = (\Etot_2 + \Etot_\gamma)^2 - (\Vec{p}_2 + \Vec{p}_\gamma)^2 c^2 \nonumber \\
              & = \Etot_2^2 + \Etot_\gamma^2 + 2 \Etot_2 \Etot_\gamma - p_2^2 c^2 - p_\gamma^2 c^2 - 2 \Vec{p}_2 \cdot \Vec{p}_\gamma c^2 . \nonumber \\ 
              & = (\Etot_2^2 - p_2^2 c^2) + (\Etot_\gamma^2 - p_\gamma^2 c^2) + 2 (\Etot_2 \Etot_\gamma - \Vec{p}_2 \cdot \Vec{p}_\gamma c^2) . \nonumber \\ 
              & = m_e^2 c^4 + 0 + 2 (\Etot_2 \Etot_\gamma - \Vec{p}_2 \cdot \Vec{p}_\gamma c^2) . \nonumber
  \end{align}
  Notons $\Vec{u}$ la direction de propagation du photon, on a $\Vec{p}_\gamma = p_\gamma \Vec{u}$, et donc :
  $$
  \Etot_2 \Etot_\gamma - \Vec{p}_2 \cdot \Vec{p}_\gamma c^2 = \Etot_\gamma (\Etot_2 - (\Vec{p}_2 \cdot \Vec{u}) c) .
  $$
  En conclusion, on trouve 
  $$
  \Etot_2 = (\Vec{p}_2 \cdot \Vec{u}) c \leq p_2 \,c < \Etot_2,
  $$
  d'où la contradiction.
  }
  \end{exo}

\begin{exo}[PC][1][colle]{Réalisation d'un faisceau de neutrinos}
Pour réaliser certaines expériences de physique des particules, il est nécessaire de disposer d'un faisceau de neutrinos, particules neutres. Or, il n'est pas simple de focaliser un faisceau de particules neutres : on ne peut pas utiliser des champs électromagnétiques comme on le fait pour des particules chargées. Le but de cet exercice est de montrer comment fabriquer un faisceau bien focalisé de neutrinos.

On utilise pour cela la désintégration d'un pion $\pi^{+}: \pi^{+} \rightarrow \mu^{+}+\nu_\mu$. On donne : $m_{\pi^{+}}=140 \mathrm{MeV} / c^2, m_{\mu^{+}}=106 \mathrm{MeV} / c^2$. La masse du $\nu_\mu$ peut être considérée comme nulle. On suppose disposer d'un faisceau de pions se propageant selon l'axe $O x$ du laboratoire, avec une énergie de 140 GeV .
\begin{quest}
  \item Que valent les paramètres $\beta$ et $\gamma$ du pion dans le référentiel du laboratoire? Donner la vitesse du neutrino, ainsi que la relation entre son énergie et son impulsion dans le référentiel du laboratoire. Qu'en est-il dans le référentiel du centre de masse?
  \item On suppose que l'impulsion du neutrino fait un angle $\theta^{\prime}$ avec l'axe $O x$ dans le référentiel du centre de masse. Que vaut l'angle $\theta$ dans le référentiel du laboratoire? Faire l'application numérique pour $\theta^{\prime}=\pi / 4, \theta^{\prime}=\pi / 2, \theta^{\prime}=3 \pi / 4, \theta^{\prime}=99 \pi / 100, \theta^{\prime}=\pi$. Qu'en concluez-vous pour la focalisation du faisceau?
  \item Montrer que l'énergie du muon et du neutrino sont fixées dans le référentiel du centre de masse. En déduire l'énergie maximale et l'énergie minimale du neutrino dans le référentiel du laboratoire.
\end{quest}
\end{exo}

\begin{exo}[PC][1][devoir]{Désintégration d'un méson (4 points)}
  % \Notions{\item Collision inélastique }
  \Enonce{
Un méson $K^{+}$ au repos dans le référentiel inertiel du laboratoire $(\Rcal)$ se désintègre en plusieurs pions chargés suivant la réaction 
\begin{equation}
  K^+ \rightarrow \alpha \pi^{ +} + \beta \pi^ -,
\end{equation}
avec $\alpha$ et $\beta$ des entiers. On donne les masses $m_{K}=\SI{493.7}{MeV.c^{-2}}$, et $m_{\pi}=\SI{139.6}{MeV.c^{-2}}$.
  }
\begin{quest}
  \QR {Montrer qu'il peut se désintégrer au plus en trois pions. En déduire les valeurs possibles de $(\alpha, \beta)$.}
  {
    Appliquons la conservation de l'énergie et de la quantité de mouvement dans le référentiel du laboratoire, où le méson est au repos. On a :
    $$
    m_K c^2 = \Etot_{\rm tot} = \sum_i \Etot_i = \sum_i \gamma_i m_\pi c^2 \geq \sum_i m_\pi c^2 = n m_\pi c^2,
    $$
    où $n=\alpha + \beta$ est le nombre total de pions produits. On en déduit 
    \eqbox{
    n \leq \frac{m_K}{m_\pi},
    }
    d'où le résultat. La conservation de la charge impose de plus 
    $$
    \alpha - \beta = 1 \Rar \alpha = \frac{n+1}{2}, \quad \beta = \frac{n-1}{2},
    $$
    d'où les valeurs possibles ,
    \eqbox{
    n = 1 \Rightarrow (\alpha, \beta) = (1,0), \quad n = 3 \Rightarrow (\alpha, \beta) = (2,1).
    }
    Les mésons ne sont pas soumis à d'autres contraintes de conservation, puisqu'ils sont composés d'un quark et d'un antiquark.
  }

  \QR {On suppose qu'un parmi trois des pions produits est immobile dans ($\Rcal$). Déterminer son énergie $\Etot_1$, ainsi que les énergies $\Etot_2$ et $\Etot_3$ des deux autres pions.}
  {
    \begin{enumerate}
      \item On a directement 
      \eqbox{
        \Etot_1 = m_\pi c^2 = \SI{139.6}{MeV}. 
      }
      \item
      La \udl{conservation de l'énergie} dans $\Rcal$ s'écrit
      $$
      m_K c^2 = \Etot_1 + \Etot_2 + \Etot_3 = m_\pi c^2 + \Etot_2 + \Etot_3 \Rar \Etot_2 + \Etot_3 = (m_K - m_\pi) c^2.
      $$
      La \udl{conservation de la quantité de mouvement} s'écrit
      $$
      \Vec{p}_1 + \Vec{p}_2 + \Vec{p}_3 = \Vec{0} \Rar \Vec{p}_2 = - \Vec{p}_3 \Rar p_2 = p_3.
      $$
      Enfin, la relation énergie-impulsion pour les pions 2 et 3 donne
      $$

      $$
      \item En repartant de la conservation de l'énergie, on trouve finalement 
      \eqbox{
        \Etot_2 = \Etot_3 = \frac{(m_K - m_\pi) c^2}{2} = \SI{177.05}{MeV}.
      }
    \end{enumerate}
  }

\end{quest}
