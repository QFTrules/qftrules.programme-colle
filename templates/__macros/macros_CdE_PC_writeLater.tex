\newcommand\immediateprotectedwrite[2]
  {%
    \begingroup
      \let\thepage\relax
      \let\protect\@unexpandable@protect
      \edef\reserved@a{\immediate\write#1{#2}}%
      \reserved@a
    \endgroup
  }
% ->->->->->-> %
\makeatother

% ----------------------------------------------- %
%    Removing \belowdisplayshortskip or \belowdisplayskip 
%    when it is at the end of a given environment.
% ----------------------------------------------- %
% https://tex.stackexchange.com/q/608903/8323


\def\miniskip{\vspace{1.5pt plus 1pt minus 1pt}}

\def\removeBelowDisplayskip
{\par%
 \ifdim\lastskip=\belowdisplayshortskip\vskip-\lastskip\miniskip\fi%
 \ifdim\lastskip=\belowdisplayskip\vskip-\lastskip\miniskip\fi
 }

\def\corrige{%
  \immediateprotectedwrite{\fich@corr}{\protect\printNumberExercise{\printNumeroDeCalul}}
  \@bsphack
  \let\do\@makeother\dospecials
  \catcode`\^^M\active \catcode`\^^I=12
  \def\verbatim@processline{%
    \immediate\write\fich@corr
      {\the\verbatim@line}}%
  \verbatim@start}
\def\endcorrige{%
  \immediateprotectedwrite{\fich@corr}{\protect\removeBelowDisplayskip{\protect\tightdotfill}}
  \immediate\write\fich@corr{}
  \@esphack}


\def\corrigeNewpage{%
\immediateprotectedwrite{\fich@corr}{\protect\newpage}
\corrige}
\def\endcorrigeNewpage{\endcorrige}
