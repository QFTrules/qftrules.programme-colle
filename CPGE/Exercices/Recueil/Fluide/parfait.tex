\begin{exo}[PC][1][colle]{Écoulement sous la couverture d’un livre}
  On souhaite décrire l’écoulement d’air sous la couverture d’un livre qui se ferme sous l’effet de la pesanteur. On note $\alpha(t)$ l’angle que forme la couverture du livre par rapport à l’horizontale. On suppose l’écoulement irrotationnel, parfait, incompressible.

  \begin{fig}[label]
    \centering
    \includegraphics[width=0.5\columnwidth]{diedre.pdf}
    %\capt{}
  \end{fig}
On considère pour commencer que la couverture du livre est immobile, d’angle constant $\alpha$.
% et on suppose l’écoulement stationnaire.
  \begin{quest}
    \item	On cherche un potentiel des vitesses de la forme $\phi(r,\theta) = A r^n \cos(n\theta)$ en coordonnées polaires, avec $n$ une constante. Montrer que ce potentiel des vitesses est compatible avec la nature de l’écoulement.
    \sol{
      L’écoulement est irrotationnel et incompressible, donc l’écoulement doit vérifier l’équation de Laplace,
      $$
      \Delta \phi = 0.
      $$
      Avec le potentiel choisi, on a bien 
      $$
      \Delta \phi = \frac{1}{r}\dr{}{r}\left( r\dr{\phi}{r} \right) + \frac{1}{r^2} \ddr{\phi}{\theta} = A n^2 r^{n-2} \cos(n\theta) - An^2 r^{n-2} \cos(n\theta) = 0.
      $$
    }
    \item Déterminer $n$ pour que ce potentiel des vitesses soit compatible avec les conditions aux limites. 
    \sol{
      Le champ des vitesses s’écrit 
      $$
      \Vec{v} =  \grad \phi = \dr{\phi}{r} \er  + \frac{1}{r} \dr{\phi}{\theta} \et = A nr^{n-1} \cos(n\theta) \er -  A n r^{n-1} \sin(n \theta) \et.
      $$
      La vitesse normale aux plaques doit être nulle,
      soit 
      $$
      v_\theta(r,0) = 0 = A n r^{n-1} \sin(0), 
      $$
      et 
      $$
      v_\theta(r,\alpha) = 0 = A n r^{n-1} \sin(n \alpha),
      $$
      soit 
      \eqbox{
      n = \frac{\pi}{\alpha}.
      }
      Le champ des vitesses a donc pour expression 
      \eqbox{
      \Vec{v}(M) = \frac{A\pi}{\alpha}r^{\frac{\pi}{\alpha}-1} \left[ \cos\left(\pi \frac{\theta}{\alpha}\right) \er + \sin\left(\pi \frac{\theta}{\alpha}\right) \et\right].
      }
      La norme de la vitesse en un point $M$ est donc constante égale à 
      $$
      v(M) = \frac{A\pi}{\alpha} r^{\frac{\pi}{\alpha}-1}.
      $$
    }
      \item Que vaut le champ de pression $p(M)$ en tout point ?
      \sol{
        On utilise la relation de Bernoulli pour un écoulement parfait, irrotationnel, incompressible, stationnaire.
$$
p_0 + \frac{1}{2} \rho v^2 = p(M) + \frac{1}{2} \rho v^2(M),
$$
et donc 
\eqbox{
p(M) = p_0.
}
    }
  \end{quest}
On suppose maintenant que l’angle de la couverture $\alpha(t)$ dépend du temps.
\begin{quest}
  \item	Quelles sont les nouvelles conditions aux limites ? En cherchant un potentiel des vitesses de la même forme que précédemment, déterminer la valeur de $n$ et l’expression complète du champ des vitesses.
  \sol{
    On a cette fois-ci 
    $$
    v_\theta(r,\alpha(t),t) = r \dot{\alpha}(t) = -A(t) n r^{n-1} \sin(n\alpha(t)),
    $$
    soit 
    \eqbox{
    n=2,
    }
    et 
    $$
    A(t) = -\frac{\dot{\alpha}(t)}{2\sin(2\alpha(t))}.
    $$
    Le champ des vitesses vaut donc 
    \eqbox{
    \Vec{v}_{\rm mob}(M,t) = \frac{r\dot{\alpha}}{\sin(2\alpha)}\left( - \cos(2\theta) \er + \sin(2\theta) \et \right) = - \frac{r\dot{\alpha}}{\sin(2\alpha)} \left( \cos(\theta) \ex - \cos(3\theta) \ey \right).
    }
    Sa norme vaut 
    $$
    v_{\rm mob} = \frac{r|\dot{\alpha}|}{\sin(2\alpha)}.
    $$
    Le potentiel des vitesses vaut 
    $$
    \phi_{\rm mob}(r,\theta,t) = - \frac{r^2 \dot{\alpha}}{2\sin(2\alpha)} \cos(2\theta).
    $$
    }
    % \item Justifier qu’en superposant le champ des vitesses trouvé ci-dessus à celui obtenu pour une plaque immobile, le champ des vitesses est encore solution du problème. En déduire l’expression générale du champ des vitesses, ainsi que la norme carrée de la vitesse.
    % \sol{
    %   Le champ des vitesses pour la plaque immobile s’écrit 
    %   $$
    % \Vec{v}(M)_{\rm im} = \frac{A}{\alpha} r^{\frac{\pi}{\alpha}-1} \left[ \cos(\pi \frac{\theta}{\alpha}) \er + \sin(\pi \frac{\theta}{\alpha}) \et\right].
    %   $$
    %   Le champ des vitesses total est donc
    %   % $$
    %   \eqbox{
    %     \Vec{v_{\rm tot}}(M,t) = \Vec{v}(M)_{\rm im} + \Vec{v}_{\rm mob}(M,t) = \frac{A}{\alpha} r^{\frac{\pi}{\alpha}-1} \left[ \cos(\pi \frac{\theta}{\alpha}) \er + \sin(\pi \frac{\theta}{\alpha}) \et\right] - \frac{r\dot{\alpha}}{\sin(2\alpha)}\left( \cos(2\theta) \er - \sin(2\theta) \et \right).
    %     }
    %     % $$
    %   La norme carrée de la vitesse vaut 
    %   $$
    %   v_{\rm tot}^2 = v_{\rm im}^2 + v_{\rm mob}^2 + 2 \Vec{v}_{\rm im} \cdot \Vec{v}_{\rm mob},
    %   $$
    %   soit 
    % \eqbox{
    %      v_{\rm tot}^2 =
    %   \left( \frac{A}{\alpha} \right)^2 r^{2\left(\frac{\pi}{\alpha}-1\right)} + \frac{r^2 \dot{\alpha}^2}{\sin^2(2\alpha)} - 2 \frac{A}{\alpha} \frac{r^{\frac{\pi}{\alpha}} \dot{\alpha}}{\sin(2\alpha)} \cos(\theta(\pi/\alpha - 2)).
    %   }
    %   Le potentiel total des vitesses vaut 
    %   $$

    %   $$
    % }
    \item On admet que la relation de Bernoulli se généralise aux écoulements instationnaires sous la forme
    \begin{equation}
\dr{\phi}{t} + \frac{1}{2} v^2 + \frac{p}{\rho} = \cte.
    \end{equation}
    Déterminer la pression $p(M,t)$ s’exerçant sous la couverture du livre.
    \sol{
    Considérons un point $\rm A$ de l’écoulement sur l’axe horizontal, où la pression vaut $p_0$, et un point $\rm B$ sous la couverture du livre. On a
    $$
    p_0 + \frac{1}{2} \rho v_{\rm A}^2 + \rho \dr{\phi_A}{t} = p(B,t) + \frac{1}{2} \rho v_{\rm B}^2 + \rho \dr{\phi_{\rm B}}{t},
    $$
    soit 
    $$
    p(B,t) = p_0 + \rho \left( \dr{\phi_A}{t} - \dr{\phi_{\rm B}}{t} \right) = p_0 
    % + \frac{1}{2} - 2 \frac{A}{\alpha} \frac{r^{\frac{\pi}{\alpha}}\dot{\alpha}}{\sin(2\alpha)} \rho \left(  1  -  \cos(\pi - 2\alpha) \right) + 
 + \rho r^2 \left( \frac{\ddot{\alpha}}{2\sin(2\alpha)} - \frac{\cos(2\alpha)\dot{\alpha}^2}{\sin(2\alpha)^2} \right)(\cos(2\alpha)-1),
    $$
    soit 
    \eqbox{
    p(M,t) = p_0 - \rho r^2 \left( \frac{\ddot{\alpha}}{4}\tan(\alpha) - \frac{\dot{\alpha}^2}{4} (1-\tan(\alpha)^2)\right).
    }
    }
\end{quest}

\begin{exo}[PC][1][devoir]{Le tube de Pitot}
  Le tube de Pitot est un des nombreux capteurs qui équipent l'avion. Il permet la mesure de la vitesse de l'avion, donnée essentielle à sa bonne conduite. Il s'agit d'un tube très fin (moins de $5 \mathrm{~mm}^{2}$ de surface) qui est placé parallèlement à la direction de l'écoulement de l'air (\Fig{pitot}). Ce tube possède deux ouvertures en $F$ et $G$. L'ouverture en $F$ est la prise dite de pression totale et celle en $G$ est la prise dite de pression statique. On mesure la différence de pression de l'air entre les deux tubes 1 et 2 avec un manomètre différentiel, ce qui permet d'obtenir la vitesse $v_{\infty}$ de l'écoulement.
  \begin{fig}[pitot]
    \centering
    \includegraphics[width=0.4\textwidth]{2023_11_08_92f046a700805e4eaca0g-1}
    \capt{Tube de Pitot.}
  \end{fig}

On considère que l'air est un fluide parfait, homogène, incompressible, de masse volumique $\rho_{\infty}$ et en écoulement stationnaire. On rappelle que les effets de la gravité sur l'air sont négligés. Loin du tube l'air a une pression $P_{\infty}$ et une vitesse $v_{\infty}$.
\begin{quest}
  \item Représenter l'allure de la ligne de courant qui aboutit en $F$ et l'allure de la ligne de courant qui longe le tube et passe à proximité de $G$.
  
  \item Déterminer, en fonction de $P_{\infty}, \rho_{\infty}$, et $v_{\infty}$, les expressions de la vitesse $v_{F}$ et de la pression $P_{F}$ du fluide en $F$ ainsi que la vitesse $v_{G}$ et la pression $P_{G}$ du fluide en $G$.
  
  \item Dans le manomètre, il y a un liquide de masse volumique $\rho_{l}$. On mesure une différence d'altitude $h$ entre les deux surfaces du liquide. Déterminer l'expression de la différence de pression, $P_{H}-P_{I}$, entre ces deux surfaces.
  
  \item Déduire des questions précédentes l'expression de la vitesse de l'écoulement $v_{\infty}$ de l'air en fonction de $\rho_{l}, \rho_{\infty}, g$ et $h$. Comment évolue $h$ lorsque la vitesse de l'air augmente ?
\end{quest}

\end{exo}

\begin{exo}[PC][2][TD]{Problème de robinets}
  \Notions{\item Relation de Bernoulli \task Débit volumique \task Torricelli}
  Une baignoire de volume $V=96 \mathrm{~L}$ se remplit en $\tau_{1}=8\,\mathrm{min}$, robinet ouvert et bonde fermée. Elle se vide en $\tau_{2}=12\,\mathrm{min}$, bonde ouverte et robinet fermé. On se propose de répondre à la question suivante :  \emph{En combien de temps la baignoire se remplit-elle complètement si la bonde et le robinet sont ouverts ?}
\begin{fig}[]
  \centering
  \includegraphics[width=0.5\columnwidth]{2023_10_11_e0b3d6ffc7f03dae9839g-1}
  % \capt{Schéma de la baignoire}
\end{fig}

  \begin{quest}
    \item Un étudiant donne la réponse suivante :
  \emph{Le robinet débite $96 / 8=12 \mathrm{~L} / \mathrm{min}$, la bonde $96 / 12=8 \mathrm{~L} / \mathrm{min}$. Bonde et robinet ouverts, le débit sera égal à $12-8=4 \mathrm{~L} / \mathrm{min}$, il faudra donc $96 / 4=24\, \mathrm{min}$ pour que la baignoire se remplisse.}
  Expliquer en quoi le raisonnement précédent est faux.
  \sol{
    On ne peut pas appliquer une simple règle de trois car le débit de vidange n'est pas constant. Notamment, le débit de vidange dépend du niveau d'eau : il sera donc différent lorsque l'on ouvre le robinet.
  }
\end{quest}
Traitons le problème en supposant que la baignoire est un récipient cylindrique de section $S$ et de hauteur $H$. La bonde ouverte fait naître un orifice de section $s \ll S$. À l'instant $t=0$ on ouvre robinet et bonde, la baignoire étant vide.
\begin{quest}
    \item
  Exprimer le débit volumique ${{\DV}}_1$ sortant par le robinet en fonction de $\tau_{1}$ et $V=SH$. Faire l'application numérique.
\sol{Le débit d'eau sortant du robinet est constant et égal à
$$
{{\DV}}_1=S H / \tau_1=12 \mathrm{~L} \cdot \mathrm{mn}^{-1}
$$
}

    \item Supposons qu'à l'instant $t$, le niveau d'eau dans la baignoire soit $z(t)$. À partir de la formule de Torricelli, trouver le débit volumique ${\DV}_2$ sortant par la bonde.
     \sol{
      En appliquant la relation de Bernoulli entre un point de la surface libre (de vitesse négligeable) et un point de l'orifice on obtient $\textstyle\frac{1}{2} \mu v^2=\mu g z$ d'où la formule de Torricelli
      $$
      v=\sqrt{2 g z} \Rightarrow {{\DV}}_2=s \sqrt{2 g z(t)}
      $$
    }

    \item En traduisant la conservation de la masse, montrer que le niveau d'eau vérifie l'équation différentielle
  \begin{equation}
    \dot{z}(t)=\frac{1}{S}\left( {{\DV}}_1-s \sqrt{2 g z(t)} \right).
  \end{equation}
  % Commenter.

  \sol{
    La masse d'eau contenue dans la baignoire vaut $m(t)=\mu S z(t)$. La conservation de la masse se traduit par
$$
\frac{\mathrm{d} m}{\mathrm{~d} t}=\mu\left(Q_{V 1}-Q_{V 2}\right)
$$
d'où
$$
\dot{z}(t)=\frac{1}{S}\left(Q_{V 1}-s \sqrt{2 g z(t)}\right)
$$
$\dot{z}>0$ donc $z$ croît : la baignoire déborde au bout d'un certain temps.

  }

    \item Définissons $z_{\infty}$ tel que ${{{\DV}}}_1=s \sqrt{2 g z_{\infty}}$. Exprimer $\tau_2$ puis calculer $z_{\infty}/H$. Conclusion ?
    
    \sol{
    Indication : Exprimer $\tau_{2}$ en fonction de $s, S, g$ et $H$ en intégrant l'équation (3.1) dans le cas où le robinet est fermé, puis éliminez $s$ dans la définition de $z_{\infty}$.
    Que représente $z_{\infty}$ ?
    Montrer que
    $
    z_{\infty}=H\left(\frac{\tau_{2}}{2 \tau_{1}}\right)^{2}.
    $

    Lorsque $Q_{V 1}=s \sqrt{2 g z}, \dot{z}=0$ : le niveau d'eau se stabilise à la valeur
$$
z_{\infty}=\frac{Q_{V 1}{ }^2}{2 g s^2}=\frac{V^2}{2 g s^2 \tau_1^2}
$$

Dans ce cas, la baignoire ne se remplit que partiellement si $z_{\infty}<H$.


    $\tau_2$ représente le temps de vidange, robinet fermé. On obtient ce temps en intégrant l'équation précédente avec $Q_{V1}=0$ :
    $$
    \dot{z}(t)=-\frac{s}{S} \sqrt{2 g z(t)} \quad \Rightarrow \quad 2 \sqrt{z}=2 \sqrt{H}-\frac{s}{S} \sqrt{2 g} t
    $$
    
    Lorsque $z=0, t=\tau_2$ d'où
    $$
    s^2=\frac{2 H S^2}{g \tau_2^2} \quad \Longrightarrow \quad z_{\infty}=H\left(\frac{\tau_2}{2 \tau_1}\right)^2
    $$
    $z_{\infty}=0,56 H$. La baignoire ne se remplit que partiellement (un peu plus de la moitié).
    L'équation différentielle peut se réécrire
    $$
    \dot{z}=\frac{1}{S}\left(s \sqrt{2 g z_{\infty}}-s \sqrt{2 g z}\right)=\frac{2 \sqrt{H}}{\tau_2}\left(\sqrt{z_{\infty}}-\sqrt{z}\right)
    $$
    }
      % \item Calculer $z_{\infty} / H$ et répondre à la question initiale.

      % \item Au bout de combien de temps la baignoire se remplit si $\tau_{1}=4 \min$ et $\tau_{2}=12\,\mathrm{min}$ ?

  \end{quest}
\end{exo}

\begin{exo}[PC][2][oral]{Hauteur d'un jet d'eau}%{1}
Un jet d'eau sort d'une buse verticale avec une vitesse $v_0\,\ez$ uniforme de norme $v_0 = \SI{2}{m.s^{-1}}$ dans la section de sortie confondue avec le disque de rayon $a$ centré en un point $O$ pris comme origine de l'espace. Le champ de pesanteur $\Vec{g}=-g\,\ez$ est uniforme, d'intensité $g=\SI{9.8}{m.s^{-2}}$.
%L'écoulement est supposé parfait, stationnaire, incompressible et homogène de masse volumique $\rho$.
\begin{center}
\includegraphics[width=0.3\columnwidth]{jet_eau.pdf}
\end{center}
\noindent En précisant les hypothèses, estimer quelle hauteur atteint le jet d'eau.
%Pour quelle valeur d'altitude $z$ le profil diverge-t-il ? Application numérique pour $v_0 = \SI{2}{m.s^{-1}}$.

\sol{

\begin{liste}
  \item[\encircle{S}] 
  \begin{itemize}
    \item \textsc{Données du problème} : la vitesse initiale $v_0 = \SI{2}{m.s^{-1}}$, l'intensité de la pesanteur $g = \SI{9.8}{m.s^{-2}}$, et le rayon $a$ de la section initiale. D'autres données importantes sont probablement la masse volumique de l'eau $\rho = \SI{e3}{kg.m^{-3}}$, et la pression atmosphérique $p_0 = \SI{1}{bar}$.
    \item \textsc{Schéma} : le schéma est déjà fourni et semble indiquer que le rayon du jet $R(z)$ joue également un rôle important.
    \item \textsc{Paramètre à déterminer} : hauteur $h$ du jet d’eau. 
  \end{itemize}  
\end{liste}
}
\sol{
  \begin{liste}
  \item[\encircle{A}] 
  \begin{itemize}
    \item \textsc{Cadre de l’étude} : il s'agit d'un problème de dynamique des fluides. 
    \item \textsc{Hypothèses} : en première approximation, il est légitime de supposer l'écoulement parfait, car la viscosité n’est pas le moteur de l’écoulement. Nous supposerons aussi l'écoulement stationnaire, et incompressible, de sorte à pouvoir utiliser les lois usuels des écoulements parfaits (conservation du débit et Bernoulli). On peut proposer un champ des vitesses de la forme $\Vec{v}(M)=v_{r}(r,z)\,\er+v_{z}(z)\,\ez$ en coordonnées cylindriques, avec $v_r\ll v_z$ pour la partie où le jet n'est pas trop évasé.
    \item \textsc{Analyse dimensionnelle} : supposons que la hauteur $h$ dépende seulement de $v_0$, $g$, et $\rho$. On a alors 
    $$
  h \sim v_0^\alpha g^\beta \rho^\gamma.
  $$
  La dimension de $h$ est alors 
  $$
  [h] = L = [v_0]^\alpha [g]^\beta [\rho]^\gamma = (L\cdot T^{-1})^{\alpha} (L\cdot T^{-2})^{\beta} (M\cdot L^{-3})^{\gamma} = L^{\alpha+\beta - 3\gamma} M^{\gamma} T^{-\alpha-2\beta}.
  $$
  On en déduit $\alpha = 2$, $\beta = -1$, et $\gamma = 0$. Donc 
  \eqbox{
    h \sim \frac{v_0^2}{g}. 
  }
\item \textsc{Analyse semi-quantitative} :
  On peut trouver le préfacteur numérique par analogie avec le problème de la vidange. Dans un fluide parfait, si l'on ne tient pas compte de la pression, chaque particule de fluide évolue dans le champ de pesanteur comme un point matériel. On peut donc considérer que chaque particule subit une chute libre. C'est d'ailleurs cohérent avec la loi de Torricelli, qui donne $v_B = \sqrt{2gh}$ ! Même si la loi de Torricelli s'applique a priori dans un problème légèrement différent, la loi de la chute libre s'applique probablement ici aussi, de sorte que $v_0 = \sqrt{2gh}$, ce qui donne 

  \eqbox{
    h = \frac{v_0^2}{2g} = \SI{20}{cm}.
  }
  
  \item \textsc{Analyse quantitative} : on peut maintenant proposer un argument quantitatif pour confirmer notre résultat. On définit pour cela une stratégie de résolution.
  \begin{enumerate}
    \item Appliquer Bernoulli sur une ligne de courant au bord du jet pour trouver la fonction $v_z(z)$. 
    \item Optionnel : appliquer la conservation de ${{\DV}}$ pour trouver le rayon $R(z)$ en fonction de $v_z(z)$.
    \item Étudier les fonctions $v_z(z)$ ou $R(z)$ pour trouver $h$.
  \end{enumerate} 
\end{itemize}
\end{liste}
  }
  \sol{
    \begin{liste}
      \item[\encircle{R}] 
      \begin{itemize}
        \item 
        On applique la loi de Bernoulli le long d'une ligne de courant  $A$-$B$ sur la surface libre du jet :
        $$
        \frac{1}{2} \rho v_A^2 + p_A + \rho g z_A = \frac{1}{2} \rho v_B^2 + p_B + \rho g z_B.
        $$
        On en déduit comme $v_A \simeq v_z(z)$ et $p_A = p_B = p_0$ que
        $\frac{1}{2}\rho v_z^2  + \rho g z = \frac{1}{2}\rho v_0^2 $. Ainsi, on a 
        \eqbox{
        v_z(z) = \sqrt{v_0^2 - 2gz}.
        }
        \item La conservation du débit volumique donne ${{\DV}} = v_z(z) S(z) = \pi R(z)^2 v_z(z) = \pi a^2 v_0$, et donc
        \eqbox{
        R(z)
        = a  \left(1-\frac{2gz}{v_0^2} \right)^{1/4}.
        }
        \item Le jet atteint sa hauteur maximale lorsque $v_z(h)=0$, soit 
        \eqbox{
        h  = \frac{v_0^2}{2g} = \SI{20}{cm}.
        }
        On remarque que $R(z)$ diverge en $z=h$.
      \end{itemize}
    \end{liste}
  }
  \sol{
    \begin{liste}
  \item[\encircle{V}] On retrouve la même expression qu'avec l'argument semi-quantitatif, ce qui est encourageant ! L'ordre de grandeur semble tout à fait raisonnable si l'on compare aux situations de la vie courante.
\end{liste}

}
\end{exo}

$p_B -p_0 = \frac{\rho_a v^2}{2}$

}
\end{quest}



\subsubsection{La clepsydre}

Un récipient à symétrie cylindrique est initialement rempli, sur une
hauteur $h_0$, d’un liquide de masse volumique $\rho$. Le
liquide s’écoule par une ouverture de section $s$ à la base du récipient. Le récipient est de rayon $R(h)$ à la hauteur $h$.
L'écoulement est supposé de plus stationnaire et irrotationnel.

\begin{fig}
\includegraphics[width=0.3\textwidth]{clepsydre.png}
\end{fig}

\begin{quest}
\item La section $s$ est négligeable devant la section $S$ de la surface libre.  Que dire de la vitesse de la surface libre du fluide par rapport à  la vitesse $v$ d’écoulement au niveau de l'ouverture ?
\sol{
Débit volumique conservé donc $sv = Sv_A$ et donc $v_A \ll v$.
}
\item Exprimer alors la vitesse $v$  en fonction de la hauteur $h$.
\sol{
Bernoulli en tout point du fluide : $\frac{\rho v^2}{2} + p_0 = p_0 + \rho g h $ et donc $v = \sqrt{2gh}$.
}
\item Que vaut le volume $\d V$ du fluide compris dans une épaisseur $\d h$ au niveau de la surface libre ? En calculant le débit volumique ${{\DV}}$ de deux façons, déterminer une équation différentielle définissant l’évolution de la hauteur $h(t)$ au cours du temps.
\sol{
$\d V = \pi R^2 \d h$ et donc
${{\DV}} = - \dd{V}{t} = -\pi R^2 \dd{h}{t} = s v = s \sqrt{2gh}$, d'où
$$
\dd{h}{t} = - \frac{s\sqrt{2gh}}{\pi R^2}
$$
}
\item Quelle devrait être la relation $R(h)$ entre le rayon $R$ du vase et la hauteur $h$ pour que $h(t)$ soit une
fonction affine du temps ? Expliquer le fonctionnement d'une clepsydre.
\sol{
On doit avoir $\sqrt{h}/R^2=\cte$, soit $R(h) = A h^{1/4}$ avec $A$ une constante.
}
\item À quelle condition est-il légitime de supposer cet écoulement stationnaire ?

\sol{
On compare les accélérations locales et convective en dessous de la clepsydre.
$$
\norme{\dr{\Vec{v}}{t}} \sim \frac{v}{\tau} \sim \frac{v v_A}{h_0}
$$
$$
  \norme{(\Vec{v}\cdot \grad \Vec{v}) \Vec{v}} \sim \frac{v^2}{h_0}
$$
Donc le rapport vaut
$$
  \frac{\norme{\dr{\Vec{v}}{t}} }{\norme{(\Vec{v}\cdot \grad \Vec{v}) \Vec{v}}} \sim \frac{v_A}{v} \sim \frac{s}{S}.
$$
Il faut $s \ll S$.
}

%\item Reprendre le calcul en tenant compte de la vitesse du fluide à la surface libre.
\end{quest}

\subsubsection{Le tube en U}
Un volume $V=\SI{20}{mL}$ d'eau remplit un tube en U de section constante $S=\SI{1}{cm^2}$.
On note $h(t)=z_B(t)-z_A(t)$ la dénivellation entre les branches droite et gauche.
%On suppose l'écoulement parfait, incompressible, homogène.
On note $x$ l'abscisse curviligne définie le long du tube, en suivant la ligne pointillée,
L'écoulement est décrit par un champ des vitesses de la forme $\vv{v}(M,t)=v(x,t)\vv{T}$
uniforme sur une section du tube, où $\vv{T}$ désigne le vecteur unitaire tangent.
On note $L$ la longueur totale de fluide dans le tube, c'est-à-dire la longueur de la ligne pointillée ci-dessous.
\begin{center}
\includegraphics[width=0.22\textwidth]{tube_U.pdf}
\end{center}

\begin{quest}
  \item Calculer numériquement $L$.
  \sol{
  \eqbox{
    L=\frac{V}{S}=\SI{20}{cm}
  }  
  }
  \item Montrer que la vitesse $v(x,t)$ ne dépend pas de $x$ en réalité.
  \sol{
  Le liquide est \fbox{incompressible} et donc le débit volumique ${{\DV}}(x,t) = v(x,t)S$ ne dépend pas de $x$, et donc $v$ non plus.
  }
\item Justifier que l'écoulement est irrotationnel.
Montrer ensuite que l'écoulement est bien décrit par un potentiel des vitesses de la forme $\phi(x,t) = v(t) x$.
\sol{
On trouve  $\rot \Vec{v} = \Vec{0}$ donc l'écoulement est irrotationnel. Il existe un potentiel des vitesses tel que $\Vec{v} = \grad \phi = v(t) \Vec{T}$. En négligeant la courbure du tube, on peut assimiler $x$ à une variable cartésienne, et donc $\Vec{T} = \ex$ et donc  $\dr{\vphi}{x} = v(t)$ estime
\eqbox{
  \phi(x,t) = v(t) x.
}
}
\item Relier la vitesse $v(t)$ à la dénivellation $h(t)$.
\sol{
La dénivellation est reliée à la vitesse par $v(t) = v_A(t) = \dd{z_A}{t}$. Or $\dd{z_B}{t} = -\dd{z_B}{t}$ et donc $\dot{h}(t)=2v(t)$, soit 
\eqbox{
v(t) = \frac{1}{2} \dd{h}{t}.
}
}
\item On admet que la relation de Bernoulli se généralise aux écoulements non stationnaires comme
$$
\rho \dr{\phi}{t} + \frac{\rho v^2}{2} + p + \rho g z = \cte.
$$
En appliquant cette relation entre les points $A$ et $B$, établir l'équation différentielle à laquelle obéit la dénivellation $h(t)$
et déterminer la période $T_0$ des oscillations. Application numérique.
\sol{
On a
$$
  \rho x_A \dd{v}{t} + \frac{\rho v^2}{2} + \rho g z_A = \rho x_B \dd{v}{t} + \frac{\rho v^2}{2} + \rho g z_B \Rightarrow  L \dd{v}{t} + g h = 0.
$$
Or $v(t) = \frac{1}{2}\dd{h}{t}$ et donc
$$
  \ddm{h}{t}{2} + \underbrace{\frac{2 g}{L}}_{\omega_0^2} h = 0.
$$
On obtient l'équation d'un OH :
\eqbox{
  \ddm{h}{t}{2} + \omega_0^2 h = 0,
}
avec lu pulsation propre $\omega_0 = \sqrt{\frac{2g}{L}} = \SI{9.9}{rad.s^{-1}}$ puis $T_0=\frac{2\pi}{\omega_0} = \SI{0.63}{s}$.
}
\end{quest}

\end{exo}

\begin{exo}[PC][2][TD]{Les vents géostrophiques}
  \Notions{\item Statique des fluides \task Force de Coriolis}
  \Annale[Physique PC]{X}{2002}
On étudie la dynamique des mouvements atmosphériques dans le référentiel terrestre $\RT$, supposé non galiléen. Le référentiel géocentrique $\RG$ est supposé galiléen. Les masses d'air évoluent dans le champ de pesanteur terrestre $\Vec{g}$. Un cyclone est une zone de basse pression atmosphérique autour de laquelle les masses d'air tournent. On souhaite expliquer le sens de rotation des cyclones, tel que celui observé sur la photographie du cyclone Idai de 2019 ci-dessous [Wikipedia]. 
\begin{fig}[cyclone]
  \centering
  \includegraphics[width=0.4\textwidth]{cyclone_ex}
  % \capt{}
\end{fig}
\begin{quest}
  \item Décrire le mouvement de $\RT$ par rapport à $\RG$ en s'appuyant sur un schéma. Définir une base cartésienne attachée au référentiel terrestre.
  \sol{
    Voir cours sur la dynamique dans le référentiel terrestre.
    \begin{center}
      \includegraphics[width=0.5\textwidth]{ref_terrestre_3}
    \end{center}
  }
  % \item Déterminer l'expression de la force volumique de Coriolis $\Vec{f_{\rm c}}$ dans un fluide de masse volumique $\rho$. Pourquoi ne discute-t-on pas de la force d'entraînement ?
  \item Quel(s) phénomène(s) physique(s) est à l'origine des mouvements verticaux de masse d'air ? 
  \sol{
  Les déplacements verticaux sont induits par les différences de température. L'air à basse altitude est plus chaud, car le sol est chauffé par le soleil, tandis que l'air est plus froid à haute altitude. Les reliefs influent également sur les mouvements des masses d'air.
  }
\end{quest}
Dans la suite, on s'intéressera uniquement aux mouvements horizontaux des masses d'air. Une particule de fluide, de vitesse $\vRp = v_x \ex + v_y \ey$ dans $\RT$, se déplace suffisamment lentement aux grandes échelles pour pouvoir appliquer la loi de la statique des fluides. Les vents vérifiant cette condition sont appelés \emph{vents géostrophiques}.
\begin{quest}
  \item Montrer par un schéma que dans
l’hémisphère nord, une masse d'air en mouvement est
déviée vers sa droite, alors que qu'elle est déviée vers la gauche dans
l’hémisphère~sud.
%Commenter alors la phrase de la p.3 : \emph{En raison de cette différence du sens de rotation, tout corps en mouvement sera dévié vers la droite dans l'hémisphère Nord et vers la gauche dans l'hémisphère Sud.}

\sol{
    On utilise la règle de la main droite pour prédire le sens de $\Fc$.
\begin{fig}[label]
    \centering
    \begin{subfig}[label][0.45]
        \includegraphics[width=\columnwidth]{./deviation_droite.pdf}
    \end{subfig}
    \begin{subfig}[label][0.45]
        \includegraphics[width=\columnwidth]{./deviation_gauche.pdf}
    \end{subfig}
    \capt{(a) Déviation vers la droite dans l'hémisphère nord. (b) Déviation vers la gauche dans l'hémisphère sud.}
\end{fig}
}

\item Écrire la loi de la statique des fluides. Avec un minimum de calcul, justifier que les vents géostrophiques se déplacent le long des isobares.
\sol{
  L'équation de la \udl{statique des fluides} dans $\RT$ non galiléen donne 
  $$
  \grad p = \rho \Vec{g} + \Vec{f}_{\rm C}.
  $$
  En projection sur l'horizontale, il reste
  $$
  \grad p  =  2 \rho \Vec{v} \pv \Vec{\Omega}_{\rm T}.
  $$
  Considérons un élément infinitésimal $\dl$ de la trajectoire d'un vent géostrophique. On a $$\dl \cdot \grad p = \d p = 2 \rho \dl \cdot (\Vec{v} \pv \Vec{\Omega}_{\rm T}).$$ Or $\dl$ est colinéaire à $\Vec{v}$, donc $\dl \cdot (\Vec{v} \pv \Vec{\Omega}_{\rm T}) = 0$. Donc 
  \eqbox{
  \d p = 0.
  } La pression $p$ est constante le long de la trajectoire d'un vent géostrophique, c'est-à-dire que les vents géostrophiques se déplacent le long des isobares.
}

  \item En déduire, en s'aidant d'un schéma, que les masses d'air tournent dans le sens direct autour d'un cyclone, aussi appelé dépression (zone de pression $p$ faible) dans l'hémisphère nord. \sol{
  Pour une dépression, le gradient de pression est dirigé vers l'extérieur de la masse d'air. La force de Coriolis est de même sens que le gradient de pression, et fait dévier la masse d'air vers la droite dans l'hémisphère nord. La masse d'air tourne donc dans le sens direct autour de la dépression.
  \begin{center}
    \includegraphics[width=0.3\textwidth]{depression.png}
  \end{center}
}

  \item Projeter la loi de la statique des fluides et en déduire l'équation des vents géostrophiques, qui relie la pression $p$ au champ aux composantes de la vitesse $v_x$ et $v_y$.
  \sol{
    La force d'inertie d'entraînement contribue au poids, qui est vertical et non n'apparaît pas dans une projection horizontale. Avec $\Vec{\Omega}_{\rm T} = \Omega(\cos \lambda \ey + \sin \lambda \ez)$, la force de Coriolis s'écrit
    $$
    \Vec{f}_{\rm C} = 2\rho \Vec{v} \pv \Vec{\Omega}_{\rm T} = 2 \rho \Omega  (v_x \ex + v_y \ey) \pv (\cos \lambda \ey + \sin \lambda \ez) = \frac{4\pi\rho}{T}(v_x \cos \lambda\, \ez - v_x \sin \lambda\, \ey + v_y \sin \lambda\,\ex) ,
    $$
    La projection sur l'horizontale de la loi de la statique des fluides donne alors  
    $$\grad p = \Vec{f}_{\rm C}  = 2\rho \Omega (-v_x \ey + v_y \ex).$$
    On peut récrire cette équation sous la forme 
    \eqbox{
    \left\{
      \begin{array}{l}
        \displaystyle\dr{p}{x} = +2\rho \Omega v_y, \\[3mm]
        \displaystyle\dr{p}{y} = -2\rho \Omega v_x.
      \end{array}
    \right.
    }
  }
\end{quest}

    \centering
    \includegraphics[width=0.6\columnwidth]{aile_avion.png}
    %\capt{}
  \end{fig}
  \begin{quest}
    \item	En analysant les lignes de courant, justifier que la vitesse de l'air est plus grande au-dessus du bord d'attaque de l'aile (extrados) qu'en dessous du bord d'attaque (intrados).
    \sol{
    L'écoulement étant incompressible, le débit volumique $\DV$ est conservé le long d'un tube de courant. En notant $\ell$ la largeur entre deux lignes de courant et $e$ la profondeur du tube de courant dans la direction perpendiculaire au plan de la figure, on a $\DV \simeq ve\ell \simeq \cte$. Plus les lignes de courant sont reserrées ($\ell$ petit) et plus la vitesse $v$ est grande. On en conclut que la vitesse est plus grande au niveau de l'extrados que de l'intrados, soit 
    \eqbox{
      v_{\rm e} > v_{\rm i}.
    }
    }
    \item On note $v_{\rm i}$ la vitesse au niveau de l'intrados et $v_{\rm e}$ la vitesse au niveau de l'extrados. Estimer numériquement $v_{\rm e}$ et $v_{\rm i}$ à partir de la figure en fonction de $v_{\infty}$.
    \sol{
    En mesant la largeur $\ell$ entre les lignes de courant, on trouve, au niveau du bord d'attaque,
    \eqbox{
      v_{\rm e} \simeq \SI{1.3}{}v_{\infty} \quad \text{et} \quad v_{\rm i} \simeq \SI{}{}v_{\infty}.
    }
    }
    \item En déduire que la pression $p_{\rm e}$ est plus faible au-dessus du bord d'attaque de l'aile que la pression $p_{\rm  i}$ en dessous. Conclure sur l'existence d'une force de portance qui permet à l'avion de se sustenter.
    \sol{
    D'après la relation de Bernoulli, on trouve à altitude $z$ constante,
    $$
    p_{\rm e} + \frac{1}{2}\rho v_{\rm e}^2 = p_{\infty} + \frac{1}{2}\rho v_{\infty}^2 =
    p_{\rm i} + \frac{1}{2}\rho v_{\rm i}^2.
    $$
    On retrouve la propriété de l'effet Venturi,
    \eqbox{
      p_{\rm e} - p_{\rm i} = \frac{1}{2}\rho (v_{\rm i}^2 - v_{\rm e}^2) < 0.
    }
    En intégrant sur toute la section $\Surf$ de l'aile, la résultante des forces de pression conduite donc à une force de portance $\Vec{F}_{\rm p}$ dirigée vers le haut.
    }
    \item Estimer la force de portance~$F_{\rm p}$ qui s'exerce sur une aile de surface alaire $S=\SI{845}{m^2}$ pour un Airbus A380 de $\SI{300}{t}$ volant à $\SI{950}{km.h^{-1}}$. Commenter.
    \sol{
    On trouve numériquement 
    $$
    p_{\rm e} - p_{\rm i} \simeq \frac{1}{2}\rho (v_{\rm i}^2 - v_{\rm e}^2) \simeq \num{0.3} \rho v_{\infty}^2 \simeq \SI{1}{bar}. 
    $$
    La force de portance vaut donc 
    \eqbox{
      F_{\rm p} \simeq (p_{\rm e} - p_{\rm i}) S \simeq \SI{2e7}{N}.
    }
    On peut comparer au poids de l'avion $P = mg = \SI{3e6}{N}$. On voit que 
    \eqbox{
      {F_{\rm p}} > P
    }
    La force de portance permet bien à l'avion de se sustenter. En réalité, la portance est sur-estimée car les rapports de vitesse sont inversées au niveau du bord de fuite. Il faudrait considérer les vitesses moyennées sur toute la section de l'aile.
    }

  \end{quest}
\end{exo}

\begin{exo}[PC][2][TD]{La clepsydre}
  \Notions{\item Relation de bernoulli \task Débit volumique \task Stationnarité approchée}
  \Annale[Physique BCPST]{G2E}{2007}
  \Annale[Composition de physique]{Agrégation spéciale}{2023}
  La clepsydre est un instrument de mesure du temps inventé en Égypte dans l'antiquité et perfectionné par les Grecs vers 270 avant notre ère. Un récipient à symétrie cylindrique est initialement rempli d’un liquide de masse volumique constante $\rho$ sur une
  hauteur $h_0$. Le  liquide s’écoule par une ouverture de section $s$ à la base du récipient sous l'effet de la pesanteur, d'intensité $g$. Le récipient est de rayon $R(h)$ à la hauteur $h$.
  De plus, l'écoulement est supposé parfait, incompressible, et stationnaire.

  \begin{fig}
    % \includegraphics[width=0.3\textwidth]{clepsydre.png}
    \includegraphics[scale=1]{clepsydre.pdf}
  \end{fig}

  \begin{quest}
  \item La section de l'ouverture $s$ est supposée négligeable devant la section de la surface libre  $S$. Que dire de la vitesse de la surface libre $v_A$ par rapport à la vitesse d’écoulement au niveau de l'ouverture $v_B$ ? Cette hypothèse est-elle justifiée sur toute la durée de l'écoulement ?
  \sol{
  L’écoulement est incompressible (car la masse volumique $\rho$ est constante) donc le
  \imp{débit volumique est conservé} le long d'un tube de courant. Comme la vitesse est uniforme sur une section, on trouve ${{\DV}} = sv_B = Sv_A$ et donc
  \eqbox{v_A = \frac{s}{S} v_B \ll v_B.}
  Cette hypothèse n'est plus justifiée lorsque le vase est presque vide, car alors $S \simeq s$.
  }
  \item Montrer sous cette hypothèse que la vitesse au niveau de l'ouverture $v_B$ vérifie la formule de Torricelli 
  \begin{equation}
    v_B = \sqrt{2gh}.
  \end{equation}
  \sol{
  On applique la relation de \imp{Bernoulli} entre un point $A$ de la surface libre et un point $B$ de l’ouverture.
\udl{Hypothèses} :
écoulement parfait, stationnaire, incompressible. Le long de la ligne de courant AB, on a
$$
\frac{p_A}{\rho} + \frac{v_A^2}{2} + g z_A = \frac{p_B}{\rho} + \frac{v_B^2}{2} + g z_B,
$$
soit  $\frac{\rho v_B^2}{2} + p_0 \simeq p_0 + \rho g h $ car $v_A \ll v_B$ et donc
\eqbox{v_B = \sqrt{2gh}.}
  }
  \bonus{
    Comment est reliée la vitesse de la surface libre $v_A$ à la fonction $h(t)$ ? Si le rayon $R$ de la clepsydre était constant, quelle équation différentielle vérifierait $h(t)$ ? La résoudre. Cette configuration est-elle pratique pour mesurer le temps écoulé ?
  \sol{Par définition, la vitesse de la surface libre vaut $v_A = -\dd{h}{t}$ en valeur absolue. La relation de conservation du débit volumique donne $sv_B = Sv_A$ et donc $v_A = \frac{s}{S} v_B = \frac{s}{\pi R^2} \sqrt{2gh}$. Si $R$ est constant, on a
  \eqbox{
    \dd{h}{t} + \frac{s \sqrt{2g}}{\pi R^2} \sqrt{h} = 0
  } 
  L'intégration donne $2\sqrt{h(t)} - 2\sqrt{h_0} = \frac{s \sqrt{2g}}{\pi R^2}  t $, soit 
  \eqbox{
  h(t) = h_0\left(1-\frac{t}{\tau}\right)^2  \quad \text{avec} \quad \tau = \frac{\pi R^2}{s} \sqrt{\frac{2h_0}{g}}.
  }
   Cette configuration n'est pas pratique pour mesurer le temps écoulé, car la hauteur d'eau $h(t)$ n'est pas une fonction affine du temps.
  }
  }
  \item Pour réaliser une horloge hydraulique, on souhaite que la vitesse de la surface libre $v_A$ soit constante. Que vaut alors la fonction $h(t)$ ? Expliquer alors le fonctionnement de la clepsydre. Que vaut le temps de vidange $\tau$ ? Quelle doit être le profil $R(h)$ du vase, qui donne le rayon $R$ à la hauteur $h$ ? 
  \bonus{Que doit valoir le rayon $r$ de l'ouverture pour une clepsydre de hauteur $h_0 = \SI{15}{cm}$ et de rayon maximal $R_0 = \SI{25}{cm}$, si l'on souhaite que la hauteur de liquide baisse de $\SI{1}{cm}$ toutes les minutes ?}
  \sol{
  Pour réaliser une horloge hydraulique, on doit avoir $v_A = -\dd{h}{t} = \cte$, et donc $h(t) = h_0-v_At$ est une fonction affine du temps. Le temps de vidange vaut alors $\tau = \frac{h_0}{v_A}$. Si $v_A$ est constant, on a $s \sqrt{2gh} = \pi R^2 v_A$ et donc le profil du vase doit vérifier 
  \eqbox{
    R(h) = r \left(  \frac{2gh}{v_A^2}\right)^{1/4},
  }
  où $s=\pi r^2$ et $r$ est le rayon de l'ouverture. Pour une clepsydre entièrement remplie d'eau de hauteur $h_0 = \SI{15}{cm}$, de rayon maximal $R_0 = \SI{25}{cm}$, et de vitesse de vidange $v_A = \SI{1}{cm.min^{-1}} = \SI{1.7e-4}{m.s^{-1}}$ :
  \eqbox{
  r = R_0\left(  \frac{v_A^2}{2gh_0}\right)^{1/4} = \SI{2.5}{mm}.
  }
  }
  % \item Reprendre la question précédente sans supposer que $v_A \ll v_B$.
  % \sol{
  
  %   Sans supposer que $v_A \ll v_B$, la formule de Torricelli est modifiée en 
  %   $$
  %   v_B = \frac{\sqrt{2gh}}{1-(s/S)^2}.
  %   $$
  %   La conservation du débit volumique donne alors $s \frac{\sqrt{2gh}}{1-(s/S)^2} = S v_A$, et donc
  %   $$
  %   -x^2 v_A -x \sqrt{2gh} + v_A = 0,
  %   $$
  %   avec $x=s/S$. La racine positive vaut $x = \frac{\sqrt{2gh}}{2} \left(1+\sqrt{1+4v_A^2/2gh}\right) = s/(\pi R^2)$, et donc 
  %   \eqbox{
  %   R(h) = R_0 \left(1-\sqrt{1-4v_A^2/2gh}\right)^{1/4}.
  %   }
  %   }
  \bonus{
  \item À quelle condition est-il légitime de supposer cet écoulement stationnaire ?

  \sol{
  On compare les accélérations locale et convective au niveau de l’ouverture de la clepsydre.
  $$
  \norme{\dr{\Vec{v}}{t}} \sim \frac{v}{\tau} \sim \frac{v v_A}{h_0},
  $$
  où $\tau \sim \frac{h_0}{v_A}$ est le temps typique de variation de la vitesse. De plus,
  $$
    \norme{(\Vec{v}\cdot \grad \Vec{v}) \Vec{v}} \sim \frac{v^2}{h_0}
  $$
  Donc le rapport vaut
  $$
    \frac{\norme{\dr{\Vec{v}}{t}} }{\norme{(\Vec{v}\cdot \grad \Vec{v}) \Vec{v}}} \sim \frac{v_A}{v} \sim \frac{s}{S}.
  $$
  Il faut donc
  \eqbox{s \ll S.}
  }
  }
  %\item Reprendre le calcul en tenant compte de la vitesse du fluide à la surface libre.
  \end{quest}

\end{exo}

% \begin{minipage}{0.68\textwidth}
On considère un fluide qui s'écoule dans une canalisation horizontale, représentée ci-dessous.
L'écoulement est supposé homogène, permanent, parfait et incompressible.

\begin{quest}
\item En appliquant la relation de bernoulli, déterminer la différence de pression $p_{1}-p_{2}$ entre l'entrée et la sortie du débitmètre.

\sol{
Avant d'appliquer le théorème de Bernoulli, il faut bien regarder si toutes les hypothèses sont bien vérifiées. Ici, on a un écoulement permanent, incompressible, homogène et parfait. On peut donc appliquer le théorème de Bernoulli entre les points $1$ et $2$, le long d’une ligne de courant horizontale :
$$
  p_1 + \frac{1}{2}\rho v_1^2 = p_2 + \frac{1}{2}\rho v_2^2.
  $$
  On en déduit 
  \eqbox{
  p_1-p_2 = \frac{1}{2}\rho \left(v_2^2-v_1^2\right).
  }
}
\item Justifier pourquoi le débit volumique est conservé. Exprimer $p_1-p_2$ en fonction de la vitesse $v_1$ uniquement.

\sol{
L’écoulement est \boxed{incompressible}, donc le débit volumique ${{\DV}}=vS$ est conservé.
On en déduit que $v_1 S_1 = v_2S_2$ et donc $v_2 = v_1 \frac{S_1}{S_2}$, d’où l’on tire 
\eqbox{
  p_1-p_2 = \frac{1}{2}\rho v_1^2 \left(\left(\frac{S_1}{S_2}\right)^2-1\right).
}
}

\end{quest}

\begin{quest}
\item On note $p_0$ la pression atmosphérique, qui règne au niveau de la surface libre du liquide. Soit $h$ la différence de hauteurs du liquide entre l'amont et l'aval. }

\begin{exo}[PC][1][devoir]{Effet de masse entraînée -- étude complète}
  \Enonce{
Lorsqu'une bulle de champagne sphérique de rayon $a(t)$ et de centre B migre vers la surface libre, elle subit de la part du liquide environnant des forces qui ne peuvent pas être modélisées uniquement par une poussée d'Archimède et une force de frottement fluide car sa vitesse $U(t)$ et son rayon $a(t)$ dépendent du temps. On adopte le modèle suivant:
\begin{itemize}
\item le référentiel $(\mathcal{R}_{\rm{O}})$ associé au repère $(\text{O}xyz)$ du verre dans lequel la vitesse de la bulle est $U(t)\,\vec{e}_z$ est galiléen;
\item on étudie le mouvement du champagne liquide autour de la bulle dans le référentiel barycentrique $(\mathcal{R}_{\rm{B}})$ associé au repère $(\text{B}xyz)$. On note $\vec{v}(M,t)$ le champ des vitesses dans le référentiel $(\mathcal{R}_{\rm{B}})$ où $M$ est repéré en sphériques par $(r,\theta,\varphi)$ de centre B;
\item l'écoulement du champagne autour de la bulle est supposé parfait et on néglige la pesanteur, il est aussi supposé incompressible et irrotationnel, décrit par le potentiel des vitesses $\phi(M,t)$ tel que $\vec{v}=\grad{\phi}$, et on cherche un potentiel des vitesses selon
\begin{equation}
\phi=\frac{\alpha(t)}{r}+\frac{\beta(t)\cos\theta}{r^2}+\gamma(t)\,r\cos\theta.
\end{equation}
\end{itemize}
  }
\begin{quest}
\QR {Exprimer les composantes $v_r$, $v_\theta$, $v_\varphi$ du champ des vitesses en fonction de $\alpha$, $\beta$, $\gamma$, $r$, $\theta$.}
{ On utilise la définition du potentiel des vitesses,
$$\Vec{v} = \Grad{\phi},$$ soit
\eqbox{
  \left\{\displaystyle
    \begin{array}{ll}
      v_r & = \dr{\phi}{r} =  -\frac{\alpha(t)}{r^2} - \frac{2\beta(t) \cos\theta}{r^3} -U\cos\theta, \\
      v_\theta & = \frac{1}{r} \dr{\phi}{\theta} = -\frac{\beta(t) \sin(\theta)}{r^3} +U \sin\theta, \\
      v_\varphi & = \frac{1}{r\sin\theta} \dr{\phi}{\varphi}  = 0.
    \end{array}
    \right.
    }
}

\QR {En exploitant le fait que loin de la bulle de gaz, le champagne est au repos dans le référentiel $(\mathcal{R}_{\rm{O}})$ lié au verre, exprimer $\gamma(t)$ en fonction de $U(t)$.}
{
  Le champ de vitesse de l’écoulement vaut $-U\ez$ à l’infini, et donc par continuité de la vitesse, 
  $$\phi(r=\infty)= -Uz = -Ur\cos\theta,$$ ce qui donne le coefficient
  \eqbox{
    \gamma(t)=-U(t).
    }
}


\QR {Sur la bulle, la condition aux limites s'écrit $v_r(r=a(t),\theta)=\dot{a}(t)$; en revanche il n'y a aucune condition sur la composante $v_\theta$ car l'écoulement est supposé parfait. En déduire les expressions de $\alpha$ et $\beta$ en fonction de $a(t)$, $\dot{a}(t)$ et $U(t)$. Vérifier alors qu'à la surface de la bulle en $r=a(t)$, on a
% \eqbox{
\begin{equation}
  v^2=\dot{a}^2(t)+\dfrac{9}{4}U^2(t)\sin^2(\theta).
\end{equation}}
{
  Dans un écoulement, la vitesse normale est continue au niveau d’une paroi imperméable, et donc 
$$v_r(r=a(t),\theta)=\dot{a}(t)$$
 est égale à la vitesse de déplacement de la paroi de la bulle. Aucune condition ne porte sur la composante $v_\theta$, car l'écoulement est supposé parfait. Or
 $$v_r(r=a(t),\theta)= \frac{\alpha(t)}{a^2} - \frac{2\beta(t) \cos\theta}{a^3} -U(t)\cos\theta = \frac{\alpha(t)}{a^2}  - \cos\theta\left[ \frac{2\beta(t) }{a^3} +U(t) \right] = \dot{a}(t) \,\forall \theta.$$ 
 Cette condition impose donc
 \eqbox{
 \alpha(t) = \dot{a}(t) a^2 \quad \text{et} \quad \beta(t) = -\frac{a^3}{2}U(t).
 }
 Le champ des vitesses a donc pour expression 
 $$
  \Vec{v}(r,\theta) = \dot{a}(t) \frac{a^2}{r^2} \er + U(t)\cos\theta \left[\frac{a^3 }{r^3} - 1 \right]\er + U(t) \sin\theta \left[-\frac{a^3 }{2 r^3} +  1 \right] \et.
 $$
 On calcule ensuite la norme carrée de la vitesse en $r=a(t)$,
 $$
 v^2 = v_r(r=a(t))^2 + v_\theta(r=a(t))^2 = \dot{a}^2(t) + U^2(t) \sin^2( \theta) \left[-\frac{a^3 }{2 a^3} +  1 \right]^2.
  $$
  On a bien 
  \eqbox{
    v^2=\dot{a}^2(t)+\dfrac{9}{4}U^2(t)\sin^2(\theta).
  }
}


\QR {Quelle est l'expression de la force volumique d'inertie dans le référentiel $(\mathcal{R}_{\rm{B}})$ lié à la bulle ? Montrer que cette force dérive d'une énergie potentielle volumique dont on donnera l'expression.}
 {Pour une particule fluide de masse $\d m$, la force d’entrainement s’écrit
$$ \d \Fe = -\d m \Vec{a}_e = -\d m \dot{U}(t) \ez = -\rho \d V \dot{U}(t) \ez,$$  soit $\Vec{f}_{\rm \! e} = -\rho \dot{U}(t) \ez = -\Grad{e_{\rm p}}$ avec l’énergie potentielle volumique
\eqbox{
  e_{\rm p} = \rho \dot{U}(t) z = \rho \dot{U}(t) r \cos\theta.
}
}
\QR {Montrer que la grandeur
$p+\rho \dot{U}r\cos\theta+\dfrac{1}{2}\rho v^{2}+\rho\dfrac{\partial \phi}{\partial t}$
prend la même valeur en tout point de l'espace.}
{
  On souhaite ici généraliser la relation de Bernoulli aux écoulements instationnaires en présence d’une force d’inertie à la place de la pesanteur. On part de l’équation d’Euler dans le référentiel $(\mathcal{R}_{\rm{B}})$,
  $$\dr{\Vec{v}}{t} + (\Vec{v}\cdot \Grad)\Vec{v} = -\frac{1}{\rho} \Grad{p}  - \dot{U}(t) \ez.$$
  En utilisant l’expression de la force d’inertie d’entrainement, on a
  $$\dr{\Vec{v}}{t} + (\Vec{v}\cdot \Grad)\Vec{v} = -\frac{1}{\rho} \Grad{p} + \Vec{g} + \frac{1}{\rho} \Grad{e_{\rm p}}.$$
  Or l’écoulement est supposé irrotationnel, donc on peut utiliser l’identité vectorielle
  $$(\Vec{v}\cdot \Grad)\Vec{v} = \Grad{\left(\frac{v^2}{2}\right)} - \Vec{v} \wedge (\rot{\Vec{v}}) = \Grad{\left(\frac{v^2}{2}\right)}.$$
  De plus, on a $\dr{\Vec{v}}{t} = \Grad{\left(\dr{\phi}{t}\right)}$. On en déduit
  $$\Grad{\left(\dr{\phi}{t} + \frac{v^2}{2} + \frac{p}{\rho} + \frac{e_{\rm p}}{\rho}  \right)} = \Vec{0},$$ soit
  \eqbox{
    p+\rho \dot{U}r\cos\theta+\dfrac{1}{2}\rho v^{2}+\rho\dfrac{\partial \phi}{\partial t} = f(t),
  }
  où $f$ est une fonction du temps uniquement.
}

\QR {En exploitant les résultats précédents, on montre (travail non demandé) que la pression en un point de la surface de la bulle est de la forme
\begin{equation}

\Enonce{

  {
  On a tout d'abord $V_{0}=N_{a}V_{a}+N_{c}V_{c}$ avec $V_{a}$, $V_{c}$, volumes de tissu dû aux parois des artères et des capillaires respectivement. $V_{a}$ correspond donc au volume compris entre deux cylindres de même axe, de longueur $L_{a}$ et de rayons respectifs $r_{a}$ et $r_{a}+e$.   \eqbox{V_{0}=2\pi \alpha\left\lbrace N_{a}L_{a}r^{2}_{a}+N_{c}L_{c}r^{2}_{c} \right\rbrace }

  \eqbox{\dfrac{dr_{c}}{dr_{a}}=- \dfrac{N_{a}L_{a}}{N_{c}L_{c}}\dfrac{r_{a}}{r_{c}}}

  \item le rôle de la pesanteur a été négligé alors qu'elle a bien sûr un effet;
  \item la résistance a été calculée en considérant uniquement une géométrie cylindrique. Cette géométrie n'est pas rigoureusement représentative de la réalité, les vaisseaux se coudent, n'ont pas des sections uniformes... De même, $\alpha$ n'est certainement pas uniforme dans tout le système. Pour les petits vaisseaux par exemple, la membrane pourra représenter une plus grande proportion car on atteint la taille minimum des cellules qui la compose (typiquement, une cellule est de l'ordre du micron) ;
  \item le caractère non newtonien supposé du sang dans les calculs menés auparavant est sans doute discutable
  \item comme $Re>>1$, l'écoulement n'est sans doute pas un simple écoulement de cisaillement.
  \end{itemize}
  En particulier, on peut supposer assez raisonnablement qu'en raison de la géométrie réelle, la résistance totale $R_{H}$ est plus élevée que celle que nous proposons. Or nous avons vu $R_{H}=\frac{P_{A}-P_{B}}{D_{v}}$. La différence de pression $P_{A}-P_{B}$ est fixée par le c\oe ur. Une augmentation de $R_{H}$ se traduit donc par une diminution de $D_{v}$. pour "compenser" cet effet, un nombre plus élevé de vaisseaux permettrait de maintenir un débit en sang suffisant pour alimenter tous les tissus.
  }
\end{quest}

\end{exo}

\begin{exo}[PC][1][devoir]{Effet Magnus}%{1}
Un objet en rotation sur lui-même modifie l'écoulement du fluide autour de lui, qui en retour engendre une force qui permet de modifier la trajectoire de l'objet. C'est ce qui est par exemple utilisé au tennis ou au ping-pong pour les coups coupés (liftés) ou encore les coups slicés. C'est aussi ce qui est à l'origine des trajectoires si particulières des ballons enroulés au football. Pour une illustration spectaculaire avec un ballon de basket, voir par exemple \url{https://youtu.be/2OSrvzNW9FE}.

Cet effet porte le nom d'effet Magnus, qui historiquement est l'effet observé sur un cylindre en rotation. On se propose ici d'étudier cet effet dans le référentiel lié au cylindre, où l'écoulement est stationnaire et supposé parfait. Le vecteur rotation du cylindre est $\vec{\Omega}=\Omega\,\vec{e}_z$, et en l'absence du cylindre l'écoulement du fluide est uniforme avec $\vec{v}(M)=v_0\,\vec{e}_x$. On admet qu'autour du cylindre en rotation, le champ des vitesses bidimensionnel décrivant l'écoulement se met sous la forme suivante en coordonnées polaires
\begin{equation}
 \vec{v}(M)=\left(A-\frac{B}{r^2}\right)\cos\theta\,\vec{e}_r+\left[\frac{a^2\Omega}{r}-\left(A+\frac{B}{r^2}\right)\sin\theta\right]\vec{e}_\theta.
 \end{equation}
\begin{quest}
\item À partir des conditions aux limites vérifiées par ce champ de vitesses pour $r\rightarrow a$ et pour $r\rightarrow \infty$, déduire les expressions des constantes $A$ et $B$ en fonction de $v_0$ et $a$, où $v_0$ désigne la vitesse uniforme de l'écoulement en l'absence du cylindre tournant.
\sol{
Conditions aux limites d'un écoulement parfait : continuité de la vitesse radiale, donc
$$
v_r(r=a,\theta) = 0 \Rightarrow (A - \frac{B}{a^2})\cos \theta = 0 \Rar B=a^2A.
$$

$$
\Vec{v}(r=+\infty) = v_0 \ex = A \cos \theta \er- A \sin \theta\et = A \ex \Rar A = v_0.
$$
On en déduit 
\eqbox{
\Vec{v}(M) = v_0\left(1-\frac{a^2}{r^2}\right)\cos\theta\,\vec{e}_r+v_0\left[\frac{a^2\Omega}{v_0r}-\left(1+\frac{a^2}{r^2}\right)\sin\theta\right]\vec{e}_\theta.
}
}

$$

\begin{cases}
  r = a \text{ ou } \theta=\pm\frac{\pi}{2}\\

\begin{cases}
  r = a \text{ ou } \theta=\pm\frac{\pi}{2}\\
\frac{\alpha}{2} = \sin\theta \text{ ou } \alpha = \pm \left( \frac{a}{r} + \frac{r}{a} \right).
\end{cases}
$$
Or la fonction $x \mapsto x+\frac{1}{x}$ est de minimum égal à $2$. En conclusion,
\begin{liste}
  \item Si \fbox{$|\alpha|<2$}, il existe deux points d'arrêt symétriques sur le cylindre d'angle $\theta$ et $\pi-\theta$ avec
  \eqbox{
  \theta = \arcsin\left(\frac{\alpha}{2}\right).
  }
  \item Si \fbox{$|\alpha|=2$}, il existe un point d'arrêt sur le cylindre sur l'axe $y$. soit en haut si $\alpha >0$ (le cylindre tourne dans le sens direct), soit en bas si $\alpha <0$ (le cylindre tourne dans le sens indirect).
  \item Si \fbox{$|\alpha|>2$}, il existe un point d'arrêt dans le fluide sur l'axe $y$ à une distance 
  \eqbox{
  r = \frac{a}{2}\left( \sqrt{\alpha^2-4}-\abs{\alpha} \right),
  }
  soit en haut si $\alpha >0$ (le cylindre tourne dans le sens direct), soit en bas si $\alpha <0$ (le cylindre tourne dans le sens indirect).
\end{liste}
}
\item On admet qu'on peut utiliser le théorème de Bernoulli dans le cas présent. En l'appliquant entre un point à l'infini et un point à la surface du cylindre, en déduire l'expression de la pression $P$ en un point de la couche fluide en contact avec la surface du cylindre en fonction de $\rho$, $v_0$, $a$, $\Omega$, $\theta$ et $P_{\infty}$.
\sol{
On applique le théorème de Bernoulli entre un point à l'infini et un point à la surface du cylindre, soit
$$
P_{\infty} + \frac{1}{2}\rho v_0^2 = P(\theta) + \frac{1}{2}\rho v(r=a)^2,
$$
d'où l'on tire 
$$
P(\theta) = P_{\infty} + \frac{1}{2}\rho v_0^2 - \frac{1}{2}\rho v_0^2\left(\alpha-2\sin\theta\right)^2,
$$
soit 
\eqbox{
  P(\theta) =  P_{\infty} + \frac{1}{2}\rho v_0^2\left(1-\alpha^2+4\alpha \sin\theta - 4 \sin^2 \theta \right).
}
}
\item Déterminer la force verticale $\vec{F}_{\rm{M}}$, dite force de Magnus, exercée par le fluide sur une portion de longueur $h$ du cylindre en fonction de $\rho$, $v_0$, $a$, $\Omega$ et $h$. On rappelle que $\textstyle\int_{0}^{2\pi}\sin^{3}\theta\,\der\theta=0$.

Mettre cette force sous la forme $\vec{F}_{\rm{M}}=\beta \vec{v}_0\pv\vec{\Omega}$ et exprimer la constante $\beta$ en fonction des paramètres du problème. Donner une explication qualitative de l'effet Magnus.
\end{quest}
\sol{
On a $\vec{F}_{\rm{M}} = F_{\rm M} \ey $ avec 
$$
F_{\rm M} = \vec{F}_{\rm{M}} \cdot \ey = \iint_S p(M) \dS \cdot \ey = h \int_0^{2\pi} p(\theta) a \d \theta \er \cdot \ey = ha \int_0^{2\pi} \d p(\theta) \sin \theta \d \theta,
$$
soit 
\begin{align*}
  F_{\rm M} & = ha \int_0^{2\pi} \left( P_{\infty} + \frac{1}{2}\rho v_0^2\left(1-\alpha^2 + 4\alpha \sin\theta - 4 \sin^2 \theta\right) \right) \sin \theta \d \theta \\ 
  & = ha \int_0^{2\pi} \left(P_{\infty} +  \frac{1}{2}\rho v_0^2\left(1-\alpha^2\right)\right) \sin \theta \d \theta + ha \int_0^{2\pi} 2 \rho v_0^2 \alpha \sin^2\theta \d \theta - ha \int_0^{2\pi} 2\rho v_0^2 \sin^3\theta \d \theta, \\
\end{align*}
soit 
\eqbox{
  F_{\rm M}  = 2 ha \pi \alpha \rho v_0^2 =  2 \rho \pi a^2 h \Omega v_0.
}
On peut mettre la force sous la forme 
\eqbox{
  \vec{F}_{\rm{M}}= 2 m_{\rm f} \vec{v}_0\pv\vec{\Omega},
}
où $m_{\rm f} = \rho \pi a^2 h$ est la masse du fluide déplacé par le cylindre.

%  que ressentirait le fluide déplacé par le cylindre dans le référentiel non galiléen lié au cylindre en rotation lorsque celui-ci 
}
\end{exo}

\begin{exo}[PC][2][colle]{L'effet Magnus}%{1}
  \Enonce{
  Un objet en rotation sur lui-même modifie l'écoulement du fluide autour de lui.
  La résultante du champ de pression engendre alors une force qui modifie la trajectoire de l'objet.  Cet effet porte le nom d'effet Magnus.
  On se propose ici d'étudier l'effet Magnus sur un cylindre en rotation en mouvement uniforme par rapport à l’air immobile. Dans le référentiel galiléen $(\Rcal)$ lié au centre du cylindre, en translation par rapport au référentiel du laboratoire, l'écoulement est stationnaire et supposé parfait. Le cylindre est de rayon $a$, de vecteur rotation $\Vec{\Omega}=\Omega\,\vec{e}_z$, et loin du cylindre l'écoulement du fluide est uniforme de champ des vitesses $\vec{v}(M)=U\vec{e}_x$. On admet qu'autour du cylindre en rotation, le champ des vitesses bidimensionnel dans le plan $(x,y)$  se met sous la forme suivante en coordonnées polaires
  \begin{equation}
   \vec{v}(M)=\left(A-\frac{B}{r^2}\right)\cos\theta\,\vec{e}_r+\left[\frac{a^2\Omega}{r}-\left(A+\frac{B}{r^2}\right)\sin\theta\right]\vec{e}_\theta.
   \end{equation}
  }
  \begin{quest}
  % \item Interpréter les constantes $A$, $B$ et $a$ en indiquant le profil de vitesse $\Vec{v}_{\! A}$, $\Vec{v}_{\! B}$ et $\Vec{v}_{\! \Omega}$ associé.   {
  Conditions aux limites d'un écoulement parfait : continuité de la vitesse radiale, donc
  $$
  v_r(r=a,\theta) = 0 \Rightarrow \left( A - \frac{B}{a^2} \right)\cos \theta = 0 \Rar B=a^2A.
  $$
  De plus, le profil de vitesse doit se raccorder à l'écoulement uniforme à l'infini, donc
  $$
  \Vec{v}(r=+\infty) = U \ex = A \cos \theta \er- A \sin \theta\et = A \ex \Rar A = A.
  $$
  On en déduit 
  \eqbox{
  \Vec{v}(M) = U\left(1-\frac{a^2}{r^2}\right)\cos\theta\,\vec{e}_r+U\left[\frac{a^2\Omega}{Ur}-\left(1+\frac{a^2}{r^2}\right)\sin\theta\right]\vec{e}_\theta.
  }

  $$
    \Vec{v}(r=a) = \left(a\Omega-2U\sin\theta\right)\vec{e}_\theta.
  $$
  On voit donc qu’il y a glissement du fluide sur la paroi du cylindre a une vitesse relative $-2U \sin \theta$, ce qui n’est pas gênant puisque la vitesse tangentielle n’est pas continue dans un écoulement parfait.

    $$\frac{a\Omega}{U} = \alpha.$$

  $$
  \Vec{v}(M) = U\left(1-\frac{a^2}{r^2}\right)\cos\theta\,\vec{e}_r+U\left[\alpha\frac{a}{r}-\left(1+\frac{a^2}{r^2}\right)\sin\theta\right]\vec{e}_\theta.

  \begin{cases}
    r = a \text{ ou } \theta=\pm\frac{\pi}{2}\\
  \alpha = \left( \frac{a}{r} + \frac{r}{a} \right)\sin\theta

  soit

    \text{ou} & \\

  $$
  Or la fonction $x \mapsto x+\frac{1}{x}$ est de minimum égal à $2$. En conclusion,
  \begin{enumerate}
    \item Si \fbox{$|\alpha|<2$}, il existe deux points d'arrêt symétriques sur le cylindre d'angle $\theta$ et $\pi-\theta$ avec

  \end{enumerate}

\begin{exo}[PC][1]{Vidange d'un réservoir}
  \Notions{\item Relation de bernoulli \task Débit volumique \task Torricelli}
%\begin{wrapfigure}{r}{0.3\textwidth}

On considère un réservoir cylindrique rempli d’eau, de masse volumique $\rho$, représenté à gauche ci-dessous. La surface libre de l’eau donne sur l'atmosphère, de pression $p_0$, et la sortie en B est ouverte. On note $d_1$ le diamètre du récipient et $d_2$ celui de l’orifice circulaire. On suppose un régime quasi-stationnaire atteint (la hauteur d’eau varie dans le temps mais la colonne d’eau est à l’équilibre mécanique).
%\begin{figure}[h!]
%\centering

\begin{center}

\includegraphics[scale=1]{vidange_mariotte.pdf}

\begin{quest}
\item En explicitant les hypothèses faites, déterminer une condition sur les diamètres $d_1$ et $d_2$ pour que $v_A$ n'excède pas 1\% de $v_B$.  Exprimer alors $v_B$ en fonction de la hauteur d’eau $h(t)$.

\sol{
L’écoulement est supposé parfait, stationnaire, incompressible et homogène.  L’incompressibilité assure la conservation du débit volumique, soit $v_A \pi d_1^2/4 = v_B \pi d_2^2/4$ et donc
$$
\frac{v_A}{v_B} = \left(\frac{d_2}{d_1}\right)^2 \leq \num{0.01} \Rightarrow \boxed{d_2 \leq \num{0.1}d_1}.
$$

On applique ensuite la relation de Bernoulli,
$
p_0 + \rho g h(t) + \frac{\rho}{2} v_A^2 \simeq p_0 + \rho g h(t)  = p_0 + \frac{\rho}{2} v_B^2
$, et donc
\eqbox{
v_B = \sqrt{2 g h(t)}.
}
}

\item On note $S_1$ et $S_2$ les sections du récipient et de l’orifice, respectivement. Trouver l'équation différentielle vérifiée par $h(t)$.

\sol{
Par définition du débit volumique,
$
{{\DV}} = \frac{\d V}{\d t} = v_B S_2
$,
avec $S_2 = \pi d_2^2/4$ la section de l’orifice et $V = h(t)S_1$ le volume d’eau dans le récipient au-dessus de l’orifice.
Donc,
$$
S_1 \dd{h}{t} = -S_2 \sqrt{2 g h} \Rightarrow \boxed{\dd{h}{t} = -\frac{S_2}{S_1} \sqrt{2 g h}.}
$$
}

\item Résoudre cette équation et déterminer le temps de vidange $\tau$. Application numérique pour $d_1=20\,d_2$ et $h_{0}=\SI{40}{cm}$.

\sol{
On a $\frac{\d h}{\sqrt{h}} = -\frac{S_2}{S_1} \sqrt{2 g} \d t$ et donc en intégrant entre $0$ et $t$,
$$
2\sqrt{h(t)} - 2\sqrt{h_0} = -\frac{S_2}{S_1} \sqrt{2 g} t \Rightarrow  \sqrt{h(t)} = \sqrt{h_0} - \frac{S_2}{S_1} \sqrt{\frac{ g}{2}} t.
$$
Le temps de vidange correspond à $h(\tau) = 0$, soit
\eqbox{
  \tau = \left(\frac{d_1}{d_2}\right)^2 \sqrt{\frac{2h_0}{ g}}  = \SI{114}{s}.
  }
}

\item Qu’est-ce qui change si l’on utilise le dispositif de la figure de droite ?

\sol{
Ce dispositif se nomme vase de Mariotte. On impose au bas du tube la pression $p_0$. Le débit est donc constant tant que le niveau de l’eau est au-dessus du bas de tube.
 La hauteur $h$ doit être remplacée par la hauteur $H$ entre les points B et C.
 La relation de Bernoulli donne alors \fbox{$v_B = \sqrt{2gH}$}, qui est constant.
}

\end{quest}
\end{exo}
