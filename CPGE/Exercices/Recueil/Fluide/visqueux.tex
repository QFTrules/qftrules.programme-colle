\begin{exo}[PC][2][appli]{Nombres adimensionnés}
  \Notions{\item Nombre de Reynolds \task Applications numériques}
  % \titreEntrainementFacultatif{}
  \hauteurLargeurCadreReponse		{8mm}{4cm}
  \dureeResolutionFacultative		{2} % 1, 2, 3 ou 4
  \typeEntrainement				{Newton} % Newton ou QCM ou AN ou vide (exercice "normal")
  \basiqueEtTransversal			{Y}
  \nombreColonnesQuestions		{1} % vide, 1, 2, 3, etc.
  \avecPlusieursQuestions			{Y} % Y ou N
  % \pourcentageDeLaPartieAGauche   {0.75}
  % \initialisationEntrainement
  % -------------------- Partie à gauche -------------------- %
  % \initialisationPartieGauche %
  % \noindent
  % \setlength{\marginparwidth}{50pt}
 On considère l'écoulement incompressible d'un fluide newtonien de viscosité cinématique $\nu$ soumis au champ de pesanteur $\Vec{g}$ dans un référentiel non galiléen en rotation à la vitesse angulaire $\Vec{\Omega}$. L'écoulement est décrit par l'équation de Navier-Stokes, que l'on peut mettre sous la forme 
 $$
  \underbrace{\frac{\partial \Vec{v}}{\partial t}}_{\small\encircle{1}} + \underbrace{(\Vec{v} \cdot \Grad{}) \Vec{v}}_{\small\encircle{2}} = - \underbrace{\frac{1}{\rho}\Grad{p}}_{\small\encircle{3}} + \underbrace{\nu \Delta \Vec{v}}_{\small\encircle{4}} + \underbrace{\Vec{g}}_{\small\encircle{5}} + \underbrace{2 \Vec{v} \pv \Vec{\Omega}}_{\small\encircle{6}} + \underbrace{\Vec{\Omega} \pv (\Vec{\Omega} \pv \Vec{r})}_{\small\encircle{7}}.
 $$
%  où $\nu = \eta/\rho$ désigne la viscosité cinématique. 
%  Il est d'usage de comparer l'ordre de grandeur des différents termes de l'équation au terme de convection ${\small\encircle{2}}=(\Vec{v} \cdot \Grad{}) \Vec{v}$, qui est le seul terme non linéaire.
  Soit $\tau$ la durée typique de l'écoulement, $U$ la vitesse typique et $L$ l'échelle spatiale typique. Pour chaque item ci-dessous, donner l'expression du nombre adimensionné en fonction des paramètres de l'écoulement, interpréter physiquement ce nombre et donner son ordre de grandeur dans une situation typique.  

% \debutEntrainement
  % ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^ %
  \begin{enonce}
  	Le nombre de Reynolds $\Re$, ordre de grandeur du rapport de ${\small\encircle{2}}$ sur ${\small\encircle{3}}$.
  \end{enonce}

  % \reponse{$r \left(\cos(\theta) \ex+\sin(\theta) \ey\right)$}

  \begin{corrige}
  	En ordre de grandeur, on a ${\small\encircle{2}} \sim U^2/L$ et ${\small\encircle{3}} \sim \rho U^2/L$. On définit donc le nombre de Reynolds comme 
    \eqbox{
    \Re = \frac{U L \rho}{\eta}.
    }
  \end{corrige}

  % ************************************** %
  \begin{enonce}
  	Le nombre de Strouhal ${\rm St}$, ordre de grandeur du rapport de ${\small\encircle{1}}$ sur ${\small\encircle{2}}$.
  \end{enonce}

  % \reponse{$r \er$}

  \begin{corrige}
    En ordre de grandeur, on a ${\small\encircle{1}} \sim U/\tau$ et ${\small\encircle{2}} \sim U^2/L$. On définit donc le nombre de Strouhal comme
    \eqbox{
    {\rm St} = \frac{L}{U \tau}.
    }
    Le nombre de Strouhal détermine notamment la périodicité des tourbillons de Von Karman, qui apparaissent derrière un obstacle pour un nombre de Reynolds intermédiaire, comme illustré ci-dessous.
    
    \begin{fig}[label]
      \centering
      \includegraphics[width=0.7\columnwidth]{karman.png}
      %\capt{}
    \end{fig}
    Le temps $\tau$ correspond alors à la période des tourbillons, la longueur $L$ à la taille de l’obstacle. Les tourbillons autour d’un obstacle de forme donnée sont gouvernés par la loi d’échelle 
    $$
    {\rm St} = f(\Re), 
    $$
    où $f$ est une fonction universelle, qui ne dépend que de la forme de l’obstacle.
  \end{corrige}

  % ************************************** %
  \begin{enonce}
  Le nombre de Froude ${\rm Fr}$, ordre de grandeur du rapport de ${\small\encircle{5}}$ sur ${\small\encircle{2}}$.
  \end{enonce}

  % \reponse{$r \left(\cos(\theta) \ex+\sin(\theta) \ey\right) +z\ez $}

  \begin{corrige}
    En ordre de grandeur, on a ${\small\encircle{5}} \sim g$ et ${\small\encircle{2}} \sim U^2/L$. On définit donc le nombre de Froude comme
    \eqbox{
    {\rm Fr} = \frac{U^2}{gL}.
    }
  \end{corrige}

  % ************************************** %
  \begin{enonce}
  	Le nombre de Rossby ${\rm Ro}$, ordre de grandeur du rapport de ${\small\encircle{6}}$ sur ${\small\encircle{2}}$.
  \end{enonce}

  % \reponse{$r \er+z\ez$}

  \begin{corrige}
    En ordre de grandeur, on a ${\small\encircle{6}} \sim U \Omega$ et ${\small\encircle{2}} \sim U^2/L$. On définit donc le nombre de Rossby comme
    \eqbox{
    {\rm Ro} = \frac{L \Omega}{U}.
    }
  \end{corrige}

\begin{exo}[PC][1][colle]{Écoulement entre deux cylindres}
  L'écoulement d'un fluide entre deux cylindres concentriques, de rayons $r_1$ et $r_2$ et tournant autour de leur axe commun $(Oz)$ aux vitesses angulaires $\omega_1$ et $\omega_2$ peut être décrit par un champ des vitesses de la forme $\Vec{v}(r,t)=(Ar+\frac{B}{r})\,\Vec{e_\theta}$.
  On donne en cylindriques le formulaire suivant
  \begin{equation}
  \begin{split}
  \grad{f}_{\text{cyl}}&=\frac{\partial f}{\partial r}\,\Vec{e_r}+\frac{1}{r}\frac{\partial f}{\partial \theta}\,\Vec{e_\theta}+\frac{\partial f}{\partial z}\,\Vec{e_z}\\
  \left.\dive{\Vec{u}}\right|_{\text{cyl}}&=\frac{1}{r}\dfrac{\partial (ru_{r})}{\partial r}+\frac{1}{r}\dfrac{\partial u_{\theta}}{\partial \theta}+\frac{\partial u_{z}}{\partial z}\\
  \left.\rot{\Vec{u}}\right|_{\text{cyl}}&=
  \left(\dfrac{1}{r}\frac{\partial u_{z}}{\partial \theta}-\frac{\partial u_{\theta}}{\partial z}\right)\Vec{e_r}+\left(\frac{\partial u_{r}}{\partial z}-\frac{\partial u_{z}}{\partial r}\right)\Vec{e_\theta}+\left(\frac{1}{r}\dfrac{\partial (r u_{\theta})}{\partial r}-\frac{1}{r}\frac{\partial u_{r}}{\partial \theta}\right)\Vec{e_z}
  \end{split}
  \end{equation}

\begin{exo}[PC][3][TD]{Ramer ou nager}
  \Notions{\item Nombre de Reynolds \task Force de traînée}
  Parmi les sports nautiques présents aux jeux olympiques, le kayak et la nage libre sont deux disciplines phares. Bien que le principe de propulsion soit le même (la main ou la pagaie plonge dans l'eau pour exercer une force vers l'avant), les vitesses atteintes sont sensiblement différentes. Pour accroître sa vitesse, un kayakiste ou un nageur augmente sa fréquence de nage $f$. On observe expérimentalement que la vitesse $V$ du sportif n'augmente pas suivant la même loi pour le kayak et la nage. Dans cet exercice, nous cherchons à expliquer cette loi.\\ 

  Le kayak est une embarcation profilée dont la partie immergée présente une faible section $S=\SI{6e-3}{m^2}$, et mue par la poussée imprimée par la pagaie double lorsque chaque pale plonge dans l'eau à la vitesse $v$ alternativement de gauche à droite. Chaque pale est de section $s=\SI{0.1}{m^2}$, et exerce sur l'eau la force $F$ fournie par le kayakiste. Le kayakiste rame de part et d'autre avec une cadence $f$, inverse de la période $T$ entre deux poussée du même côté, ce qui lui permet d'atteindre des vitesses de l'ordre de $V\simeq \SI{4}{m.s^{-1}}$.
  \begin{quest}
    \item	Estimer le nombre de Reynolds $\Re$ associé à l'écoulement autour du kayak. En déduire la nature de l'écoulement, l'ordre de grandeur du coefficient de traînée $C_x$, et l'expression des forces de traînée que l'eau exerce sur la pagaie, notée $F_p$, et le kayak, notée $F_k$.
    \sol{
     L'eau est de masse volumique $\rho = \SI{1000}{kg.m^{-3}}$, et de viscosité $\eta = \SI{1e-3}{Pa.s}$. La longueur transverse au mouvement est $L=\sqrt{S}\simeq \SI{7.7e-2}{m}$.
      Le nombre de Reynolds vaut donc 
      \eqbox{
      \Re = \frac{\rho V L}{\eta} \sim \SI{2e5}{}.
      }
       L'écoulement est donc turbulent, et le coefficient de traînée est constant et vaut environ $C_x = \num{0.5}$. Les forces de traînée ont alors pour expression 
       $F_p = \frac{1}{2} \rho v^2 s C_x$, et $F_k = \frac{1}{2} \rho V^2 S C_x$.
    }
    
    \item Par le principe des actions réciproques, montrer que  $v\ll V$ en pratique. En déduire que la pale peut être considérée comme immobile dans l'eau pendant la poussée du kayakiste. 
    \sol{
    À l'équilbre, la force que l'eau exerce sur la pagaie compense celle qui freine le kayak, et donc $F_p=F_k$. On en déduit 
    $$
    \frac{1}{2} \rho v^2 s C_x = \frac{1}{2} \rho V^2 S C_x \quad \Rightarrow \quad \boxed{v = \sqrt{\frac{S}{s}}V \simeq \num{0.25} V \ll V.}
    $$
    En bonne approximation, la pale est donc immobile dans l'eau pendant la poussée du kayakiste.
    }
    \item En déduire que la distance $\ell$ dont avance le kayak pendant l'immersion de la pagaie est indépendant de la cadence $f$, et dépend uniquement des paramètres géométriques de la pagaie, du kayak et du sportif. Estimer la valeur de $\ell$ sachant que la pagaie est de longueur $L=\SI{2}{m}$ et que la hauteur des kayaks standards impose au kayakiste d'immerger la pagaie avec un angle d'attaque $\alpha = \SI{45}{\degree}$ à la fois en avant et de côté.
    \sol{
    Comme la pagaie est immobile dans l'eau pendant la poussée, le point de contact entre la pagaie et l'eau reste fixe pendant toute la durée de la poussée. Le kayak avance d'une distance $\ell$ déterminée uniquement par la longueur de la pagaie $L$, la longueur de son bras, et la hauteur du kayak au-dessus de l'eau. L'hypothénuse du trièdre formé par les trois côtés de longueur $\ell$, en avant, sur le côté, et en hauteur, vérifie $(L/2)^2 = 3(\ell/2)^2$, soit  
    \eqbox{
    \ell  \simeq \frac{L}{\sqrt{3}} \simeq \SI{1.2}{m}.
    }
    }
    \item On note $E=2F\ell$ l'énergie fournie par le kayakiste à chaque cycle de coups de pagaie à gauche et à droite, de période $T=1/f$. Par des considérations énergétiques, montrer que 
    \begin{equation}
      V \sim \alpha f^{\frac{1}{3}},
    \end{equation}
    où $\alpha$ est une constante à déterminer. \AN~pour un kayakiste de force $F = \SI{200}{N}$.
    \sol{
      L'énergie vaut $E = \SI{480}{J}$. Le TEC appliqué au kayak sur un cycle de durée $T$, où le kayak se déplace d'une distance $VT$, s'écrit :
      $$
    \Delta E_c = E = F_k V T = C_x \rho V^3 S/2f, \Rar V^3 = \frac{2E}{C_x \rho S} f.  
      $$
      Cette loi est de la forme annoncée avec
      \eqbox{
        \alpha =\left( \frac{2E}{\rho S C_x} \right)^{\frac{1}{3}}
        % = \left( \frac{2*480}{1E3 *6E-3 *0.5} \right)^{\frac{1}{3}}
         \simeq \SI{6.8}{m.s^{-2/3}}.
        }
    }
  \end{quest}

  La nage libre est une discipline olympique où le nageur se propulse en crawl, en poussant l'eau vers l'arrière avec ses mains à la vitesse $v$, alternativement de gauche à droite. La partie immergée du corps du nageur présente une section transverse à l'écoulement $S=\SI{8e-2}{m.s^{-1}}$ plus grande que celle d'un kayak, tandis que la section de l'avant-bras $s=\SI{3e-2}{m.s^{-1}}$ est inférieure à celle d'une pagaie, ce qui explique les vitesses $V\simeq\SI{2}{m.s^{-1}}$ plus faibles atteintes par le nageur. 
  \begin{quest}
    \item	Estimer le nombre de Reynolds $\Re$ associé à l'écoulement autour du nageur. En déduire la nature de l'écoulement, l'ordre de grandeur du coefficient de traînée $C_x$, et l'expression des forces de traînée que l'eau exerce sur la main, propulsive, notée $F_m$, et le nageur,  notée $F_n$.
    \sol{
      L'eau est de masse volumique $\rho = \SI{1000}{kg.m^{-3}}$, et de viscosité $\eta = \SI{1e-3}{Pa.s}$. La longueur transverse au mouvement est $L=\sqrt{S}\simeq \SI{0.28}{m}$.
      Le nombre de Reynolds vaut donc 
      \eqbox{
      \Re = \frac{\rho V L}{\eta} \sim \SI{5.6e5}.
      }
       L'écoulement est donc turbulent, et le coefficient de traînée est constant et vaut environ $C_x = \num{0.5}$. Les forces de traînée ont alors pour expression 
       $F_p = \frac{1}{2} \rho v^2 s C_x$, et $F_k = \frac{1}{2} \rho V^2 S C_x$.
    }
    \item  Par le principe des actions réciproques, montrer que  $V\sim v$ en pratique. En déduire que le temps de propulsion $t$, pendant lequel chaque main reste immergée sous l'eau, est indépendant de la cadence~$f$ dès lors que $2ft<1$, et dépend uniquement des paramètres physiologiques du nageur.
    \sol{
    À l'équilbre, la force que l'eau exerce sur la main compense celle qui freine le nageur, et donc $F_m=F_n$. On en déduit 
    $$
    \frac{1}{2} \rho v^2 s C_x = \frac{1}{2} \rho V^2 S C_x \quad \Rightarrow \quad \boxed{v = \sqrt{\frac{S}{s}}V \simeq \SI{1.6}{} V \sim V.}
    $$
    En bonne approximation, la pagaie est donc immobile dans l'eau pendant la poussée du kayakiste.
    }
    \item On note $P=2Ft$ la quantité de mouvement produite par le nageur à chaque cycle de nage, de période $T=1/f$. Par des considérations énergétiques, montrer que 
    \begin{equation}
      V \sim \beta f^{\frac{1}{2}},
    \end{equation}
    où le paramètre $\beta$, appelé indice de nage, est une constante à déterminer. \AN~pour un nageur de force $F = \SI{100}{N}$ et de temps de propulsion $t=\SI{0.4}{s}$.
    \sol{
      La quantité de mouvement vaut $P = \SI{80}{kg.m.s^{-1}}$. Le PFD appliqué au corps du nageur et intégré sur un cycle de durée $T$ s'écrit :
      $$
    \Delta p = P = F_n T = C_x \rho V^2 S/f, \Rar V^2 = \frac{P}{C_x \rho S} f.  
      $$
      Cette loi est de la forme annoncée avec
      \eqbox{
        \beta =\left( \frac{2P}{\rho S C_x} \right)^{\frac{1}{2}} \simeq \SI{1.4}{m.s^{-1/2}}.
        }
    }
  \end{quest}
\end{exo}

\begin{exo}[PC][3][TD]{Modèle microscopique de la viscosité}
  \Notions{\item Cinétique des gaz \task Gaz parfait}
  \Annale[Physique PC]{CCMP}{2025}
  Nous considérons l'écoulement de Couette plan d'un gaz parfait. En plus de la vitesse moyenne de l'écoulement $\Vec{v}(M,t) = v_x(z) \ex$, les particules sont animés de mouvements désordonnés de vitesse moyenne $u$. Soit $\ell$ le libre parcours moyen, $n$ le nombre de particules par unité de volume, et $m$ leur masse. Pour simplifier le problème, on suppose que la distribution des vitesses des particules à l'échelle microscopique est isotrope dans les trois directions de l'espace $(Ox,Oy,Oz)$.

  \begin{quest}
    \item Que vaut la quantité de mouvement que transporte une particule à la cote $z+\ell$ ? En supposant que les particules cèdent toutes leur quantité de mouvement, en déduire la quantité de mouvement totale transférée par les particules de cote $z+\ell$ à celles de cote $z$ sur une section horizontale $S$ pendant une durée élémentaire $\d t$.
    \sol{
    La quantité de mouvement transportée par une particule à la cote $z + \ell$ vaut 
    $$
    p_x = m v_x(z+\ell).
    $$
    Comme $\ell$ représente la distance moyenne entre deux collisions, on peut considérer que toutes les particules situées à la cote $z+\ell$ cédent leur quantité de mouvement aux particules situées à la cote $z$, lorsque celles-ci se déplacent du haut vers le bas. Le nombre de particules de ces particules dans le cylindre de section $S$ et de hauteur $u \d t$ vaut
    $$
    \d N = \frac{1}{6} n  \d V = \frac{1}{6} n u S\d t.
    $$
    La quantité de mouvement totale transférée par ces particules est donc
    \eqbox{
    \d P_{x,+} = \d N \, p_x = \frac{1}{6} n u S \, m v_x(z+\ell) \, \d t.
    }

    }
    \item Même question pour la quantité de mouvement totale transférée par les particules de cote $z-\ell$ à celles de cote $z$. En déduire la variation de quantité de mouvement des particules dans la couche de fluide de section $S$ comprise entre les cotes $z\pm \ell$, dans l'hypothèse des milieux continus.
    \sol{
    De la même façon, la quantité de mouvement totale transférée par les particules situées à la cote $z-\ell$ aux particules situées à la cote $z$ est
    \eqbox{
    \d P_{x,-} = \frac{1}{6} n u S \, m v_x(z-\ell) \, \d t.
    }
        La variation de quantité de mouvement des particules dans la couche de fluide de section $S$ comprise entre les cotes $z\pm \ell$ est donc
    \eqbox{
    \d P_x = \d P_{x,+} - \d P_{x,-} = \frac{1}{6} n u S m \left[ v_x(z+\ell) - v_x(z-\ell) \right] \d t \simeq \frac{1}{3} n u S m \ell \dr{v_x}{z} \d t,
    }
    dans l'hypothèse des milieux continus, où la fonction $v_x(z)$ est supposée suffisamment régulière sur l'échelle microscopique $\ell$.
    }
    \item En déduire la force de viscosité $\Vec{F}$ subie par ces particules et l'expression de la viscosité dynamique $\eta$ d'un gaz parfait.
    \sol{
    La force de viscosité subie par ces particules est
    \eqbox{
    \Vec{F} = \frac{\d P_x}{\d t} \e_z = \frac{1}{3} n u m \ell S \dr{v_x}{z} \e_z.
    }
    En identifiant cette expression à la définition de la viscosité dynamique $\Vec{F} = \eta S \dr{v_x}{z} \e_z$, on trouve
    \eqbox{
    \eta = \frac{1}{3} n m u \ell.
    }
    }
    
    \item \AN. On considère de l'air à $\SI{20}{\degree C}$, de masse molaire moyenne $M=\SI{29}{g.mol^{-1}}$ et de pression $p=\SI{1}{atm}$.  On pourra assimiler le dioxygène et le diazote à des sphères dures de diamètre $d = \SI{3.7e-10}{m}$. Calculer $n$, $\ell$, $u$,  et $\eta$. Comparer à la viscosité expérimentale $\eta_{\rm exp} = \SI{1.8e-5}{Pa.s}$.
    \sol{
      \begin{enumerate}
        \item La densité particulaire de l'air s'obtient à partir de l'équation d'état des gaz parfaits $p = n k_B T$, où $k_B = \SI{1.38e-23}{J.K^{-1}}$ est la constante de Boltzmann, et $T=\SI{293}{K}$ la température. On trouve
        \eqbox{
        n = \frac{p}{k_B T} \simeq \SI{2.5e25}{m^{-3}}.
        }
        \item Le libre parcours moyen s'obtient à partir de la section efficace de collision $\sigma = \pi d^2$, où $d \simeq \SI{3.7e-10}{m}$ est le diamètre effectif d'une molécule de dioxygène et de diazote. On trouve
        \eqbox{
        \ell \sim \frac{1}{n \sigma} \simeq \SI{7e-8}{m}.
        }
        \item La vitesse quadratique moyenne s'obtient à partir de l'énergie cinétique moyenne par particule $\frac{1}{2} m u^2 = \frac{3}{2} k_B T$. En toute rigueur, c'est la vitesse moyenne et non la vitesse quadratique moyenne qui intervient ici. On peut se contenter d'une analyse dimensionnelle. On trouve
        \eqbox{
        u \sim \sqrt{\frac{k_B T}{m}} = \sqrt{\frac{RT}{M}} \simeq \SI{5e2}{m.s^{-1}}.
        }
        \item La viscosité dynamique s'obtient en remplaçant les résultats précédents dans l'expression microscopique de la viscosité. On trouve
        \eqbox{
        \eta = \frac{1}{3} n m u \ell \simeq \SI{2e-5}{Pa.s}.
        }
        L'ordre de grandeur est correct.
      \end{enumerate}
    }
  \end{quest}
\end{exo}

\begin{exo}[PC][1]{Écoulement de Couette plan}
  \Notions{\item Écoulement de Couette \task Équation de Navier-Stokes}
Un fluide homogène de masse volumique $\rho$ et de viscosité dynamique $\eta$ s'écoule en régime stationnaire entre deux parois planes situées en $z=\pm\textstyle\frac{e}{2}$. Celle du dessous est fixe, tandis que celle du dessus se déplace à la vitesse $U\,\ex$. On suppose que le champ des vitesses est de la forme $\Vec{v}(M)=v_x(z)\,\ex$
et le champ de pression de la forme $p(z)$. On néglige la pesanteur.
\begin{quest}
\item Montrer que le champ de pression ne dépend pas de $z$. Montrer ensuite que le champ des vitesses vérifie l'équation 
\begin{equation}
  \ddd{v_x}{z}=0.
\end{equation}
Déterminer $v_{x}(z)$ à partir des conditions aux limites.
\sol{

\parag{Hypothèses}
On suppose l’écoulement stationnaire : l’accélération locale $\dr{\Vec{v}}{t}$ est nulle.
Comme toujours pour un fluide newtonien, on suppose l’écoulement incompressible : $\Dive{\Vec{v}}=\Vec{0}$.
En raison de la géométrie du problème, on cherche un champ de vitesse de la forme $\vv{v}(M,t)=v(y)\vv{e_x}$, qui est bien de divergence nulle.
Comme $v$ est dirigé suivant $x$ mais ne dépend que de $y$, l’accélération convective $(\Vec{v}\cdot \Grad{})\Vec{v}$ est nulle également.
L'équation de Navier-Stokes se réduit à
\begin{equation*}\label{eq:ns}
-\grad{p}+\eta\Delta\vv{v}=\Vec{0}.
\end{equation*}

\parag{Détermination du champ de pression}
En projetant l’\Eq{ns} sur la verticale $\vv{e_y}$, on trouve
\begin{equation*}
-\dfrac{\partial p}{\partial z}= 0,
\end{equation*}
qui s’intègre pour trouver $p(z)=\cte$.

\parag{Détermination du champ de vitesse} D’après l’\Eq{vcouette}, on a
\begin{equation*}
\frac{\d^2 v_x}{\d z^2}=0 \quad\Rightarrow \quad v_x(z)=az+b.
\end{equation*}
Les constantes $a$ et $b$ s’obtiennent
à partir des conditions aux limites sur la vitesse,
\begin{equation*}
\begin{cases}
v_x(z=-e/2)=0=-ae/2+b,\\
v_x(z=e/2)=U=ae/2+b,
\end{cases}
\Rightarrow  \quad v_x(z)=\frac{U}{e}(z+e/2).
\end{equation*}
}

\item La paroi mobile est de surface $S$. Calculer la force visqueuse $\Vec{F}_{\rm  f \rightarrow  p}$ exercée par le fluide sur la plaque.
\sol{
Force visqueuse
$$
  \Vec{F_v} = -\eta S \dr{v_x}{z} \ex = -\eta S \frac{U}{e} \ex.
$$
Par les actions réciproques, la force exercée par l'opérateur vaut $\Vec{F}=-\Vec{F_v}$.
}
\end{quest}
\end{exo}

\begin{exo}[PC][1][TD]{Écoulement de Poiseuille plan}
  \Notions{\item Écoulement de Poiseuille \task Équation de Navier-Stokes \task Écoulement plan}
Un fluide de masse volumique $\rho$ et de viscosité dynamique $\eta$ s'écoule en régime stationnaire entre deux parois fixes et planes situées en $z=\pm e/2$, avec un champ des vitesses de la forme $\Vec{v}(M)=v_x(z)\,\ex$
et un champ de pression de la forme $p(x,z)$. On note $L$ la longueur de la conduite et on néglige la pesanteur.
\begin{quest}
\item 
Montrer que le champ de pression ne dépend pas de $z$. Montrer ensuite que le champ des vitesses vérifie l'équation 
\begin{equation}
  \ddd{v_x}{z} =  \frac{C}{\eta}, \quad \text{avec } C = \frac{\der p}{\der x}=\cte.
\end{equation}
 \sol{

\udl{Hypothèses}
\begin{liste}
  \item écoulement stationnaire : l’accélération locale $\dr{\Vec{v}}{t}$ est nulle.
  \item pour appliquer l'équation de Navier-Stokes à un fluide newtonien,  l’écoulement doit être incompressible, ce qui est le cas ici car 
  
  $$\Dive{\Vec{v}}=\dr{v_x}{x} = 0.$$
  \item l’accélération convective  est nulle car 
  $$
  (\Vec{v}\cdot \Grad{})\Vec{v} = v_x \dr{v_x}{x} \ex = \Vec{0}.
  $$
\end{liste}
L'équation de Navier-Stokes se réduit alors à
% \begin{equation*}\label{eq:ns}
  % $$
  \eqbox{\label{eq:ns}
    -\grad{p}+\eta\Delta\vv{v}=\Vec{0}.
  }
% $$
% \end{equation*}

\udl{Détermination du champ de pression} : en projetant l’\Eq{ns} sur la verticale $\ez$, on trouve
\begin{equation*}
-\dr{p}{z}= 0,
\end{equation*}
qui s’intègre pour trouver $p=p(x)$. En projetant l’\Eq{ns} sur l’horizontale $\ex$, on trouve
\begin{equation*}
\dd{p}{x}=\eta\frac{\d^2 v_x}{\d z^2}.
\end{equation*}
Or le terme de gauche ne dépend que de $x$ et le terme de droite que de $z$, donc les deux termes doivent être égaux à une constante $C$. On a donc 
% \begin{equation*}
  \eqbox{
    \frac{\d^2 v_x}{\d z^2}=\frac{C}{\eta}.
  }
% \end{equation*}
}


\item On pose $\Delta p=p(x=0)-p(x=L)$ la surpression entre l'amont et l'aval. Exprimer $v_{x}(z)$.
\sol{
On résout
\begin{equation*}
\frac{\d^2 v_x}{\d z^2} = \frac{C}{\eta} \quad\Rightarrow \quad v_x(z)=\frac{C}{2\eta}z^2 + az+b.
\end{equation*}
Les constantes $a$ et $b$ s’obtiennent
à partir des conditions aux limites sur la vitesse,
\begin{equation*}
\begin{cases}
v_x(z=-e/2)=0=\frac{C}{8\eta}e^2- a\frac{e}{2}+b,\\
v_x(z=e/2) = 0 = \frac{C}{8\eta}e^2 + a\frac{e}{2} + b,
\end{cases}
\Rightarrow  \quad 
\begin{cases}
  a &= 0,\\ 
  b &= -\frac{C}{8\eta}e^2.
  \end{cases}
\end{equation*}
De plus, $\dd{p}{x} = C$ donne $\Delta p = CL$ par intégration, et donc
\eqbox{
  v_x(z)=\frac{\Delta p}{2\eta L}\left(z^2-\frac{e^2}{4}\right).
}
}
\item Les parois sont de surface $S$. Calculer la force visqueuse $\Vec{F}_{\rm  f \rightarrow  p}$ exercée par le fluide sur une des plaques.
 \sol{
  La force visqueuse a pour expression
  $$
    \Vec{F_v} = -\eta S \dd{v_x}{z} \ex = - \frac{\Delta p S}{L}z \ex.
  $$
  Par les actions réciproques, la force exercée par le fluide sur la plaque supérieure vaut $\Vec{F}_{\rm  f \rightarrow  p} = - \Vec{F_v}(z=e/2)$, soit 
  \eqbox{
    \Vec{F}_{\rm  f \rightarrow  p} = \frac{\Delta p S e}{2L} \ex.
  }
  On peut remarquer que la force totale qui s'exerce sur les deux plaques vaut 
  $$
  \Vec{F}_{\rm  tot} = 2\Vec{F}_{\rm  f \rightarrow  p} = \frac{\Delta p}{L} Se \,\ex ;
  $$
  elle s'écrit comme le gradient de pression $ \frac{\Delta p}{L}$ fois le volume $V=Se$ de la conduite. Cela est cohérent car la force visqueuse compense  la force de pression. Elle est donc indépendante de la viscosité $\eta$ !
 }
\end{quest}
\end{exo}

\begin{exo}[PC][1]{Plaque oscillante et diffusion visqueuse}
Une plaque infinie confondue avec le plan $z=0$ oscille à la vitesse $\Vec{v_0}(t) =  U\cos(\omega t)\,\ex$ à la surface d’un fluide de masse volumique $\rho$ et de viscosité $\eta$ occupant l’espace $z<0$.
%On néglige la pesanteur et le champ de pression est supposé uniforme. Le champ des vitesses est de la forme $\Vec{v}(M,t)=v_x(z,t)\ex$.

Montrer que moyennant certaines hypothèses, la vitesse dans le fluide $v_x(z,t)$ vérifie une équation de diffusion, et la résoudre.

\sol{
On néglige la pesanteur et le champ de pression est supposé uniforme.
On suppose que le champ des vitesses est de la forme qu'en surface, soit $\Vec{v}(M,t)=v_x(z,t)\ex$, dont l'accélération convective vaut
$$
(\Vec{v} \cdot \grad) \Vec{v} = v_x \dr{}{x} \Vec{v} = \Vec{0}.
$$
L'équation de Navier-Stokes se réduit donc à
\begin{equation*}\label{eq:ns}
\eta\Delta\vv{v}=\rho \dr{\Vec{v}}{t} \Rightarrow \dr{v_x}{t} = \nu \drm{v_x}{z}{2}.
\end{equation*}
Il s'agit bien d'une équation de diffusion.


L'équation de diffusion, qui est linéaire, admet nécessairement comme solution une fonction qui oscille à la même pulsation $\omega$.
Pour faciliter les calculs, travaillons en notation complexe,
on cherche une solution de la forme $\underline{v_x}(z,t)=T_{0}+f(z)\,\e^{i\omega t}$, où la fonction spatiale $f(z)$ reste à déterminer.
L'équation de diffusion donne alors
\begin{equation}
i\omega f\e^{i\omega t}=\nu\frac{\d^2 f}{\d z^2}\e^{i\omega t}\quad\Rightarrow\quad \frac{\d^2 f}{\d z^2}-i\frac{\omega}{\nu}f = 
\end{equation}
En cherchant $f$ sous la forme $f(z) \propto e^{-ikz}$, nous trouvons
\begin{equation}
k^2=i\frac{\omega}{\nu} \Rightarrow k = k=\pm\frac{1+i}{\sqrt{2}}\sqrt{\frac{\omega}{\nu}} = \pm\frac{1+i}{\delta},
\end{equation}
où $\delta = \sqrt{2\nu/\omega}$ est homogène à une distance. Comme $k$ peut avoir deux valeurs, on obtient $f$ comme combinaison linéaire
\begin{equation}
f(z)=\underline{A}\,\exp\left[(1+i)\frac{z}{\delta}\right]+\underline{B}\,\exp\left[-(1+i)\frac{z}{\delta}\right].
\end{equation}
Une divergence de l'amplitude en $z\rightarrow\infty$ n'étant pas possible physiquement, nécessairement $\underline{A}=0$. On obtient ensuite $\underline{B}$ grâce aux conditions aux limites,
\begin{equation}
\underline{v_x}(z=0,t)=f(z=0)e^{i\omega t}=\underline{B}e^{i\omega t} = U e^{i\omega t}.
\end{equation}
Il vient finalement
\begin{equation}
\begin{split}
\underline{v_x}(z,t)=U\,\exp\left[-(1+i)\frac{z}{\delta}\right]\e^{i\omega t},
\end{split}
\end{equation}
puis en prenant la partie réelle
\begin{equation}
v_x(z,t) = U\,\e^{-z/\delta}\cos\left(\omega t-\dfrac{z}{\delta}\right)\qquad\text{où}\qquad\delta=\sqrt{\frac{2\nu}{\omega}}=\sqrt{\frac{\nu \tau}{\pi}}.
\end{equation}
La grandeur $\delta$ est appelée \imp{longueur de pénétration} de l'onde dans le milieu.
Elle est responsable d'une part d'une atténuation exponentielle de l'onde avec la profondeur, et d'autre part d'un déphasage de l'onde croissant avec la profondeur $\Delta\varphi = z/\delta$. Qualitativement, $\delta$ donne la distance typique sur laquelle les variations de température sont sensibles.
%Pour le sol, $\nu = \SI{\item 0e-7}{m^\item s^{-1}}$ .
}

  L’écoulement incompressible d’un fluide newtonien est décrit par l’équation de Navier-Stokes,
  $$
  \rho\dr{\Vec{v}}{t} + \rho \left( \Vec{v} \cdot \grad \right) \Vec{v} = \Vec{0} = -\grad p + \eta \Delta \Vec{v} + \rho \Vec{g}.
  $$
  \item \udl{Champ de pression}. On projette sur l’axe de l’écoulement, et on obtient
  $$
  0 = -\dr{p}{x} + \eta \ddd{v_x}{z} + \rho g \sin \alpha \Rar \dr{p}{x} = \eta \ddd{v_x}{z} + \rho g \sin \alpha.
  $$
  Le membre de droite ne dépend que de $z$, ce qui implique que $\dr{p}{x}$ est une constante vis-à-vis de $x$. Comme l’écoulement est invariant par translation suivant $x$, on doit avoir $\dr{p}{x}=0$, et donc 
 $ p(z)$ ne dépend pas de $x$.
\end{itemize}
}
\item Déterminer la vitesse $v_x(z)$ à deux constantes $A$ et $B$ près. Quelle est la condition aux limites en $z=0$ ? En $z=h$ ?
\sol{
On repart de l’expression précédente. On a 
$$
\ddd{v_x}{z} = -\frac{\rho g \sin\alpha}{\eta}.
$$
Par intégration, on trouve, 
$$
v_x(z) = -\frac{\rho g \sin\alpha}{2\eta} z^2 + A z + B.
$$
Les conditions aux limites sont les suivantes, 
\eqbox{
  \left\{
    \begin{array}{llll}
      z=0 : & \text{vitesse nulle} : & v_x(z=0) & = 0, \\
      z=h : & \text{force visqueuse nulle} : & \dd{v_x}{z}(z=h) & = 0. \\
    \end{array}
    \right.
  }
}
\item En déduire l’expression complète de la vitesse $v_x(z)$. Représenter le profil de vitesse dans l’écoulement. Quelles analogies voyez-vous avec les écoulements de Couette et de Poiseuille ?
 \sol{
On en déduit les constantes d’intégration,
$$
\left\{
\begin{array}{ll}
  B & = 0,\\
A & = \frac{\rho g \sin\alpha}{\eta} h.  \\
\end{array}
\right.
$$
L’expression complète du champ de vitesse est alors
\eqbox{
  v_x(z) = \frac{\rho g \sin\alpha}{2\eta} z \left( 2h - z \right).
}
L’écoulement est similaire à l’écoulement de Couette dans la mesure où une seule plaque est fixe, tandis que la surface libre du fluide est mobile. L’écoulement est similaire à l’écoulement de Poiseuille dans la mesure où la vitesse est quadratique en $z$. L’écoulement est induit ici par la pesanteur, mais pas par un gradient de pression.
}
\item On suppose que la couche de lave est de largeur $L$ selon l'axe $y$. Calculer le débit volumique de lave $\DV$ et en déduire la vitesse moyenne $v_{\rm moy}$ de l'écoulement.
\sol{
Le débit volumique à travers une section $S=Lh$ de l’écoulement est donné par
$$
\DV = \iint_S \Vec{v} \cdot \dS =  \int_0^L \d y \int_{0}^{h} v_x(z) \, \d z = \frac{\rho g \sin\alpha L}{2\eta} \int_{0}^{h} z \left( 2h - z \right) \, \d z, 
$$
On trouve alors
\eqbox{
  \DV = \frac{\rho g \sin\alpha L h^3}{3\eta}.
  }
La vitesse moyenne est donnée par $\DV = v_{\rm moy} S$, soit
% $$
\eqbox{
  v_{\rm moy} = \frac{\rho g h \sin\alpha}{3\eta}.
  }
% $$
}
\item \AN~: déterminer $v_{\rm{moy}}$ pour une couche de  lave de viscosité $\eta=\SI{2e4}{SI}$, de masse volumique $\rho=\SI{2,7}{g.cm^{-3}}$, de hauteur $h=\SI{30}{cm}$ et un angle $\alpha=\SI{5}{^\circ}$. Évaluer également le nombre de Reynolds $\Re$ et commenter.
\sol{
On trouve $v_{\rm{moy}} = \SI{1.2}{cm.s^{-1}}$. Le nombre de Reynolds vaut $\Re = \frac{\rho v_{\rm{moy}} h}{\eta} = \num{4.7e-4}$. L’écoulement est donc laminaire, ce qui justifie de négliger l’accélération convective.
}
\end{quest}
\end{exo}

\begin{exo}[PC][3][TD]{Rhéomètre piézoélectrique}
  \Notions{\item Équation de Navier-Stokes \task Débit volumique \task Force de pression}
On considère l'écoulement incompressible d'un fluide newtonien, de masse volumique $\rho \sim \SI{e3}{kg.m^{-3}}$ et de viscosité dynamique $\eta \sim \SI{1}{Pl}$, placé entre deux lames de verres parallèles et circulaires de rayon $R$, collées sur deux céramiques piézoélectriques. La lame supérieure est motorisée et impose une variation sinusoïdale de l'épaisseur $h(t) = h_0 + \veps \cos(2\pi f t ) $ d'amplitude $\veps \sim \SI{10}{\micro m} \ll h_0 \ll R$ à une fréquence $f$ comprise entre \SI{0,1}{Hz} et \SI{10}{kHz}. On note $\gamma =\textstyle\frac{1}{h}\frac{\d h}{\d t}$ le taux de déformation. La lame inférieure est reliée à un capteur qui permet de mesurer la force normale $\vv{F}=F(\gamma)\,\ez$ exercée par le fluide sur la plaque.
Le champ des vitesses est supposé de la forme $\vv{v}(M,t)=v_r(r,z,t)\vv{e_r}$, et on néglige la pesanteur.

On admet que la céramique motrice L'épaisseur du film liquide est suffisamment faible pour que l'on puisse négliger la gravité.

\begin{quest}
\item Par un raisonnement en ordres de grandeur, justifier que si la fréquence est inférieure à une limite que l'on déterminera, on peut négliger le terme d'accélération locale devant la force volumique de viscosité. Cette condition est-elle vérifiée ici ?
\sol{
  L'écoulement incompressible d'un fluide newtonien est décrit par l'équation de Navier-Stokes.
Le terme d'accélération locale dans l'équation de Navier-Stokes est de l'ordre de 
$$\left|\rho \dr{\Vec{v}}{t}\right| \sim \rho \omega v.$$ La force volumique visqueuse est elle de l'ordre de 
$$\left|\eta \Deltav{v}\right| \sim \frac{\eta v}{\veps^2}.$$ Le rapport entre ces deux termes est négligeable à condition que
$$
\frac{\left| \rho \dr{\Vec{v}}{t}\right|}{\left|\eta \Deltav{v}\right|} \ll 1,
$$

soit 
\eqbox{
  \frac{\rho \omega \veps^2}{\eta}  \ll 1.
}
L'application numérique donne au maximum, pour $f = \SI{10}{kHz}$, la valeur 
$$
  \frac{\rho \omega \veps^2}{\eta} \sim \SI{e-7}{} \ll 1.
$$
}
\item Estimer le nombre de Reynolds $\Re$. Justifier que l'on peut négliger le terme d’accélération convective devant la force volumique de viscosité.
\sol{
Le nombre de Reynolds a pour expression 
\eqbox{
  \Re = \frac{\rho v \veps}{\eta} = \frac{\rho \omega \veps^2}{\eta} \sim \SI{e-7}{} \ll 1.
}
Le nombre de Reynolds s'interprète comme l'ordre de grandeur du rapport entre le terme convectif et la force volumique de viscosité. On peut donc négliger le terme d'accélération convective devant la force volumique de viscosité.
}
\item Par intégration de l'équation de Navier-Stokes, établir l'expression de $v_r(r,z,t)$ en fonction de $\eta$, $z$, $h(t)$ et du gradient radial de pression $\dr{p}{r}(r,t)$. On admet qu'ici le laplacien vectoriel se simplifie en
% $$

\begin{equation}
  \Deltav{v} = \ddr{v_r}{z} \,\er.
\end{equation}
% $$
\sol{
  Ici l'énoncé ne guide pas mais il faut justifier que le terme convectif est également négligeable devant la terme visqueux. Pour cela, on peut évaluer le nombre de Reynolds $\Re = \frac{\rho U L}{\eta}$. Dans un régime à bas Reynolds ($\Re \ll 1$), on peut négliger ce terme convectif. On peut alors aussi approximer le laplacien vectoriel à $\Delta \Vec{v} = \ddr{v}{z} \er$.

L'équation de Navier-Stokes donne alors :
$$
\rho \left[ \dr{\Vec{v}}{t} + (\Vec{v} \cdot \grad) \Vec{v} \right] = \Vec{0} =  -\grad p + \eta \Delta \Vec{v}.
$$
Par projection sur les axes $\er$ et $\ez$, on obtient
$$
\left\{
\begin{array}{ll}
  0 & = -\dr{p}{r} + \eta \ddr{v_r}{z}, \\[2mm]
  0 & = -\dr{p}{z}. \\
\end{array}
\right.
$$
On en déduit que $p(r,t)$ est indépendant de $z$, et que la vitesse vérifie
$$
\ddr{v_r}{z} = \frac{1}{\eta} \dr{p}{r} \Rightarrow v_r(r,z,t) = \frac{1}{\eta} \dr{p}{r}(r,t) \frac{z^2}{2} + A(r,t) z + B(r,t).
$$
Les conditions aux limites des écoulements visqueux imposent que la vitesse est nulle sur les parois, et donc
$$
  \left\{
\begin{array}{ll}
    v_r(r,0,t) & = 0 \Rightarrow B(r,t) = 0, \\[2mm]
    v_r(r,h(t),t) & = 0 \Rightarrow A(r,t) = -\frac{1}{2\eta} \dr{p}{r}(r,t) h(t).
\end{array}
\right.
$$
On obtient finalement
\eqbox{
  v_r(r,z,t) =\frac{1}{2\eta} \dr{p}{r}(r,t) z(z-h(t)).
}
}
\item En utilisant l'incompressibilité de l'écoulement, montrer que le débit volumique du liquide à travers la surface latérale d’un cylindre d'axe $(Oz)$ et de rayon $r < R$, orientée vers l'extérieure du cylindre, s'écrit 
\begin{equation}
\DV=-\frac{\pi r h(t)^3}{6\eta}\dr{p}{r}(r,t).
\end{equation}


\sol{
  On calcule le débit volumique sortant du cylindre en rouge ci-dessous.
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.6\columnwidth]{rheo_DV}
    %\capt{}
  \end{fig}
Le débit volumique est le flux du vecteur vitesse, soit
\begin{align*}
  \DV = \displaystyle \iint \Vec{v} \cdot \Vec{\d S} =  \int_0^{h(t)} \int_0^{2\pi} v_r(r,z,t) r \d\theta \d z & =2\pi r  \frac{1}{2\eta} \dr{p}{r}(r,t) \int_0^{h(t)} z(z-h(t)) \d z \\ & = \frac{\pi r }{\eta} \dr{p}{r}(r,t) \left( \left[\frac{z^3}{3}\right]_0 ^{h(t)} -  h(t)\left[\frac{z^2}{2} \right]_0^{h(t)} \right)
\end{align*}
d'où le résultat ,
\eqbox{
  \DV = -\frac{\pi r h(t)^3}{6\eta} \dr{p}{r}(r,t).
}
}
\bonus{
\item En déduire l'expression du gradient de pression $\textstyle\dr{p}{r}(r,t)$ en fonction de $\eta$, $h(t)$, $r$ et du taux de déformation $\gamma(t)$. Déterminer l'expression du champ de pression $p(r,t)$ en admettant qu'en $r=R$ la pression est égale à la pression atmosphérique $p_0$.
\sol{
  Le volume de fluide enfermé dans une section cylindrique de rayon $r$ vaut $V=\pi r^2h(t)$, et donc le débit volumique s'écrit aussi 
  $$\DV = \frac{\d V}{\d t} = \pi r^2 \dd{h}{t}(t) = \pi r^2 h(t) \gamma(t).$$ 
  En égalisant les deux expressions du débit volumique, on trouve
$$
 -\frac{\pi r h(t)^3}{6\eta} \dr{p}{r}(r,t) = \pi r^2 h(t) \gamma(t) \Rightarrow
\dr{p}{r} = -6\eta  r \frac{\gamma(t)}{h(t)^2}, 
$$
d'où l'on tire par intégration,
\eqbox{p(r,t) = p_0 + \frac{3\eta \gamma(t)}{h(t)^2} (R^2-r^2),}
en utilisant la condition aux limites $p(R,t)=p_0$.
}
\item Calculer enfin la résultante des forces de pression $\vv{F}$ qui s'exerce sur la lame inférieure de la cellule. En déduire comment mesurer la viscosité dynamique du fluide avec cet appareil.
\sol{
  Il s'exerce une pression $p$ sur la face du dessus et une pression $p_0$ sur la face du dessous de la lame inférieure en $z=0$.
  La résultante des forces de pression s'écrit donc
  $$
\Vec{F} = \iint (p_0-p) \d S \, \ez = 2\pi \int_0^R  \frac{3\eta \gamma(t)}{h(t)^2} (R^2-r^2) r \d r \ez.
  $$
  En approximant $h(t)$ par $h_0$, on trouve
  \eqbox{
   \Vec{F} \simeq  \frac{3\pi \eta R^4}{2 h_0^2}  \gamma \,\ez.
  }
  La force est proportionnelle au taux de déformation $\gamma$ et à la viscosité dynamique $\eta$. On peut donc mesurer la viscosité dynamique du fluide par régression linéaire de la force de pression $\vv{F}$ en fonction du taux de déformation $\gamma$.
}
}
\end{quest}

\begin{exo}[PC][1][devoir]{Glissement d'un liquide sur une paroi solide en microfluidique}
% \Annale[Physique PC]{CCINP}{2021}
\Enonce{
L'étude d'écoulements au sein de microsystèmes, c'est-à-dire de canaux de taille de l'ordre de quelques dizaines de micromètres, présente aujourd'hui un enjeu industriel majeur. Des recherches portent sur l'utilisation de surfaces hydrophobes texturées pour lesquelles l'expérience met en évidence des propriétés de glissement des liquides à l'échelle de la microfluidique, donc une réduction des effets de friction de l'écoulement sur les parois solides.
On propose ici d'étudier l'influence du glissement d'un liquide sur une paroi solide dans le cas d'un écoulement de Poiseuille plan.

Le liquide considéré est assimilé à un fluide newtonien de masse volumique $\rho$ et de viscosité dynamique $\eta$. L'écoulement est stationnaire, incompressible et unidirectionnel. Il s'effectue entre deux plans, fixes dans le référentiel d'étude galiléen, de longueur $L$ selon $\ex$, de largeur $\ell$ selon $\ey$, situés en $z=-h / 2$ et $z=+h / 2$ (\Fig{Poiseuille_microflu}).

\begin{fig}[Poiseuille_microflu]
  \centering
  \includegraphics[width=0.7\textwidth]{poiseuille_plan}
  % \includegraphics[width=0.7\textwidth]{2023_12_06_63fa7679f8cb8ccd5cfeg-12}
  \capt{Écoulement dans un canal rectangulaire (écoulement de Poiseuille plan).}
\end{fig}

L'écoulement est induit par un gradient de pression horizontal appliqué dans la direction $\ex$ de la forme $\textstyle\frac{\partial p}{\partial x}=-K$, avec $K>0$ une constante. Les effets de la pesanteur sont par ailleurs négligés. Dans ces conditions, le champ des vitesses en tout point $\rm M$ entre ces deux plans est de la forme $\Vec{v}(M)=v_{x}(z) \ex$ et l'équation de Navier-Stokes se réduit à la forme: 
\begin{equation}
  \label{eq:Navier_Stokes_Poiseuille}
  \rho(\Vec{v} \cdot \grad) \Vec{v}=-\grad p + \eta \Deltav{v}.
\end{equation}
}

\begin{quest}
\QR {Rappeler le sens physique de chacun des termes de l'équation de Navier-Stokes \Eq{Navier_Stokes_Poiseuille}.}
{
  Voir cours : terme convectif, force volumique de pression, force volumique de viscosité.
}

\QR {Calculer le nombre de Reynolds Re pour un écoulement d'eau dans un canal d'une hauteur de $10\,\mu \mathrm{m}$ à une vitesse typique de $1\,\mathrm{mm} \cdot \mathrm{s}^{-1}$. Comment se simplifie dans ces conditions l'équation de Navier-Stokes?}
{Le nombre de Reynolds vaut 
\eqbox{
  Re = \frac{\rho v h}{\eta} = \frac{10^3 \times 10^{-3} \times 10^{-5}}{10^{-3}} = 10^{-2}.
}
Comme $Re\ll 1$, on peut négliger la convection et l'équation de \bsc{Navier-Stokes} se simplifie en l’équation de Stokes 

\eqbox{-\vv{\textrm{grad}}p + \eta \Deltav{\vv v}=0}.
}

\QR {Justifier que la forme supposée du champ des vitesses (écoulement parallèle à $\ex$) conduit à la même simplification indépendamment de la valeur du nombre de Reynolds.}
{
  Comme $\vv v=v_x(z)\vv u_x$, on a $\left(\vv v.\vv{\textrm{grad}}\right)\vv v=v_x.\dfrac{\partial \vv v}{\partial x}+v_y.\dfrac{\partial \vv v}{\partial y}+v_z.\dfrac{\partial \vv v}{\partial z}=\vv 0$, donc on retrouve bien \eqbox{-\vv{\textrm{grad}}p + \eta \Deltav{\vv v}=0}.
}

\QR {Montrer finalement que l'équation de Navier-Stokes se réduit ici à : 
\begin{equation}
  -\frac{\partial p}{\partial x}+\eta \frac{\partial^{2} v_{x}}{\partial z^{2}}= 0.
\end{equation}}
{
  En projetant l'équation de \bsc{Navier-Stokes} sur $\vv u_x$, on obtient 
  \eqbox{
  {-\dfrac{\partial p}{\partial x}+\eta\dfrac{\partial^2v_x}{\partial z^2}=0}.
  }
}
\end{quest}
\Enonce{
La condition de non glissement habituellement adoptée dans les écoulements à une échelle macroscopique suppose la vitesse relative liquide-solide nulle. L'existence d'un glissement observé à l'échelle micrométrique impose de fait une modification des conditions aux limites pour le champ des vitesses en $z= \pm h / 2$. Navier puis Maxwell ont proposé d'écrire ces conditions sous la forme :

\begin{equation}
  v_{x}=-L_{\rm g} \frac{\partial v_{x}}{\partial z}\text { en } z = \pm h/2,
\end{equation}

où la longueur $L_{g}$ est appelée longueur de glissement.}

\begin{quest}
\QR {Déduire de ces conditions aux limites que 

\begin{equation}
v_{x}(z)=\frac{K L_{\rm g} h}{2 \eta}+\frac{K}{2 \eta}\left(\frac{h^{2}}{4}-z^{2}\right).  
\end{equation}
}
{
  Comme $\dfrac{\partial P}{\partial x}=-K$, on a $\dfrac{\partial^2v_x}{\partial z^2}=-\dfrac{K}{\eta}$, d'où $\dfrac{\partial v_x}{\partial z}=-\dfrac{Kz}{\eta}+A$ et $v_x(z)=-\dfrac{Kz^2}{2\eta}+Az+B$.

  D'après les conditions aux limites, on a $-\dfrac{Kh^2}{8\eta}+A\dfrac{h}{2}+B=-L_g\left(-\dfrac{Kh}{2\eta}+A\right)$ et $-\dfrac{Kh^2}{8\eta}-A\dfrac{h}{2}+B=L_g\left(\dfrac{Kh}{2\eta}+A\right)$

On en déduit que $-\dfrac{Kh^2}{4\eta}+2B=\dfrac{L_gKh}{\eta}$, d'où $B=\dfrac{L_gKh}{2\eta}+\dfrac{Kh^2}{8\eta}$.

De plus, on a $Ah=-2AL_g$, d'où $A=0$.

On obtient finalement 
\eqbox{v_x(z)=\dfrac{KL_gh}{2\eta}+\dfrac{K}{2\eta}\left(\dfrac{h^2}{4}-z^2\right)}.

}

\QR {À quelles positions théoriques $z_{0}$ le champ des vitesses devrait s'annuler ? Dans l'hypothèse où $L_{\rm g} \ll h$, montrer que $L_{g}=\left|z_{0}\right|-\textstyle\frac{h}{2}$. Justifier alors l'appellation de longueur de glissement attribuée à la longueur $L_{\rm g}$.}
{
  La vitesse s'annule lorsque $\dfrac{KL_gh}{2\eta}+\dfrac{K}{2\eta}\left(\dfrac{h^2}{4}-z_0^2\right)=0$, soit \eqbox{z_0=\pm \sqrt{L_g h+\dfrac{h^2}{4}}}.

Lorsque $L_g<<h$, on a $\vert z_0 \vert =\dfrac{h}{2}\sqrt{1+\dfrac{4L_g}{h}}\simeq \dfrac{h}{2}+L_g$.

On en déduit que 
\eqbox{L_g=\vert z_0 \vert-\dfrac{h}{2}}.

On parle de longueur de glissement, car il s'agit de la distance entre la position où la vitesse s'annule, et les parois du canal.
}

\QR {Exprimer le débit volumique $\DV$ à travers une section droite comprise entre les deux plans en fonction de $K$, $L_{\rm g}$, $h$, $\eta$ et $\ell$.}
{
  Le débit volumique s'écrit $D_v=\displaystyle\int_{-h/2}^{h/2}\left(\dfrac{KL_gh}{2\eta}+\dfrac{K}{2\eta}\left(\dfrac{h^2}{4}-z^2\right)\right)\ell dz$, soit 

$D_v=\dfrac{K\ell L_gh^2}{2\eta}+\dfrac{K\ell h^3}{8\eta}-\dfrac{K\ell}{2\eta}\left[\dfrac{z^3}{3}\right]_{-h/2}^{h/2}$ , d'où \eqbox{D_v=\dfrac{K\ell h^2}{2\eta}\left(L_g+\dfrac{h}{6}\right)}.
}

\QR {Exprimer $K$ en fonction de $L$ et de la perte de charge $\Delta p=p(x=0)-p(x=L)$ imposée entre l'entrée $(x=0)$ et la sortie $(x=L)$ du canal.}
{
  En intégrant, on a $P(L)-P(0)=-KL$, d'où \eqbox{K=\dfrac{\Delta P}{L}}.
}

\QR {En déduire que $\Delta p=R_{h} \DV$. Donner l'expression de $R_{\rm h}$ en fonction de $\eta, L_{\rm g}$ et des dimensions du canal. Justifier le nom de «résistance hydraulique » qu'on lui donne.}
{
  On a $K=\dfrac{2\eta D_v}{\ell h^2\left(L_g+\dfrac{h}{6}\right)}$, d'où $\Delta P=\dfrac{2\eta L D_v}{\ell h^2\left(L_g+\dfrac{h}{6}\right)}$, et \eqbox{R_h=\dfrac{2\eta L}{\ell h^2\left(L_g+\dfrac{h}{6}\right)}}.

On parle de résistance hydraulique car le débit volumique est l'équivalent de l'intensité, et la différence de pression est l'équivalent de la tension, donc on retrouve bien l'équivalent de la loi d'\bsc{Ohm} $U=RI$.
}

\QR {En considérant que $L_{g}=0,\!5 h$, calculer le rapport $R_{\rm h} / R_{\rm h, n g}$ où $R_{\rm h, n g}$ désigne la résistance hydraulique en l'absence de glissement $\left(L_{\rm g}=0\right)$. Conclure sur l'effet du glissement sur la résistance hydraulique.}
{
  Pour $L_g=\dfrac{h}{2}$, on a $R_h=\dfrac{3\eta L}{\ell h^3}$ et $R_{h,ng}=\dfrac{12\eta L}{\ell h^3}$, donc \eqbox{\dfrac{R_h}{R_{h,ng}}=\dfrac{1}{4}}.

Le glissement permet d'avoir une conduite qui s'oppose quatre fois moins à l'écoulement qu'en l'absence de glissement.
}

\end{quest}

\paragraph{Généralités}
\begin{quest}
  \QR {Définir la notion de ligne de courant associée aux particules de fluide. Est-ce une description de nature eulérienne ou lagrangienne?}
  {
    Une ligne de courant est la courbe telle qu'en chacun de ses points la particule de fluide coïncidente à une \textbf{vitesse tangente à la ligne}. C'est un \textbf{point de vue eulérien} comme une photo instantanée (différent de la trajectoire d'une particule).

  }
  \QR {Dans la \Fig{aile} sont représentées les lignes de courant pour un profil donné. En analysant ces lignes de courant, expliquer pourquoi l'écoulement stationnaire de l'air, supposé parfait, homogène et incompressible, génère une force de portance de l'avion.

  \centering
  \includegraphics[width=0.5\columnwidth]{2025_01_24_e263637573f0bd280f3fg-03}
  \capt{
    Lignes de courant autour du profil d'une aile d'avion
  }
\end{fig}
  }
{
  Considérons deux lignes de courants infiniment grandes $(A_\infty,A)$ et $(B_\infty,B)$.

  % \begin{wrapfigure}[5]{r}{50mm}
  %    \includegraphics[width=5cm]{portance}
  % \end{wrapfigure}

  \begin{fig}[label]
    \centering
    \includegraphics[width=0.5\columnwidth]{ligne_courant_aile.png}
    %\capt{}
  \end{fig}
  
  Les lignes de courant étant plus resserrées en $A$ qu'en $B$, cela signifie que la vitesse y est plus grande (conséquence de $\textrm{div} \vv v=0$ pour un écoulement incompressible).
  
  Par ailleurs, l'écoulement étant supposé stationnaire de l'air supposé parfait, homogène et incompressible, en utilisant le \textbf{théorème de \bsc{Bernoulli}} généralisé à un écoulement rotationnel, en négligeant les différences d'altitude, sur chacune des lignes de courant : 
  
  $$P_A+\rho\frac{v_A^2}{2}=P_\infty+\rho\frac{v_\infty^2}{2}\textrm{ et }P_B+\rho\frac{v_B^2}{2}=P_\infty+\rho\frac{v_\infty^2}{2}\textrm{ donc }P_A+\rho\frac{v_A^2}{2}=P_B+\rho\frac{v_B^2}{2}$$
  
  Sachant que $v_A>v_B$, on peut alors conclure que $\resm{P_A<P_B}$ et cette dépression va générer une force de portance.
}
  
  \QR {En pratique, pour étudier la répartition de pression \(P\) le long de l'intrados et de l'extrados, on définit une pression adimensionnée appelée coefficient de pression :
  
  \begin{equation}
    C_{P}=\frac{P-P_{\infty}}{\frac{1}{2} \cdot \rho_{\infty} \cdot v_{\infty}^{2}},
  \end{equation}

où \(P_{\infty}, \rho_{\infty}\) et \(v_{\infty}\) sont respectivement la pression, la masse volumique et la vitesse de l'écoulement incident loin de l'aile. On représente \(C_{P}\) pour l'intrados et l'extrados en fonction de \(x / L\) la position par rapport à la corde (\Fig{pression}). Justifier à quelle courbe, \(C_{P 1}\) ou \(C_{P 2}\), on associe l'intrados et l'extrados.

  \centering
  \includegraphics[width=0.5\columnwidth]{2025_01_24_e263637573f0bd280f3fg-03(1)}
  \capt{
    Coefficient de pression sur l'intrados et l'extrados.
  }
\end{fig}
  }
{
  D'après l'expression précédente $P_{extrados}<P_{intrados}$ donc en utilisant l'expression des coefficients de pression : 

  \eqbox{C_{P1}=C_{Pintrados}>C_{P2}=C_{Pextrados}}
}

\QR {Pourquoi observe-t-on toujours \(C_{P} \leq 1\) ?}
 {
  En reprenant le théorème de \bsc{Bernoulli} :

  $$P+\rho\frac{v^2}{2}=P_\infty+\rho\frac{v_\infty^2}{2}\Rightarrow P-P_\infty =\rho\frac{v_\infty^2}{2}\left(1-\frac{v^2}{v_\infty^2}\right)\Rightarrow \resm{C_P=\frac{P-P_\infty}{\rho\frac{v_\infty^2}{2}}=1-\frac{v^2}{v_\infty^2}\leq 1}$$
}

\end{quest}
\Enonce{
Par intégration de la différence des coefficients de pression entre intrados et extrados sur toute la corde du profil, on obtient le coefficient de portance, adimensionné,

% \[
\begin{equation}
  C_{z}=\frac{F_{z}}{\frac{1}{2} \cdot S_{r e f} \cdot \rho_{\infty} \cdot v_{\infty}^{2}},
\end{equation}
% \]

avec \(S_{\text {réf }}\) la surface de l'aile et \(F_{z}\), la force de portance exercée sur l'aile. Cette force de portance est perpendiculaire à l'écoulement (figure 4). \QR {Le coefficient de portance \(C_{z}\) est-il plus important pour une incidence de \(2^{\circ}\) ou \(6^{\circ}\) ? Justifier.}
{
  Pour un angle d'incidence de 6°, les écarts entre les coefficients de pression de l'intrados et de l'extrados sont plus grands (courbes en pointillés) que pour 2°.     \paragraph{Tube de Pitot}
    \Enonce{
    Le tube de Pitot est un des nombreux capteurs qui équipent l'avion. Il permet la mesure de la vitesse de l'avion, donnée essentielle à sa bonne conduite. Il s'agit d'un tube très fin (moins de \(5 \mathrm{~mm}^{2}\) de surface) qui est placé parallèlement à la direction de l'écoulement de l'air (\Fig{pitot}). Ce tube possède deux ouvertures en \(F\) et \(G\). L'ouverture en \(F\) est la prise dite de pression totale et celle en \(G\) est la prise dite de pression statique. On mesure la différence de pression de l'air entre les deux tubes 1 et 2 avec un manomètre différentiel, ce qui permet d'obtenir la vitesse \(v_{\infty}\) de l'écoulement.

    \begin{fig}[pitot]
      \centering
      \includegraphics[width=0.5\columnwidth]{2025_01_24_e263637573f0bd280f3fg-07}

      \QR {Représenter l'allure de la ligne de courant qui aboutit en \(F\) et l'allure de la ligne de courant qui longe le tube et passe à proximité de \(G\).}
      {
        La ligne de courant qui aboutit en $F$ se termine en ce point qui est un point d'arrêt par symétrie (le fluide est au repos dans le tube). Celle qui passe en G est en fait très peu perturbée depuis l'infini car le tube est très fin.

        \begin{center}
        \includegraphics[width=9cm]{pitot_aile_avion.png}
        \end{center}
        
      }
      
      \QR {Déterminer, en fonction de \(P_{\infty}, \rho_{\infty}\), et \(v_{\infty}\), les expressions de la vitesse \(v_{F}\) et de la pression \(P_{F}\) du fluide en \(F\) ainsi que la vitesse \(v_{G}\) et la pression \(P_{G}\) du fluide en \(G\).}
      
      {
        Les conditions sont réunies pour appliquer le théorème de \bsc{Bernoulli} le long d'une ligne de courant, en négligeant les effets de pesanteur. On suit alors les 2 lignes de courant précédentes.

        Entre l'infini et $F$, on peut écrire : $P_{\infty}+\rho\frac{v_\infty^2}{2}=P_{F}+\rho\frac{v_F^2}{2}$ or $v_F=0$ donc \eqbox{P_F=P_{\infty}+\rho\frac{v_\infty^2}{2}}.
        
        Entre l'infini et $G$, on obtient : $P_{\infty}+\rho\frac{v_\infty^2}{2}=P_{G}+\rho\frac{v_G^2}{2}$. Or la ligne de courant passant par $G$ est peu perturbée par le tube donc $\resm{v_G\approx v_\infty}$ et donc $\resm{P_G\approx P_{\infty}}$.
      }

      \QR {Dans le manomètre, il y a un liquide de masse volumique \(\rho_{l}\). On mesure une différence d'altitude \(h\) entre les deux surfaces du liquide. Déterminer l'expression de la différence de pression, \(P_{H} -P_{I}\), entre ces deux surfaces.}
      
      {
        Le liquide dans le tube est à l'équilibre. On peut donc utiliser la statique des fluides : $P_{inf}=P_{sup}+\rho_l g \Delta z$, ce qui donne ici : \eqbox{P_H-P_I=\rho_l g h}.

      }

      \QR {Déduire des questions précédentes l'expression de la vitesse de l'écoulement \(v_{\infty}\) de l'air en fonction de \(\rho_{l}, \rho_{\infty}, g\) et \(h\). Comment évolue \(h\) lorsque la vitesse de l'air augmente?}
      {
        Si on néglige les effets de pesanteur sur l'air, on peut écrire : $P_H\approx P_F$ et $P_I\approx P_G=P_\infty$. On injecte dans la relation précédente :

        $$P_\infty+\rho_\infty \frac{v_\infty^2}{2}-P\infty=\rho_l g h\Rightarrow \resm{v_\infty=\sqrt{\frac{2\rho_l g h}{\rho_\infty}}}$$
        
        Lorsque la vitesse de l'air augmente, $h$ augmente.
      }

    \end{quest}

  \QR {Montrer que le champ électrique entre les armatures a pour expression : \(\vec{E} =-\frac{\sigma}{\epz} \ez \).}
  {
    Démonstration classique du cours, dont voici les grandes lignes. On calcule dans un premier temps le champ créé par une plaque seule. On utilise les symétries et invariances pour montrer que le champ est de la forme $\vv E=E(z)\vv u_z$ (pour des armatures infinies). On applique le théorème de \bsc{Gauss} sur un cylindre droit dont les faces du haut et du bas sont en $z$ et $-z$ pour exploiter à nouveau une symétrie. On obtient $\vv E=\pm\frac{\sigma}{2\varepsilon_0}\vv u_z$ de part et d'autre.

    Reste à combiner l'effet des 2 plaques : le champ créé par la plaque chargée $+\sigma$ vaut $+\frac{\sigma}{2\varepsilon_0}\vv u_z$ au-dessus de la plaque, et $-\frac{\sigma}{2\varepsilon_0}\vv u_z$. On a l'effet opposé pour l'autre plaque. Par superposition, on obtient que le champ est nul à l'extérieur du condensateur et vaut \eqbox{\vv E=-\frac{\sigma}{\varepsilon_0}\vv u_z} entre les armatures.
    
  }

  \QR {Déterminer l'expression de la capacité \(C\) du condensateur plan.
  
  
  On admet que la capacité d'un condensateur plan placé dans un milieu diélectrique de permittivité relative \(\varepsilon_{r}\) est obtenue en remplaçant, dans l'expression de la capacité \(C\) obtenue à la question précédente, \(\epz\) par \(\epz  \varepsilon_{r}\).}
  
  {
        On calcule d'abord $U=V(z=e)-V(z=0)=\int dV=-\int\vv{E}.\vv{dl}=\frac{\sigma}{\varepsilon_0}\int_{0}^{e}dz
    \Leftrightarrow U=\frac{\sigma}{\varepsilon_0}e$
    
    Or par définition, $C=\frac{Q_1}{V_1-V_2}$, donc ici,
    $C=\frac{Q}{\frac{Qe}{S\varepsilon_0}}$, soit finalement :
    
    
    \eqbox{C=\frac{S\varepsilon_0}{e}}
    
  }
