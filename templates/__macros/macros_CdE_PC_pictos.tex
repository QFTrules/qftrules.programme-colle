%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Include Graphical files  %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\includeCDCGraphics}[3]
{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\pathToGraphicalFile{#1/#2}
\IfFileExists{#2}
	% then
    {\def\pathToGraphicalFile{#2}}
    % else
    {
        \IfFileExists{__macros/#1/#2}
        	% then
        	{\def\pathToGraphicalFile{__macros/#1/#2}}
        	% else
        	{
                \IfFileExists{/__macros/#1/#2}
                	% then
                	{\def\pathToGraphicalFile{/__macros/#1/#2}}
                	% else
                	{
                		\IfFileExists{../__macros/#1/#2}
                			% then
                			{\def\pathToGraphicalFile{../__macros/#1/#2}}
                			% else
                			{
                				\IfFileExists{../../__macros/#1/#2}
                				% then
                				{\def\pathToGraphicalFile{../../__macros/#1/#2}}
                				% else
                				{}
                			}
                	}
        	}
	}
%	
\includegraphics[#3]{\pathToGraphicalFile}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%
}
