\section{Présentation du matériel}

\Enonce{
Les câbles coaxiaux sont utilisés pour le transport de signaux informatifs, par exemple pour la télévision, la téléphonie, ou internet. Un câble coaxial a l’intérêt de transporter des signaux de fréquence plus élevées que les fils électriques communs. On présente une coupe d'un câble
ci-dessous.

%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{figure}[h!]
\centering
\includegraphics[scale=0.3]{coax.png}
\caption{Constitution d’un câble coaxial.}
\label{fig:coax}
\end{figure}
%%%%%%%%%%%%%

Un câble coaxial est constitué, de l'intérieur vers l'extérieur, d'une tresse en cuivre (l'âme), d'un isolant électrique (diélectrique), d'une deuxième tresse de cuivre (gaine) et enfin d'une enveloppe plastique isolante pour protéger
l'ensemble.
%On s’intéresse à la propagation d’une onde électrique dans un câble coaxial.
}

\section{Étude de la réflexion}

\Enonce{
On rappelle la définition du coefficient de réflexion
% \begin{equation}
$r = V_r/V_i$
% \end{equation}
où la tension incidente $V_i$ et la tension réfléchie $V_r$ sont celles en sortie du câble ($x=L$). On suppose qu’on place une impédance réelle en sortie du câble, de résistance $R$.
Le coefficient de réflexion en amplitude en bout de câble vaut alors
\begin{equation}
\label{eq:r}
r = \frac{R - Z_c}{R + Z_c},
\end{equation}
avec $Z_{c} = \SI{50}{\Omega}$ l'impédance caractéristique du câble.
}


\subsection{Observations qualitatives}
\Enonce{
On envoie de nouveau signal carré de fréquence $f \simeq \SI{100}{kHz}$ dont l’amplitude varie entre $\SI{0}{V}$ et $\SI{5}{V}$.
}
% un signal carré de hauteur $\SI{5}{V}$ et de fréquence $\SI{100}{kHz}$.

\begin{quest}
  % \item Observer le signal en entrée du câble et expliquer.
  \QR {À partir de l’\Eq{r}, prédire la forme de l’onde en sortie $V_s$ et en entrée $V_e$ lorsque $R=0$ (court-circuit), $R=Z_c$ (adaptation d’impédance), et $R=\infty$ (à vide). Vérifier qualitativement vos prédictions à l’oscilloscope.}
  {
    On étudie les ondes dans le câble pour les instants $t>0$, et on note $V_i$ la tension incidente constante envoyée en entrée du câble par le GBF.  L’interface en sortie du câble engendre une onde réfléchie et une onde transmise. La tension en entrée du câble vaut donc 
    $$
    V_e(t) = V_i + V_r(t,x=0),
    $$
    avec, puisque $V_r$ est une onde progressive,
    $$
    \left\{
      \begin{array}{ll}
        V_r(t<2\tau,x=0) = 0 \\
        V_r(t>2\tau,x=0) = V_r(t-\tau,x=L) = r V_i\\
      \end{array}
    \right.
    $$
    où on a utilisé la définition du coefficient de réflexion. On en déduit que 
    \eqbox{
      V_e(t) = \left\{
        \begin{array}{ll}
          V_i, & \text{si } t<2\tau, \\
          (1+r)V_i, & \text{si } t>2\tau.
        \end{array}
      \right.
    }
    On détermine de même la tension en sortie du câble,
    % $$
    \eqbox{
      V_s(t) = \left\{
        \begin{array}{ll}
          0, & \text{si } t<\tau, \\
          (1+r)V_i, & \text{si } t>\tau, \\
        \end{array}
        \right.
        }
  }
  \sol{
    \begin{enumerate}
      \item  $\boxed{Z=0 \Leftrightarrow r=-1}$ et donc l’onde est entièrement réfléchie et en opposition de phase avec l’onde incidente. Les signaux valent 
      \eqbox{
      V_e(t) = \left\{
        \begin{array}{ll}
          V_i, & \text{si } t<2\tau, \\
          0, & \text{si } t>2\tau.
        \end{array}
      \right.
      \quad \text{et} \quad
      V_s(t) = 0~\forall t.
      }
      \begin{fig}[label]
        \centering
        \includegraphics[width=0.7\columnwidth]{Z=0.pdf}
        %\capt{}
      \end{fig}
      \item $\boxed{Z=Z_c \Leftrightarrow r=0}$ et donc l’onde est entièrement transmise. Il y a adaptation d’impédance. Les signaux valent
      \eqbox{
      V_e(t) = \left\{
        \begin{array}{ll}
          V_i, & \text{si } t<2\tau, \\
          V_i, & \text{si } t>2\tau.
        \end{array}
      \right.
      \quad \text{et} \quad
      V_s(t) = \left\{
        \begin{array}{ll}
          0, & \text{si } t<\tau, \\
          V_i, & \text{si } t>\tau.
        \end{array}
      \right.
      }
      \begin{fig}[label]
        \centering
        \includegraphics[width=0.7\columnwidth]{Z=Zc.pdf}
        %\capt{}
      \end{fig}
      
      \item $\boxed{Z=\infty \Leftrightarrow r=1}$ et donc l’onde est entièrement réfléchie et en phase avec l’onde incidente. Les signaux valent
      \eqbox{
      V_e(t) = \left\{
        \begin{array}{ll}
          V_i, & \text{si } t<2\tau, \\
          2V_i, & \text{si } t>2\tau.
        \end{array}
      \right.
      \quad \text{et} \quad
      V_s(t) = \left\{
        \begin{array}{ll}
          0, & \text{si } t<\tau, \\
          2V_i, & \text{si } t>\tau.
        \end{array}
      \right.
      }
      \begin{fig}[label]
        \centering
        \includegraphics[width=0.7\columnwidth]{Z=inf.pdf}
        %\capt{}
      \end{fig}
    \end{enumerate}
      
   }
\end{quest}

\subsection{Mesure du coefficient de réflexion}


\begin{quest}
  \QR {Exprimer $r$ en fonction de la tension incidente $V_i$, de la variation algébrique de tension d’entrée $\Delta V_e$ dû au retour de l’onde réfléchie, et du coefficient d’atténuation $\beta$. En déduire un protocole pour mesurer $r$ à partir du signal en entrée du câble observé à l’oscilloscope.}
  {
Le saut de tension vaut aussi

  $$
  \Delta V_e = V_e(t>2\tau) - V_e(t<2\tau) = V_r(x=0).
  $$
  En raison de l’atténuation, on a 
  $$
  V_r(x=0) = V_r(x=L)10^{-\beta L/20} = rV_i(x=L)10^{-\beta L/20} = rV_i(x=0)e^{-\beta L/10}.
  $$
  On a donc la relation 
  \eqbox{
    r = \frac{\Delta V_e}{V_i}e^{\beta L/10}.
  }
  On peut ainsi déterminer $r$ à partir de grandeurs mesurables.
  }
  \item Déterminer alors $r$ expérimentalement pour plusieurs valeurs de résistance $R$ en sortie du câble, et comparer à l’\Eq{r}. On pourra faire varier $R$ entre $\SI{10}{\Omega}$ et $\simeq\SI{10}{k\Omega}$ en progressant par multiples de $\sqrt{10}$.
\end{quest}
