\begin{exo}[PC][1][devoir]{Détente d'un gaz dans une tuyère}
Une tuyère est une conduite cylindrique rigide d'axe de révolution $(O x)$ et de section variable $S(x)$. On considère un écoulement d'air dans une tuyère dont les parois sont calorifugées. On suppose que l'air se comporte comme un gaz parfait diatomique de masse molaire $M=29 \mathrm{~g} \cdot \mathrm{mol}^{-1}$ caractérisé par un coefficient $\gamma=\num{1,4}$ et que la vitesse de l'air en un point d'abscisse $x$ de l'écoulement s'écrit $\Vec{c}=c(x) \ex$.
% Figure 5.23. Schéma d'une tuyère.
L'air entre dans la tuyère avec une vitesse $c_1=100 \mathrm{~m} \cdot \mathrm{s}^{-1}$, une température $T_1=1500 \mathrm{~K}$, une pression $p_1=50$ bar et un volume massique $v_1=\SI{8,71e-2}{m^3.kg^{-1}}$ et en sort à une température $T_2=500 \mathrm{~K}$, une pression $p_2=1$ bar et avec un volume massique
$v_2=\SI{1,44e-2}{m^3.kg^{-1}}$.
\begin{quest}
\item Montrer en appliquant le premier principe industriel sur un système ouvert que l'on définira que la quantité $h(x)+\frac{1}{2} c^2(x)$ se conserve dans la tuyère, i.e. est indépendante de $x$.
\sol{
\udl{Système ouvert} : volume de contrôle défini entre les abscisses $0$ (section d'entrée $S_1$) et $x$ (section de sortie $S(x)$).
La tuyère est calorifugée donc $q=0$ et ne comporte aucune pièce mécanique donc $w_u=0$. En négligeant la variation d'énergie potentielle de pesanteur, le premier principe industriel s'écrit:
$$
\left(h(x)+\frac{1}{2} c^2(x)\right)-\left(h_1+\frac{1}{2} c_1^2\right)=q+w_u=0 .
$$
Cette relation est vraie pour toute abscisse $x$. On en déduit que la quantité \fbox{$h(x)+\frac{1}{2} c^2(x)$} se conserve dans la tuyère.
}
\item Déterminer la vitesse de sortie du gaz $c_2$ en fonction des températures $T_1$ et $T_2$, de $c_1$, $R$, $M$ et de $\gamma$. Application numérique.
\sol{
 Utilisons la conservation de la quantité $h(x)+\frac{1}{2} c^2(x)$ entre l'entrée et la sortie de la tuyère :
$$
h_2+\frac{1}{2} c_2^2=h_1+\frac{1}{2} c_1^2 \quad \text { soit } \quad c_2=\sqrt{c_1^2+2\left(h_1-h_2\right)} .
$$
L'air se comporte comme un gaz parfait donc $h_2-h_1=c_p\left(T_2-T_1\right)$ avec $c_p=\frac{R \gamma}{M(\gamma-1)}$ la capacité thermique massique à pression constante de l'air.
\udl{Remarque } :
Il faut être en mesure de retrouver les expressions des capacités thermiques massiques $c_P$ et $c_V$ à partir de la définition du coefficient $\gamma: \gamma=\frac{C_P}{C_V}$ et de la loi de Mayer : $C_P-C_V=n R$ où $n$ est la quantité de matière. Sous forme massique, ces deux équations se réécrivent :
$$
\gamma=\frac{c_P}{c_V} \quad \text { et } \quad c_P-c_V=\frac{R}{M}
$$
En résolvant ce système de deux équations à deux inconnues, il vient :
$$
c_V=\frac{R}{M(\gamma-1)} \quad \text { et } \quad c_P=\frac{R \gamma}{M(\gamma-1)}
$$


La vitesse d'éjection du gaz est donc :
\eqbox{c_2=\sqrt{c_1^2+\frac{2R \gamma}{M(\gamma-1)}\left(T_1-T_2\right)}=\SI{1,4e3}{m.s^{-1}}.}
La vitesse d'éjection est supersonique.
}
\end{quest}
Pour une tranche de gaz comprise entre les abscisses $x$ et $x+\mathrm{d} x$, la variation de l'enthalpie massique s'écrit $\mathrm{d} h(x)=T(x) \mathrm{d} s(x)+v(x) \mathrm{d} p(x)$, où $v(x)$ est le volume massique du gaz à l'abscisse $x$.
\begin{quest}
\item Exprimer la variation d'entropie massique du gaz entre l'entrée $1$ et la sortie $2$ de la tuyère en fonction des températures $T_1$ et $T_2$ et pressions $p_1$ et $p_2$.
\sol{

La variation d'entropie massique sur une tranche de longueur $\mathrm{d} x$ est :
$$
\mathrm{d} s(x)=\frac{\mathrm{d} h(x)}{T(x)}-\frac{v(x)}{T(x)} \mathrm{d} p(x) .
$$
L'air est considéré comme un gaz parfait. L'enthalpie du gaz ne dépend donc que de la température et $\mathrm{d} h(x)=c_P \mathrm{~d} T(x)$ où $\mathrm{d} T(x)$ est la variation de température du gaz entre $x$ et $x+\mathrm{d} x$. De plus, l'équation d'état des gaz parfaits donne $\frac{v(x)}{T(x)}=\frac{R}{M p(x)}$.
La variation d'entropie se réécrit :
$$
\mathrm{d} s(x)=c_P \frac{\mathrm{d} T(x)}{T(x)}-\frac{R}{M} \frac{\mathrm{d} p(x)}{p(x)}
$$
L'intégration de cette relation entre l'entrée et la sortie de la tuyère donne la variation ds de l'entropie massique
\eqbox{
s_2-s_1=c_P \ln \left(\frac{T_2}{T_1}\right)-\frac{R}{M} \ln \left(\frac{p_2}{p_1}\right)
}
% Attention!
% Malgré l'intégration de $\mathrm{d} s(x)$ entre l'entrée et la sortie de la tuyère, ds reste un infiniment petit correspondant à la variation d'entropie du gaz entre les instants $t$ et $t+\mathrm{d} t$.
}
\item En appliquant le $2^{\rm nd}$ principe industriel, calculer numériquement l'entropie créée lors du passage du gaz dans la tuyère. La détente est-elle réversible ?
\sol{
$\triangleright$ Appliquons le deuxième principe industriel :
$$
s_2-s_1=s_{\mathrm{c}}+s_{\mathrm{e}}
$$
or, $s_{\mathrm{e}}=0$ car la tuyère est calorifugée. Il s'ensuit que l'entropie massique créée durant d $t$ est :
$$
s_{\mathrm{c}}=c_P \ln \left(\frac{T_2}{T_1}\right)-\frac{R}{M} \ln \left(\frac{p_2}{p_1}\right)= \num{19.2} \mathrm{~J} \cdot \mathrm{K}^{-1} \cdot \mathrm{kg}^{-1} .
$$
}
% \item Calculer le rapport des sections de sortie et d'entrée $S_2 / S_1$ permettant le bon fonctionnement de la tuyère.
% \sol{
% $\triangleright$ En régime stationnaire, le débit massique $D_m$ se conserve entre l'entrée et la sortie de la tuyère. Il vient alors:
% $$
% D_m=\frac{S_1 c_1}{v_1}=\frac{S_2 c_2}{v_2}
% $$
% On en déduit le rapport des sections permettant le bon fonctionnement de la tuyère :
% $$
% \frac{S_2}{S_1}=\frac{c_1 v_2}{c_2 v_1}=1,2
% $$
% La section de sortie doit donc être supérieure à la section d'entrée de la tuyère.
% Complément
% On peut montrer que pour que la tuyère réalise la détente du gaz, celle-ci doit tout d'abord être convergente jusqu'à ce que l'écoulement devienne supersonique puis être divergente jusqu'à la sortie du gaz.
% }
\end{quest}
\end{exo}

    \end{equation}

        U = um,

      \end{enumerate}

\begin{exo}[PC][1][devoir]{Cycle thermodynamique de Brayton}%{1}
% \ccp{2020}
\Annale[Physique PC]{CCINP}{2020}
Pour leur propulsion, les avions sont équipés majoritairement de réacteurs. Cette dénomination
usuelle désigne en fait des turboréacteurs qui appartiennent à la catégorie des turbomachines encore
appelées générateurs ou turbines à gaz. Les turbomachines présentent plusieurs avantages par
rapport aux moteurs à pistons, avec notamment un rapport puissance-poids environ trois fois
supérieur. En effet, le nombre de pièces mobiles est réduit et leur mouvement est très simple, ce qui
permet de les alléger. Ces machines sont inégalables lorsque de grandes puissances sont requises
avec des contraintes d’espace ou de poids. Leur inconvénient majeur est que leur efficacité et leur
réactivité chutent très rapidement à faible puissance : ils ne sont donc pas adaptés au domaine
automobile par exemple.

%%%%%%%%%%%%%%%%%
\begin{fig}[turbo]
\includegraphics[scale=0.2]{turboreacteur.png}
\capt{Schéma de principe d’un turboréacteur simple flux.}
\end{fig}

Les constituants principaux d’un turboréacteur (\Fig{turbo}) sont un compresseur, une chambre de combustion et
une turbine. Dans cette partie on étudie un turboréacteur dit simple flux (figure 12) pour lequel le
gaz entrant dans le réacteur passe dans un diffuseur pour en diminuer la vitesse avant d’être
comprimé par le compresseur. Le gaz comprimé arrive dans une chambre de combustion où il est
chauffé avant d’être détendu partiellement dans la turbine qui fournit la puissance nécessaire au
compresseur. En sortie de turbine, le gaz reste à une pression relativement élevée par rapport à la
pression extérieure et il est détendu dans une tuyère, ce qui permet de l’accélérer : c’est cette
accélération qui permet la propulsion de l’avion.
Le turboréacteur simple flux est principalement utilisé dans l’aviation militaire.



Le turboréacteur fonctionne selon le cycle théorique ouvert de Brayton. Les conditions d’étude de
ce cycle sont les suivantes : l’air est considéré comme un gaz parfait. Sa capacité thermique massique à pression
constante $c_p$ est supposée constante, comme le rapport $\gamma$ entre les capacités thermiques
isobare et isochore. L'enthalpie massique a donc pour expression $h=c_p T$. On prendra $\gamma = \num{1.35}$ et $c_p = \SI{1.1}{ kJ.kg^{-1}.K^{-1}}$ ;
 les variations d’énergie potentielle sont négligeables. Dans ce cas, on rappelle l'expression du premier principe industriel :
 \begin{equation}
  \Delta h+\Delta\left(\frac{v^2}{2} \right) = w_u + q_e,
 \end{equation}
 avec $w_u$ le travail utile et $q_e$ le transfert thermique échangé avec l'extérieur, et $\Delta$ indique une variation entre l'entrée et la sortie. La variation d’énergie cinétique est supposée négligeable uniquement entre l’entrée du compresseur et la sortie de la turbine.

\begin{liste}


\item En entrée du diffuseur, l’air est à l’état (1) : $(P_1, T_1)$. On considère que le diffuseur est idéal, ce qui
revient à dire que l’énergie cinétique du gaz après traversée du diffuseur est négligeable devant les
autres termes énergétiques et que la traversée du diffuseur est adiabatique et réversible, sans travail utile ni transfert thermique.
\item En entrée du
compresseur, l’air se trouve à l’état (2) : $(P_2, T_2)$ et est amené à l’état (3) : $(P_3 = 10P_2, T_3)$ par une
compression adiabatique réversible.
\item Dans la chambre de combustion, l’air, mélangé au carburant, subit un échauffement isobare
réversible jusqu’à l’état (4) : $(P_4, T_4 = \SI{1 400}{K})$. Bien que les compositions du gaz à l’entrée et à la
sortie de la chambre de combustion soient différentes, pour simplifier la modélisation, on suppose
que celle-ci sert uniquement à réchauffer l’air et que les propriétés de l’air ne sont pas modifiées par
ce changement de composition. 
\item L’air parvient dans la turbine où il subit une détente adiabatique réversible jusqu’à l’état (5) :
$(P_5, T_5)$.
\item Enfin, il se détend de façon adiabatique et réversible dans la tuyère et arrive dans l’état (6) :
$(P_6,T_6)$. Aucun travail utile ni transfert thermique n'est fourni.
\end{liste}


On considère un avion qui vole avec une vitesse de croisière $V_a = \SI{260 }{m.s^{-1}}$ par rapport à l’air
considéré au repos. À cette altitude, l’air est à la pression de $\SI{34.5}{kPa}$ et à la température de $\SI{-40}{\dC}$.
L’air entre dans le compresseur avec un débit massique $D_m = \SI{45}{kg.s^{-1}}$.

\QR{
   Donner l’expression de la température $T_2$ en fonction de $T_1$, $V_a$ et $c_p$. Effectuer l’application
numérique.
}{
Dans le diffuseur, il n'y a ni pièce mobile ni transfert thermique, donc en négligeant la pesanteur, il vient : $\Delta h+\Delta(\frac{v^2}{2})=0$.

La vitesse en sortie et négligeable et celle en entrée vaut $V_a$, et on a un gaz supposé parfait, donc $\Delta h=c_p\Delta T$. On en déduit :

\eqbox[*]{T_2=T_1+\frac{V_a^2}{2 c_p}}
\AN~: $T_2 = 264$ K.
}

\QR{
   Donner l’expression de la pression $P_2$ en fonction de $P_1$, $T_1$, $T_2$ et $\gamma$. Effectuer l’application numérique.
}{
La transformation est adiabatique et réversible
donc isentropique : on peut utiliser la loi de Laplace

$$P_2^{1-\gamma}T_2^\gamma=P_1^{1-\gamma}T_1^\gamma,$$
soit 
\eqbox{
P_2 = P_1\left(\frac{T_1}{T_2} \right)^{\frac{\gamma}{1-\gamma}}
}


\AN~: $P_2 = 55,9$ kPa
}

\QR{ Établir l’expression du travail massique utile $w_{\rm comp}$ fourni à l’air par le compresseur. En
prenant $T_3 = \SI{480}{K}$, calculer la puissance $P_{\rm comp}$ fourni à l'air par le compresseur.
}{
Le compresseur est adiabatique et on néglige pesanteur et énergie cinétique. Le premier principe s'écrit donc :

$$\Delta h=c_p(T_3-T_2)=w_{comp}$$

On a donc $w_{comp}=c_p(T_3-T_2)=\SI{238}{kJ.kg^{-1}}$. Pour avoir la puissance, on multiplie par le débit massique :

\eqbox{
\mathcal{P}_{comp}=D_mc_p(T_3-T_2).
}
\AN~: $\mathcal{P}_{comp}=\SI{10.7}{MW}$.
}

\QR{  Sachant que le travail fourni par la détente du gaz dans la turbine est intégralement reçu par le
compresseur, déterminer l’expression de la température $T_5$ en fonction de $T_2$, $T_3$ et $T_4$. Calculer la valeur de $T_5$. En déduire la valeur de la pression $P_5$.

$$c_p(T_5-T_4)=-c_p(T_3-T_2),$$
soit 
\eqbox{
T_5=T_4+T_2-T_3.

\eqbox{
P_5 =P_4\left(\frac{T_4}{T_5} \right)^{\frac{\gamma}{1-\gamma}}.
}

\AN~: $P_5=2,93.10^5$ Pa
}

$$c_p(T_6-T_5)+\frac 1 2 v_s^2=0,$$
soit 
\eqbox{
v_s=\sqrt{2c_p(T_5-T_6)}.
}

\AN~: $v_s=1,05.10^3$ m.s$^{-1}$

}

\eqbox{
\mc{P} = D_m(v_s-V_a)V_a
}

\AN~:$\mathcal{P}=9,24$ MW.

}

\QR{ Calculer le rendement $\eta$ du turboréacteur, qui correspond au rapport entre la puissance de la
force propulsive, et la puissance qui sert à chauffer le gaz dans la chambre de combustion $P_{\rm chamb} = \SI{45.5}{MW}$. Comparer avec le rendement d’autres machines thermiques.
}{
Le rendement est 
\eqbox{
  \eta=\frac{\mathcal{P}}{\mathcal{P}_{chamb}}.
}

\AN~: $\eta=20,3\%$

Ce rendement est inférieur aux machines thermiques motrices usuelles (moteur de voiture : rendement de 30 à 40\%).

}

\begin{exo}[PT][1]{Fonctionnement d'une hélice}

On considère un écoulement homogène, parfait et incompressible. L'hélice est
animée d'un mouvement de rotation autour de son axe fixe, à vitesse angulaire
constante. On néglige son épaisseur ce qui revient à supposer que $S_1 = S_2 = S$. On
néglige les effets de la pesanteur. Le mouvement du fluide autour de l'hélice est
supposé stationnaire dans le référentiel terrestre galiléen et à symétrie de révolution autour de $Ox$. On considère un tube de courant représenté sur la figure ci-dessous qui englobe l'hélice. La pression $p_0$ est supposée uniforme autour du
tube de courant. On appelle $D_m$ le débit massique. On rappelle que :
\begin{equation}
\d h = T \d s + \dfrac{\d p}{\mu}.
\end{equation}

\begin{quest}
\item Exprimer $p_2 - p_1$ en fonction de $v_A$, $v_B$ et $\mu$.

\sol{
Parfait + incompressible $\rightarrow$ Bernouilli, entre AI et JB. Incompressible et stationnaire donc conservation du débit volumique $D_v =v_1S_1 = v_2S_2$ donc $v_1=v_2=v$.
$$
\dfrac{p_0}{\mu} + \dfrac{v_A^2}{2} + g z_A = \dfrac{p_1}{\mu} + \dfrac{v_I^2}{2} + g z_I
$$
donc $p_2-p_1=\dfrac{\mu}{2}(v_A^2-v_B^2)$.
}

\item Exprimer la force exercée par l'hélice sur le fluide de deux façons et en déduire la vitesse $v$ du fluide au niveau de l'hélice en fonction de $v_A$ et $v_B$.

\sol{
Méthode 1 : système ouvert IJ $\rightarrow$ système fermé IJ + masse $\d m = D_m \d t$ qui entre en $t$ puis sort en $t+\d t$.

On a $\vec p_IJ(t) +\d m \vec{v_1} = \vec p_IJ(t+\d t) +\d m \vec{v_2} $ en régime permanent donc $\d \vec p = \vec 0$.

Actions extérieures : force $\vec F$ de l'hélice, force de pression $(p_1-p_2)S \vec{e_x}$. Somme $=0$ donc $\vec F = (p_2-p_1)S \vec{e_x} = \dfrac{\mu S}{2}(v_A^2-v_B^2) \vec{e_x}$\\

Méthode 2 : système ouvert AB. $\d \vec p/\d t = D_m(\vec v_B-\vec v_A)$.

Actions de pression car surface fermée et pression $p_0$ uniforme. Donc $\vec F  = D_m(\vec v_B-\vec v_A)$.\\

On en déduit $v = \dfrac{v_A+v_B}{2}$.

}

\item Déterminer la puissance de la force exercée par l'hélice sur le fluide en fonction de $D_m$, $v_A$ et $v_B$.

\sol{
Méthode 1 : 1er principe sur système IJ refermé
$\d m (h_2-h_1) = \delta W_i + \delta Q
$ avec $\delta Q=0$ car adiabatique réversible et $W_i = P \d t$ travail indiqué. Donc
$$
P = D_m(h_2-h_1) = \dfrac{p_2-p_1}{\mu} = \dfrac{D_m}{2}(v_B^2-v_A^2).
$$

}
\end{quest}

L’eau liquide est supposée indilatable et incompressible, et de capacité thermique massique isobare constante $c$.
% On donne ci-dessous quelques propriétés thermodynamiques de l’eau.
%\begin{center}
%\includegraphics[width=0.85\textwidth]{extrait_table.pdf}
%\end{center}

  L’eau est évaporée dans le générateur de vapeur (GV) par absorption de la chaleur produite dans le réacteur, puis elle entraîne une turbine couplée à un alternateur produisant l'électricité. Elle est ensuite liquéfié dans un condenseur (eau de rivière ou atmosphère via une tour aéroréfrigérante), puis comprimée avant d'être renvoyée vers le générateur de vapeur.
Le cycle thermodynamique correspondant est le \emph{cycle de Rankine}, schématisé sur la \Fig{REP}, et comprenant les transformations suivantes :
\begin{enumerate}%[label=\protect\circled{\arabic*}]
\item  $\rm A-B$ : échauffement isobare dans le générateur de vapeur à la pression $p_2 =\SI{55}{bar}$ jusqu’à un état de liquide saturant $A’$, puis vaporisation isobare jusqu’à un état de vapeur saturante $B$;
\item $\rm B-C$ : détente adiabatique réversible dans la turbine, de $p_2$ à la pression $p_1 = \SI{430}{mbar}$. Le fluide est diphasé en $C$ ;
\item $\rm C-D$ : liquéfaction isobare dans le condenseur jusqu’à un état de liquide saturant $D$;
\item $\rm D-A$ : compression adiabatique réversible dans la pompe de $p_1$ à $p_2$, de travail utile négligeable.
\end{enumerate}

On place les points successivement en fonction des transformations suivies (2 isentropiques et deux isobares).
\begin{center}

Le seul point délicat est la transformation A $\rightarrow$ D qui est isentropique. \item En s’aidant du premier principe industriel, exprimer puis calculer à partir du diagramme $(p,h)$ le travail massique $w_{BC}$ reçu par l'eau dans la

\sol{
\begin{liste}
\item \underline{Turbine} : transformation  $B \rightarrow C$ adiabatique et donc $w_{\rm{u}} = w_{BC} =  \Delta h =h_C-h_B = -\SI{750}{kJ.kg^{-1}}$.
\item \underline{Générateur de vapeur} : transformation $A \rightarrow B$ sans travail utile, donc $q_c = q_{AB} = \Delta h = h_B-h_A = \SI{2480}{kJ.kg^{-1}}$.
\end{liste}

}

Comparer au rendement de Carnot et commenter.
\sol{

}

\begin{exo}[PC]{Échangeur thermique}%{1}

Considérons un échangeur thermique possédant deux entrées et deux sorties parcouru par deux liquides identiques de capacité calorifique massique $c=\SI{4.8}{kJ.K^{-1}.kg^{-1}}$. Le liquide entrant par $E_1$ sort par $S_1$ avec un débit massique $D_{m,1} = \SI{10}{kg.s^{-1}}$ et le liquide entrant par $E_2$ sort par $S_2$ avec un débit massique $D_{m,2}=\SI{2}{kg.s^{-1}}$. Le liquide arrivant sur l'entrée $1$ a une température $T_{E1}=\SI{290}{\dC}$ plus importante que le liquide arrivant sur l'entrée $2$ notée $T_{E2} = \SI{180}{\dC}$. On suppose que les parois externes de l'échangeur thermique sont calorifugées, de sorte que les liquides n'échangent du transfert thermique qu'entre eux.
\begin{fig}[label]
  \centering
  \includegraphics[width=0.4\columnwidth]{echangeur2.pdf}
  %\capt{}
\end{fig}
\begin{quest}
\item  Un échangeur thermique fournit-il ou reçoit-il du travail utile au fluide ? En appliquant le premier principe industriel, version \guill{puissance}, aux liquides $1$ puis $2$, déterminer une relation entre les températures d'entrée $T_{E1}$, $T_{E2}$ et de sortie  $T_{S1}$, $T_{S2}$.
\sol{
Le $1$er principe pour $1$ donne $D_{m,1}(h_{S1}-h_{E1}) = D_{m,1}c(T_{S1}-T_{E1})  =  P_{2\rightarrow1}$. Le $1$er principe pour $2$ donne $D_{m,2}(h_{S2}-h_{E2}) = D_{m,2}c(T_{S2}-T_{E2})  =  P_{1\rightarrow2} = - P_{2\rightarrow1}$. On en déduit $D_{m,1}(T_{S1}-T_{E1})   + D_{m,2}(T_{S2}-T_{E2}) =0$.
}
\item En supposant que le temps de contact entre les deux liquides est long, déterminer une autre relation entre $T_{S1}$ et $T_{S2}$.
\sol{
$T_{S1}=T_{S2}=T_S$.
}
\item Sous cette hypothèse, calculer alors la température de sortie $T_{S}$.
\sol{
On a alors $T_S = \frac{D_{m,1}T_{E1} + D_{m,2}T_{E2}}{D_{m,1}+D_{m,2}}=\SI{545}{K}=\SI{272}{\dC}$.
}
\end{quest}
En réalité, les parois de l'échangeur ne sont pas parfaitement calorifugées et chaque liquide perd une puissance thermique $P=\SI{50}{kW}$ dans l'échangeur.
\begin{quest}

\item En supposant que le temps de contact entre les deux liquides est long, calculer la nouvelle température de sortie $T_{S}$ en présence de pertes.
\sol{
Le $1$er principe pour $1$ donne $D_{m,1}(h_{S1}-h_{E1}) = D_{m,1}c(T_{S1}-T_{E1})  =  P_{2\rightarrow1} - P$. Le $1$er principe pour $2$ donne $D_{m,2}(h_{S2}-h_{E2}) = D_{m,2}c(T_{S2}-T_{E2})  =  P_{1\rightarrow2} - P = - P_{2\rightarrow1} - P$. On en déduit $D_{m,1}(T_{S1}-T_{E1})   + D_{m,2}(T_{S2}-T_{E2}) = - 2P$.

On a alors $T_S = - \frac{2P}{c(D_{m,1}+D_{m,2})}\frac{D_{m,1}T_{E1} + D_{m,2}T_{E2}}{D_{m,1}+D_{m,2}}=\SI{543}{K}=\SI{270}{\dC}$.
}
\end{quest}
\end{exo}
