\begin{tikzpicture}[]
	\def \a {2};%distance source-miroir
	\def \L {8};%distance miroir-écran
	\def \divergence {16};
	\def \deviation {6};%angle entre les miroirs 
	\coordinate (O) at (0,0);
	\coordinate (S) at (0:\a);%source primaire
	\coordinate (SS) at ({180-\deviation}:\a);%S1
	\coordinate (SSS) at ({180+\deviation}:{\a});%S2
	\coordinate (A) at({90-.5*\deviation}:{(\a*sin(\divergence))/cos(\divergence-.5*\deviation)});
	\coordinate (B) at({-90+.5*\deviation}:{(\a*sin(\divergence))/cos(\divergence-.5*\deviation)});
	\coordinate (C) at (\L,{\a*sin(\deviation)+(\L+\a*cos(\deviation))*tan(\divergence-\deviation)});
	\coordinate (D) at (\L,{\L*tan(\deviation)});
	\coordinate (E) at (\L,{-\L*tan(\deviation)});	
	\coordinate (F) at (\L,{-\a*sin(\deviation)-(\L+\a*cos(\deviation))*tan(\divergence-\deviation)});

	\fill[white] (-\a,2)rectangle(\L,-2);%masque blanc pour l'opacité
	%faisceaux
	\fill[red!50!white,opacity=0.5] (S)--(A)--(O)--(B)--cycle;
	\fill[red!50!white,opacity=0.5] (A)--(C)--(E)--(O)--cycle;
	\fill[red!50!white,opacity=0.5] (B)--(F)--(D)--(O)--cycle;

	%miroirs
	\draw[thick](0,0)--({90-.5*\deviation}:{(\a*sin(\divergence))/cos(\divergence-.5*\deviation)});
	\draw[thick](0,0)--({-90+.5*\deviation}:{(\a*sin(\divergence))/cos(\divergence-.5*\deviation)});
	
	\draw[thin,dashed](0,0)--({-90-.5*\deviation}:1.6);
	\draw[thin,dashed](0,0)--({-90+.5*\deviation}:1.6);
	\draw[->](-100:1.5)arc(-100:{-90-.5*\deviation}:1.5);
	\draw[->](-80:1.5)arc(-80:{-90+.5*\deviation}:1.5);
	\draw (0,-1.5)node[below]{\footnotesize $\alpha$};
	\draw[dashed,red](SS)--(A)--(C);
	\draw[dashed,red](SS)--(E);
	\draw[dashed,red](SSS)--(D);
	\draw[dashed,red](SSS)--(F);
	\draw ({\L/2},0) node{\footnotesize champ d'interférence};
	%sources principales et primaires
	\draw (S)node{\tiny •};
	\draw (SS)node{\tiny •}node[above]{\footnotesize S$_2$};
	\draw (SSS)node{\tiny •}node[below]{\footnotesize S$_1$};
	\draw (\a,0) node[pin={[pin edge={black,-},font=\footnotesize]90:source}] {};
