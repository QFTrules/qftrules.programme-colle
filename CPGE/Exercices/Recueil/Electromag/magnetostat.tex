\begin{exo}[PC][3][TD]{Bobine de Helmholtz}
	\Notions{\item Spire de courant \task Principe de superposition \task Carte de champ magnétique}
Une bobine est constituée de deux spires circulaires coaxiales de même rayon $R$ et de centres $\rm O_1$ et $\rm O_2$ distants de $d$. On place le centre $\rm O$ du repère à mi-distance de $\rm O_1$ et $\rm O_2$. Les spires sont parcourues par un courant de même intensité $I$, soit circulant dans le même sens, comme sur la \Fig{helm}a, soit circulant dans des sens opposés, comme sur la \Fig{helm}b. On souhaite déterminer le champ magnétique en un point $\rm M$ situé sur l'axe $(O_1O_2)$ à une distance $z$ du centre $\rm O$ de la bobine. 
\begin{fig}[label]
	\centering
	\begin{subfig}[helm][0.3]
		\includegraphics[width=\columnwidth]{magnetic_field_2loop_anti}
	\end{subfig}\hspace{1cm}
	\begin{subfig}[label][0.3]
		\includegraphics[width=\columnwidth]{magnetic_field_2loop}
	\end{subfig}
	\capt{(a) Configuration Helmholtz. (b) Configuration anti-Helmholtz.}
\end{fig}
\begin{quest}
	\item	Pour chacune des configurations, déterminer les symétries et invariances de la distribution de courant, et représenter la carte de champ magnétique dans un plan en coupe passant par l’axe $(O_1O_2)$.
\end{quest}
On rappelle l'expression de la composante axiale du champ magnétique en un point ${\rm M}(x)$ situé sur l'axe~$(Ox)$ d'une spire circulaire de rayon $R$, de centre $\rm O$, parcourue par un courant $I$,
\begin{equation}
B_{\rm s}(x) = \frac{\mu_0 I}{2}\frac{R^2}{(x^2+R^2)^{3/2}}.
\end{equation}
Configuration Helmholtz.
\begin{quest}
	\item En se servant du principe de superposition, déterminer le champ magnétostatique $B_{\rm h}(z)$ créé par la bobine de Helmholtz en un point $\rm M$  sur l’axe situé à la distance $z$ du centre $\rm O$ de la bobine.
	\sol{
	Posons le champ sous la forme 
	$$
	B_s(x) = \frac{\mu_0 I}{2R} f(x/R),
	$$
	avec
	$$
	f(y) = (1+y^2)^{-3/2}. 
	$$
	Le principe de superposition donne 
	$$
	B_{\rm h}(z) = B_s(z-d/2) + B_s(z+d/2),
	$$
	soit 
	\eqbox{
		B_{\rm h}(z)=\frac{\mu_0 I}{2R} \left[ f\left(\frac{d/2+z}{R}\right) + f\left(\frac{d/2-z}{R}\right) \right].
	}
	}
	\item Dans l'hypothèse où $z \ll R$, développer l'expression de $B_{\rm h}(z)$ à l'ordre 3 en $z/R$.
	\sol{
	Le développement limité de la fonction $f(y)$ au voisinage de $y=\frac{d}{2R}$ donne 
	$$
	f\left( \frac{d}{2R} + \veps\right) = f\left( \frac{d}{2R}\right) + \veps f'\left( \frac{d}{2R}\right) + \frac{\veps^2}{2} f''\left( \frac{d}{2R}\right) + \frac{\veps^3}{6} f'''\left( \frac{d}{2R}\right) + o(\veps^3).
	$$
	Par sommation, nous trouvons que les termes impaires en $\veps = z/R$ se simplifient, soit 
	% $$
	\eqbox{
		B_h(z) = \frac{\mu_0 I}{R} \left[ f\left(\frac{d}{2R}\right)  + \frac{z^2}{2R^2} f''\left(\frac{d}{2R}\right) + o(\veps^3)  \right].
		}
	% $$

	}
	\item Montrer qu'en peut rendre le champ magnétostatique uniforme sur l'axe jusqu'à l'ordre $3$ en $z/R$ en plaçant les spires à une distance $d=R$. Déterminer la valeur du champ magnétique au centre $B_0$.
	\sol{
		La dérivée seconde de $f$ vaut 
	$$
	f''(y) = -3\frac{4y^2-1}{(1+y^2)^{7/2}},
	$$
	Pour obtenir un champ uniforme jusqu'à l'ordre 3 en $z/R$, il faut que $f''(d/2R) = 0$, soit 
	\eqbox{
	d = R.
	}
	La valeur du champ au voisinage de l'origine est donc 
	\eqbox{
	B_0 = \frac{\mu_0 I}{R} f\left( \frac{1}{2}\right) = \frac{8\mu_0 I}{5\sqrt{5}R}. 
	}
	}
\end{quest}
Configuration anti-Helmholtz.
\begin{quest}
	\item Reprendre les questions précédentes pour la configuration anti-Helmholtz, \Fig{helm}b. On notera $B_a(z)$ le champ sur l’axe. Montrer en particulier que le gradient de champ magnétostatique sur l'axe est uniforme jusqu’à l'ordre $2$ en $z/R$ pour une valeur de $d$ que l'on déterminera.
	\sol{
	Cette fois-ci, ce sont les termes pairs en $z/R$ qui se simplifient, soit
	\eqbox{
		B_a(z) = \frac{\mu_0 I}{R} \left[ \frac{z}{R} f'\left(\frac{d}{2R}\right) + \frac{z^3}{3R^3} f'''\left( \frac{d}{2R}\right) + o(\veps^3)  \right].
	}
	Le gradient de champ vaut 
	$$
	\dd{B_a}{z}(z) = \frac{\mu_0 I}{R^2} \left[ 
	f'\left(\frac{d}{2R}\right) + \frac{z^2}{R^2} f'''\left( \frac{d}{2R}\right) + o(\veps^2)  \right].
	$$
	La dérivée troisième de $f$ vaut
	$$
	f'''(y) = 15\frac{(3-4y^2)}{(1+y^2)^{9/2}}.
	$$
	Le gradient de champ est donc uniforme jusqu'à l'ordre 2 en $z/R$ si $f'''\left( \frac{d}{2R}\right) = 0$, soit
	\eqbox{
	d = \sqrt{3}R.
	}
	La valeur du gradient de champ est alors 
	\eqbox{
	\dd{B_a}{z} = \frac{\mu_0 I}{R^2} f'\left(\frac{\sqrt{3}}{2}\right) = \frac{48\sqrt{3}}{49\sqrt{7}} \frac{\mu_0 I}{R^2}.
	}
	}	
\end{quest}
	
\end{exo}

	\pourcentageDeLaPartieAGauche   {0.50}
	\initialisationPartieGauche  
	On considère deux fils conducteurs infinis parallèles à l'axe $(\ptO z)$,
	distants de $2a$ et parcourus par des courants de même intensité $I$ circulant en sens inverse.
	\initialisationPartieDroite 
\vspace{-1cm}
\begin{center}
\subimport{/home/eb/Dropbox/CPGE/Physique/Exercices/Recueil/_fiches/fiche_MAG03/_images/}{MAG03_deux_fils_CBD_v1.tex}

\begin{quest}
	\item	
	% \begin{enonce}
		Lequel de ces trois plans est plan de symétrie pour la distribution des courants ?
		\begin{listeQCM3Colonnes}
			\item le plan $(x \ptO y)$
			\item le plan $(y \ptO z)$
			\item le plan $(x \ptO z)$
		\end{listeQCM3Colonnes}
		\bigskip
	% \end{enonce}

	\begin{corrige}
		Dans une symétrie par rapport au plan $(x \ptO y)$, les fils restent inchangés mais les courants sont inversés  : c'est donc un plan d'antisymétrie.

		Dans une symétrie par rapport au plan $(y \ptO z)$, on permute les fils de gauche et de droite, les courants circulant dans le sens inverse de la situation initiale : il s'agit ici encore, d'un plan d'antisymétrie.

		Seul, le plan $(x \ptO z)$ laisse les fils inchangés ainsi que les sens des courants : c'est donc bien un plan de symétrie pour la distribution des courants. 
	\end{corrige}

	% \begin{enonce}
		\item 
		L'analyse des symétries permet de dire qu'en un point $\ptA$ de l'axe $(\ptO x)$, le champ $\vv{B}(\ptA)$ est : 
		\begin{listeQCM3Colonnes}
			\item parallèle à $(\ptO x)$
			\item parallèle à $(\ptO y)$
			\item parallèle à $(\ptO z)$
		\end{listeQCM3Colonnes}
		\bigskip
	% \end{enonce}

	\begin{corrige}
		Pour le point $\ptA$ sur l'axe $(\ptO x)$, le plan $(x \ptO y)$ est un plan de symétrie pour la distribution des courants et laisse $\ptA$ invariant. Le vecteur champ magnétique, vecteur axial, est perpendiculaire à tout plan de symétrie, donc on a $\vv{B}(\ptA) \perp (x \ptO z)$. Donc, $\vv{B}(\ptA)$ est parallèle à  $(\ptO y)$. 
	\end{corrige}

	\item
	% \begin{enonce}
		L'analyse des symétries permet de dire qu'en un point $\ptD$ de l'axe $(\ptO y)$, le champ $\vv{B}(\ptD)$ est : 
		\begin{listeQCM3Colonnes}
			\item parallèle à $(\ptO x)$
			\item parallèle à $(\ptO y)$
			\item parallèle à $(\ptO z)$
		\end{listeQCM3Colonnes}
		\bigskip
	% \end{enonce}

	\begin{corrige}
		Pour le point $\ptD$ sur l'axe $(\ptO y)$, les plans $(x \ptO y)$ et $(y \ptO z)$ sont des plans d'antisymétrie pour la distribution des courants et laissent $\ptD$ invariant. Le vecteur champ magnétique, vecteur axial, est contenu dans tout plan d'antisymétrie, donc $\vv{B_{\text{tot}}} \in (x \ptO y)\cap (y \ptO z)$, soit $\vv{B_{\text{tot}}}$ est parallèle à  $(\ptO y)$. 
	\end{corrige}
\end{quest}			
\end{exo}

\begin{exo}[PC][2][TD]{Câble parcouru par un courant non uniforme}
	\Notions{\item Symétries et invariances \task Théorème d’Ampère \task Équations de Maxwell}

On considère un câble électrique cylindrique de rayon $R$ et d'axe $(Oz)$ parcouru par une densité volumique de courant de la forme
\begin{equation}
	\j(M)=j_{0}\left(1-\frac{r^2}{R^2}\right)\,\ez.
\end{equation}
\begin{quest}
\item Exprimer la constante $j_0$ en fonction de l'intensité du courant $I$.
\sol{
	L'intensité du courant électrique traversant une section $S$ est donnée par 
	$$
	I = \iint_S \j(M) \cdot \d S.
	$$
	Sur la section totale du câble, on trouve alors,
$$I = 2\pi \int_0^R j(r) r \d r = 2\pi j_0 \int_0^R \left(1-\frac{r^2}{R^2}\right) = 2\pi j_0 \left(\frac{R^2}{2} -\frac{R^2}{4} \right) = \frac{\pi j_0 R^2}{2},$$ d'où l'on tire
\eqbox{
	j_0 = \frac{2I}{\pi R^2}.
	}
}
\item Évaluer le champ magnétostatique $\B(M)$ créé par ce câble en tout point de l'espace.
\sol{
	\begin{itemize}
		\item 
		\udl{Symétries} : le plan $(M,\er,\ez)$ est plan de symétrie, et donc 
		$$\Vec{B}(M) = B_\theta(M) \et.$$
		
		\item \udl{Invariances} : rotation autour de $(Oz)$ et translation selon $(Oz)$. Donc les composantes cylindriques du champ magnétique ne dépendent pas de $z$ ni de $\theta$, soit
		\eqbox{
			\Vec{B}(M) = B_\theta(r) \et.
		}
		\item \udl{Contour d’Ampère} : $\Gamma$ = $\{$cercle de rayon $r$ d'axe $(Oz)$$\}$. La circulation du champ magnétique le long de $\Gamma$ se calcule comme 
		$$
		\oint_\Gamma \Vec{B} \cdot \d \Vec{l} = B(r) 2\pi r.
		$$
		\item \udl{Théorème d’Ampère} : le théorème d'Ampère s'écrit
		$$
		\oint_\Gamma \Vec{B} \cdot \d \Vec{l} = \mu_0 I_{\rm \Gamma}.
		$$
		Le courant enlacé dépend de la position du point $\rm M$ par rapport à la surface du câble.
		\begin{itemize}
			\item 
			Si \fbox{$r<R$}, on a 
			$$
			I_{\rm \Gamma} = \iint_{r<R} \j(M) \cdot \d S = 2\pi \int_0^r j(r') r' \d r' = 2\pi j_0 \int_0^r \left(1-\frac{r'^2}{R^2}\right) r' \d r' = j_0 \left(r^2-\frac{r^4}{2R^2}\right).
			$$
			On en déduit
				\eqbox{
					\Vec{B}(M) = \frac{\mu_0 j_0}{2} \left(r-\frac{r^3}{2R^2}\right) \et.
					}
			\item 
			Si \fbox{$r>R$}, on a simplement $I_\Gamma = I$ et donc
			\eqbox{
				\Vec{B}(M) = \frac{\mu_0 I}{2 \pi r} \et.
				}
		\end{itemize}
	\end{itemize}


}
\item Vérifier explicitement que le champ obtenu vérifie l'équation de Maxwell-Ampère.
\sol{
	La forme du champ permet de simplifier l'écriture du rotationnel en coordonnées cylindriques,
	$$
	\rot \B = \frac{1}{r}\dr{rB_\theta}{r}\ez.
	$$
	Pour $r<R$, on a donc  
	$$\rot \B = \mu_0 j_0 \left(1- \frac{r^2}{R^2}\right)\ez = \mu_0 j(r) \ez,$$
	et l'équation de Maxwell-Ampère est vérifiée. Pour $r>R$, on a 
	$$\rot \B = \Vec{0},$$
	et l'équation de Maxwell-Ampère est aussi vérifiée, puisqu'il n'y a pas de courant en dehors du câble. En conclusion, on a bien
	\eqbox{
	\rot \B = \mu_0 \Vec{j}.
	}
	}
\end{quest}
\end{exo}

\begin{exo}[PC][3][TD]{Disque de Rowland}
	\Notions{\item Calcul intégral \task Spire de courant}
	Soit un disque de rayon $R$, chargé uniformément, de densité surfacique de charge $\sigma$. On fait tourner ce disque autour de son axe de révolution  $(Oz)$ à la vitesse angulaire $\omega$. 
	\begin{fig}[label]
		\centering
		\includegraphics[width=0.5\columnwidth]{Rowland.png}
		%\capt{}
	\end{fig}

	\begin{quest}
		\item	On décompose le disque par en un ensemble de spires élémentaires de rayons $r$ et d'épaisseur $\mathrm{d} r$.
 		Montrer qu’on peut associer à chaque spire une distribution de courant d’intensité $\d I$ à exprimer.
		\sol{
			La charge élémentaire d'une spire est $\d q = \sigma \d S = \sigma 2\pi r \d r $. Cette charge effectue un tour complet en un temps $T = \frac{2\pi}{\omega}$, et donc l'intensité du courant élémentaire vaut 
			$$
			\d I = \frac{\d q}{T} = \frac{2\pi \sigma r \d r}{T},
			$$
			soit
			\eqbox{
				\d I = \sigma \omega r \d r.
			}
		}
		\item Une spire circulaire de rayon $r$ et parcourue par un courant d'intensité $I$ produit un champ magnétique sur l’axe d’expression
		$$
		\Vec{B}_{\rm s}(M)=\frac{\mu_0 I}{2 r} \sin ^3(\vphi) \,\ez,
		$$
		où  $\vphi$ est l’angle sous lequel est vue la spire depuis le point M. Déterminer le champ magnétostatique élémentaire $\Vec{\d B}$ créé par une spire de d'épaisseur $\d r$ en fonction de $\mu_0$, $r$, $R$, $\omega$, $\sigma$ et $\d r$.
		\sol{
			On adapte avec les notations,
			% \eqbox{
				$$
				\d \Vec{B} = \frac{\mu_0 \d I}{2 r} \sin ^3(\vphi) \,\ez = \frac{\mu_0 \sigma \omega \d r}{2} \sin ^3(\vphi) \,\ez,
				$$
				soit avec $\d r = z/d \tan(\vphi) = z/\cos(\theta)^2 \d \theta$,
				\eqbox{
					\d \B = \frac{\mu_0 \sigma \omega}{2} \frac{\sin^3(\vphi)}{\cos^2\theta} \d\theta \,\ez.
				}		
		}
		\item En déduire le champ magnétostatique total $\B(M)$.
		\sol{
			On obtient le champ totale par intégration,
			$$
			\Vec{B}(M) = \int \d \Vec{B}.
			$$
			soit
			\eqbox{
				\Vec{B}(M) = \frac{\mu_0 \sigma \omega}{2} \frac{(\sqrt{R^2+z^2}-z)^2}{\sqrt{R^2+z^2}} \,\ez.
			}
			La composante du champ sur l'axe s'écrit donc 
			$$
			B(z) = B_0 f\left( \frac{z}{R} \right), \text{ avec }
			\left\{
			\begin{array}{l}

\begin{exo}[PC][3][TD]{Champ magnétique hors axe d'une spire de courant}
	\Notions{\item Théorème d’Ampère \task Conservation du flux \task Équations de Maxwell \task Spire de courant}
\begin{minipage}{\textwidth}
\begin{wrapfigure}{r}{0.4\textwidth}
\vspace{-0.8cm}
\begin{center}
\includegraphics[scale=1]{loop_mag.pdf}
\end{center}
\end{wrapfigure}
On considère une spire circulaire d’axe $(Oz)$, de rayon $R$, et parcouru par un courant constant d’intensité $I$.
% La spire est d'axe $Oz$ et le courant parcourt la spire dans le sens trigonométrique.
En un point $M$ de l'axe $(Oz)$ où la spire est vue sous un angle $\theta$, le champ magnétostatique a pour expression  $\Vec{B}_{\rm axe} = B_0\, \ez $ avec


\begin{equation}
B_0 = \dfrac{\mu_0 I}{2R} \sin^3\!\theta.
\end{equation}
\end{minipage}
\vspace{0.45cm}
%\begin{center}
%\includegraphics[scale=1.2]{loop_mag.pdf}
%\end{center}

\begin{quest}

	\item Justifier la direction et le sens de $\Vec{B}_{\rm axe}$. Représenter le graphe de la fonction $B_0(z)$ en fonction de la hauteur $z$ par rapport au plan de la spire.
	\item  On définit un contour carré $\Gamma$ dont un des côtés est l’axe $(Oz)$, et les autres côtés sont rejetés à l’infini, où le champ est nul. Vérifier que cette l’expression du champ magnétostatique donnée est cohérente avec le théorème d'Ampère appliqué à ce contour.
	%On calculera pour cela la circulation de $ \vec{B}$ autour d'un contour rectangulaire infini dont un des côtés suit l'axe $Oz$. De plus, exprimer $B_0$ en fonction de $z$, et


	 \sol{
	 Tout plan contenant l'axe $(Oz)$ est plan d'antisymétrie de la distribution de courant, d'où l'orientation de $\Vec{B}$.


Le champ $B$ décroît suffisamment rapidement pour que l'intégrale sur les bords autre que l'axe $(Oz)$ s'annule.
La circulation sur ce contour d'Ampère vaut donc
$$\Gamma = \int_{(Oz)} \Vec{B}\ps \vec{\d l}$$
Or $z=R/\tan \theta$ donc  $\d z = -R/\sin^2(\theta) \d \theta$. On peut ensuite intégrer sur $\theta$,
$$ \Gamma = -R \int_\pi^0 \dfrac{B_0}{\sin^2(\theta)} \d \theta = -\dfrac{\mu_0 I}{2}\int_\pi^0 \sin(\theta)\,\d \theta = \mu_0 I,$$
en accord avec le théorème d'Ampère.

On utilise finalement la relation trigonométrique $\sin(\theta) = \dfrac{R}{\sqrt{R^2+z^2}}$ pour obtenir

$$B_0(z) = \dfrac{\mu_0 I}{2} \dfrac{R^2}{(R^2+z^2)^{3/2}}.$$
	 }

	\item On considère maintenant un point ${\rm M}(r,\theta,z)$ voisin de l'axe $(Oz)$ situé à une distance~$r\ll R$ de l'axe $(Oz)$. Justifier que le champ au point $\rm M$ est de la forme
%	\begin{equation}
	$\B(M) = B_z(r,z) \ez + B_r(r,z) \er$.
%	\end{equation}
En calculant le flux de $\B$ sur une surface bien choisie, montrer que %pour $r\ll R$,
\begin{equation}
B_r(r,z) \simeq - \dfrac{r}{2}\dfrac{\d B_0}{\d z}(z).
\end{equation}
%En déduire la fonction $B_r(r,z)$.

	\sol{
La symétrie autour de l'axe (Mz) ne s'applique plus et donc à priori le champ comprend une composante radiale $B_r$.

Comme surface d'intégration, on chosit un cylindre élémentaire de rayon $r$ et de longueur $\d z$ centré autour de l'axe $(Oz)$ et qui contient $M$. Puisque le champ magnétique est à flux conservatif, le flux sur cette surface fermée est nul. De plus, pour $r$ suffisamment petit, le champ axial sur les faces avant et arrière du cylindre est environ égal à $B_0$, donc
$$\Phi = \int_{\rm cylindre} \vec{B} \cdot \vec{\d S} = -B_0(z)\pi r^2 + B_0(z+\d z)\pi r^2 + B_r(r,z) 2\pi r \d z = 0,$$
d'où l'on tire la relation de l'énoncé.

Explicitement, il vient
$$
B_r(r,z) = \dfrac{3}{4} \mu_0 I   \dfrac{rzR^2}{(R^2+z^2)^{5/2}}.
$$
	}

	\item On considère maintenant un point $\rm M$ dans le plan de la spire $z=0$. On souhaite déterminer le champ longitudinal $B_z(r,0)$ pour $r\ll R$. À partir de l'équation de Maxwell-Ampère, montrer que
	\begin{equation}
	B_z(r,z) \simeq B_0(z) - \dfrac{r^2}{4} \dfrac{\d^2 B_0}{\d z^2}(z).
	\end{equation}
	En déduire $B_z(r,0)$. Que vaut $B_r(r,z)$ dans le plan de la spire $z=0$ ?
	%Le montrer par le calcul et par une considération de symétrie.

	\sol{On s'aide cette fois-ci d'une équation locale, l'équation de Maxwell-Ampère,
	$$\vec{\text{rot}} \vec{B} = \vec{0}.$$
En coordonnées cylindriques, et projeté selon $\vec{u_\theta}$, cette équation donne
$$\dfrac{\partial B_r}{\partial z} - \dfrac{\partial B_z}{\partial r}=0 \Rightarrow \dfrac{\partial B_z}{\partial r} = \dfrac{\partial B_r}{\partial z} = -\dfrac{r}{2}\dfrac{\d^2B_0}{\d z^2},$$
d'où la formule de l'énoncé par intégration. Explicitement, on trouve

	$$
	B_z(r,0) = \dfrac{\mu_0 I}{2R} \left(1+\dfrac{3r^2}{4R^2} \right).
	$$


	Enfin, le champ radial $B_r$ vaut zéro dans le plan de la spire d'après l'expression de la question 2). Cela est attendu puisque le plan contenant la spire est un plan de symétrie de la distribution de courant, et donc le champ est purement axial.

	}

\end{quest}
\end{exo}
