\subsection{Position du problème}

Soient deux référentiels $\Rcal$ et $\Rcal'$, liés respectivement aux observateurs $O$ et $O’$. Soient $(Oxyz)$ et $(O'x'y'z')$ leurs repères respectifs. Afin de relier les mouvements
d’un même point dans ces deux référentiels, il faut préciser le mouvement relatif de l’un par rapport à l’autre. Par convention, le référentiel $\Rcal$ est celui dans lequel on souhaite \emph{in fine} décrire le mouvement ; le référentiel $\Rcal'$ est celui dans lequel le mouvement est le plus simple.
 % Pour bien les distinguer, donnons-leur des noms.
% \begin{definition}
On appelle par convention $\Rcal$ le \imp{\nomR}~et $\Rcal'$ le \imp{\nomRprime}.
% \end{definition}

 Nous pouvons imaginer le repère $(O'x'y'z')$ comme un solide indéformable. Les mouvements de $\Rcal'$ par rapport à $\Rcal$ sont donc les mêmes que ceux que vous avez étudier en PCSI en cinématique du solide : les mouvements de \imp{translation} et de \imp{rotation uniforme} autour d'un axe fixe dans $\Rcal$.

\begin{definition}[Translation]\label{def:}
  Dans un mouvement de translation, les axes de $\Rcal'$ se déplacent en restant \imp{parallèles} à ceux de~$\Rcal$. Le mouvement de $\Rcal'$ est alors décrit entièrement par la vitesse de son origine $O'$ dans $\Rcal$, notée $\vR(O')$.
\end{definition}
\begin{danger}
  Les axes de $\Rcal'$ restent fixes, mais le mouvement de l'origine $O'$ peut être quelconque (rectilign, circulaire, autre). La vitesse $\vR(O')(t)$ est une fonction quelconque du temps.
\end{danger}
On peut ainsi rencontrer des mouvements de translation :
\begin{liste}
  \item rectiligne uniforme : par exemple un ascenseur montant à vitesse constante $\vR(O') = U\ez$ ;
  \item rectiligne non uniforme : par exemple un train se déplaçant en ligne droite uniformément accéléré, de vitesse $\vR(O') = a t \ex$, comme sur la \Fig{rect} ;
  \item circulaire uniforme : par exemple une nacelle d'une grande roue de rayon $R$ tournant à la vitesse angulaire constante $\omega$, de vitesse $\vR(O') = R \omega \et$.
  \item circulaire non uniforme : par exemple la masse d'un pendule simple de longueur $\ell$, de vitesse $\vR(O') = \ell \dot{\theta} \et$, comme sur la \Fig{circu}.
  \item quelconque : par exemple un skieur sur une piste droite bosselée.
  % , comme sur la \Fig{quelconque}.
\end{liste}

\begin{definition}[Rotation uniforme]\label{def:}
  Dans un mouvement de \imp{rotation uniforme} autour d'un axe fixe, les axes de $\Rcal'$ tournent à vitesse angulaire constante autour d'un axe fixe $(\Delta)$ dans $\Rcal$ passant par $O'$. Le mouvement de $\Rcal'$ est alors décrit entièrement par le vecteur rotation autour de $(\Delta)$, notée $\OR = \Omega \Vec{e}_{\!\Delta} = \Vec{\cte}$.
\end{definition}

\subsection{Transformation de Galilée}

Soit un point matériel $M$ quelconque, dont le mouvement est étudié ou bien dans le référentiel absolu $\Rcal$ d'origine $O$, ou bien dans le référentiel relatif $\Rcal'$ d'origine $O'$. Le passage d'un référentiel à un autre en mécanique newtonienne est donnée par la transformation de Galilée.

\begin{theorem}[Transformation de Galilée]\label{form:galile}
La relation de passage d'un référentiel $\Rcal$ à un autre $\Rcal'$ en mécanique newtonienne est donnée par :
% \begin{equation*}
$$
% \left\{
\begin{cases}
   t = t' & \text{(caractère absolu du temps),} \\
  \Vec{OM}  = \Vec{OO'} + \Vec{O'M} & \text{(relation de Chasles).}
\end{cases}
% \right.
$$

% \end{equation*}
\end{theorem}
\begin{remarque}
  La relation de Charles est une simple loi de géométrie euclidienne. La caractère absolu du temps est l'hypothèse de départ de la mécanique newtonienne, mais cette hypothèse est fausse et ne permet pas de décrire correctement la réalité pour les vitesses proches de celle de la lumière ! La mécanique relativiste est la théorie correcte.
\end{remarque}

\subsection{Loi de composition des vitesses}

\parag{Formule de dérivation} L'observateur $O$ lié à $\Rcal$ mesure une vitesse, appelée vitesse absolue,
\begin{equation}
\vR(M) = \dR{\Vec{OM}}{t}{\Rcal}.
\end{equation}
Par définition, l'opérateur $\dR{}{t}{\Rcal}$ signifie que la base cartésienne $(\ex,\ey,\ez)$ liée à $\Rcal$ est considérée comme fixe lors de la dérivation.
L'observateur $O'$ lié à $\Rcal'$ mesure quant à lui une vitesse, dite vitesse relative,
\begin{equation}
\vRp(M) = \dR{\Vec{O'M}}{t}{\Rcal'},
\end{equation}
où l'opérateur $\dR{}{t}{\Rcal'}$ signifie que la base cartésienne $(\ex',\ey',\ez')$ liée à $\Rcal'$ est considérée comme fixe lors de la dérivation. Pour faire le lien entre les vitesses absolue et relative, nous devons relier les opérateurs $\dR{}{t}{\Rcal'}$  et $\dR{}{t}{\Rcal}$.

\begin{prop}\label{prop:deriv_R}
Les dérivés temporelles sont identiques dans deux référentiels $\Rcal$ et $\Rcal'$ en translation,
\begin{equation*}\label{eq:deriv_R}
\dR{}{t}{\Rcal} = \dR{}{t}{\Rcal'}.
\end{equation*}
\end{prop}

\subsection{Loi de composition des accélérations}

Il vient
\begin{align*}
\aR(M) & = \dR{\vR(M)}{t}{\Rcal} & (\text{définition accélération})\\
& = \dR{\vRp(M)}{t}{\JVert{\Rcal}}  + \dR{\vR(O')}{t}{\Rcal}   & (\text{\Eq{relat_trans}}) \\
& = \dR{\vRp(M)}{t}{\JVert{\Rcal'}} + \dR{\vR(O')}{t}{\Rcal}     & (\text{\Prop{deriv_R}}) \\
& =  \aRp(M) ~~~~~~~~+ \aR(O') .  & (\text{définition accélération})
\end{align*}

\begin{theorem}[Composition des accélérations en translation]\label{form:compoa}
Les accélérations d'un point $M$ dans deux référentiels $\Rcal$ et $\Rcal'$ en \imp{translation} sont reliées par
\begin{equation*}
\aR(M) = \aRp(M) + \aR(O').
\end{equation*}
\end{theorem}

\noindent Dans le cas de référentiels en translation, l'accélération d'entraînement de $\mathcal{R'}$ par rapport à $\Rcal$ s'identifie à l'accélération de l'origine $O'$ mesurée dans $\Rcal$. Elle ne dépend pas du point d'étude $M$.

\section{Référentiels en rotation uniforme autour d'un axe fixe}

Nous considérons maintenant un référentiel $\Rcal'$ en rotation uniforme autour d'un axe fixe $(\Delta)$ par rapport à $\Rcal$. Sans perte de généralité, nous supposerons que les origines $\rm O = O'$ sont confondues, et que l'axe $(\Delta)$ correspond à l'axe porté par $\ez=\ez'$. Nous pouvons donc attacher une base cartésienne $(\ex,\ey,\ez)$ à $\Rcal$ et voir la base cartésienne $(\ex',\ey',\ez')$ liée à $\Rcal'$ comme une base cylindrique dans $\Rcal$ attachée à $\Rcal'$. Soit $\OR = \Omega \ez$ le vecteur rotation de $\Rcal'$ par rapport à $\Rcal.$



\subsection{Loi de composition des vitesses}


\parag{Formule de dérivation} Pour déterminer les lois de composition des vitesses et des accélérations, nous devons trouver l'équivalent de la formule de dérivation en référentiel mobile pour les référentiels en rotation.

\begin{rappel}{Formule de dérivation en référentiel mobile en rotation uniforme}
Soit $\Vec{u} = a \ex' + b \ey' + c \ez'$ un vecteur exprimé dans la base attachée à $\Rcal'$.
% \begin{equation}
% \Vec {O'M} = x' \ex' + y' \ey' + z' \ez''.
% \end{equation}
En dérivant par rapport au temps $t'=t$ dans le référentiel $\Rcal$, il vient
\begin{align}\label{eq:deriv_formu}
\dR{\Vec {u}}{t}{\Rcal}  = \JVert{\dd{a}{t} \ex' + \dd{b}{t} \ey' + \dd{c}{t} \ez'} +  a \JRouge{\dR{\ex' }{t}{\Rcal}} + b  \JRouge{\dR{\ey'}{t}{\Rcal}}  + c  \JRouge{\dR{\ez'}{t}{\Rcal}}
 % = \JVert{\dR{\Vec {u}}{t}{\Rcal'}} +  a \JRouge{\dR{\ex' }{t}{\Rcal}} + b  \JRouge{\dR{\ey'}{t}{\Rcal}}  + c  \JRouge{\dR{\ez'}{t}{\Rcal}}.
\end{align}
Or d'après le cours de cinématique du point,
$$
\begin{cases}
  \dR{\ex'}{t}{\Rcal} = \dot{\theta} \ey' = \Omega \ey' = \Omega \ez' \pv \ex' = \OR \pv \ex', \\
\dR{\ey'}{t}{\Rcal} = -\dot{\theta} \ex' = -\Omega \ex' = \Omega \ez' \pv \ey' = \OR \pv \ey', \\
\dR{\ez'}{t}{\Rcal} = \Vec{0} = \OR \pv \ez'. \\
\end{cases}
$$
On en déduit que
\begin{align}
  % \label{eq:deriv_formu}
\dR{\Vec {u}}{t}{\Rcal}
 = \JVert{\dR{\Vec {u}}{t}{\Rcal'}} +  a \JRouge{\OR \pv \ex'} + b  \JRouge{\OR \pv \ey'}  + c  \JRouge{ \OR \pv \ez'} = \dR{\Vec {u}}{t}{\Rcal'} + \OR \pv \Vec{u}.
\end{align}
% Or les trois termes ne rouge sont nuls si $\Rcal'$ est en translation par rapport à $\Rcal$, puisque les vecteurs de base $(\ex',\ey',\ez')$ sont fixe dans $\Rcal$, les axes restant constamment parallèles.
 % ; seule la position de l'origine $O'$ dépend du temps.
\end{rappel}
 Cette formule est en fait très générale, et s'applique quel que soit le mouvement de $\Rcal'$ par rapport à $\Rcal$.
%  également dans un référentiel auxiliaire en rotation autour d'un axe fixe $\Delta$ passant par $O'$. Nous devons par contre déterminer la vitesse d'entraînement d'un point $M$ dans cette nouvelle situation. Pour cela, exprimons la vitesse du point coïncident $\PC$. On peut montrer que
% \begin{equation}\label{eq:ve_rot}
% \vR(\PC) =  \OR \pv \Vec{O’\PC}.
% \end{equation}
%où $H$ désigne le projeté orthogonal de $\PC$ sur l'axe de rotation $\Delta$. Or $\Vec{O'H}$ est colinéaire à $\OR$ par définition de $H$, et donc on peut remplacer $H$ par $O'$ dans l'\Eq{ve_rot}.
% Par définition du point coïncident, on a de plus $M^*=M$. On en déduit la formule suivante.


\begin{prop}\label{prop:deriv_Rrota}
Les dérivés temporelles dans deux référentiels en mouvement \imp{quelconque} $\Rcal$ et $\Rcal'$ vérifient
\begin{equation*}
\dR{\Vec{u}}{t}{\Rcal} = \dR{\Vec{u}}{t}{\Rcal'} + \OR \pv \Vec{u}
\end{equation*}
% pour tout vecteur $\Vec{u}$.
\end{prop}


\parag{Loi de composition des vitesses}

Reprenons l'\Eq{relat_trans} en utilisant que $\vR(O') = \Vec{0}$, on a
% nous pouvons exprimer $\vR(M)$ en fonction de $\vRp(M)$, suivant
\begin{equation}\label{eq:relat_trans}
\vR(M)  = \dR{\Vec {O'M}}{t}{\Rcal} = \dR{\Vec {O'M}}{t}{\Rcal'} + \OR \pv \Vec {O'M} = \vRp(M) + \OR \pv \Vec {O'M}.
\end{equation}
% Il nous reste à relier la dérivée temporelle de $\Vec {O'M}$ dans~$\Rcal$ (calculée en supposant les vecteurs $\ex,\ey$ et $\ez$ fixes) à cette même dérivée dans $\Rcal'$ (calculée en supposant les vecteurs $\ex',\ey'$ et $\ez'$ fixes).
% On aboutit à la relation
% \begin{equation}\label{eq:relat_trans}
% \vR(M)  = \vR(O') +  \vRp(M).
% \end{equation}

\begin{theorem}[Composition des vitesses en rotation uniforme]\label{form:compov}
Les vitesses d'un point $M$ dans deux référentiels $\Rcal$ et $\Rcal'$ en \imp{rotation uniforme} sont reliées par
\begin{equation*}
\vR(M) = \vRp(M) + \OR \pv \Vec {O'M}
\end{equation*}
\end{theorem}

\begin{exemple}{Ressort en rotation}
  \begin{minipage}{0.4\textwidth}
    \includegraphics[scale=0.6]{masse_ressort_ref.pdf}
  \end{minipage}\hfill
  \begin{minipage}{0.58\textwidth}
    Le point matériel $M$ est astreint à se osciller radialement le long d'une tige en rotation. Sa vitesse dans $\Rcal'$ a donc pour position $\Vec{OM'}(t) = r(t) \ex'$. La tige tourne dans le référentiel $\Rcal$ du laboratoire à une vitesse angulaire constante $\Omega$. Déterminer sa vitesse absolue $\vR(M)$.
  \end{minipage}

\end{exemple}
\examplesol{
La vitesse relative vaut $\vRp(M) = \dot{r}\, \ex'$. La vitesse absolue vaut ensuite
$$
\vR(M) = \vRp(M) + \OR \pv \Vec {O'M} =  \dot{r} \ex' + \Omega \ez \pv r \ex' = \dot{r} \ex' + \Omega r \ey'.
$$
 C'est l'expression connue de la vitesse exprimée dans les coordonnées polaires.
}


\subsection{Loi de composition des accélérations}

En utilisant la \Formu{compov} et le fait que $\OR = \Vec{\cte}$ dans $\Rcal$ pour une rotation uniforme, il vient
\begin{align*}
% \tag{\text{définition accélération}}
\aR(M) = \dR{\vR(M)}{t}{\Rcal}
% \tag{\text{\Formu{compov}}}
 = \dR{\vRp(M)}{t}{\Rcal} + \OR \pv \dR{\Vec{O’M}}{t}{\Rcal}.
\end{align*}
Le premier terme, issu de la dérivée de la vitesse dans $\Rcal$, donne d’après \Prop{deriv_Rrota},
\begin{align*}
% \tag{\text{\Prop{deriv_Rrota}}}
\dR{\vRp(M)}{t}{\Rcal}  =\dR{\vRp(M)}{t}{\Rcal’} + \OR \pv  \vRp(M)  = \aRp(M) +  \JVert{\OR \pv  \vRp(M)}.
\end{align*}
Le second terme donne, là aussi en utilisant \Prop{deriv_Rrota},
\begin{align*}
% \tag{\text{\Formu{ve_rot}}}
\OR \pv  \dR{\Vec{O'M}}{t}{\Rcal} = \JVert{\OR \pv  \vRp(M)} + \JRouge{\OR \pv  (\OR \pv \Vec{O'M})}.
\end{align*}
% puisque par hypothèse la rotation est \imp{uniforme}, et donc
% \begin{equation}
%  \dR{\OR}{t}{\Rcal} = \Vec{0}.
% \end{equation}
On reconnaît deux termes : ceux en rouge qui dépendent de la position $\rm M$ ; ceux en vert qui dépendent de la vitesse $\vRp(M)$.

\begin{liste}
  \item \udl{Terme en position}. Soit $\rm H$ le projeté orthogonal de $\rm M$ sur l'axe de rotation $(\Delta)$.
  %  et $(\er,\et)$ une base polaire attachée à $\rm M$ dans le plan orthogonal à $(\Delta)$.
  % Son expression peut se simplifier de la façon suivante en utilisant la formule du double produit vectoriel, $\Vec {a}\pv(\Vec {b}\pv \Vec {c})=\Vec {b}\left(\Vec {a}\cdot\Vec {c}\right)-\Vec {c}(\Vec {a}\cdot\Vec {b})$.
  On décompose la position relative comme $\Vec{O'M} = \Vec{O'H} + \Vec{HM}$, où $\Vec{O'H}$ est colinéaire à $\OR$ et $\Vec{HM}$ est orthogonal à $\OR$. On utilise la formule du double produit vectoriel pour obtenir
  \begin{equation}
  \begin{split}
    \OR \pv  (\OR \pv \Vec{O'M}) & = (\OR \cdot \Vec{O'M}) \OR - \Omega^2\,\Vec{O’M} \\ & =  (\OR \cdot \Vec{O'H}) \OR   -  \Omega^2 \Vec{O’H} -\Omega^{2} \Vec {HM} \\ 
    & = -\Omega^{2} \Vec {HM}
  \end{split}
  \end{equation}
  car $\Vec{O'H}$ est colinéaire à $\OR$. On voit que cette accélération est \imp{centripète}.
% \begin{equation}
% \OR \pv  (\OR \pv \Vec{O'M}) = \Omega \ez \pv  (\Omega \ez \pv (\Vec{O'H} + \Vec{HM})) = \Omega \ez \pv \Omega r \et   =  -\Omega^{2} r \,\er = -\Omega^{2} \Vec {HM}.
% \end{equation}
% \begin{equation}
% \OR \pv  (\OR \pv \Vec{O'M}) = \OR (\Omega z) - \Omega^2 \Vec{O’M} =  \Omega^2 z \ez  -  \Omega^2 \Vec{O’M}  = -\Omega^{2} \Vec {HM},
% \end{equation}
% \begin{danger}
%   Attention aux cas où les origines $O$ et $O'$ ne sont pas confondues, car
% $r$ représente la distance entre le point $M$ et l'axe de rotation $(\Delta)$, et non la distance entre le point $M$ et l'axe $(Oz)$ !
% \end{danger}
 \item \udl{Terme en vitesse}. Les terme en vert $\OR \pv  \vRp(M)$ est présent \imp{deux} fois. La somme de ces deux termes donne l'accélération dite de Coriolis. L'accélération de Coriolis dépend de la vitesse relative $\vRp(M)$ : elle est donc nulle si le point matériel $M$ est immobile dans $\Rcal'$.
\end{liste}

 % Le terme en vert, qui apparaît deux fois, dépend de la vitesse de $M$ dans $\Rcal'$ : il s'agit de l'accélération de Coriolis.
\begin{definition}[Accélération de Coriolis]\label{def:ae}
On appelle \imp{accélération de Coriolis} du point $M$ dans le référentiel $\mathcal{R'}$ le vecteur
\begin{equation*}
\aC  = 2\OR \pv  \vRp(M).
\end{equation*}
\end{definition}

En combinant toutes les contributions au vecteur accélération dans $\Rcal$, nous aboutissons à la loi suivante.
%, qui est en fait générale.
\begin{theorem}[Composition des accélérations en rotation uniforme]\label{form:compoa}
Les accélérations d'un point $M$ dans deux référentiels $\Rcal$ et $\Rcal'$  en \imp{rotation uniforme} sont reliées par
\begin{equation*}
\aR(M) =  \aRp(M) -\Omega^{2} \Vec{HM}  +\aC.
\end{equation*}
\end{theorem}

\noindent De même que pour la vitesse, l'accélération de $O'$ dans $\Rcal$ est nulle, soit $\aR(O')=\Vec{0}$.

\begin{exemple}{Le manège}
On considère un enfant représenté par un point matériel $\rm M$ sur un manège. Soit $\Rcal$ le référentiel lié au sol, de repère $(Oxyz)$, et $\Rcal'$ le référentiel lié à une origine $\rm O'=O$ de l'axe du manège, en rotation à la vitesse angulaire constante $\Omega$ autour de l’axe $(Oz)$. L'enfant est à une distance $r$ de l'axe $(Oz)$.
\begin{quest}
  \item	Définir une base cartésienne $(x',y',z')$ attachée à $\Rcal'$, puis une base locale cylindrique $(r,\theta,z)$ attachée à $\rm M$. Schématiser la situation. 
  \item Déterminer l'accélération de Coriolis $\aC$ et l'accélération absolue $\aR(M)$, lorsque :
  \begin{enumerate}[label=(\alph*)]
    \item l'enfant est assis sur un cheval en bois qui oscille de haut en bas avec une vitesse relative $\vRp(M) = v_0 \cos(\omega t) \ez$ ;
    \item l'enfant marche en cercle sur le socle du manège à la vitesse relative $\vRp(M) = r \omega \et$.
  \end{enumerate}
\end{quest}

  \item  Avec $\OR = \Omega\, \ez$, on trouve l'accélération de Coriolis  $\aC = 2\OR \pv  \vRp(M) = 2 \Omega\, \ez \pv v_0 \cos(\omega t) \ez  = \Vec{0}$ ! L'accélération relative vaut $\aRp(M) = \dd{\vRp(M)}{t} = -v_0\omega \sin(\omega t) \ez$, d'où l'accélération absolue
  $$
\aR(M) = \aRp(M) -\Omega^{2} r \,\er +\aC = -v_0\omega \sin(\omega t) \ez -\Omega^{2} r \,\er.
  $$

  \item L'accélération de Coriolis vaut $\aC = 2\OR \pv  \vRp(M) = 2 \Omega\, \ez \pv v_0 \et = -2r\Omega \omega \,\er.$ L'accélération relative vaut ici $\aRp(M) = -r\omega^2 \er$, et donc l'accélération absolue vaut

\section{Formules générales}

\subsection{Notion de point coïncident}

Les vecteurs vitesses de $\rm M$ dans $\Rcal$ et $\Rcal'$ diffèrent en général. Notons que l'\Eq{relat_trans} est spécifique au mouvement de translation entre référentiels. Pour interpréter physiquement et généraliser ce résultat à d'autres mouvements, nous définissons les notions de point coïncident et de vitesse d'entraînement.

\begin{definition}[Point coïncident]
On appelle \imp{point coïncident} en $\rm M$ le point $\PC$ fixe dans $\Rcal'$ qui
coïncide à l’instant $t$ avec le point mobile $\rm M$.  Mathématiquement, le point coïncident vérifie $\PC(t)=\rm M(t)$ avec $\vRp(\PC)=0$.
\end{definition}

\begin{figure}[h!]
\centering
\includegraphics[scale=1]{coincid.pdf}
\caption{Soient le référentiel $\Rcal$ lié au sol et le référentiel $\Rcal'$ lié centre du cerceau qui roule sans glisser (qui ne tourne pas avec lui). $\Rcal'$ est en translation par rapport à $\Rcal$. On indique la position de cerceau à deux instants voisins $t_1$ (bleu plein) et $t_2$ (rouge tireté). Le point mobile $M$ lié à la roue tourne avec elle et donc se trouve en position $M(t_1)$ puis en position $M(t_2)$ au-dessus par rapport au sol. Le point coïncident en $t_1$, noté $\PC(t_1)$, est fixe dans $\Rcal'$ et donc se déplace avec le centre $C$, en direction de la roue mais sans tourner avec elle.}
\label{fig:coinci}
\end{figure}
%%%%%%%%%%%%%


Le point $\PC$ est donc rigidement lié au référentiel relatif $\Rcal'$. Le mouvement du point coïncident $\PC$ représente le mouvement d'entraînement de $\Rcal'$ par rapport à $\Rcal$. On peut alors définir les notions de vitesse et d'accélération d'entraînement.

\begin{definition}[Vitesse d'entraînement]\label{def:ve}
On appelle \imp{vitesse d'entraînement} du point $M$ la grandeur
% vecteur vitesse du point coïncident en $M$, noté $\PC$,
\begin{equation*}
\ve  = \vR(\PC) = \begin{cases}
\vR(O') & \text{(translation),}\\
\OR \pv \Vec {O'M} & \text{(rotation uniforme).}
\end{cases}
\end{equation*}
\end{definition}
La vitesse d'entraînement dépend du point $M$ lorsque le mouvement de $\Rcal$ par rapport à $\Rcal'$ est une rotation uniforme mais pas une translation. Comme nous avons également que $\aRp(\PC)=0$, l'accélération d'entraînement a les expression suivantes.

\begin{definition}[Accélération d'entraînement]\label{def:ae}
On appelle \imp{accélération d'entraînement} du point $M$ la grandeur
\begin{equation*}
\ae  = \aR(\PC) 
%= \dR{\ve}{t}{\Rcal} 
=
 \begin{cases}
\aR(O') & \text{(translation),}\\
- \Omega^2 \Vec{HM} & \text{(rotation uniforme).}
\end{cases}
\end{equation*}
\end{definition}
% \begin{danger}
%   Attention, l'accélération d'entraînement n'est pas la dérivée de la vitesse d'entraînement !
% \end{danger}
L'accélération d'entraînement dépend du point $M$ lorsque le mouvement de $\Rcal$ par rapport à $\Rcal'$ est une rotation uniforme mais pas une translation.

\subsection{Lois de composition}

La définition de la vitesse et de l'accélération d'entraînement conduit aux formules générales suivantes.

\begin{theorem}[Composition des vitesses]\label{form:compov}
Les vitesses d'un point $M$ dans deux référentiels quelconques $\Rcal$ et $\Rcal'$ sont reliées par
\begin{equation*}
\vR(M) = \vRp(M) + \ve.
\end{equation*}
\end{theorem}

\begin{theorem}[Composition des accélérations]\label{form:compoa}
Les accélérations d'un point $M$ dans deux référentiels quelconques $\Rcal$ et $\Rcal'$ sont reliées par
\begin{equation*}
\aR(M) = \aRp(M) +  \ae + \aC.
\end{equation*}
\end{theorem}
