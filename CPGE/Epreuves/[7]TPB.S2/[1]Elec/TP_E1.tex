\begin{protocole}
  \item Montrer que la tension de sortie $s(t)$ vérifie l'équation différentielle
  \begin{equation}
    \frac{\d s(t)}{\d t} + a \, s(t) = b \, e(t),
  \end{equation}
  où l'on déterminera les constantes $a$ et $b$ en fonction de $R$ et $C$.

  \sol{
    Le courant traversant la résistance et le condensateur, a pour expression $t=C \frac{d s}{d t}$. La loi des mailles donne $e(t)=s(t)+R i$ soit :

\eqbox{
  s^{\prime}(t)+\frac{1}{\tau} s(t)=\frac{1}{\tau} e(t)
  }

en posant $\tau=R C$ temps caractéristique du circuit $R C$, on obtient finalement 
\eqbox{
a=b=\frac{1}{\tau}.
}
  }
  \item Le circuit est alimenté par un échelon de tension entre $\SI{0}{V}$ à $E=\SI{10}{V}$ à l'instant $t=0$. Déterminer l'expression de $s(t)$. 
  \sol{
    Si la tension d'entrée est un échelon de tension d'amplitude $E$, l'équation différentielle devient :

$$
s^{\prime}(t)+\frac{1}{\tau} s(t)=\frac{1}{\tau} E
$$


La solution générale de cette équation est $s_g(t)=\mathrm{A} e^{-\frac{1}{6}}$ et la solution particulière $s_{\mathrm{p}}=E$ donc $s(t)=E+\mathrm{A} e^{-\frac{1}{t}}$.

Le condensateur est initialement déchargé, soit $s\left(t=0^{-}\right)=0$. La tension étant continue aux bomes d'un condensateur (par continuité de l'énergie) nous avons donc également $s\left(t=0^{+}\right)=0$ d'où $0=E+\mathrm{A}$.

Finalement :

% $$
\eqbox{
  s(t)=E\left(1-e^{-t/\tau}\right).
  }
% $$

  }
  % Cette fonction est appelée réponse indicielle du circuit.
  \item Déterminer et tracer le diagramme de Bode asymptotique de $s(t)$.
  \sol{
  La fonction de transfert du circui est donnée par la loi du pont diviseur de tension, soit 
  $$
  \udl{H} = \frac{s(t)}{e(t)} = \frac{1/jC\omega}{R+1/jC\omega} = \frac{1}{1+j\omega R C},
  $$
  et donc 
  \eqbox{
    \udl{H}(\omega) = \frac{1}{1+j\omega \tau}.
  }
  Le diagramme de Bode asymptotique de $\udl{H}(\omega)$ s'obtient en calculant le gain $G_{\rm dB}(\omega)$ et la phase $\varphi(\omega)$ de $\udl{H}(\omega)$, dans les limites basse et haute fréquence. On trouve pour le gain,
  $$
  G_{\rm dB}(\omega) = 10 \log |\udl{H}(\omega)|^2 = -10 \log \left( 1+\left(\omega \tau\right)^2 \right) \sim \begin{cases}
    -20 \log(\omega) & \text{si } \omega \tau \gg 1,\\
    0 & \text{si } \omega \tau \ll 1.
  \end{cases}
  $$
  Il s'agit d'un filtre passe-bas d'ordre $1$ de pulsation de coupure $\omega_c = 1/\tau$.
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.8\columnwidth]{gain_RC}
    %\capt{}
  \end{fig}
  La phase est donnée par
  $$
  \varphi(\omega) = \arg \udl{H}(\omega) = -\arctan(\omega \tau) \sim \begin{cases}
    -\frac{\pi}{2} & \text{si } \omega \tau \gg 1,\\
    0 & \text{si } \omega \tau \ll 1.
  \end{cases}
  $$
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.8\columnwidth]{phase_RC}
    %\capt{}
  \end{fig}
  }
\end{protocole}

\subsection{Le circuit RC}
 \begin{protocole}
  \item Réaliser le montage de la \Fig{RC}. 
  \sol{
    \udl{ATTENTION} : bien positionner la masse et la sortie $s(t)$ ! Souvent, la sortie est prise au niveau de la borne du condensateur qui devrait être reliée à la masse. Éviter cette erreur de branchement !
  }
  \item Tracer la réponse indicielle $s(t)$ sur LatisPro et imprimer la courbe.
  \sol{


    \udl{ATTENTION} : veillez à bien alimenter l'entrée par un créneau entre $\SI{0}{V}$ et $\SI{10}{V}$, i.e. de mettre \emph{level} à $\SI{10}{V}$ et \emph{offset} à $\SI{5}{V}$. Choisir une fréquence très grande devant la pulsation de coupure du circuit (par exemple quelques $\SI{100}{Hz}$). Bien choisir les paramètres d'acquisition : notamment, prendre une durée d'acquisition $T$ de l'ordre de quelques $\tau$, et non de plusieurs secondes !
    \begin{fig}[label]
      \centering
      \includegraphics[width=0.8\columnwidth]{s.jpg}
      %\capt{}
    \end{fig}

    Voici à quoi ressemble la réponse indicielle du circuit RC. 

  }
  \item Quelle méthode connaissez-vous pour déterminer la constante de temps~$\tau$ du circuit ? Sachant que $C = \SI{10}{nF}$, déterminer la valeur de $R$.
  \sol{
    \begin{itemize}
      \item 
      \udl{Tangente à l'origine}
      
      L'intersection de la tangente à l'origine et de la valeur finale a lieu à l'instant $\tau$. En effect l'équation de la tangente à l'origine s'écrit,
       $$s_{\text {tan }}(t)=s^{\prime}(0) t+s(0).$$
      
      Ici, $s(0)=0$ et $s^{\prime}(0)=\frac{E}{\tau}$ done 
      \eqbox{
        s_{\text {tan }}(t)=\frac{E}{\tau} t.
      }
      Il est alors évident que cette droite coupe donc l'asymptote, d'équation $y=E$, au temps $\tau$.
      \begin{fig}[label]
        \centering
        \includegraphics[width=0.8\columnwidth]{tangente_s.jpg}
        %\capt{}
      \end{fig}

      \item \udl{Valeur particulière de $s(\tau)$} : on a
      \eqbox{
      s(\tau)=E\left(1-\frac{1}{\mathrm{e}}\right) \approx \num{0,63} E:
        }
        en repérant avec les curseurs de l'oscilloscope le temps au bout duquel la tension correspond a $63 \%$ de la valeur finale, on peut déduire le temps $\tau$.
      
      \begin{fig}[label]
        \centering
        \includegraphics[width=0.8\columnwidth]{lecture_s.jpg}
        %\capt{}
      \end{fig}
      Connaissant $C$ et $\tau$, on peut en déduire la valeur de $R$ par
      \eqbox{
      R = \frac{\tau}{C}.
      }
    \end{itemize}
      }
      
  \item Tracer le diagramme de Bode expérimental de $s(t)$, point par point, en expliquant votre méthode de mesure. On tracera le diagramme en gain et en phase.
  \sol{
    \udl{ATTENTION} : pour tracer le diagramme de Bode, il faut alimenter le circuit par un signal \emph{sinusoïdal} de fréquence $f$ connue. On mesure ensuite les amplitudes $E$ et $S$ des signaux d'entrée et de sortie, respectivement, ainsi que le déphasage $\varphi$ entre la sortie et l'entrée. On répète cette opération pour différentes fréquences $f$, en couvrant au moins 2 décades centrées autour de la fréquence de coupure $f_c=1/2\pi\tau$. Après avoir calculé le gain en dB, on obtient les graphes suivants.
    \begin{fig}[label]

      \includegraphics[width=0.8\columnwidth]{gain_RC.jpg}
      %\capt{}
    \end{fig}
    
    \begin{fig}[label]
      \centering
      \includegraphics[width=0.8\columnwidth]{phase_RC.jpg}
      %\capt{}
    \end{fig}
  }
  \item Déterminer la bande passante du circuit, en expliquant votre méthode de mesure. Comparer avec la valeur attendue. 
  \sol{
    Des courbes précédentes nous pouvons déduire de différentes manières la bande passante.
    \begin{itemize}
      \item repérage de la fréquence telle que $G_{\mathrm{dB}}\left(f_{\mathrm{c}}\right)=G_{\mathrm{dB}, M a x}-3 \mathrm{~dB}$;
      \item repérage de la fréquence correspondant à l'intersection de l'asymptote et l'axe des abscisses;
      \item repérage de la fréquence où le déphasage est $\varphi\left(f_c\right)=-\frac{\pi}{4}$.
    \end{itemize}
      
      Les différentes mesures sont cohérentes et la largeur de la bande passante obtenue vaut $[0 ; \SI{2,7}{Hz}]$.
  }

 \subsection{La boîte sans nom}
 Le candidat dispose d'un circuit inconnu, appelé boîte sans nom (BSN).
 \begin{protocole}
  \item Alimenter la BSN par un échelon de tension allant de $\SI{0}{V}$ à $\SI{10}{V}$. Tracer la réponse du circuit $s(t)$ sur LatisPro et imprimer la courbe.
  \sol{
    Voici la réponse indicielle de la BSN, obtenue en alimentant le circuit par un échelon de tension de $\SI{10}{V}$. On remarque que la réponse est très proche de celle du circuit RC, mais que la tension est discontinue à l'instant $t=0$. Le circuit se comporte donc comme un circuit aux temps longs (basse fréquence) mais de façon très différente du circuit RC aux temps courts (haute fréquence).
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.8\columnwidth]{reponse_BSN}
    %\capt{}
  \end{fig}
  Sachant de plus que la BSN est un circuit actif, on peut supposer que la BSN contient un ALI avec des résistances et une capacité, mais que la tension de sortie n'est pas mesurée directement aux bornes de la capacité ; peut-être à la sortie de l'ALI ?

  En supposant une loi exponentielle pour la réponse indicielle, on peut proposer la loi suivante pour la réponse indicielle de la BSN,
  \eqbox{
    s(t) = E \left(1 - 2e^{-t/\tau}\right).
  }
  Avec la méthode de la tangente à l'origine, on peut d'ailleurs montrer que la constante de temps $\tau$ du circuit a la même valeur que pour le circuit RC.
  }
  \item Tracer le diagramme de Bode expérimental de la BSN, point par point, en expliquant votre méthode de mesure. On tracera le diagramme en gain et en phase.
  \sol{
  Voici le diagramme de Bode expérimental de la BSN, obtenu en alimentant le circuit par un signal sinusoïdal de fréquence $f$ connue. Pour le gain :
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.8\columnwidth]{gain_BSN}
    %\capt{}
  \end{fig}
  On remarque que le gain est constant à toute fréquence ! (sauf au-delà du $\SI{}{MHz}$, car l'ALI ne se comporte plus de façon idéale). 
  Pour la phase :
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.8\columnwidth]{phase_BSN}
    %\capt{}
  \end{fig}
  On remarque que la phase évolue de $0$ à $-\pi$ en passant par $-\pi/2$ à la fréquence de coupure. Elle est donc égale au double de la phase du circuit RC. On en conclut que la BSN est un circuit d'ordre $2$.
  }
  \item De quel type de circuit s'agit-il ? Comparer au circuit RC. Sur quelle plage de fréquence la BSN se comporte-t-elle comme un circuit RC ?
  \sol{
  D'après l'étude du diagramme de Bode, la BSN est un circuit d'ordre 
  \begin{center}
    \fbox{passe-tout déphaseur.}
  \end{center}
  Le comportement est identique à celui du circuit RC pour les basses fréquences, i.e. lorsque $\omega \tau \ll 1$.\\ 

  \udl{Pour aller plus loin.} Voici le circuit complet du passe-tout déphaseur.
  \begin{fig}[label]
    \centering
    \includegraphics[width=0.8\columnwidth]{passe_tout}
    %\capt{}
  \end{fig}
  On peut montrer que pour un ALI idéal, la fonction de transfert du passe-tout déphaseur est donnée par
  \eqbox{
    \udl{H}(\omega) =  \frac{1-j\omega \tau}{1+j\omega \tau}.
  }
  Le diagramme de Bode du passe-tout déphaseur est donc donné par
  \eqbox{
    G_{\rm dB}(\omega) = 0 \quad \text{et} \quad \varphi(\omega) = -2\arctan(\omega \tau).
  }
  L'équation différentielle vérifiée par la tension de sortie est donnée par
  \eqbox{
    \frac{\d s}{\d t} + \frac{s}{\tau}  = -\frac{\d e}{\d t} + \frac{e}{\tau}.
  }
  Pour un échelon de tension, on résout l'équation différentielle pour $t>0$,
  % \eqbox{
  $$
    \frac{\d s}{\d t} + \frac{s}{\tau}  = \frac{E}{\tau},
  $$
  de solution,
  $$
  s(t) = A e^{-t/\tau} + E.
  $$
  La condition initiale s'obtient en intégrant cette équation sur un intervalle infinitésimal $[-\veps,\veps]$, et en utilisant le fait que $s(t)$ et $e(t)$ sont discontinues à l'instant $t=0$, mais subissent des discontinuités finies, soit 
  % $$
  \begin{align*}
    \int_{-\veps}^\veps \left(\frac{\d s}{\d t} + \frac{s}{\tau}\right) \d t =  \int_{-\veps}^\veps \left( -\frac{\d e}{\d t} + \frac{e}{\tau}\right) \d t \Rightarrow_{\veps \rightarrow 0} \int_{-\veps}^\veps \frac{\d s}{\d t} \d t =  \int_{-\veps}^\veps  -\frac{\d e}{\d t} \d t\, \\ 
     \Rightarrow_{\veps \rightarrow 0} s(0^ +) - s(0^ -)  = s(0^ + ) = A + E = -\left(e(0^ +) - e(0^ -)\right) = - E \\ 
      \Rightarrow  A = - 2E.
  \end{align*}
  On en déduit la réponse indicielle du passe-tout déphaseur,
  $$
  s(t) = -2E e^{-t/\tau} + E = E\left(1 - 2e^{-t/\tau}\right).
  $$

  \udl{Pour aller plus loin.} La réponse indicielle $s_i(t)$ d'un circuit quelconque a un intérêt calculatoire majeur, puisqu'elle permet de déterminer directement la fonction de transfert du circuit $\udl{H}(\omega)$. Le lien entre ces deux grandeurs est donné par la transformée de Fourier : on a 
  \eqbox{
    \udl{H}(\omega) = TF[s_i](\omega) = \int_{-\infty}^{+\infty} s_i(t) e^{-j\omega t} \d t.
  }
  On peut donc déterminer la fonction de transfert du circuit en effectuant la transformée de Fourier de la réponse indicielle sur LatisPro, puis calculer le gain $G_{\rm dB}(\omega)$ et la phase $\varphi(\omega)$ du circuit. 
  }
  
 \end{protocole}
