  \section{Rappels sur la polarisation}
  \subsection{Polarisation d'une OPPH}
  
\Enonce{
  La lumière est une onde électromagnétique ($\vv{E}$, $\vv{B}$). Dans toute la suite, on ne considère que des OPPH, dans le vide, de vecteur d'onde $\vv{k}=k\vv{e_z}$ avec $k=\frac{\omega}{c}$. La structure des OPPH est telle que les vecteurs $(\vv{k},\vv{E},\vv{B})$ forment un trièdre direct. L'onde est donc transverse.

  \begin{figure}[H]
\centering
\includegraphics[width=0.7\columnwidth]{trièdre.pdf}
\caption{Définition de la polarisation d'une OPPH.}
\end{figure}

La \imp{polarisation} de l'onde désigne la courbe décrite par la pointe du vecteur champ électrique $\vv{E}$ dans un plan d'onde donné. On distingue trois types de polarisations : rectiligne, circulaire et elliptique.
% Par exemple, si l'extrémité du vecteur champ électrique décrit un cercle en projection sur le plan d'onde, on parle de polarisation circulaire. On ne se soucie pas du champ magnétique ici puisqu'en connaissant $\vv{k}$ et $\vv{E}$ on peut en déduire $\vv{B}$ (trièdre direct).

% \begin{danger}
% On veillera à ne pas confondre cette direction de polarisation avec la direction de propagation de l'onde (selon $\vv{k}$). L'onde électromagnétique plane progressive harmonique (OPPH) est une onde transverse: la direction de polarisation est orthogonale à la direction de propagation de l'onde.
% \end{danger}
}
\subsection{État de polarisation}

\Enonce{
On paramétrise le champ électrique de l'OPPH sous la forme
\begin{equation}
\label{eq:paramétrisation_E}
\vv{E}=E_{0x}\cos(\omega t -kz)\vv{e_x}+E_{0y}\cos(\omega t -kz+\varphi)\vv{e_y}.
\end{equation}
où la phase  constante $\varphi$ caractérise l'état de polarisation.
%. Selon les valeurs de $\varphi$, il existe plusieurs cas.

\parag{Polarisation rectiligne}
 \fbox{$\varphi=0$}. On peut écrire $E_{0x}=E_{0}\cos\alpha$ et $E_{0y}=E_{0}\sin\alpha$ et on remarque que
% \begin{equation}
$ E_y(t)=E_x(t)\tan\alpha$
 % \end{equation}
 qui est l'équation d'une droite de pente $\tan\alpha$ dans le plan d'onde. Il s'agit d'une polarisation rectiligne de vecteur polarisation $\Vec{\veps} = \cos\alpha\, \ex + \sin\alpha\, \ey$.

\parag{Polarisation circulaire} \fbox{$\varphi=\frac{\pi}{2}$ et $E_{0x} = E_{0y} = E_0$}. Alors
% \begin{equation}
$\vv{E} =E_{0}\cro{\cos(\omega t -kz)\vv{e_x}-\sin(\omega t -kz)\vv{e_y}}$
% \end{equation}
et donc d'après $\cos^2+\sin^2=1$ on observe que
% \begin{equation}
${E_x}^2+{E_y}^2=E_0^2$
% \end{equation}
qui est l'équation cartésienne d'un cercle dans le plan d'onde. La polarisation est circulaire.

\parag{Polarisation elliptique}
\fbox{$0<\varphi<\pi$ et $E_{0x}$ et $E_{0y}$ quelconques}. Le vecteur champ électrique décrit une ellipse dans le plan d'onde. Lorsque $\varphi=\pi/2$ on a
% \begin{equation}
% \begin{split}
% \vv{E}&=E_{0x}\cos(\omega t -kz)\vv{e_x}+E_{0y}\cos(\omega t -kz+\frac{\pi}{2})\vv{e_y}\\
% &=E_{0x}\cos(\omega t -kz)\vv{e_x}-E_{0y}\sin(\omega t -kz)\vv{e_y}
% \end{split}
% \end{equation}
% et donc d'après $\cos^2+\sin^2=1$ on observe que
% \begin{equation}
$\left(\frac{E_x}{E_{0x}}\right)^2+\left(\frac{E_y}{E_{0y}}\right)^2=1$
% \end{equation}
et donc l'ellipse est de demi-grand axe $E_{0x}$ suivant $(Ox)$ et de demi-petit axe $E_{0y}$ suivant $(Oy)$. Dans le cas général, les axes de l'ellipse ne sont pas confondus avec les axes du repère.

% Le champ vibre en parcourant au cours du temps cette ellipse. On parle de polarisation elliptique.
\begin{figure}[h!]
  \centering
  \includegraphics[width=0.95\columnwidth]{schéma_polarisation.pdf}
  \caption{Récapitulatif des différents types de polarisation avec la convention de l'Éq.~\eqref{eq:paramétrisation_E}. En changeant $\varphi$ en $-\varphi$, les polarisations droites deviennent gauches. Les axes de l'ellipse sont indiqués en vert (traits pointillés).}
\end{figure}

\parag{Polarisation droite ou gauche ?}
Pour déterminer le sens de parcours de l'ellipse, il faut se placer à deux instants judicieux: $t=0$ et $t=T/4$. On garde le cas simple $\varphi=\pi/2$ mais le raisonnement est identique si $\varphi$ est quelconque. On se place dans le plan d'onde $z=0$ sans perte de généralité.
À $t=0$, $\vv{E}=E_{0x}\cos(0)\vv{e_x}+E_{0y}\cos(\frac{\pi}{2})\vv{e_y}=E_{0x}\vv{e_x}$. Après un quart de période, $\omega T/4=\pi/2$ donc $\vv{E}=E_{0x}\cos(\frac{\pi}{2})\vv{e_x}+E_{0y}\cos(\pi)\vv{e_y}=-E_{0y}\vv{e_y}$.
On en déduit donc le sens de parcours de l'ellipse: ici une elliptique droite. Le sens est défini dans le quadrant supérieur droit $E_x>0,E_y>0$.

% $E_x^2+E_y^2=E_0^2$
% qui est l'équation cartésienne d'un cercle de rayon $E_0$. On parle de polarisation circulaire.


\parag{Lumière non polarisée}
Il s'agit du cas où la phase $\varphi(t)$ est une fonction aléatoire du temps $t$. Aucune direction du champ électrique n'est privilégiée dans le plan $z=\cte$.% on écrit alors
% \begin{equation}
% \vv{E}=E_{0x}\cos(\omega t -kz)\vv{e_x}+E_{0y}\cos\left(\omega t -kz+\varphi(t)\right)\vv{e_y}.
% \end{equation}
  }


\section{Action d'un polariseur}


\subsection{Description d'un polariseur}

\Enonce{
\begin{figure}[H]
  \centering
  \includegraphics[width=\columnwidth]{polariseur.pdf}
  \caption{Principe d'un polariseur: il laisse seulement passer la composante du champ électrique dans la direction perpendiculaire aux chaînes de polymères (ici représentée par une grille). Ici la lumière est initialement non polarisée et ressort avec une polarisation rectiligne (vibration sinusoïdale du champ électrique selon une même direction fixe dans un plan d'onde).
  % mage tirée de Wikipédia.
  }
\end{figure}

Un polariseur est un élément optique formé de chaînes de polymères conducteurs étirés dans une direction privilégiée. Lorsqu'un champ électrique arrive sur le polariseur, la composante selon la direction privilégiée est absorbée car elle met en vibration les chaînes du polymère. En revanche dans la composante orthogonale aux chaînes de polymères, qui définit l'\imp{axe du polariseur}, est intégralement transmise : la lumière en sortie est alors polarisée rectilignement dans la direction  de l'axe du polariseur, de vecteur unitaire $\Vec{e_P}$. L'action d'un polariseur $P$ sur un champ électrique $\E$ est donc  une projection
\begin{equation}\label{eq:polar}
  \E \underset{(P)}\mapsto  \left(\E \ps \Vec{e_P}\right) \Vec{e_P}.
\end{equation}
%On dit que le polariseur est un milieu dichroïque: c'est un milieu anisotrope qui absorbe la lumière selon une direction privilégiée.


On pourra voir une animation ici : \url{https://www.f-legrand.fr/scidoc/simul/elecmag/ondeEMpolariseur.html}.
%, avec en rouge le champ électrique, en bleu le champ magnétique et le polariseur au centre avec son axe en noir.
}

\begin{quest}
  \todo {Montrer qu'un polariseur transforme une polarisation quelconque en entrée (non polarisée, elliptique, circulaire, rectiligne) en une polarisation rectiligne en sortie.}
  \sol{
  Le champ électrique en sortie du polariseur s’écrit 
  $$
  \Vec{E}_{\rm sortie} = \left(\Vec{E}_{\rm entrée} \ps \Vec{e_P}\right) \Vec{e_P} = E_{\rm sortie}(z,t) \Vec{e_P}.
  $$
  Le champ électrique en sortie est porté par un vecteur constant $\Vec{e_P}$ : il est donc \fbox{polarisé rectilignement} suivant $\Vec{e}_p$.
  }
\end{quest}


\subsection{Axe d'un polariseur et angle de Brewster}

\Enonce{
Nous allons montrer comment il est possible d'identifier de façon absolue l'axe d'un polariseur en utilisant la réflexion vitreuse.

En calculant les coefficients de réflexion $r$ et de transmission $t$ à la surface d'un dioptre, on peu montrer que le coefficient de réflexion de l'onde polarisée dans le plan d'incidence $r_{\parallel}$ s'annule pour un angle d'incidence $i_B$ appelé \imp{angle de Brewster} et noté $i_{\rm{B}}$, qui vérifie
% \begin{equation}
\eqbox[brewster]{
  \tan(i_{\rm{B}})=\frac{n_2}{n_1}
  }
% \end{equation}
où $n_2$ est l'indice du milieu transmis et $n_1$ l'indice du milieu incident. Le faisceau réfléchi est alors  polarisé rectilignement dans la direction orthogonale au plan d'incidence.
Cet effet peut être utilisé respectivement pour  mesurer un indice optique ou déterminer l'axe d'un polariseur.
}

\begin{enumerate}
  \item Lorsque la lumière est polarisée dans le \fbox{plan d'incidence}, orthogonal au plan du dioptre, l'intensité lumineuse varie peu. Le coefficient de réflexion dépend peu de l'ange d'incidence. 
  \item Lorsque la lumière est polarisée dans le \fbox{plan du dioptre}, orthogonal au plan d'incidence, l'intensité lumineuse varie fortement avec l'angle d'incidence. À l'angle de Brewster, le coefficient de réflexion s'annule et l'intensité lumineuse réfléchie est nulle sur l'écran.
\end{enumerate}

  }
  \QR {Déterminer l'angle de Brewster en observant l'extinction du faisceau. Préciser la polarisation de la lumière éteinte. En déduire l'indice du verre et commenter sur les incertitudes.}{

    \eqbox{

\subsection{Loi de Malus}

\Enonce{
% La \imp{loi de Malus} donne l'intensité lumineuse $I(\theta)$ en sortie d'un polariseur en fonction de l'angle $\theta$ que fait le champ électrique en entrée avec l'axe du polariseur.
 On prépare une lumière dans une polarisation rectiligne suivant $\ex$ en passant dans un premier polariseur $P$. On place derrière un second polariseur, appelés analyseur $A$, dont l'axe est incliné d'un angle $\theta$.% (\Fig{malus}).
}

 \begin{quest}
  \QR {Schématiser le montage. Représenter l'action du polariseur et de l'analyseur sur le champ électrique incident par une formule similaire à \Eq{polar}. En déduire que l'intensité lumineuse en sortie de l'analyseur vérifie la \imp{loi de Malus}
  % \begin{equation}\label{eq:malus}
  \eqbox[malus]{
    I = I_0 \cos^2\theta.
    }
  % \end{equation}ç
  }
  {

Le champ électrique est transformé par le polariseur $P$ en une polarisation rectiligne suivant $\ex$ d'amplitude $E_0$. En appliquant la formule de projection \Eq{polar} pour l'analyseur $A$, on trouve que le champ électrique en sortie de l'analyseur vaut
  \begin{equation}
    \E_0 = E_0 \,\ex \underset{(A)}{\mapsto} \E = (E_0 \, \ex \ps \Vec{e_A}) \Vec{e_A} = E_0 \cos \theta  \,\Vec{e_A}
  \end{equation}
  Or l'intensité lumineuse $I$ est reliée à l'amplitude du champ éléctrique par 
  $$
  I = \frac{1}{2} c \epsilon_0 E^2.
  $$
On a donc
  \eqbox{
    I = I_0 \cos^2\theta,
  }
  où $I_0$ est l'intensité associée au champ électrique incident $E_0$. Le principe de l'expérience est représentée sur la figure ci-dessus.
    \begin{fig}[malus]
    \includegraphics[width=\columnwidth]{malus-law.pdf}
    % \capt{Vérification de la loi de Malus.}
  \end{fig}
  }

  Voir le schéma ci-dessous.
}
\begin{danger}
 Toujours placer un filtre (anti-calorique ou interférentiel) après une source de lumière blanche pour éviter d'endommager les éléments optiques !
\end{danger}
\QR {
Vérifier quantitativement en sortie de l'analyseur la loi de Malus \Eq{malus} en effectuant une modélisation affine sur \texttt{Python}. On utilisera une photodiode comme capteur lumineux. Expliquer pourquoi il est important que le capteur soit linéaire.}
{
  La photodiode renvoie en sortie une tension $U$ (en V). La photodiode est un capteur \fbox{linéaire}, comme nous l'avons vu dans le \refTP{5}, i.e. que la tension $U$ à ses bornes est proportionnelle à l’intensité lumineuse $I$ et donc à $\cos^2\theta$ d’après la loi de Malus. Pour vérifier la loi de Malus, on peut effectuer la régression affine 
  \eqbox{
  U = a \cos^2\theta + b.
  }
  La valeur de la pente $a$ n'est pas importante : elle dépend des détails du montage. On s'attend à ce que l'ordonnée à l'origine $b$ soit nulle, ce qui correspond à une extinction totale de la lumière lorsque $\theta = \pi/2$. En réalité, on trouve une valeur de $b$ légèrement positive, en raison de la lumière parasite dans la pièce. Pour valider la loi de Malus, il faut quantifier la qualité de la régression à l'aide du khi-deux réduit. On rappelle qu'on doit obtenir 
  \eqbox{
  \chi^2_{\rm{red}} \sim 1
  }
  pour 
  valider la régression. 

  \begin{enumerate}
    \item \udl{Mesures} : la tension $U$ est mesurée au voltmètre pour différentes valeurs de l'angle $\theta$ de l'analyseur. On peut faire varier $\theta$ de $0$ à $90^\circ$ par pas de $10^\circ$ par exemple.
    \item \udl{Incertitude sur la tension} : on peut identifier deux contributions :
      \begin{itemize}
        \item \udl{Fluctuations de l'affichage} : on trouve typiquement  $2\%$ de la valeur moyenne affichée, soit
        $$
      \incert{U}_{\rm{fluct}} = \num{0.02} \cdot U.
        $$
        \item \udl{Précision de l'appareil} : la notice constructeur indique $1\% + 2{\rm d}$, soit 
        $$
        \incert{U}_{\rm{app}} = \num{0.01} \cdot U + \SI{2}{mV}.
        $$
      \end{itemize}L'incertitude-type totale 
      s'obtient par composition de ces deux sources d'incertitudes indépendantes, soit
      \eqbox{
\incert{U} = \sqrt{\incert{U}_{\rm{fluct}}^2 + \incert{U}_{\rm{app}}^2}.
      } 
      \item \udl{Incertitude sur l'angle} : l'incertitude de lecture sur l'angle $\theta$ est de l'ordre d'une demi-graduation, soit $\SI{0.5}{^\circ}$. En réalité, l'incertitude est plus grande, car l'aiguille du polariseur n'est pas forcément bien droite ni rigide, et le repérage de l'extinction à l'œil lorsque polariseur et analyseur sont croisés est moins précis que cela. On peut proposer,
      \eqbox{
\incert{\theta} = \SI{2}{^\circ}.
      }
  \end{enumerate}
}

\section{Action des lames à retard}


\subsection{Description d'une lame à retard}

\Enonce{
Une lame à retard est une lame fait d'un matériau dit biréfringent, dans lequel les ondes polarisées dans des directions orthogonales bien précises se propagent à des vitesses différentes. Si ces directions de polarisation sont $\ex$ et $\ey$, les ondes associées voient des indices optiques différents $n_x$ et $n_y$. Les axes $\ex$ et $\ex$ sont appelés \imp{axes neutres} de la lame à retard. Supposons que le champ électrique dans la lame s’écrive, avec $k_0 = \frac{2\pi}{\lambda_0}$ le vecteur d’onde dans le vide,
\begin{equation}
\begin{cases}
E_{e,x}=E_{0x}\cos(\omega t- n_x k_{0}z), \\
E_{e,y}=E_{0y}\cos(\omega t- n_y k_{0}z+\varphi_{\rm{e}}),
\end{cases}
\end{equation}
où $z=0$ correspond à l’entrée de la lame.
En sortie d’une lame d’épaisseur $e$, le champ vaut donc
\begin{equation}
\begin{cases}
E_{s,x}=E_{0x}\cos(\omega t -n_x k_{0}e) \\
E_{s,y}=E_{0y}\cos(\omega t- n_y k_{0}e+\varphi_{\rm{e}}) = E_{0y}\cos(\omega t-n_x k_{0}e+\vphi_s) .
\end{cases}
\end{equation}
où le déphasage en sortie vaut $\vphi_s = (n_y-n_x) k_{0}e+\varphi_{\rm{e}}$.
La variation de déphasage entre les composantes $E_{y}$ et $E_{x}$ entre l’entrée et la sortie de la lame vaut donc
\begin{equation}
\Delta \vphi = \varphi_{\rm{s}} - \varphi_{\rm{e}} = \frac{2\pi}{\lambda_0}(n_y-n_x)e = 2\pi\frac{\delta}{\lambda_0},
\end{equation}
avec $\delta=(n_y-n_x)e$ la différence de marche. Il est donc possible de modifier l'état de polarisation en jouant sur la valeur de $e$. Il existe principalement deux types de lames à retard :
\begin{liste}
  \item Lame \imp{quart d'onde} : différence de marche introduite $\delta=\pm \frac{\lambda_0}{4}$ d'où une variation de déphasage de $\Delta \varphi=\frac{2\pi}{\lambda_0}\delta=\pm \frac{\pi}{2}$.
\item Lame \imp{demi onde} = différence de marche introduite $\delta= \pm \frac{\lambda_0}{2}$ d'où une variation de déphasage de $\Delta \varphi= \frac{2\pi}{\lambda_0}\delta= \pm \pi$.
\end{liste}
\begin{remarque}
  Attention, les lames ne sont quart d'onde ou demi onde que pour une longueur d'onde $\lambda_0$ donnée !
  Pour une animation du rôle des lames quart d'onde et demi onde, voir par exemple \url{http://ressources.univ-lemans.fr/AccesLibre/UM/Pedago/physique/02/optiondu/ondepola.html}
\end{remarque}
  
}

\sol{
\begin{enumerate}
  \item \udl{Rectiligne} : supposons le champ en entrée est polarisé avec un angle $\alpha$ par rapport à l’axe $x$, soit $\E_{\rm e} = E_0 \cos(\omega t) \Vec{e}_{\alpha}$,
  $$
  \left\{

\end{array}
  \right.$$
  Le champ en sortie vaut alors, à une phase globale près,
  $$
  \left\{
\begin{array}{ll}
  E_{s,x} = E_0 \cos \alpha \cos(\omega t) = E_{e,x} \\ 
  E_{s,y} = E_0 \sin \alpha \cos(\omega t \pm \pi) = - E_0 \sin \alpha \cos(\omega t) = - E_{e,y}
\end{array}
  \right.$$
  Il s’agit d’une polarisation rectiligne symétrique d’angle $-\alpha$, i.e. symétrique de la polarisation initiale par rapport aux axes neutres de la lame.
\begin{fig}[label]
  \centering
  \includegraphics[scale=1.1]{rectiligne_demionde.pdf}
  %\capt{}
\end{fig}

  \item \udl{Circulaire} : supposons que la polarisation est circulaire gauche. Le champ en entrée est de la forme 
  $$
  \left\{
\begin{array}{ll}
  E_{e,x} = E_0 \cos(\omega t)\\ 
  E_{e,y} = E_0 \sin(\omega t)
\end{array}
  \right.$$
  En sortie, on obtient
  $$
  \left\{
\begin{array}{ll}
  E_{s,x} = E_0 \cos(\omega t) = E_{e,x} \\ 
  E_{s,y} = E_0 \sin(\omega t \pm \pi) = - E_0 \sin(\omega t) = - E_{e,y}
\end{array}
  \right.$$
  Il s’agit d’une polarisation circulaire droite. Le sens de parcours de la polarisation est donc opposé.
  \begin{fig}[label]
    \centering
    \includegraphics[scale=1.1]{circulaire_demionde.pdf}
    %\capt{}
  \end{fig}
  \item \udl{Elliptique} : la polarisation est également elliptique en sortie. Le sens de parcours de la polarisation est inversé, tout comme pour la polarisation circulaire.
\end{enumerate}
}
\end{quest}

\subsection{Action d'une lame quart d'onde}

\Enonce{
On étudie l'action d'une lame \imp{quart d'onde} sur les états de polarisation. Dans la suite, vous vérifierai toutes vos affirmations expérimentalement.
}


\parag{Polarisation elliptique}


\begin{quest}
\todo {Expliquer comment produire une polarisation elliptique à partir d'une polarisation rectiligne avec une lame quart d'onde. Préciser notamment l'orientation des axes de polarisation par rapport aux axes neutres.}
\sol{
  Supposons le champ en entrée est polarisé avec un angle $\alpha$ par rapport à l’axe $x$, soit
  $$
  \left\{
\begin{array}{ll}
  E_{e,x} = E_0 \cos \alpha \cos(\omega t),\\ 
  E_{e,y} = E_0 \sin \alpha \cos(\omega t).
\end{array}
  \right.$$
  Le champ en sortie vaut alors (à une phase globale près)
  $$
  \left\{
\begin{array}{ll}
  E_{s,x} = E_0 \cos \alpha \cos(\omega t) = E_{e,x} \\ 
  E_{s,y} = E_0 \sin \alpha \cos(\omega t \pm \frac{ \pi}{2}) = \mp E_0 \sin \alpha \sin(\omega t).
\end{array}
  \right.$$
  Il s’agit d’une polarisation elliptique lorsque $\alpha$ est quelconque. Pour produire une polarisation elliptique, il suffit donc de faire passer une \fbox{polarisation rectiligne d'orientation quelconque} à travers une \fbox{lame quart d'onde}.
}
% \todo Réciproquement, expliquer comment produire une polarisation rectiligne à partir d'une polarisation elliptique.

\QR {Proposer un protocole pour déterminer les axes d'une vibration elliptique quelconque avec un polariseur.}{
  \begin{enumerate}
    \item Faire tourner le polariseur jusqu'à trouver un minimum de la lumière en sortie. L'axe du polariseur coïncide alors avec un des axes de l'ellipse.
    \item Faire tourner le polariseur d'environ $90^\circ$ par rapport à l'orientation précédente pour trouver une seconde extinction de la lumière en sortie. L'axe du polariseur coïncide alors avec l'autre axe de l'ellipse.
  \end{enumerate}
}
\end{quest}

\parag{Polarisation circulaire}
\begin{quest}
\todo {Expliquer comment produire une polarisation circulaire à partir d'une polarisation rectiligne. Préciser notamment la direction de polarisation par rapport aux axes neutres de la lame quart d'onde.}
\sol{
  D’après les calcus précédents, le champ en sortie d'une lame quart d'onde est polarisé circulairement si la polarisation rectiligne en entrée vérifie
  \eqbox{
\alpha = \pm \frac{\pi}{4},
  }
  donc si l’onde en entrée est polarisée à $\SI{45}{^\circ}$ des axes neutres de la lame. L’onde est circulaire droite si l’angle est de $+\SI{45}{^\circ}$ (l’axe rapide est alors à droite de l’axe lent) et gauche si l’angle est de $-\SI{45}{^\circ}$ (l’axe rapide est alors à gauche de l’axe lent).

  \begin{fig}[label]
    \centering
    \includegraphics[scale=1.1]{rectiligne_quartonde.pdf}
    %\capt{}
  \end{fig}
  }
\todo {Réciproquement, expliquer comment passer d'une polarisation circulaire à une polarisation rectiligne.}
\sol{
On part d’une polarisation circulaire (supposée gauche) en entrée,
$$
\left\{
  \begin{array}{ll}
    E_{e,x} = E_0 \cos(\omega t),\\ 
    E_{e,y} = E_0 \sin(\omega t).
  \end{array}
\right.
$$
Le champ en sortie de la lame vaut 
$$
\left\{
  \begin{array}{ll}
    E_{s,x} = E_0  \cos(\omega t) \\ 
    E_{s,y} =  E_0  \sin(\omega t + \frac{ \pi}{2}) = E_0 \cos(\omega t).
  \end{array}
  \right.
$$
On obtient une polarisation rectiligne d’axe $\ex + \ey$, qui en particulier fait un angle de $\SI{45}{^\circ}$ avec les axes neutres de la lame. Si la polarisation circulaire en entrée était droite, on obtiendrait une polarisation rectiligne d’axe $\ex - \ey$.
\begin{fig}[label]
  \centering
  \includegraphics[scale=1.1]{circulaire_quartonde.pdf}
  %\capt{}
\end{fig}
}
\end{quest}
On cherche à déduire des résultats précédents un protocole d'analyse d'une lumière polarisée circulairement.
\begin{quest}
\QR {Que se passe-t-il lorsqu'on place un analyseur (polariseur en sortie) après une onde polarisée circulairement?}
{
Le champ en sortie d’un polariseur dont l’axe faut un angle $\alpha$ avec l’axe $\ex$ vaut 
% $$
\begin{align*}
  \E_s =(\E_e \cdot \Vec{e_P}) \Vec{e_P} & = (E_x \cos \alpha  + E_y \sin \alpha) \Vec{e_P} \\ 
  & = E_0 (\cos(\omega t) \cos \alpha + \sin(\omega t) \sin(\alpha)) \Vec{e}_P,
\end{align*}
% $$
soit 
% $$
\eqbox{
  \E_s = E_0 \cos(\omega t - \alpha) \Vec{e_P}.
  }
% $$
Le champ en sortie est bien sûr \fbox{polarisé rectilignement} suivant l’axe du polariseur, mais l’effet du polariseur sur la norme du champ se traduit par un simple déphasage $\alpha$. En particulier, l’intensité lumineuse en sortie ne dépend pas de la direction du polariseur !
}
\QR {Expliquer pourquoi le seul usage d'un analyseur ne permet pas d'affirmer avec certitude que l'onde initiale est polarisée circulairement.}
{
  On obtiendrait le même résultat avec une lumière \fbox{non polarisée} en entrée. On ne peut donc pas distinguer une lumière non polarisée d’une lumière polarisée circulairement avec un analyseur seul.
}
\QR {Justifier que l'utilisation d'une lame quart d'onde permet de lever l'ambiguïté précédente, et réaliser le montage.}
{
  En plaçant une \fbox{lame quart d’onde en entrée de l’analyseur}, on peut transformer une polarisation circulaire en une polarisation rectiligne, ce qui n’est pas possible avec une lumière non polarisée. On peut donc affirmer avec certitude que la lumière est polarisée circulairement si on peut l’éteindre en sortie de l’analyseur en présence d’une lame quart d’onde.
}

\end{quest}

\section{Mesure d'un pouvoir rotatoire\bonus}

\subsection{Description du phénomène}

\Enonce{
Il existe aussi des matériaux qui présentent une biréfringence circulaire : le matériau fait tourner l'axe de polarisation rectiligne de la lumière. L'effet est plus faible que pour la biréfringence linéaire, qui intervient dans les lames quart d'onde et demi-onde.

La biréfringence circulaire existe non seulement dans des matériaux cristallins mais aussi dans des solutions dites optiquement actives, lorsque les espèces dissoutes sont des molécules chirales. Dans ce cas, l'angle $\alpha$ de rotation de l'axe de polarisation rectiligne de la lumière vérifie la \imp{loi de Biot}
% \begin{equation}
\eqbox[biot]{
  \alpha = [\alpha] \ell c,
}
% \end{equation}
où $[\alpha]$ est le pouvoir rotatoire spécifique, $\ell$ la longueur de solution traversée et $c$ la concentration de la solution.
Le pouvoir rotatoire $\alpha$ peut être positive ou négative : deux molécules énantiomères ont des pouvoirs rotatoires opposés.
}

\subsection{Mise en évidence expérimentale}

\Enonce{
Nous illustrerons cette loi sur une solution de glucose de concentration massique $c=\SI{300}{g.L^{-1}}$ placée dans une cuve de longueur $\ell = \SI{20}{cm}$. L'énantiomère naturel du glucose est dextrogyre, ce qui signifie qu'il fait tourner l'axe de polarisation vers la droite ($[\alpha]>0$) ; il a un pouvoir rotatoire spécifique $[\alpha] = \SI{52}{\degres.dm^{-1}.g^{-1}.cm^3}$.
}

\begin{quest}
  \QR Placer un polariseur et un analyseur croisés de sorte à éteindre complètement la lumière en sortie. Placer une cuve contenant la solution de glucose entre les deux polariseurs. Que se passe-t-il ? Expliquer.
  {
    On voit réapparaître de la lumière. La solution de glucose fait tourner l'axe de polarisation rectiligne de la lumière d’un angle $\alpha$ par rapport à l’analyseur croisé. D’après la loi de Malus, l’intensité en sortie est de la forme $I = I_0 \sin^2\alpha$, qui est non nulle. 
  }
  \QR {En utilisant une photodiode pour repérer précisément l'extinction, déterminer la valeur expérimentale de $\alpha$ et comparer à la valeur attendue.}{
    On retrouve l'extinction en tournant l'analyseur d'un angle 
    \eqbox{
      \alpha = \SI{31}{\degres} \pm \SI{2}{\degres}.
    }
    La valeur attendue d'après la loi de Biot est de
    \eqbox{
      \alpha_{\rm{att}} = [\alpha] \ell c = \SI{31.2}{\degres}.
    }
    L'accord est bon.
  }
\end{quest}
