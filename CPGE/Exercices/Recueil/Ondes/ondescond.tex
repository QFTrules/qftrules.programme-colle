On cherche une solution sous la forme d'une OPPH généralisée se propageant dans la direction $(Oz)$, d'expression 
  \begin{equation}
    \E*(z,t) = \E*_0 \,\e^{i(\omega t - \udl{k} z)}.
  \end{equation}

  \end{quest}

  On considère pour commencer que l'onde est polarisée rectilignement selon l'axe $(Ox)$, soit $\E*_0 = E_0 \,\ex$.
  \begin{quest}
    \item	En déduire le champ électrique réel $\E(z,t) = \re\,\Vec{\udl{E}}(z,t)$. Comment s'appelle ce type d'onde ?
    \item Déterminer l'expression du champ magnétique complexe $\B*(z,t)$ puis du champ réel $\B(z,t)$.
    \item En déduire l'expression du vecteur de Poynting $\Poy(z,t)$ et de sa moyenne temporelle. Interpréter.
    % \item Quelle distance terrestre totale peut parcourir l'onde radio après réflexion sur la ionosphère ?
  \end{quest}

  \begin{quest}
    \item	Reprendre les questions précédentes pour cette polarisation.
  \end{quest}
\end{exo}

\begin{exo}[PC][3][colle]{Observation d'un pulsar}
  On donne le retard de propagation $\d \tau$ entre deux ondes de fréquences proches $\nu$ et $\nu + \d \nu$ issues d'un pulsar sous la forme, 
  \begin{equation}
    \dd{\nu}{\tau}=-\SI{5}{Mhz.s^{-1}}.
  \end{equation}
On modélise l'espace interstellaire par un plasma dilué homogène de densité $n=\SI{2e6}{m^{-3}}$.
\begin{quest}
  \item	Quelles sont les fréquences du pulsar que l'on peut capter?
  \item Estimer la distance Terre-Pulsar.
  \item A chaque période de rotation du pulsar, sa période diminue d'une variation relative $\frac{\Delta T}{T}=-10^{-15}$. En déduire la masse du pulsar.
\end{quest}
\textsc{Données} :
\begin{tabular}{ll}
  moment d'inertie d'une boule : & $J=\textstyle\frac{2}{5} m R^2$, \\ 
  puissance rayonnée par le pulsar : & 5 fois celle du Soleil.
\end{tabular}

\end{exo}

\begin{exo}[PSI][1][TD]{Réflexion à la surface d'un plasma}
  \begin{minipage}[c][2.5cm][t]{0.6\textwidth}
  Le demi-espace $z<0$ est vide, tandis qu'un plasma de pulsation caractéristique $\omega_{0}$ occupe le demi-espace $z>0$. Une onde progressive monochromatique se propageant selon
  les $z$ croissants arrive en incidence normale en $z=0$.
  \end{minipage}
  \hfill
  \begin{minipage}[c][3.5cm][t]{0.3\textwidth}
  \begin{center}
   \includegraphics[scale=1]{plasma.pdf}
  \end{center}
  \end{minipage}
  
   \begin{quest}
    \item Justifier qualitativement que l'on puisse supposer l'onde incidente polarisée rectilignement selon $\ex$ sans perte de généralité.
  
    \sol{
    Le problème est invariant par rotation autour de $Oz$ et l'incidence est normale. Il n'y a donc pas de plan d'incidence privilégié.}
  
    \item L'onde incidente (en $z<0$) est de la forme
    \begin{equation}
    \Vec{E}_{i} = E_{0} e^{i(\omega t - k z)} \,\ex
    \end{equation}
    Rappeler l'expression de $k$ et en déduire celle du champ magnétique incident $\Vec{B}_{i}$.
  
    \sol{
    La relation de dispersion dans le vide est
  
    \begin{equation*}
     k^{2}c^{2}=\omega^{2}
    \end{equation*}
  
    donc $k = \pm \dfrac{\omega}{c}$. Comme la propagation de l'onde incidente se fait dans le sens des $z$ croissants, $\Vec{k} = k\,\Vec{e}_{z}$ et
  
    \begin{equation*}
     k = \dfrac{\omega}{c}
    \end{equation*}
  
    La relation de structure donne
  
    \begin{equation*}
     \Vec{B}_{i} = \frac{\Vec{k} \wedge \Vec{E}_{i}}{\omega} = \frac{E_{0}}{c} e^{i(\omega t - k z)}\,\ey
    \end{equation*}
    }
  
    \item On suppose qu'ne onde réfléchie se propage également dans le demi-espace $z<0$,
    \begin{equation}
   \Vec{E}_{r} = \rho E_{0} e^{i(\omega t + k z)} \,\ex,
    \end{equation}
    avec $\rho$ le coefficient de réflexion en amplitude.
  
    Donner l'expression du champ magnétique réfléchi $\Vec{B}_{r}$.
  
    \sol{
    La propagation se fait selon les $z$ décroissants, donc $\Vec{k}_{r} = - \dfrac{\omega}{c}\,\Vec{e}_{z}$. Donc
  
    \begin{equation}
     \Vec{B}_{r} = - \frac{\Vec{k} \wedge \Vec{E}_{r}}{\omega} = - \rho \frac{E_{0}}{c} e^{i(\omega t + k z)}\,\ey
    \end{equation}
    }
  
    \item Rappeler la relation de dispersion du plasma. Rappeler ou retrouver l'expression de $\omega_0$ à partir du modèle du plasma. Quelle est la condition de propagation dans un ce milieu ? Préciser l'expression des champs électrique et magnétique en $z>0$ en introduisant
    un coefficient de transmission en amplitude $\tau$.
  
    \sol{
    $$
    \omega_0 = \sqrt{\dfrac{ne^2}{m \varepsilon_0}}.
    $$
  La relation de dispersion s'obtient à partir du calcul du vecteur densité de courant (principe fondamental de la dynamique appliqué à un ion du plasma) puis des
    équations de Maxwell (Faraday et Ampère). Elle est de la forme
  
    \begin{equation}
     k_{t}^{2} = \frac{\omega^{2} -\omega^{2}_{0}}{c^{2}}
    \end{equation}
    Ainsi l'onde est evanescente si $\omega < \omega_{0}$, le plasma agit comme un filtre passe-haut.
  
    Comme précedemment
  
    \begin{equation}
     \begin{cases}
      \Vec{E}_{t} = \tau E_{0} e^{i(\omega t - k_{t} z)} \,\ex \\
      \Vec{B}_{t} = \tau \frac{E_{0}}{\omega} k_{t} e^{i(\omega t - kt z)}\,\ey
     \end{cases}
    \end{equation}
    }
  
    \item En admettant qu'il n'existe pas de densité surfacique de courant en $z=0$, donner un système d'équations reliant $\rho$ et $\tau$.
  
    \sol{
  Conditions aux limites :
    \begin{itemize}
     \item[\scriptsize\textbullet] $E_{x}$ est continu en $z=0$, donc
     \begin{equation}
      1 + \rho = \tau \label{eq:plasma1}
     \end{equation}
     \item[\scriptsize\textbullet] $B_{y}$ est continu en $z=0$ car $\Vec{j}(0) = \Vec{0}$. Donc
     \begin{equation}
      \frac{1}{c} - \frac{\rho}{c} = \tau \frac{k_{t}}{\omega} \label{eq:plasma2}
     \end{equation}
    \end{itemize}
    }
  
    \item En déduire $\rho$ et $\tau$. Représenter les coefficients de réflexion et de transmission en puissance en fonction de $x=\omega/\omega_0$.
    \sol{
    Coefficient de réflexion :
    \begin{eqnarray}
     \eqref{eq:plasma1} + c\,\eqref{eq:plasma2} \qquad \Longrightarrow \qquad \tau = \frac{2}{\frac{ck_{t}}{\omega}+1} \\
     \mathrm{d'où} \hspace{4cm} \rho = \frac{1-\frac{k_{t}c}{\omega}}{1+\frac{k_{t}c}{\omega}}
    \end{eqnarray}
  
    Si le milieu est non-transparent, $k_{t}$ est imaginaire pur. Donc $\left|\rho\right|^{2} = 1$.
    }
   \end{quest}
  \end{exo}

\begin{exo}[PC][2][TD]{Propagation d'une onde dans le plasma interstellaire}
Le plasma interstellaire est constitué d'électrons de masse $m$, de charge électrique $-e$, de densité particulaire $n$, et d'ions de charge électrique $+e$, et de même densité particulaire $n$. %La densité de charge totale est nulle.
Le mouvement des ions est négligé et celui des électrons, non relativistes, est décrit par le vecteur vitesse $\vv{v}$. Avec ces hypothèses, on cherche des solutions aux équations de Maxwell  sous la forme d'OPPH de vecteur d'onde $\vv{k}$, dont le champ électrique est noté
\begin{equation}
 \underline{\vv{E}}(r,t)=\vv{E_0}\,\e^{i(\omega t-\vv{k}\cdot\vv{r})}.
 \end{equation}
\begin{quest}
\item Montrer que le champ magnétique de l'onde est aussi décrit par une onde plane de mêmes pulsation et vecteur d'onde. Déterminer la nature du trièdre $(\vv{E},\vv{B},\vv{k})$ de l'onde.
\sol{
L'équation de Maxwell-Faraday en complexe conduit à la relation de structure
$$
\Vec{\udl{B}} = \frac{\Vec{k}}{\omega} \pv \Vec{\udl{E}}  = \frac{1}{\omega} (\Vec{k}\pv \Vec{\udl{E_0}} )\e^{i(\omega t-\vv{k}\cdot\vv{r})}.
$$
Le champ magnétique est donc décrit par une OPPH de même $\omega$ et $k$. Le trièdre $(\vv{E},\vv{B},\vv{k})$  est direct d'après les propriétés du produit vectoriel.
}
\item Déterminer l'amplitude $\Vec{j_0}$ du vecteur densité volumique de courant exprimé en notation complexe comme $\udl{\j}(\vv{r},t)=\Vec{j_0}\,\e^{i(\omega t-\vv{k}\cdot\vv{r})}$ en fonction de $\vv{E_0},\omega,c,k$ et $\varepsilon_0$. En étudiant le mouvement des électrons, exprimer la constante $\alpha$ telle que $\udl{\j}=-i\frac{\alpha}{\omega}\vv{E}$.
\sol{
Pour la première formule, il est naturel d'utiliser l'équation de Maxwell-Ampère, soit
$$
\rot \udl{\B} = \mu_0 \udl{\j} + \frac{1}{c^2} \dr{\udl{\E}}{t}, \Rar -i \Vec{k} \pv \Vec{\udl{B_0}} = \mu_0 \Vec{j_0}  + i \frac{\omega}{c^2} \Vec{\udl{E_0}}
$$
donc
$$
\Vec{j_0} = -\frac{i}{\mu_0 \omega} \Vec{k} \pv (\Vec{k}\pv \Vec{\udl{E_0}}) - i \frac{\omega}{\mu_0 c^2} \Vec{\udl{E_0}} = i\left(\frac{c^2 \epz k^2}{\omega} - \epz \omega\right) \Vec{\udl{E_0}}.
$$
Pour la deuxième relation, on utilise la relation constitutive (démonstration dans le cours),
$$
\dr{\udl{\j}}{t} = i\omega \udl{\j} = \frac{ne^2}{m} \udl{\E} \Rar \udl{\j} = -i\frac{ne^2}{m\omega}.
$$
On a donc $\alpha = \frac{ne^2}{m}$.
}
\item En déduire la relation de dispersion $\omega(k)$ liant la pulsation de l'onde et la norme de son vecteur d'onde. En posant $\alpha=\varepsilon_0 c^2 K^2$, calculer les vitesses de phase et de groupe de l'onde en fonction de $k$ et $K$. Quelle est la relation liant ces vitesses ?
\sol{
On a
$$
-i\frac{\alpha}{\omega} = i\frac{c^2 \epz k^2}{\omega} -i \epz \omega \Rar -\alpha = \epz c^2k^2 - \epz \omega^2 \Rar \omega^2 = c^2 k^2 + \frac{\alpha}{\epz} = c^2 k^2 + c^2 K^2,
$$
soit
\eqbox{\omega(k) = c\sqrt{K^2+k^2}.}

On en déduit ensuite les vitesses de phase et de groupe.
 $$
v_\vphi = \frac{\omega}{k} = c\sqrt{1+(\frac{K}{k})^2} >c, \qquad v_g = \dd{\omega}{k} = \frac{ck}{\sqrt{K^2+k^2}} = \frac{c}{\sqrt{1+(\frac{K}{k})^2}} < c.
 $$
 On remarque que $v_\vphi v_g =c^2$. Cette propriété est vérifiée lorsque
 $$
c^2 = v_\vphi v_g = \frac{\omega}{k}\dd{\omega}{k} = \dd{\omega^2}{k^2},
 $$
 i.e. quand $\omega^2 = c^2k^2 + \cte$ où $\cte$ est homogène à une pulsation au carré. Lorsque $\cte > 0$, on peut écrire $\cte = \omega_p^2$ et on obtient la relation de dispersion du plasma. Lorsque $\cte<0$, on peut écrire $\cte = -\omega_0^2$ et alors $k = c \sqrt{\omega^2 + \omega_0^2}$ est toujours réel.
}
\item Deux trains d'ondes de longueurs d'onde $\lambda_1$ et $\lambda_2$ sont émis au même instant par un objet stellaire situé à une distance $L$. En supposant $K^2\lambda_1^2$ et $K^2\lambda_2^2\ll 1$, montrer que ces signaux sont reçus avec un décalage $\Delta t=t_2-t_1$ à déterminer en fonction de $L,K,c,\lambda_1$ et $\lambda_2$.
\sol{
On suppose $K^2\ll k^2$ pour les deux trains d'ondes, donc $v_g = \frac{c}{\sqrt{1+(\frac{K}{k})^2}} \simeq c $ : on est dans le régime faiblement dispersif. Le décalage dans les instants de réception est donné au premier ordre non nul par
$$
\Delta t = t_2-t_1 = \frac{L}{v_{g,2}} - \frac{L}{v_{g,1}} = \frac{L}{c}\left(\sqrt{1+(\frac{K}{k_2})^2} - \sqrt{1+(\frac{K}{k_1})^2}\right) \simeq  \frac{LK^2}{2c} \left(\frac{1}{k_2^2} - \frac{1}{k_1^2}\right)
= \frac{LK^2}{8\pi^2 c}(\lambda_2^2-\lambda_1^2).
$$
}
\end{quest}
\end{exo}

\begin{exo}[PC][3][TD]{Équation de propagation dans un métal}
  \Notions{\item Onde électromagnétique dans un métal \task Relation de dispersion}
  On se propose de déterminer l'équation de propagation générale d'une onde électromagnétique dans un conducteur ohmique, sans approximation, par deux méthodes. On notera $\sigma_0$ la conductivité électrique en régime continu et $\tau$ le temps typique de collision.
  \begin{quest}
    \item Rappeler la loi d'Ohm généralisée en notation complexe et déterminer la relation de dispersion des OPPH généralisées en s'aidant des équations de Maxwell. À l'aide de manipulations algébriques et en passant aux notations réelles, en déduire l'équation de propagation.
    \sol{
      La relation de dispersion s'écrit 
      $$
      \underline{k}^2=\frac{\omega^2}{c^2}\left(1-\frac{i\udl{\sigma}}{\omega\varepsilon_0}\right) = \frac{\omega^2}{c^2}\left(1-i\frac{\sigma_0}{\omega\varepsilon_0(1+i\omega \tau)}\right) \Rar \udl{k}^2+i\omega \tau \udl{k}^2  = \frac{\omega^2}{c^2} + \frac{i\omega^3 \tau}{c^2}  - i\mu_0 \sigma_0 \omega.
      $$
      En multipliant par $\udl{\E}$ et en passant aux notations réelles, on trouve 
      \eqbox{
      \Delta \E + \tau  \dr{}{t} \Delta\E - \frac{1}{c^2} \ddr{\E}{t} - \frac{\tau}{c^2}\drm{\E}{t}{3} - \mu_0 \sigma_0 \dr{\E}{t} = \Vec{0}.
      }
    }
    \item Sans passer par la notation complexe, déterminer la loi constitutive qui lie la densité de courant $\j$ et le champ électrique $\E$ dans un conducteur ohmique en régime variable. À l'aide des équations de Maxwell, en déduire l'équation de propagation.
  \end{quest}
  \end{exo}

\begin{exo}[PC][3][TD]{Effet de peau dans un fil électrique}
  \Notions{\item Effet de peau \task Géométrie d'un conducteur}
  On considère un fil cylindrique de rayon $R$, d'axe $(Oz)$, constitué d'un conducteur ohmique de conductivité $\sigma_0$ en régime continu, et de permittivité $\varepsilon_0$. On étudie le conducteur dans l'ARQS en régime harmonique à la pulsation $\omega$. On cherche une solution complexe pour le champ électrique de la forme $\udl{\E}(r,t) = \udl{E}(r) e^{i\omega t} \ez$ en coordonnées cylindriques $(r,\theta,z)$.
  \begin{quest}
    \item	Déterminer l'équation différentielle vérifiée par la fonction $\udl{E}(r)$. On introduira une distance caractéristique $\delta$ à exprimer. On rappelle l'expression du Laplacien d'une fonction $f(r)$ dans les coordonnées cylindriques,
    \begin{equation}
      \Delta f = \ddd{f}{r} + \frac{1}{r} \dr{f}{r}.
    \end{equation}
     \sol{
     Le Laplacien d'une fonction $f(r)$ dans les coordonnées cylindriques est donné par
     $$
     \Delta f = \frac{1}{r} \dr{}{r}\left(r \dr{f}{r}\right).
     $$
     En injectant l'expression de $\udl{\E}$ dans l'équation de propagation, on trouve
     $$
     \frac{1}{r} \dr{}{r}\left(r \dr{\udl{E}}{r}\right) + i\omega\mu_0\sigma_0 \udl{E} = 0.
     $$
     En posant $\delta = \sqrt{\frac{2}{\omega\mu_0\sigma_0}}$, on trouve l'équation différentielle suivante pour $\udl{E}$ :
     $$
     \frac{1}{r} \dr{}{r}\left(r \dr{\udl{E}}{r}\right) + (1+i)\frac{\udl{E}}{\delta^2} = 0.
     $$
    }
    \item Résoudre cette équation différentielle dans l'hypothèse où $\delta \ll R$. En déduire l'expression du champ réel $E(r)$ en fonction de l'amplitude $E_0$ du champ sur le bord du fil, en $r=R$.
    \item Calculer l'intensité électrique $I$ circulant à travers le câble, et en déduire la résistance linéique $R_\ell$ du fil. Commenter sa dépendance en la pulsation $\omega$.
  \end{quest}
\end{exo}

\begin{exo}[PC]{Effet Faraday}
On considère dans tout le problème un plasma d'ions supposés immobiles et d'électrons de charge $-e$ de masse $m$ et de densité volumique $n_0$.
\begin{quest}
\item \textbf{Pulsation plasma.} On suppose que la densité volumique des électrons n'est pas uniforme et s'écrit $n(\vv{r})=n_0+\delta n$ avec $\delta n\ll n_0$. Un champ électrique $\vv{E}(\vv{r},t)$ règne dans le plasma.
\begin{enumerate}
\item En utilisant l'équation de Maxwell-Gauss, l'équation de conservation de la charge et l'équation du mouvement d'un électron placé dans le champ électrique $\vv{E}$, déterminer l'équation différentielle vérifiée par $\delta n$.
\item En déduire que si $\vv{E}$ est sinusoïdal de pulsation $\omega\neq\omega_{\rm{p}}=\sqrt{\frac{n_0e^2}{m\varepsilon_0}}$, la densité volumique d'électrons reste égale à $n_0$ en régime forcé.
\end{enumerate}
\item \textbf{Pulsation cyclotron.} Montrer que le mouvement d'un électron dans un champ magnétique $\vv{B}_0$ uniforme dans le plan orthogonal à $\vv{B}_0$ est circulaire de pulsation $\omega_{\rm{c}}=\frac{eB_0}{m}$.
\textit{Dans la suite, une onde électromagnétique plane se propage dans un plasma. Dans l'équation du mouvement d'un électron, on tiendra uniquement compte du champ électrique $\vv{E}$ de l'onde et du champ magnétique $\vv{B_0}$ uniforme.}

\begin{exo}[PC][2][TD]{Couche anti-reflet}
À la surface d'un verre d'indice $n$ on dépose une couche d'épaisseur $e$ et d'indice $N$ de telle sorte que l'espace est divisé en 3 domaines: l'air ($x<0$) d'indice 1, la couche $(0<x<e$) d'indice $N$ et le verre $(x>e)$ d'indice $n$. Une OPPH de pulsation $\omega$ polarisée rectilignement sur $\vv{e_z}$ arrive dans l'air sous incidence normale et on cherche les conditions pour qu'il n'y ait pas d'onde réfléchie dans l'air.
 \begin{quest}
\item Donner en notation complexe la forme attendue du champ électrique dans chaque région.
\sol{
On suppose qu'il n'y a pas d'onde réfléchie dans l'air.\\
Dans l'air : champ incident $\udl{\E_0}=\udl{E_{i}} e^{i(\omega t - k_0 x)}\ez$.\\
Dans la couche : superposition des champs transmis et réfléchis : $\udl{\E_1}=\udl{E_{t1}} e^{i(\omega t - k_1 x)}\ez + \udl{E_{r1}} e^{i(\omega t + k_1 x)}\ez$.\\
Dans le verre : onde transmise uniquement : $\udl{\E_2}=\udl{E_{t}} e^{i(\omega t - k_2 (x-e))}\ez$.\\
On a $k_0 = \frac{\omega}{c}$, $k_1 = N \frac{\omega}{c}$ et $k_2 = n \frac{\omega}{c}$.
}
\item En déduire la forme du champ magnétique dans les trois milieux.
\sol{
Relation de structure pour une OPPH : $\Vec{B} = \frac{\Vec{k}}{\omega} \pv \E = \frac{n}{c} \Vec{u} \pv \E$ avec $\Vec{u} = \pm \ex$ suivant la direction de propagation. Attention à bien appliquer cette formule sur chaque OPPH séparément (et pas directement sur $\udl{\E_1}$).
}
\item Établir 4 relations entre les amplitudes du champ électrique dans les différentes régions, puis en déduire que
\begin{equation*}
\frac{(N+1)(N-n)}{(N-1)(N+n)}=\e^{i2Nk_{0}e},
\end{equation*}
où $k_{0}$ est le module du vecteur d'onde dans le vide.
\sol{
On utilise la continuité des champs électrique et magnétique à chaque interface. On repère l'interface air-couche en $x=0$. On obtient alors
$$
\begin{cases}
\udl{E_{i}} = \udl{E_{t1}} + \udl{E_{r1}} \\
\udl{E_{i}} = N(\udl{E_{t1}} - \udl{E_{r1}}) \\
\end{cases}\qquad
\begin{cases}
\udl{E_{t1}} e^{-i N k_0 e)} + \udl{E_{r1}} e^{+i N k_0 e)} = \udl{E_{t}}\\
N(\udl{E_{t1}} e^{-i N k_0 e)} - \udl{E_{r1}} e^{+i N k_0 e)}) = n\udl{E_{t}} \\
\end{cases}
$$
Le premier système donne $\udl{E_{t1}}=\frac{N+1}{2N}\udl{E_{i}}$ et $\udl{E_{r1}}=\frac{N-1}{2N}\udl{E_{i}}$. On en déduit
$$
\begin{cases}
\udl{E_{i}}(\frac{N+1}{2N} e^{-i N k_0 e)} + \frac{N-1}{2N} e^{+i N k_0 e)}) = \udl{E_{t}}\\
\udl{E_{i}}(\frac{N+1}{2} e^{-i N k_0 e)} - \frac{N-1}{2} e^{+i N k_0 e)}) = n\udl{E_{t}} \\
\end{cases}
$$
Or la conservation de l'énergie impose $\udl{E_{i}} = \udl{E_{t}}$, et donc on trouve la relation attendue.
}
\item En déduire les valeurs qu'il faut choisir pour $e$ et $N$. Comment expliquer qu'il subsiste des reflets violet et rouge sur un objectif ainsi traité lorsqu'on l'observe à la lumière du Soleil ?
\sol{
En l'absence d'absorption, $\frac{(N+1)(N-n)}{(N-1)(N+n)}$ est réel et donc $\e^{i2Nk_{0}e}=\pm 1$, soit $2Nk_0 e = p\pi$ avec $p$ entier, et donc
\eqbox{eN = p \frac{\lambda_0}{4}.}
Cette condition dépend de la longueur d'onde $\lambda_0$ (et dans une moindre mesure, l'indice optique $N$ aussi). En fonction du signe de $\e^{i2Nk_{0}e}$, on trouve alors deux valeurs possibles pour $N$ (l'une supérieure à $n$ et l'autre inférieure) puis pour $e$, données par l'équation ci-dessus.
}
\end{quest}
\end{exo}
