\begin{exo}[PC][1][oral]{Dissolution d’un grain de sucre/}
  Un grain de sucre sphérique se solubilise dans l'eau. On donne le coefficient de diffusion
   $D=\SI{5e-10}{m.s^{-2}}$, la masse volumique
  $\rho=\SI{1.5e3}{kg.m^{-3}}$, le rayon initial du grain
  $R_0=\SI{2}{mm}$, et la
   masse volumique de sucre dissoute dans la solution $c(r,t)$. Laplacien en coordonnées sphériques : $\Delta f(r) = \frac{1}{r^2}\, \frac{\d}{\d r}\left( r \dd{f}{r}\right)$.
  \begin{quest}
  \item Écrire l'équation de diffusion. Donner les conditions aux limites.
  \sol{
Par isotropie, $c(r,t)$ ne dépend que de $r$. L’équation de diffusion de particules s’écrit
\eqbox{\dr{c}{t} = D \Delta c =  \frac{D}{r}\, \frac{\d^2 (r\, f)}{\d r^2}.}
La condition initiale est $c(r=R_0,t=0) = \rho$ et $c(r>R_0,t=0) = 0$. Les conditions aux limites sont $c(r=R(t),t) =\rho$ et $c(r=+\infty,t) = 0$.
  }
  \item Estimer le temps caractéristique $\tau$ de diffusion. Que peut-on en déduire ?
  \sol{
  $\tau\sim \frac{R_0^2}{D} \sim \SI{1e4}{s}$. Ce temps est très long devant la durée typique d’une expérience, donc le régime est quasi-stationnaire. On peut négliger le terme $\dr{c}{t}$ dans l’équation de diffusion.
  }
  \item Déterminer la concentration $c(r,t)$ en fonction de $R(t)$ rayon du grain à l'instant $t$ et $\rho$.
  \sol{
  On résout alors
  $$
\frac{D}{r}\, \frac{\d^2 (r\, c)}{\d r^2} = 0 \Rar  c(r,t) = A + \frac{B}{r}.
  $$
  Condition aux limites : $A=0$ et $c(r=R(t),t) = \frac{B}{R(t)} = \rho$, d’où
  \eqbox{c(r,t) = \rho \frac{R(t)}{r}. }
  }
  \item Avec un bilan de masse, donner l'équation différentielle vérifiée par $R(t)$ puis une expression littérale de $\tau$ et faire l'AN.
  \sol{
  Bilan de masse. Soit $S(t)$ la surface du sucre de rayon $R(t)$. La densité de courant de masse vaut
  $$
  \jN = -D \grad c = - D \dr{c}{r} \er = \frac{D \rho R(t)}{r^2} \er.
  $$

   La variation de la masse de sucre vaut
  $$
  \dd{m}{t} = -\int_S \rho \jN\ps\dS.
  \Rar
  \rho \dd{}{t}\left(\frac{4}{3}\pi R(t)^3 \right) = \rho \times (4 \pi R^2) \dd{R}{t} = -(4\pi R^2) \times \frac{D \rho R}{R^2} \Rar \dd{R}{t} = \frac{D}{R}.
  $$
  Temps caractéristique $\tau \sim \frac{\rho R_0^2}{D}$, comme attendu.
  }
\end{quest}
\end{exo}

\begin{exo}[PC][1][colle]{Sublimation d’une comète}
  Comète modélisée par une Doule.
\begin{quest}
\item Déterminer le rayon de la comète rom
\item La comète est en orbite autour du Soleil et reçoit une densité de courant thermique $\jth=\SI{30}{W.m^{-2}}$ Calculer la puissance thermique reçue par la comète.
\item La comète est constituée d'un noyau de glace entouré d'une croûte rocheuse d'épaisseur $e$.
Déterminer la masse de glace sublimée par unité de temps.
\item En considérant l'aspect thermique déterminer l'épaisseur e sachant que celle-a' est négligeable par rapport au reote de la boule.
\end{quest}
Données
- Masse volumique de la comèté / mase de la comè te
- Enthalpie molaire de griblimation.
- Conductivité theamique
- Masse molaire de l'eau: $18 \mathrm{~g} \cdot \mathrm{mol}^{-1}$
\end{exo}
