\def\axes{

  \pgfmathsetmacro\ymax{\xmax+\DNy*\d} % maximum of y = ct axis

  \pgfmathsetmacro\ymaxp{(\xmax/\d+\DNyp)*\D} % maximum of y' = ct' axis
  \pgfmathsetmacro\Nylines{\Nlines+\DNy} % number of world lines at constant ct'
