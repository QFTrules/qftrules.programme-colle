\begin{exo}[PC][1][colle]{Luminosité des étoiles}

\begin{exo}[PC][3][TD]{Démonstration des lois de Stefan et de Wien}
\Notions{\item Loi de Planck \task Loi de Stefan \task Loi de Wien}
  Un rayonnement d'équilibre thermique vérifie la loi de Planck, de sorte que sa puissance surfacique spectrale  a pour expression 
  \begin{equation}
    \dd{\vphi_0}{\lambda}(\lambda, T) = \frac{2\pi hc^2}{\lambda^5} \frac{1}{e^{hc/\lambda kT}-1}~\text{(en $\SI{}{W.m^{-2}.nm^{-1}}$)}.
  \end{equation}
  On souhaite démontrer les lois de Stefan et de Wien vues en cours à partir de cette expression.
  \begin{quest}
    \item	\udl{Loi de Wien}. 
    
    La longueur d'onde $\lambda_m$ à laquelle la puissance surfacique spectrale est maximale vérifie 
    \begin{equation}
      \dd{\vphi_0}{\lambda}(\lambda_m, T) = 0.
    \end{equation}
    Calculer cette dérivée et en déduire la loi de Wien $\lambda_m T \simeq \SI{2.9e-3}{m.K}$.
    \item \udl{Loi de Stefan}. 
    
    La puissance totale émise par un corps noir à l'équilibre thermodynamique est donnée par l'intégrale de la puissance surfacique spectrale sur toutes les longueurs d'onde, soit 
    \begin{equation}
      \vphi_0(T) = \int_0^{+\infty} \dd{\vphi_0}{\lambda}(\lambda, T) \,\d \lambda.
    \end{equation}
    Calculer cette intégrale et en déduire la loi de Stefan $\vphi_0(T) = \sigma T^4$.
  \end{quest}
  \textsc{Données} : La solution de l'équation $e^{-x} + \num{0.2} x -1 =0$ vaut $\SI{4.965}{}$.  On a $h = \SI{6.63e-34}{J.s}$, $c = \SI{3e8}{m.s^{-1}}$, $k_{\rm B} = \SI{1.38e-23}{J.K^{-1}}$, $\int_{0}^{+\infty} \frac{x^3 \d x }{e^x-1} = \frac{\pi^4}{15}$. 
\end{exo}

\begin{exo}[PC][2][TD]{Rayonnement thermique d'une sphère}
\Notions{\item $1^{\rm er}$ principe  \task Loi de Stefan \task Loi de Wien}
On considère une sphère pleine métallique homogène de rayon $R$ placée dans le vide. On suppose qu'un régime quasi-stationnaire est atteint de sorte que la température de la sphère $T(t)$ mas dépend lentement du temps tout en étant uniforme. On assimile la sphère à un corps noir à l'équilibre thermique, de sorte que le rayonnement émis par la sphère est un rayonnement d'équilibre thermique. La sphère est de masse volumique $\rho$ et de capacité calorifique massique $c$.
\begin{quest}
  \item	Rappeler la définition d'un corps noir. À partir de la loi de Stefan, déterminer la puissance $\mc{P}(t)$ que la sphère rayonne vers l'extérieur en fonction de sa température $T(t)$ et de son rayon $R$.
  \sol{
    La loi de Stefan donne la puissance surfacique émise par un corps noir
    $$
  \phi_0(T) = \sigma T^4.
    $$
    On en déduit la puissance totale émise par la sphère connaissant sa surface $S = 4 \pi R_s^2$ :
    \eqbox{
      \mc{P}(t) = 4 \pi R^2 \sigma T(t)^4.
    }
  }
  \item Estimer numériquement cette puissance pour une sphère de rayon $R=\SI{10}{cm}$ et de température initiale $T_0 = \SI{1200}{K}$. On donne la constante de Stefan : $\sigma=\SI{5,67e-8}{W.m^{-2}.K^{-4}}$. Déterminer également la longueur d'onde $\lambda_m$ pour laquelle le rayonnement est maximal.
  \sol{
La loi de Wien donne 
\eqbox{
  \lambda_m = \frac{2900}{1200} = \SI{2,4}{\micro\meter}.
}
Ce rayonnement correspond au proche infrarouge.
  }
  \item Par un bilan d'énergie, déterminer l'équation différentielle vérifiée par la température $T(t)$. Résoudre cette équation différentielle en introduisant la température initiale $T_0$.
  \sol{
  L’énergie interne de la sphère vaut 
  $$
  U(t) = \rho c V T(t) = \rho c \frac{4}{3} \pi R_s^3 T(t).
  $$
  La sphère perd une puissance $\mc{P}(t)$ par rayonnement, et donc 
  $$
\dd{U}{t} = - \mc{P} \Rar \rho c \frac{4}{3} \pi R_s^3 \dd{T}{t} = - 4 \pi R_s^2 \sigma T(t)^4,
  $$
  ce qui donne l’équation différentielle
  \eqbox{
    \dd{T}{t} + \frac{3 \sigma}{R \rho c} T^4 = 0.
  }
On peut résoudre cette équation par séparation des variables. On trouve alors
$$
\frac{\d T}{T^4} = -\frac{3 \sigma}{R_s \rho c} \d t \Rar \frac{1}{3 T_0^3} - \frac{1}{3T(t)^3} = \frac{3 \sigma}{R_s \rho c} t,
$$
et donc 
\eqbox{
T(t) = T_0 \left(1 + \frac{R\rho c}{9 \sigma T_0^3} t\right)^{-1/3}.
}
  }
  \item Montrer que la température décroît sur un temps caractéristique $\tau$, fonction de $T_0$, $R$, $\rho$ et $c$, tel que  
  \begin{equation}
    T(t) = T_0 \left(1 + \frac{t}{\tau}\right)^{-1/3}.
  \end{equation}
  Exprimer $\tau$ puis l'évaluer pour $\rho=\SI{8e3}{kg.m^{-3}}$ et $c=\SI{9e2}{J.K^{-1}.kg^{-1}}$.
  % L'évaluer numériquement.
  \sol{
    On trouve 
    % $$
    \eqbox{
      \tau = \frac{9 \sigma T_0^3}{R \rho c} = \SI{6}{s}.
      }
    % $$
  }
  \bonus{
  \item Au bout de combien de temps la température est réduite de moitié ? \AN~?
  \sol{
    On trouve $T(t) = \frac{T_0}{2}$ pour $\left(1 + \frac{t}{\tau} \right)^{-1/3}= \frac{1}{2}$, soit  $1 + \frac{t}{\tau} = 8 \Rar \frac{t}{\tau}= 7$ et donc 
    \eqbox{
      t = 7\tau = \SI{42}{s}. 
    }
  }
  }
\end{quest}
\end{exo}

\begin{exo}[PC][1][TD]{Atmosphère transparente au rayonnement terrestre}
  \Notions{\item Bilan radiatif \task Loi de Stefan \task Loi de Wien}
On étudie l'effet de serre en prenant en compte  la transmission partielle dans l'atmosphère d'une fraction~$\beta$ du rayonnement émis par la Terre. On suppose par contre que l'atmosphère est totalement transparente au rayonnement solaire. On note $P_{\rm{S}\to\rm{T}}$ la puissance totale du rayonnement solaire reçue par la Terre, $P_{\rm{A}}$ la puissance émise par l'atmosphère vers la Terre et vers l'espace, et $P_{\rm{T}}$ la puissance émise par la Terre vers l'atmosphère. On note $A\simeq \num{0.3}$ l'albédo de la Terre.



\begin{quest}
\item On présente ci-dessous le bilan radiatif de la Terre pour une atmosphère \emph{totalement absorbante} au rayonnement terrestre ($\beta=0$). Comment est modifié ce bilan pour une atmosphère \emph{partiellement transparente} au rayonnement terrestre ($0<\beta<1$) ?  Représenter le schéma-bilan associé.

\begin{fig}
\includegraphics[width=0.5\textwidth]{schema_terre_atmosphere.pdf}
\capt{Bilan radiatif pour une atmosphère \emph{totalement absorbante} au rayonnement terrestre ($\beta = 0$).}
\end{fig}

\sol{
  Voici le bilan radiatif pour une atmosphère \emph{partiellement absorbante} au rayonnement terrestre, de coefficient de transmission $\beta$.
\begin{fig}
\includegraphics[width=0.5\textwidth]{schema_terre_atmosphère_trans.pdf}
% \capt{Bilan radiatif pour une atmosphère \emph{partiellement absorbante} au rayonnement terrestre.}
\end{fig}
}

\item  Écrire les conditions d’équilibre radiatif pour les deux systèmes \{Terre\} et \{atmosphère\}. En déduire la relation entre $\mc{P}_{\rm{T}}$ et $\mc{P}_{\rm{S}\to\rm{T}}$, en fonction de $A$ et $\beta$.


\sol{
Le conditions d’équilibre radiatif s'écrivent pour la Terre et son atmosphère comme
\begin{equation}
\begin{cases}
(1-A)P_{\rm{S}\to\rm{T}}+P_{\rm{a}}=P_{\rm{T}},\\
(1-\beta)P_{\rm{T}} = 2P_{\rm{a}}.
\end{cases}
\end{equation}
Donc $(1-A)P_{\rm{S}\to\rm{T}}+(1-\beta)P_{\rm{T}}/2=P_{\rm{T}}$, et  $(1-A)P_{\rm{S}\to\rm{T}} = (1+\beta)P_{\rm{T}}/2$, et donc
\eqbox{P_{\rm{S}\to\rm{T}}  = \frac{1+\beta}{2(1-A)} P_{\rm{T}} .}
}

\item On suppose que le Soleil et la Terre se comportent comme des corps noirs de températures respectives $T_S = \SI{5800}{K}$ et $T_T$.  Déterminer la longueur d'onde $\lambda_{\rm m,S}$ du pic d'émission du spectre solaire. En proposant une valeur raisonnable pour $T_T$, montrer que la Terre émet principalement dans l’infrarouge thermique. Justifier alors que l'atmosphère puisse absorber partiellement le rayonnement terrestre tout en étant totalement transparente au rayonnement solaire.
\sol{
Expérimentalement $T_T \sim \SI{300}{K}$, et la loi de Wien donne alors
\eqbox{
\lambda_{\rm m, T} =  \frac{2900}{T_T} = \SI{9.7}{\mu m}.
}
Le Soleil émet essentiellement de la lumière visible. La Terre émet essentiellement du rayonnement infrarouge. Pour le Soleil, la loi de Wien donne 
\eqbox{
\lambda_{\rm m, S} = \frac{2900}{T_S} = \SI{500}{nm}.
}

L'absorbance dépend de la longueur d'onde, et donc l'atmosphère peut tout à fait absorber le rayonnement terrestre en absorbant pas ou peu le rayonnement solaire, les deux rayonnements ayant un spectre très différent.
}
\item  On représente ci-dessous le schéma de la Terre et du Soleil, où $R_S$ désigne le rayon du Soleil et  $d_{\rm{TS}}$ la distance Terre-Soleil, avec en pontillé le cône de lumière solaire intercepté par la Terre, dont la surface émettrice est représentée en orange. En déterminant le rayon $r$ de cette surface émettrice, montrer que la puissance solaire reçu par la Terre s'écrit
\begin{equation}
\mc{P}_{\rm{S}\to\rm{T}} = \varphi_S \cdot 4 \pi R_{\rm T}^2  \left(\frac{R_{\rm{S}}}{d_{\rm{TS}}}\right)^2.
\end{equation}

 \begin{fig}[label]
  \centering
  \includegraphics[width=0.6\columnwidth]{soleil_terre.pdf}
  \capt{Schéma de la Terre et du Soleil.}
\end{fig}

\sol{
  La loi de Stefan donne 
  \eqbox{
  \phi_{\rm S} = \sigma T_S^4, \quad \phi_{\rm T} = \sigma T_T^4.
  }
Le Soleil est une boule de rayon $R_S$ qui émet dans tout l’espace de façon isotrope. La puissance émise par le Soleil vaut $P_S = \varphi_S \times 4\pi R_S^2 = \varphi_{\rm{S}\to\rm{T}} \times 4\pi d_{TS}^2$. On en déduit 
\eqbox{
\varphi_{\rm{S}\to\rm{T}} = 4\pi R_{\rm T}^2\varphi_S \left(\frac{R_{\rm{S}}}{d_{\rm{TS}}}\right)^2.
}

}
\item Exprimer la puissance émise par la Terre $P_{\rm T}$ en fonction du flux émis $\varphi_{\rm T}$. À l'aide de la loi de Stefan, exprimer $\vphi_{\rm T}$ en fonction de $T_{\rm T}$, et $\vphi_{\rm S}$ en fonction de $T_{\rm S}$.  Montrer finalement que la température à la surface de la Terre a pour expression
%En remplaçant par les expressions établies précédemment et en simplifiant on obtient
\begin{equation}
T_{\rm{T}}=T_{\rm{S}}  \sqrt{\frac{R_{\rm{S}}}{d_{\rm{TS}}}} \left[\frac{1-A}{2(1+\beta)}\right]^{1/4}.
\end{equation}

\sol{
    On a de même 
  \eqbox{
    P_{\rm T} = 4\pi R^2 \varphi_T.
  }
Pour déterminer la puissance solaire totale incidente sur la Terre $P_{\rm{S}\to\rm{T}}$, il faut remarquer que le flux solaire n’est capté que sur surface $\pi R_T^2$ correspondant à la surface du disque terrestre, i.e. à la surface projetée de la Terre. On a donc
\eqbox{
P_{\rm{S}\to\rm{T}}=\varphi_{\rm{S}\to\rm{T}}\times \pi R_{\rm{T}}^2.
}
On combine les équations entre elles. On trouve 
$$
P_{\rm S \to T} = \varphi_{\rm{S}\to\rm{T}} \pi R_{\rm{T}}^2 = \varphi_S \left(\frac{R_{\rm{S}}}{d_{\rm{TS}}}\right)^2 \pi R_{\rm{T}}^2 = \sigma T_S^4 \left(\frac{R_{\rm{S}}}{d_{\rm{TS}}}\right)^2 \pi R_{\rm{T}}^2 .
$$
Or $P_{\rm{S}\to\rm{T}}  = \frac{1+\beta}{2(1-A)} P_{\rm{T}}  = \frac{1+\beta}{2(1-A)} \sigma T_{\rm T}^4$, d’où le résultat,
\eqbox{
T_{\rm{T}}=T_{\rm{S}}  \sqrt{\frac{R_{\rm{S}}}{d_{\rm{TS}}}} \left[\frac{1-A}{2(1+\beta)}\right]^{1/4}.
}
% L'application numérique donne $T_{\rm{T}}=\SI{291}{K}=\SI{18}{\celsius}$. L'accord avec la réalité est très bon.
}

\item Estimer $\beta$ à partir de la \Fig{absorp}. En déduire une valeur numérique de $T_{\rm T}$. Comment évoluent $\beta$ et $T_{\rm T}$ si les quantités de CO$_2$ et de H$_2$O vapeur augmentent ? Expliquer l'origine du réchauffement climatique.
 \sol{
  Le paramètre
$\beta$ correspond à la fraction de rayonnement sous la courbe du corps noir à $\SI{288}{K}$ qui est transmise dans l'IR (région bleue). On peut mesurer l'aire transmise (fenêtre atmosphérique) et l'aire totale (courbe du corps noir) en découpant puis en pesant le papier. On calcule le rapport et on trouve \fbox{$\beta \simeq \num{0.3}$}, d'où 
\eqbox{
T_{\rm T} = \SI{285}{K} = \SI{12}{\dC},
}
ce qui est cohérent avec la température moyenne mesurée sur la planète.
Si la quantité de CO$_2$ et d'eau vapeur augmente, on s'attend à davantage d'absorption et donc $\beta$ diminue. Si $\beta$ diminue, $T_{\rm T}$ augmente : il s'agit du réchauffement climatique. En réalité, cet argument naïf ne peut pas expliquer le réchauffement parce que les gaz à effet de serre absorbent déjà le rayonnement à $\SI{100}{\%}$ dans leur gamme de longueur d'onde d'absorption. Pour expliquer le réchauffement, il faut prendre en compte la différence de température entre le haut de l'atmosphère et la surface terrestre.
}

\end{quest}

\begin{exo}[PC][1][devoir]{Effet de serre}
  \Notions{\item Loi de Stefan \task Loi de Wien \task Bilan radiatif}
  On étudie l'effet de serre en supposant l'atmosphère opaque au rayonnement émis par la Terre et totalement transparente au rayonnement solaire. On note $P_{\rm{S}\to\rm{T}}$ la puissance totale du rayonnement solaire reçue par la Terre, $P_{\rm{a}}$ la puissance émise par l'atmosphère vers la Terre et vers l'espace, et $P_{\rm{T}}$ la puissance émise par la Terre vers l'atmosphère. On note $A\simeq \num{0.3}$ l'albédo de la Terre.
  
  
  
  \begin{quest}
    \item Représenter le schéma-bilan radiatif de la Terre et de l'atmosphère.
    
    
    \sol{
      \begin{fig}
      \includegraphics[width=0.5\textwidth]{schema_terre_atmosphere.pdf}
      \capt{Bilan radiatif pour une atmosphère totalement absorbante au rayonnement terrestre.}
      \end{fig}
      }
  
  \item  Écrire les bilans radiatifs pour les deux systèmes \{Terre\} et \{atmosphère\}. En déduire la relation entre $P_{\rm{T}}$ et $P_{\rm{S}\to\rm{T}}$. Où intervient implicitement la condition d'équilibre radiatif ?
  
  
  \sol{
  La condition d’équilibre radiatif impose 
  $$
      P_{\rm abs} = P_{\rm émis},
  $$
  pour les systèmes $\{$Terre$\}$ et $\{$atmosphere$\}$. D’après le schéma-bilan ci-dessus, on obtient donc
  $$
  \begin{cases}
  (1-A)P_{\rm{S}\to\rm{T}}+P_{\rm{a}}=P_{\rm{T}}~\text{(Terre)},\\
  P_{\rm{T}} = 2P_{\rm{a}}~(\text{atmosphère}).
  \end{cases}
  $$
  Donc $(1-A)P_{\rm{S}\to\rm{T}}+P_{\rm{T}}/2=P_{\rm{T}}$,  soit $(1-A)P_{\rm{S}\to\rm{T}} = P_{\rm{T}}/2$ et donc
  \eqbox{P_{\rm{T}} = 2(1-A) P_{\rm{S}\to\rm{T}}.}
  }
  
  \item On suppose que le Soleil se comporte comme un corps noir de température $T_S = \SI{5800}{K}$. Quelle est alors la nature du rayonnement émis par le Soleil ? Déterminer la longueur d'onde $\lambda_{\rm m,S}$ du pic d'émission du spectre solaire.
  
  \sol{
    La loi de Wien donne la longueur d’onde $\lambda_m$ correspondant au maximum d’émission d’un corps noir ; elle impose 
    $$
      \lambda_m T = \SI{2.9e3}{K.\micro m}.
    $$
    
    En assimilant le Soleil à un corps noir de température $T_S = \SI{5800}{K}$, on a
