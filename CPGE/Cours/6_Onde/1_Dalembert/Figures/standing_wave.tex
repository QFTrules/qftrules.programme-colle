  \draw[xcol,JoliBleu,very thick,samples=100,smooth,variable=\x,domain=0:\L]
    plot(\x,{\A*cos(360/\lam*\x)})  node[right]{$f(x)$};

      plot(\x,{-\A*cos(360/\lam*\x)}) ;
  
  \foreach \j in {-0.75,-0.5,...,0.75}{
  \draw[xcol,JoliBleu,densely dashed,samples=100,smooth,variable=\x,domain=0:\L]

    }
  \foreach \i [evaluate={\x=(\i-0.75)*\lam;}] in {1,...,\n}{

    \draw[blue!50!black!60,thin,dashed] (\x+3*\lam/4,\ymax) --++ (0,-1.4*\s);
    \draw[red!50!black!60, thin,dashed] (\x,-\ymax) --++ (0,1.4*\s);
    \fill[red!50!black] (\x,0) circle(0.03);
