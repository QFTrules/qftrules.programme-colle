\begin{exo}[PC][1][TD]{Propriétés des OPPH acoustiques}
    \Notions{\item OPPH \task Impédance acoustique}
    Une OPPH acoustique de pulsation $\omega$ se propage dans une direction portée par le vecteur unitaire~$\Vec{n}$. On note $\Vec{k} = k \Vec{n}$ le vecteur d’onde.
    \begin{quest}
        \item Exprimer la surpression complexe $\udl{p_1}(M,t)$ à la position $\Vec{r}$ à l’instant $t$. Quelle relation lie $k$ et $\omega$ ?
        \item Déterminer le champ de vitesse complexe $\udl{v}_1(M,t)$ et justifier que les ondes acoustiques sont longitudinales. Déterminer l’impédance acoustique $\udl{Z}$.
        \sol{
            L'équation d'Euler s'écrit sous forme vectorielle et en notation complexe
            \begin{equation}
                \rho_0 \dr{\udl{\Vec{v_1}}}{t} = - \grad \udl{p_1} \quad\Rightarrow\quad \rho_0 i\omega\underline{\vv{v_1}}=i\vv{k}\underline{p_1} \quad\Rightarrow\quad  \underline{\vv{v_1}}=\frac{\udl{p_1}}{\omega \rho_0}\vv{k}
            \end{equation}
            La vitesse $\Vec{v}$ est colinéaire au vecteur d'onde $\Vec{k}$, et donc les ondes acoustiques sont longitudinales. L’impédance acoustique est alors donnée par
                \eqbox{
                    \udl{Z} = \frac{\udl{p_1}}{\udl{v_1}} = \rho_0 c.
                }
        }
        \item Déterminer la surdensité $\udl{\rho_1}(M,t)$ de deux façons différentes.
        \sol{
         La conservation de la masse donne 
         $$
         i\omega \udl{\rho_1} = ik \rho_0 \udl{v_1} = \frac{ik}{c} \udl{p_1} \Rar \udl{\rho_1} = \frac{\udl{p_1}}{c^2}  = \rho_0 \chi_S \udl{p_1}.
         $$
         On retrouve cette expression à partir de l’expression de la compressibilité isentropique $\chi_S$.
        }
    \end{quest}

\end{exo}

\begin{exo}[PC][1][colle]{Impédance acoustique d’une onde stationnaire}
    Considérons une OPS acoustique, dont la surpression est de la forme $p_1(x,t) = f(x)g(t)$.
    \begin{quest}
        \item Rappeler la forme des fonctions $f(x)$ et $g(t)$ pour que cette OPS soit solution de l’équation de d’Alembert. Comment s’écrit la surpression $\udl{p_1}(x,t)$ en notation complexe ?
        \sol{
            % \eqbox{
            $p_1(x,t) = p_m \cos(kx - \vphi) \cos(\omega t) = \frac{p_m}{2} \left[ \cos(\omega t - kx  + \vphi) + \cos(\omega t + kx - \vphi) \right] = \re \udl{p_1}(x,t),$
            % }
            avec 
            \eqbox{
                p_1(x,t) = \frac{p_m}{2} \left[ e^{i(\omega t - kx  + \vphi)} + e^{i(\omega t + kx - \vphi)} \right].
            }
        }
        \item Que vaut alors le champ de vitesse complexe $\udl{v_1}(x,t)$ ? En déduire l’expression de l’impédance acoustique $\udl{Z}$
        \sol{
            L’équation d’Euler linéarisée donne 
            $$
            \rho_0 \dr{v_1}{t} = -\dr{p_1}{x} = kp_m \sin(kx+\vphi) \cos(\omega t + \psi) \Rar v_1(x,t) = \frac{kp_m}{\omega \rho_0} \sin(kx+\vphi) \sin(\omega t + \psi) 
            $$
        }
    \end{quest} 
\end{exo}

\begin{exo}[PC][3]{Onde acoustique dans un pavillon sonore}
On cherche à modéliser la propagation d'une onde acoustique dans un pavillon de trompette, que l'on modélisera par un pavillon de révolution exponentiel de section circulaire variable donnée par $S(x)=S_{0}\,\e^{x/a}$. Le pavillon contient de l'air de masse volumique au repos $\rho_0$. On continuera de supposer que dans le cadre de l'approximation acoustique, une tranche d'air suit une évolution adiabatique réversible et on notera $\chi_S$ la compressibilité isentropique.
\begin{quest}
\item En effectuant un bilan de masse sur la tranche d'air située entre les sections $S(x)$ et $S(x+\d x)$, déterminer l'équation de conservation de la masse dans le pavillon. Vérifier que si $S=\cte$ on retrouve bien la forme usuelle établie dans le cours.
\item En utilisant l'équation d'état du fluide pour l'évolution thermodynamique considérée et l'équation d'Euler, montrer que l'équation de propagation dans le pavillon pour la surpression acoustique $P_1(x,t)$ s'écrit
\begin{equation*}
\frac{\partial^2 P_1}{\partial x^2}-\frac{1}{c^2}\frac{\partial^2 P_1}{\partial t^2}=-\frac{\partial P_1}{\partial x}\frac{\partial\ln S}{\partial x}.
\end{equation*}
\item En cherchant une solution sous la forme d'une OPPH se propageant selon les $x$ croissants avec $k\in\mathbb{C}$, établir la relation de dispersion des ondes acoustiques dans le pavillon.
\item En déduire qu'il existe une pulsation de coupure $\omega_{\rm{c}}$ dont on précisera l'expression en dessous de laquelle aucune propagation n'est possible. Donner dans ce cas la forme de la surpression $P_1$.
\item Calculer la section maximale que doit avoir le pavillon pour transmettre les sons de fréquence supérieure à $f_{\rm{c}}=\SI{500}{Hz}$ si sa section minimale est $S_0=\SI{3}{cm^2}$ et sa longueur est $x_{0}=\SI{10}{cm}$.
\end{quest}
\end{exo}

\begin{exo}[PC][1][TD]{Isolation passive de bruit}
    \Notions{\item Onde acoustique \task Impédance acoustique}
On considère un tuyau supposé infini et de section $S$ constante dans lequel on place une membrane infiniment mince, imperméable et de masse surfacique uniforme $\sigma$. Cette membrane sépare deux fluides parfaits de masse volumique $\rho_1$ et $\rho_2$ dans lesquels l'onde se propage à la vitesse $c_1$ et $c_2$. On suppose qu'une OPPH arrive en incidence normale sur la membrane.
\begin{quest}
\item Proposer une forme pour l'onde incidente (en vitesse et en pression), puis pour les ondes réfléchies et transmises en faisant intervenir les coefficients de transmission en amplitude en vitesse $\underline{r}$ et $\underline{t}$.
\sol{
    Dans le milieu incident l'onde incidente se superpose avec l'onde réfléchie donc
% \begin{equation}
    $$
\begin{cases}
\underline{v_1}=\underline{A_{\rm{i}}}\e^{i(\omega t-kx)} + \underline{A_{\rm{r}}}\e^{i(\omega t+kx)},\\
\underline{p_1}=Z\underline{A_{\rm{i}}}\e^{i(\omega t-kx)} \JRouge{-}Z \underline{A_{\rm{r}}}\e^{i(\omega t+kx)},
\end{cases}
% \end{equation}
$$
avec $Z=\rho_1 c_1$,
et dans le milieu où l'onde est transmise
% \begin{equation}
    $$
\begin{cases}
\underline{v_1}'=\underline{A_{\rm{t}}}\e^{i(\omega t-k'x)},\\
\underline{p_1}'=Z'\underline{A_{\rm{t}}}\e^{i(\omega t-k'x)},
\end{cases}
% \end{equation}
$$
avec $Z' = \rho_2 c_2$.
Par définition on a $\underline{r}=\frac{\underline{A_{\rm{r}}}}{\underline{A_{\rm{i}}}}$ et $\underline{t}=\frac{\underline{A_{\rm{t}}}}{\underline{A_{\rm{i}}}}$, soit 
% \begin{equation}
    % \eqbox{
    $$
        \begin{cases}
            \underline{v_1}=\underline{A_{\rm{i}}}\left( \e^{i(\omega t-kx)} + \udl{r} \e^{i(\omega t+kx)} \right),\\
            \underline{p_1}=Z\underline{A_{\rm{i}}}\left(\, \e^{i(\omega t-kx)} \udl{r} \e^{i(\omega t+kx)}\right),
        \end{cases}\quad
        \begin{cases}
            \underline{v_1}'=\udl{A_{\rm i}} \udl{t} \e^{i(\omega t-k'x)},\\
            \underline{p_1}'=Z'\udl{A_{\rm i}} \udl{t} \e^{i(\omega t-k'x)}.
        \end{cases}
        $$
    % }
    % \end{equation}
}
\item Déterminer les expressions des coefficients $\underline{r}$ et $\underline{t}$, d'abord en supposant la pression continue à l'interface, puis en prenant en compte l'inertie de la membrane, en fonction de $Z_1,Z_2,\omega$ et $\sigma$. Commenter sur le type de filtrage réalisé par la membrane.
\sol{
On rappelle que pour une membrane imperméable sans masse, la pression $\udl{p_1}$ et la vitesse $\udl{v_1}$ sont continues. Ici, l'imperméabilité assure encore la continuité de la vitesse en $x=0$, soit 
\eqbox{
    \udl{v_1}(x=0)=\udl{v_1}'(x=0),
}
ce qui donne 
$$
\udl{A_{\rm i}}\left(1+\udl{r}\right)=\udl{A_{\rm i}}\udl{t} \Rar \udl{r}=\udl{t}-1.
$$
La condition sur la pression s'obtient à partir du PFD appliqué sur une section $S$ de la membrane. Par continuité, la vitesse de la membrane vaut $\Vec{v_1}(x=0,t)$, soit  
$$
m \dr{\udl{v_1}}{t}(x=0) = S \udl{p_1}(x=0,t)  - S \udl{p_1}'(x=0,t) \Rar \udl{p_1}(x=0,t) = \udl{p_1}'(x=0,t)  + \sigma i \omega \udl{v_1}(x=0), 
$$
et donc $Z \left(1-\udl{r}\right) = 2Z -  Z\udl{t} = (Z' + i \sigma \omega)\udl{t}.$
On en déduit  le coefficient de transmission en amplitude en vitesse,
\eqbox{
\udl{t} =  \frac{2Z}{Z' + Z - i\omega \sigma} = \frac{\udl{t}_0}{1 + i\frac{\omega}{\omega_c}},
% = \frac{2Z(Z'+Z) + 2iZ\omega\sigma}{(Z'+Z)^2 + \omega^2 \sigma^2}.
}
avec $\udl{t_0} = \frac{2Z}{Z' + Z}$ le coefficient de transmission en l'absence de membrane et $\omega_c = \frac{Z' + Z}{\sigma}$ la fréquence de coupure. Ce coefficient est complexe, ce qui signifie que la membrane déphase l'onde transmise. La membrane agit alors comme un filtre passe-bas de fréquence de coupure $\omega_c$. Les hautes fréquences sont coupées et la membrane permit d'isoler phoniquement les deux milieux. Cette isolation est possible même lorsqu'il y a adaptation d'impédance $Z=Z'$, où $\udl{t_0}=1$. On a alors $\omega_c = \frac{2\rho c}{\sigma}$.
}
\end{quest}
\end{exo}

\begin{exo}[PSI][2][TD]{Onde acoustique d’une sphère pulsante}
 Une sphère pulsante de centre fixe O, dont le rayon $a(t)=a_0 + a_1 \cos(\omega t)$ varie sinusoïdalement avec une amplitude $a_1 \ll a_0 \ll \lambda$, émet des ondes sonores dans tout l'espace rempli d'air, de masse volumique $\rho_0$, où la célérité des ondes sonores vaut $c$. Le problème étant à symétrie sphérique, on cherche des champs de la forme $p_1(M,t)=p_1(r,t)$ et $\Vec{v_1}(M,t)=v_1(r,t)\er$ en coordonnées sphériques. On rappelle que pour un champ scalaire $f(r,t)$ ne dépendant que de $r$ en coordonnées sphériques, le laplacien s'écrit
    \begin{equation*}
        \Delta f= \frac{1}{r}\frac{\partial^2 (rf)}{\partial r^2}.
    \end{equation*}

    \begin{quest}
        \item Rappeler quelle équation d’onde vérifie les ondes acoustiques. Déterminer la forme générale des solutions $p_1(r,t)$ de cette équation et interpréter.

        \sol{
        Le champ de surpression est solution de l'équation de d'Alembert qui s'écrit ici :
        \begin{equation*}
            0=\Delta p_1 - \frac{1}{c^2}\frac{\partial^2 (p_1)}{\partial t^2}=\frac{1}{r}\frac{\partial^2 (rp_1)}{\partial r^2}-\frac{1}{c^2}\frac{\partial^2 (p_1)}{\partial t^2}
        \end{equation*}
        On multiplie par $r$ et on rentre $r$ dans la dérivée temporelle (c'est une constante par rapport au temps, on a bien le droit !). On obtient donc :
        \begin{equation*}
            \frac{\partial^2 (rp_1)}{\partial r^2}-\frac{1}{c^2}\frac{\partial^2 (rp_1)}{\partial t^2}
        \end{equation*}
        Donc la fonction $\beta(r,t)=rp_1(r,t)$ est solution de l'équation de d'Alembert dans le cas unidimensionnel. Donc la solution générale s'écrit (voir cours) :
        \begin{equation*}
            \beta(r,t)=rp_1(r,t)=f(r-ct) + g(r+ct) \Rightarrow p_1(r,t)=\frac{1}{r}(f(r-ct) + g(r + ct))
        \end{equation*}
        La surpression apparaît alors comme la superposition de deux signaux se propageant selon $\vec{u_r}$ et $-\vec{u_r}$ avec une atténuation en $1/r$. Or, s'il n'y a pas d'obstacle, il n'y a pas d'onde selon $-\vec{u_r}$ donc on a $p_1$ de la forme :
        \begin{equation*}
            \boxed{p_1(r,t)=\frac{1}{r}f(r-ct)}
        \end{equation*}
        }

        \item Dans la suite, on pose $\omega= k c$ et on cherche une solution de la forme $p_1(r,t)=\frac{A}{r}\mathrm{cos}(\omega t - kr - \alpha)$.
            \begin{enumerate}

                \item Déterminer l'expression correspondante du champ de vitesses. En déduire les expressions simplifiées qu'on adoptera dans la suite :
                \begin{equation*}
                    v_1(r \ll \lambda,t)=\frac{A}{\rho_0 \omega r^2}\,\sin(\omega t - kr - \alpha) \ \ ; \ \ v_1(r \gg \lambda,t)=\frac{kA}{\rho_0 \omega r}\,\cos(\omega t - kr - \alpha).
                \end{equation*}

            \sol{L'équation d'Euler généralisée relie $v_1$ à $p_1$. Avec la forme proposée pour $p_1$, on obtient que :
        \begin{equation*}
            \rho_0 \frac{\partial v_1}{\partial t}=-\frac{\partial p_1}{\partial r}= \frac{A}{r^2}\cos (\omega t - kr - \alpha) - \frac{kA}{r}\sin (\omega t - kr - \alpha)
        \end{equation*}
        On intègre alors par rapport au temps (attention à rajouter la constante qui dépend de la position $r$). La valeur temporelle de $v_1$ doit être nulle, ce qui impose de prendre la fonction constante nulle. On obtient finalement que :
        \begin{equation*}
            \boxed{v_1= \frac{A}{\rho_0 \omega r^2}\sin (\omega t - kr - \alpha) + \frac{kA}{\rho_0 \omega r}\cos (\omega t - kr - \alpha)}
        \end{equation*}
        Le rapport des deux amplitudes en facteur des deux membres de $v_1$ vaut $1/2\pi * \lambda/r$, d'où les expressions approchées qui négligent un terme ou l'autre en fonction de $r$.
                }

                \item Justifier la condition aux limites approchée $v_1(a_0,t)=\dot{a}(t)$. En déduire les expressions de $A$ et $\alpha$.

        \sol{

        La membrane étant imperméable, les vitesses normales doivent être continues en $r=a_0+a_1\cos(\omega t)$ donc :
            \begin{equation*}
                v_1(a_0+a_1\cos(\omega t))=-a_1 \omega \sin (\omega t)
            \end{equation*}
            Vu que $a_1 << \lambda$, on peut considérer que la condition est en $r=a_0$. En utilisant l'expression simplifiée de $v_1(r,t)$ pour $r << \lambda$, on obtient :
            \begin{equation*}
                \boxed{A=\rho_0 \omega^2 a_0^2 a_1}\ \ \ \ \text{et} \ \ \ \ \boxed{\alpha=\pi-ka_0}
            \end{equation*}}
            \end{enumerate}


        \item Exprimer la puissance moyenne rayonnée à travers une sphère de centre O et de rayon $r \gg \lambda$. Commenter l'influence de $r$. Comparer la surface nécessaire pour émettre un son grave et aigu de même puissance.



            \sol{

            La puissance rayonnée à travers la sphère de rayon $r$ s'exprime à l'aide du vecteur $\Vec{R}=p_1v_1\vec{u_r}$.
        \begin{equation*}
            \mathcal{P} =  \iint \limits_\mathrm{S} \vec{\Pi} \cdot \mathrm{d}\vec{S}=4\pi r^2 p_1(r,t)v_1(r,t)=\frac{A^2}{\rho_0 c r^2}\cos^2(\omega t -kr - \alpha)
        \end{equation*}
        Puis, en prenant la valeur moyenne et en remplaçant $A$ par sa valeur, on obtient finalement que :
        \begin{equation*}
            \boxed{<\mathcal{P}>=\frac{2\pi \rho_0 \omega^4 a_0^4 a_1^2}{c}}
        \end{equation*}
        On remarque que cette puissance moyenne est indépendante de la distance $r$ à l'émetteur : la décroissance des champs en $1/r$ assure une conservation de l'énergie sonore : en effet, la décroissance de l'énergie sera en $1/r^2$ mais la croissance des surfaces sphériques est en $r^2$ ! De plus, la puissance croît en $\omega^4$ (Tiens puissance de Larmor, ça vous dit quelque chose ?). Les sons aigus sont donc favorisés par rapport aux sons graves ! Si l'on écoute une musique, il est donc nécessaire d'utiliser une enceinte plus spécifique aux sons graves, sinon ces sons seront atténués.
        }


        \item Exprimer l'impédance complexe $\underline{Z}=\udl{p_1}/\udl{v_1}$ au niveau de la membrane et la comparer à l'impédance $Z_{\mathrm{OPP}}$ des ondes planes progressives. En déduire pourquoi on peut considérer qu'on a un nœud de pression à l'extrémité d'un tuyau ouvert dont la section a pour dimension typique $a_0 \gg \lambda$.
        \sol{
        Sur la membrane, les champs complexes sont de la forme :
        $$
        \begin{cases}
            \underline{p}_1=\frac{A}{a_0}\mathrm{exp}(j\omega t - jkr - j\alpha) \\
            \underline{v}_1=\frac{A}{\rho_0 \omega a_0^2}\mathrm{exp}(j\omega t - jkr -j\alpha - j\pi/2)
        \end{cases}
        $$
        En faisant simplement le rapport des deux, on obtient :
        \begin{equation*}
            \boxed{\underline{Z}_R=\frac{\underline{p}_1}{\underline{v}_1}=j\rho_0 \omega a_0}
        \end{equation*}
        Le rapport de cette impédance avec celui des ondes planes progressives vaut alors $\frac{\omega a_0}{c}$ soit $2\pi \frac{a_0}{\lambda}<<1$. On remarque donc que l'impédance due au rayonnement est très inférieure à l'impédance des ondes planes progressives.

        Essayons d'interpréter ce résultat : au bout d'un tuyau, les ondes sonores se propagent sur une demi-sphère de rayon $a_0$ (par exemple !). Avec notre modèle, on a alors une onde rayonnée dans l'atmosphère qui respecte la condition aux limites $\underline{Z}_R=\frac{\underline{p}_1}{\underline{v}_1}$ à l'embouchure. Sachant qu'avant l'embouchure, la structure de l'onde était plane, on a donc apparition d'un coefficient de réflexion de l'OPPH sur l'embouchure :
        \begin{equation*}
            r_{embouchure}=\frac{Z_{OPP}-\underline{Z}_R}{Z_{OPP} + \underline{Z}_R} \approx 1
        \end{equation*}
        car $\underline{Z}_R << Z_{OPP}$. En fait, tout se passe alors si on avait $Z_R=0$, soit $p_1=0$ à l'embouchure : c'est la condition utilisée lorsqu'on dit qu'un tuyau est "ouvert" (sur l'atmosphère). L'exercice explique donc pourquoi lorsqu'un tuyau est "ouvert" sur l'atmosphère, on peut donc considérer que la surpression est nulle à l'ouverture.

        }
    \end{quest}
\end{exo}

\begin{exo}[PSI][1][TD]{Tuyau d'orgue}
    \Notions{\item Approximations acoustiques \task Modes propres \task Intensité acoustique}
On considère un tuyau d'orgue d'axe $(Ox)$ rempli d'air de masse volumique $\rho_0$, où la célérité des ondes sonores vaut $c$. On note $p_1(x,t)$ la surpression acoustique et $v_1(x,t)$ la vitesse particulaire. La célérité du son est notée. L'extrémité du tuyau est fermée en $x = 0$ et ouverte en $x = L$. On cherche $p_1(x,t)$ sous la forme d'une onde plane stationnaire, de surpression $p_1(x,t) = p_{\rm m} \cos(\omega t) \cos(kx + \vphi)$.

\begin{quest}

	\item Rappeler le cadre des approximations acoustiques. Écrire les trois équations de l'acoustique : équation d'Euler, conservation de la masse et expression du coefficient de compressibilité. En déduire que $v_1$ vérifie l'équation de d'Alembert et exprimer la compressibilité $\chi_0$ en fonction des données du problème.
% \sol{
% Les approximations acoustiques sont valables si les variations de pression et de vitesse sont faibles devant les valeurs moyennes, soient $v_1 \ll c$ et $p_1 \ll p_0^$. Les équations de l'acoustique sont :
% \begin{liste}
%     \item Équation d'Euler linéarisée : $\rho_0 \dr{v_1}{t} = -\dr{p_1}{x}$,

    \item Déterminer la vitesse particulaire $v_1(x,t)$ en fonction de $p_{\rm m}$, $\rho_0$, $c$, $t$ et $x$.
\sol{
On utilise  l'équation d'Euler linéarisée pour obtenir la vitesse particulaire,
\begin{equation*}
    \rho_0 \dr{v_1}{t} = -\dr{p_1}{x} = p_{\rm m} k \cos(\omega t) \sin(kx + \vphi),
\end{equation*}
soit 
$$
v_1(x,t) = \frac{p_{\rm m} k}{\rho_0 \omega} \sin(\omega t) \sin(kx + \vphi).
$$
En utilisant la relation de dispersion, on trouve,
\eqbox{
v_1(x,t) = \frac{p_{\rm m}}{\rho_0 c} \sin(\omega t) \sin(kx + \vphi).
}
On remarque que la vitesse et la surpression sont déphasés dans le temps et l'espace, mais leurs amplitudes vérifient $p_{\rm m} = \rho c v_{\rm m}$ comme pour les OPPH.
}
    \item Justifier pourquoi l'on cherche une solution sous forme d'onde stationnaire. Écrire les conditions aux limites qui portent sur la surpression $p_1$ ou la vitesse $v_1$ en $x=0$ et $x=L$. En déduire la quantification du vecteur d'onde $k$ et la forme des modes propres. Déterminer la fréquence $\nu_0$ du fondamental et les fréquences des harmoniques.
\sol{
Les  conditions aux limites imposent un nœud de surpression ou de vitesse. Les seules solutions à présenter des nœuds sont les ondes stationnaires. Les conditions aux limites sont 
\begin{equation*}
    \begin{cases}
        v_1(x =0,t) = 0,\\
        p_1(x =L,t) = 0.
    \end{cases}
\end{equation*}
On en déduit $\vphi = 0~[\pi]$ pui $\cos(kL) = 0$, soit 
\eqbox{
k_n = \frac{(2n+1)\pi}{2L},~  n \in \mathbb{N}.
}
Le mode fondamental $n=0$ vérifie $k_0 = \frac{\pi}{2L}$, soit une longueur d'onde $\lambda = 4L$. Le mode $n=1$ vérifie $k_1 = \frac{3\pi}{2L}$, soit une longueur d'onde $\lambda = \frac{4L}{3}$. Les fréquences des modes propres vérifient $\nu_n = \frac{k_nc}{2\pi}$, soit

\eqbox{
\nu_n = \frac{c}{4L}(2n+1),~ n \in \mathbb{N}.
}
La fréquence du fondamental vaut 
\eqbox{
\nu_0 = \SI{140}{Hz}.
}
}

\item L'amplitude du déplacement des particules est $\xi_{\rm m} = \SI{0.4}{mm}$. Que vaut le déplacement $\xi(x,t)$ en fonction de la vitesse $v_1(x,t)$ ? En
déduire l'amplitude $p_m$ de la surpression acoustique à la fréquence fondamentale $\nu_0$. \AN déterminer l'intensité acoustique $I$ en dB sachant que la la pression de référence vaut $p_{\rm ref}=\SI{20}{\micro Pa}$.
\sol{
Par définition de la vitesse particulaire, on a 
$$
v_1 = \dr{\xi_1}{t} = \frac{p_{\rm m}}{\rho_0 c} \sin(\omega t) \sin(kx + \vphi).
$$
ce qui donne $\xi(x,t) = -\frac{p_{\rm m}}{\rho_0 \omega c} \cos(\omega t) \sin(kx + \vphi)$. L'amplitude de déplacement vaut donc $\xi_{\rm m} = \frac{p_{\rm m}}{\rho_0 \omega c}$, soit
\eqbox{
p_{\rm m} = \xi_{\rm m} \rho_0 \omega c = \SI{1.5e3}{Pa}
}
L'intensité acoustique s'obtient ensuite par 
% $$
\eqbox{
    I = 20 \log\left( \frac{p_{\rm m}}{p_{\rm ref}} \right) = \SI{120}{dB}.
    }
% $$
}

\end{quest}

 $\begin{array}{lll}
\textsc{Données :}
& \text{masse volumique} & :  \rho_0 = \SI{1.3}{kg.m^{-3}} \\
& \text{célérité du son} & : c = \SI{340}{m.s^{-1}} \\
& \text{longueur du tuyau} & : L=\SI{60}{cm}
\end{array}
$

\end{exo}

\begin{exo}[PSI][3][TD]{Onde sonore dans un pavillon}
    \Notions{\item Approximations acoustiques \task Conditions aux limites \task Équation de propagation }
Un pavillon est une surface de révolution d'axe $(Ox)$ et de section $S(x)$ variable (\Fig{pavillon}). Le pavillon a pour extrémités $x=0$ et $x=L$. On étudie la propagation d'une onde sonore dans ce pavillon.
%\begin{figure}[h!]
\begin{fig}[pavillon]
  \centering
  \includegraphics[scale=1]{pavillon.pdf}
  \capt{Représentation d'un pavillon sonore de section variable $S(x)$.}

\begin{quest}
	\item À l'équilibre la tranche de fluide comprise entre $x$ et $x+\mathrm{d}x$ a une masse volumique $\rho_0$ et une pression $p_0$. On note $\xi(x,t)$ le déplacement de cette masse élémentaire suite au passage de l'onde, avec $\frac{\partial \xi}{\partial x} \ll 1$. Déterminer la dilatation du fluide $\delta$, c'est-à-dire la variation relative de volume lors du passage de l'onde, en fonction de $S$, $\xi$ et de leurs dérivées spatiales.

% \end{equation*}

	\item La transformation du fluide est supposée isentropique. En déduire la surpression en fonction de $S$, $\xi$ et de leurs dérivées spatiales. % (attention ici $\mc{S}$ est l'entropie).
  % selon
\sol{
% \begin{equation*}
    On a 
    $$
    \chi \simeq - \frac{\delta}{p_1} \Rar p_1 = - \frac{\delta}{\chi_S}.
    $$

    % }
    $$
    En développant, on obtient aussi 
    % $$
    \eqbox{
        p_1 = -\frac{1}{\chi} \frac{\partial \xi}{\partial x} - \frac{\xi}{\chi} \frac{\d \ln S}{\d x}.
    }
    % $$
% \end{equation*}
}
	\item En appliquant le principe fondamental de la dynamique, trouver une relation entre $\xi(x,t)$ et $p_1(x,t)$.
  % vérifient l'équation
\sol{
% \begin{equation*,

    \eqbox{
        \rho_0 \frac{\partial^2 \xi}{\partial t^2} = - \frac{\partial p_1}{\partial x}
    }
% \end{equation*}
}
\item Montrer que l'équation de propagation de l'onde sonore est
	\begin{equation*}
	\frac{\partial^2 p_1}{\partial x^2} - \frac{1}{c^2}\frac{\partial^2 p_1}{\partial t^2} + \frac{\mathrm{d} \ln S}{\mathrm{d} x} \frac{\partial p_1}{\partial x} =0,
\end{equation*}
où l'on donnera l'expression de la célérité $c$. Comparer à l'équation de propagation libre.
\sol{
    On dérive l’expression de $p_1$ deux fois par rapport à $t$ et on utilise le PFD, soit 
    $$
    \ddr{p_1}{t} = -\frac{1}{\chi} \dr{}{x}\ddr{\xi}{t} - \frac{1}{\chi} \dd{\ln S}{x} \ddr{\xi}{t} = \frac{1}{\rho_0 \chi} \ddr{p_1}{x} + \frac{1}{\rho_0 \chi} \dd{\ln S}{x} \dr{p_1}{x}.
    $$
    On en déduit l’équation de propagation avec $c=1/\sqrt{\rho_0 \chi}$. L’équation de propagation libre est modifiée par les conditions de bord imposées par la géométrie du pavillon, et n’est plus celle de d’Alembert.
}
