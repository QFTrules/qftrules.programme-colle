\begin{exo}[PC][1][TD]{Résolutions d’équations différentielles}
Sans se soucier de la signification physique des symboles ci-dessous, résoudre les équations différentielles suivantes. Déterminer les constantes d’intégration lorsque les conditions aux limites sont données.
\begin{quest}
  \item $\ddd{T}{x} = 0$,  conditions aux limites $T(x=0)=T_1$ et $T(x=L)=T_2$.
  \sol{
  On procède en deux étapes.
  \begin{liste}
    \item \imp{Résolution de l'équation.} On intègre cette équation à une constante près, sans se soucier des bornes d'intégration. Ici l'équation différentielle linéaire  est d'ordre $2$, et il faut donc l'intégrer deux fois, ce qui donne
    $$
    \dd{T}{x} = A, \text{ puis } T(x) = A x + B,
    $$
    avec $A$ et $B$ deux constantes.
    \item \imp{Conditions aux limites.} On utilise les conditions aux limites pour déterminer les constantes d'intégration $A$ et $B$. Ici, l'énoncé impose
    $$
    \begin{cases}
      T(x=0)&=T_1 \\
      T(x=L)&=T_2
    \end{cases} \Rar   \begin{cases}
                          B&=T_1 \\
                          AL+B&=T_2
                        \end{cases} \Rar \begin{cases}
                                            A&=(T_2-T_1)/L \\
                                            B&=T_1
                                          \end{cases}
    $$
  \end{liste}
  On en déduit l'expression de $T(x)$,
  \eqbox{T(x) = (T_2-T_1)\frac{x}{L} + T_1.}
  }
  \item $\dd{p}{z} = - \frac{p}{H}$, avec $H>0$, conditions aux limites $p(z=0)=p_0$.
  \sol{
  Nous avons ici une équation différentielle linéaire d'ordre $1$. L'intégration est moins immédiate que dans le premier exemple car $p$ intervient dans les membres de gauche et de droite. Plusieurs méthodes sont possibles.
  \begin{enumerate}
    \item \udl{Séparation des variables.} On manipule les dérivées comme des taux d'accroissement, puis on intègre. En assimilant $\d p$ à une petite variation de $p$ et $\d z$ une petite variation de $z$, on peut écrire
    $$
    \dd{p}{z} = - \frac{p}{H} \Rar \frac{\d p}{p} = - \frac{\d z}{H}.
    $$
    On intègre ensuite en assimilant $\d p$ et $\d z$ à des mesures d'intégration. Comme une condition aux limites est donnée en $z=0$, il peut être pertinent d'intégrer sur la distance variant entre $0$ et $z$. Pour éviter des conflits de notation, on renomme $z'$ la variable d'intégration.
    $$
    \int_{p_0}^{p(z)} \frac{\d p}{p} = - \int_0^z \frac{\d z'}{H} \Rar \ln\left(\frac{p(z)}{p_0}\right) = - \frac{z}{H},
    $$
    d'où l'on tire
    \eqbox{p(z) = p_0 \,e^{-z/H}.}
    \item \udl{Recherche de solution exponentielle.} On peut chercher les solutions d'équations différentielles linéaires homogènes (sans second membre) sous la forme d'exponentielle. On pose $p(z)=Ae^{a z}$ avec $A$ et $a$ des constantes. On injecte ensuite cette expression dans l'équation différentielle pour déterminer $a$ :
    $$
    \dd{p}{z} = - \frac{p}{H} \Rar A a e^{az} = - \frac{A}{H}e^{az} \text{ pour tout z } \Rar a = -\frac{1}{H}.
    $$
    Nous avons trouvé une seule solution pour $a$. Comme l'équation différentielle est linéaire et d'ordre $1$, on sait que l'espace vectoriel des solutions est de dimension $1$, et donc on a trouvé toutes les solutions.
    On détermine ensuite $A$ avec les conditions aux limites, soit ici $p(z=0) = A = p_0$, d'où le résultat.
  \end{enumerate}
  }
    \item $\dd{u_C}{t}+\frac{u_C}{\tau}= 0$, avec $\tau > 0 $, condition initiale $u_C(t=0)=E$.
    \sol{
    \eqbox{u_c(t) = Ee^{-t/\tau}}
    }
    \item $\dd{u_C}{t}+\frac{u_C}{\tau}= \frac{E}{\tau}$,  avec $\tau,E > 0 $, condition initiale $u_C(t=0)=0$.
    \sol{
    \eqbox{u_c(t) = E(1-e^{-t/\tau})}
    }
  \item $\ddd{x}{t}+\omega_0^2 x= 0$,  avec $\omega_0 > 0 $, conditions initiales $x(t=0)=x_0$ et $\dd{x}{t}(t=0)=0$.
  \sol{
  \eqbox{x(t)=x_0 \cos(\omega_0 t).}
  }
  \item $\ddd{V}{x}-\frac{V}{\delta^2}= 0$,  avec $\delta > 0 $, conditions aux limites $V(x=-a)=-V_0$ et $V(x=a)=V_0$.
  \sol{
  \eqbox{V(x) = V_0 \frac{\sinh(x/\delta)}{\sinh(a/\delta)}}
  }
    \item $\ddd{x}{t}+ \frac{\omega_0}{Q} \dd{x}{t}+ \omega_0^2 x= 0$,  avec $\omega_0, Q>0 $, conditions initiales $x(t=0)=x_0$ et $\dd{x}{t}(t=0)=0$.
    \sol{
    La solution dépend du signe du discriminant et donc du régime. Revoir cours de PCSI.
    }
  \item $\ddd{f}{t}+2i\Omega\sin\lambda\,\dd{f}{t}+\omega_0^2 f=0$, avec $\omega_0>0$ et $ \Omega,\lambda \in \R$.
  \sol{
  Discriminant $\Delta = -4\Omega^2\sin^2\lambda-4\omega_0^2<0$, donc racines $-i\Omega \sin \lambda\pm i\omega$ avec $\omega=\sqrt{\Omega^2\sin^2\lambda+\omega_0^2}$. Solution de la forme
  \eqbox{
  f(t) = e^{-i\Omega \sin \lambda t}(A\cos(\omega t) + B \sin \omega t).
  }
  }
  \item $\dd{}{r}\left(r^2\dd{T}{r}\right)=0$, conditions aux limites $T(r=R) = T_0$ et $T(r=0)$ fini.
  \sol{
  Pour s'aider les premières fois, on peut poser $A = r^2\dd{T}{r}$. On voit alors que
  $$
\dd{A}{r}=0,
  $$
  ce qui indique que $A=\cte$ est une constante. On exprime alors la dérivée de $T$ restante en fonction de $A$ :
  $$
  \dd{T}{r} = \frac{A}{r^2},
  $$
  puis on primitive, ce qui donne
  \eqbox{T(r)=-\frac{A}{r}+B,}
   avec $B=\cte$ une constante.\\

   On utilise ensuite les conditions aux limites. Ici on peut voir directement que la condition aux limites $T(r=0)$ fini impose que $A=0$ ; dans le cas contraire, le terme $-\frac{A}{r}$ diverge en $r=0$ ! On utilise ensuite la condition $T(r=R)=B=T_0$, et on conclut que
   \eqbox{T(r) = T_0.}
  }
    \item $\frac{1}{r}\dd{}{r}\left(r\dd{v}{r}\right)=A$, avec $A>0$ constante, conditions aux limites $v(r=R) = 0$ et $v(r=0)$ fini.
    \sol{
    On a par intégrations successives :
    $$
\dd{}{r}\left(r\dd{v}{r}\right) = A r \Rar r\dd{v}{r} = A \frac{r^2}{2} + B \Rar \dd{v}{r} = A \frac{r}{2} + \frac{B}{r} \Rar v(r) = \frac{Ar^2}{2} + B \ln(r) + C.
    $$
    Le terme $B=0$ doit s'annuler, puis on applique la condition initiale $v(r=R)=0$. On trouve
    \eqbox{
    v(r) = \frac{A}{2}(r^2-R^2).
    }
    }
\end{quest}
\end{exo}

\begin{exo}[PC][1][colle]{Projection de vecteurs}
  Sans se soucier de la signification physique des symboles, décomposer les vecteurs représentées sur les schémas ci-dessous dans les bases définies sur le dessin.

  \begin{minipage}{0.45\textwidth}
    \includegraphics[scale=0.7]{pave.pdf}
    \begin{quest}
      \item Vecteurs $\Vec{N}$, $\Vec{T}$ et $\Vec{P}$ dans les bases $(\ex,\ez)$ et $(\ex’,\ez’)$.
    \end{quest}
  \end{minipage}\hfill
  \begin{minipage}{0.45\textwidth}
    \includegraphics[scale=0.7]{cerceau_projection.pdf}
  \begin{quest}
    \item Vecteurs $\Vec{OM}$ et $\Vec{P}$ dans les bases $(\ex,\ez)$ et $(\er,\et)$.
  \end{quest}
    \end{minipage}

\sol{
    \begin{enumerate}[label=\arabic*)]
      \item
      La base $(\ex',\ez')$ s'obtient à partir de $(\ex,\ez)$ par rotation d'angle $-\alpha$. On peut mieux visualiser cela en dessinant les bases à partir de la même origine, comme ci-dessous.

      \begin{center}
        \includegraphics{rotation_base}
      \end{center}  On a donc
      \begin{align*}[left=\empheqlbrace]
        \ex' & = \cos \alpha \,\ex -  \sin \alpha \,\ez, \\
        \ez' & = \sin \alpha \,\ex +  \cos \alpha \,\ez.
      \end{align*}
      Pensez à contrôler ces formules, en étudiant la limite où $\alpha \rightarrow 0$ (le cosinus s'approche de $1$ et donc $\ex$ s'aligne avec $\ex'$ et $\ez$ avec $\ez'$) et en vérifiant le signe des projections (l'angle $\alpha>0$ est non orienté ici et donc la composante suivant $\ez$ de $\ex'$ est négative, ce qui est cohérent avec le fait que $\ex'$ pointe plutôt vers le bas alors que $\ez$ pointe vers le haut).\\


      On exprime ensuite les vecteurs $\Vec{N}$, $\Vec{T}$ et $\Vec{P}$ dans la base la plus adaptée à chacun, soit $\Vec{N}=N\ez'$,  $\Vec{T}=-T\ex'$ et $\Vec{P} = -P\ez$, où $N$, $T$ et $P$ sont positifs. Puis on exprime les vecteurs $\ez'$, $\ex'$ et $\ez$ dans l'autre base, soit $\Vec{N} = N \cos \alpha \,\ex + N \sin \alpha \,\ez$, $\Vec{T} = - T \sin \alpha \,\ex + T \cos \alpha \,\ez $ et $\Vec{P} = P \sin \alpha \,\ex' - P \cos \alpha \,\ez'.$
      Ainsi,
      \eqbox{
      \left\{
      \begin{array}{lll}
        \Vec{N} & = N \cos \alpha \,\ex + N \sin \alpha \,\ez & = N\ez', \\
        \Vec{T} & = - T \cos \alpha \,\ex + T \sin \alpha \,\ez & = -T\ex' \\
        \Vec{P} & = -P\ez & = P \sin \alpha \,\ex' - P \cos \alpha \,\ez'.
      \end{array}
      \right.
      }

      \item On utilise la relation de Chasles pour $\Vec{OM}$.
      \eqbox{
      \left\{
      \begin{array}{lll}
        \Vec{OM} & = a(1+\cos \theta)\er -a \sin \theta \et\\
        \Vec{P} & = -P\ez = -P \sin \theta \,\er - P \cos \theta \,\et.
      \end{array}
      \right.
      }
    \end{enumerate}
    }

  \begin{minipage}{0.45\textwidth}
    \includegraphics[scale=1.1]{poids_projection.pdf}
    \begin{quest}
      \item Vecteurs $m \Vec{G_T}$, $m \Vec{g}$ et $\Vec{F}_{\rm ie}$ dans la base $(\ex,\ey)$.
    \end{quest}
  \end{minipage}\hfill
  \begin{minipage}{0.45\textwidth}
    \includegraphics[scale=0.7]{dipolaire.pdf}
  \begin{quest}
        \item Vecteurs $\Vec{NM}$ et $\Vec{PM}$  dans la base $(\er,\et)$.
  \end{quest}
    \end{minipage}

      \sol{
      \begin{enumerate}[label=3)]
      \item   On utilise un angle alterne-interne.
      \eqbox{
      \left\{
      \begin{array}{lll}
        \Vec{F}_{\rm ie} & = F_{\rm ie} \ex, \\
        m \Vec{g} & = -mg\cos(\alpha+\lambda) \ex - mg \sin(\alpha+\lambda) \ez, \\
        m \Vec{G_T} & = -m G_T \cos \lambda \ex - m G_T \sin \lambda \ey.
      \end{array}
      \right.
          }
          \end{enumerate}
          \begin{enumerate}[label=4)]
        \item
      \eqbox{
      \left\{
      \begin{array}{lll}
        \Vec{PM} & = \Vec{PO} + \Vec{OM} = - \frac{d}{2}\ez + r \er = (- \frac{d}{2}\cos \theta +r)\er + \frac{d}{2}\sin \theta \et , \\
        \Vec{NM} & = \Vec{NO} + \Vec{OM} = \frac{d}{2}\ez + r \er = (\frac{d}{2}\cos \theta +r)\er + \frac{d}{2}\sin \theta \et,
      \end{array}
      \right.
          }
        \end{enumerate}

          }


\end{exo}

\begin{exo}[PC][1][TD]{Produits vectoriels}
      Sans se soucier de la signification physique des symboles ci-dessous, calculer les produits vectoriels suivants et représenter tous les vecteurs dans l'espace.
    \begin{quest}
      \item $I\dl \pv \Vec{B}$ avec $\dl = r\d\theta\, \et$ et $\Vec{B} = B \ez$.
      \sol{
      \begin{liste}
        \item \imp{Méthode 1.}
      Les vecteurs sont décomposés ici dans la base cylindrique $(\er,\et,\ez)$. Dans cette base, on peut les représenter par les vecteurs colonnes suivants :
      $$
      I\dl = \begin{pmatrix}
        0 \\ Ir\d \theta \\ 0
    \end{pmatrix}, \qquad \B = \begin{pmatrix}
      0 \\ 0 \\ B
  \end{pmatrix}.
      $$
      On peut ensuite calculer le produit vectoriel avec la « règle du gamma »,
      $$
      I\dl \pv \Vec{B} =
       \begin{pmatrix}
        0 \\ Ir\d \theta \\ 0
    \end{pmatrix} \pv  \begin{pmatrix}
      0 \\ 0 \\ B
  \end{pmatrix} =  \begin{pmatrix}
    I r B \d \theta  \\ 0 \\ 0
\end{pmatrix},
      $$
      soit \eqbox{  I\dl \pv \Vec{B} = I r B \d \theta \er.}
      \item \imp{Méthode 2.} Lorsque les vecteurs n'ont qu'une seule composante dans une base donnée, comme c'est le cas ici, il est préférable d'utiliser le caractère orthonormé direct de la base. On écrit ici que  $\et \pv \ez = \er$, ce qui donne le résultat.
    \end{liste}
      }
      \item $\Vec{\mc{M}}\pv \Vec{B}$ avec $\Vec{\mc{M}} = \mc{M}\ez$ et $\Vec{B} = B\cos \theta \, \ez + B\sin \theta \, \ey$.
      \sol{
      Ici un des vecteurs a deux composantes, et on privilégiera alors la représentation en vecteurs colonnes, méthode 1. La méthode 2 fonctionne aussi bien entendu, mais on devra d'abord développer le produit vectoriel en somme de produits vectoriels plus simples, de la forme $\ez\pv \ez = \Vec{0}$ et $\ez \pv \ey = - \ex$.
      }
      \item $\Vec{OM}\pv m\Vec{v}$ avec $\Vec{OM} = r \er$ et $\Vec{v} = \dot{r} \er + r \dot{\theta} \et$.
      \item $\Vec{OM}\pv m\Vec{g}$ avec $\Vec{OM} = \ell \er$ et $\Vec{g} = - g\ez$.
      \item $2m \Vec{v} \pv \Vec{\Omega} $ avec $\Vec{\Omega} = -\Omega\cos \lambda\, \ex + \Omega \sin \lambda \,\ez$ et $\Vec{v} = \dot{x}\ex+\dot{y}\ey$.
      \sol{
      $\Fc = 2m \Vec{v} \pv \Vec{\Omega} \simeq -2m\Omega(-\cos \lambda \ex + \sin \lambda \ez) \pv (\dot{x}\ex + \dot{y}\ey) = -2m\Omega[-\cos\lambda\,\dot{y} \ez + \sin\lambda\,(\dot{x}\ey-\dot{y}\ex)]$
      }
      \item $\frac{\Vec{E}\pv \Vec{B}}{\mu_0}$ avec $\E=E_0 \cos(\omega t) \ex$ et $\B = B_0 \cos(\omega t) \ey$.
        \item $\Vec{k}\pv(\Vec{k}\pv\udl{\E})$ avec $\udl{\E}=E_0 e^{i(\omega t - kz)}\ex$ et $\Vec{k} = k\ez$.
    \end{quest}

\end{exo}

\begin{exo}[PC][1][colle]{Analyse dimensionnelle}
Déterminer la dimension physique et l’unité dans le système SI des grandeurs suivantes. Lorsque la dimension physique n'est pas parlante ou évidente, donner une grandeur physique de même dimension.
\sol{
Les sept grandeurs de base du système SI sont :
\begin{liste}
  \item la masse ${\rm M}$ d'unité $\SI{}{kg}$ ;
  \item la temps ${\rm T}$ d'unité $\SI{}{s}$ ;
  \item la longueur ${\rm L}$ d'unité $\SI{}{m}$ ;
  \item la température ${\rm \Theta}$ d'unité $\SI{}{K}$ ;
  \item l'intensité électrique ${\rm I}$ d'unité $\SI{}{A}$ ;
  \item la quantité de matière ${\rm N}$ d'unité $\SI{}{mol}$ ;
  \item l'intensité lumineuse ${\rm J}$ d'unité $\SI{}{cd}$ (le candela).
\end{liste}
Quelques remarques.
\begin{itemize}
  \item 
  Nous notons les dimensions physiques avec une police droite, e.g. $\rm M$ et non $M$, pour éviter de les confondre avec une grandeur physique de même symbole.
  \item Attention à ne pas confondre une dimension physique et une unité physique, e.g. $\rm m$ est l'unité de longueur mais $\rm M$ est la dimension \guill{masse}.
  \item 
  Pour déterminer la dimension de grandeurs physiques, penser à utiliser des lois, équations, ou formules connues qui relient la grandeur recherchée avec d'autres grandeurs plus simples, dont la dimension est connue.
\end{itemize}
}
\begin{quest}
    \item $\sqrt{\frac{k}{m}}$, avec $k$ raideur d'un ressort, et $m$ masse.
  \sol{
    On reconnaît la pulsation propre d'un ressort,
    $$
    \omega_0 = \sqrt{\frac{k}{m}}.
    $$
On a donc
  \eqbox{\left[\sqrt{\frac{k}{m}}\right] = {\rm T}^{-1}.}
  On peut retrouver la dimension de cette expression car la raideur d'un ressort a pour dimension
  $$
[k] = \frac{[F]}{\rm L} = \frac{\rm M\cdot L\cdot T^{-2}}{\rm L} = {\rm M \cdot T^{-2}}.
  $$
  }
  \item $\sqrt{\frac{T}{\mu}}$, avec $T$ tension d'une corde, et $\mu$ en $\SI{}{kg.m^{-1}}$.
  \sol{
    Il s'agit de la célérité des ondes mécaniques le long d'une corde vibrante,
    $$
    c = \sqrt{\frac{T}{\mu}}.
    $$
    On a donc
  \eqbox{\left[\sqrt{\frac{T}{\mu}}\right]=\SI{}{L.T^{-1}}.}
  On peut retrouver cette expression a partir des dimensions des grandeurs,
  $$
  [T] = [F] = {\rm M \cdot L \cdot T^{-2}},
  $$
  et
  $$
  [\mu] = {\rm M \cdot L^{-1}}.
  $$
  On a donc bien,
  $$
  \left[ \sqrt{\frac{T}{\mu}} \right] = \sqrt{\frac{{\rm M \cdot L \cdot T^{-2}}}{{\rm M \cdot L^{-1}}}} = \sqrt{{\rm L^2 \cdot T^{-2}}} = {\rm L\cdot T^{-1}}.
  $$

  }
  \item $RC$, avec $R$ résistance électrique, et $C$ capacité électrique.
  \sol{
    Proposons deux méthodes.
    \begin{enumerate}
      \item 
      L'équation différentielle vérifiée par la tension $u_C$ aux bornes d'un condensateur dans un circuit RC série s'écrit, en l'absence de source de tension,
      $$
\dd{u_c}{t} + \frac{u_c}{RC} = 0.
$$
   La grandeur à étudier est le temps caractéristique $\tau=RC$. On a donc 
   \eqbox{
   \left[ RC \right] = \rm T.
   }
   
   \item Pour le démontrer sans recourir à l'équation différentielle, utilisons les définitions de $R$ et $C$.
   \begin{liste}
    \item La résistance est définie par la loi d'Ohm $U=RI$ et a donc pour dimension 
    $$\left[R\right] = \SI{}{U.I^{-1}}.$$
    \item Une capacité électrique est définie par la caractéristique du condensateur $Q=CU$, et la charge est reliée à l'intensité électrique par $I=\dd{Q}{t}$.
    Une capacité électrique a donc pour dimension $$\left[C\right] = \SI{}{Q.U^{-1}} = \SI{}{I.T.U^{-1}}.$$
    
     On en déduit bien que
    \eqbox{\left[RC\right] = \SI{}{T}.}
  \end{liste}
\end{enumerate}
  }
  \item $\frac{1}{R}\sqrt{\frac{L}{C}}$, avec  $R$ résistance, $C$ capacité électrique, et $L$ auto-inductance.
  \sol{
    Proposons deux méthodes.
    \begin{enumerate}
      \item 
      La tension $u_C$ aux bornes d'une circuit RLC série vérifie, en l'absence de source de tension,
      $$
  \ddd{u_c}{t} + \frac{R}{L} \dd{u_c}{t} + \frac{u_c}{LC} = 0 \Longleftrightarrow 
  \ddd{u_c}{t} + \frac{\omega_0}{Q} \dd{u_c}{t} + \omega_0^2 u_c = 0
  $$
  La grandeur à étudier est le facteur de qualité, 
   $$Q=\frac{1}{R}\sqrt{\frac{L}{C}}.$$ 
   On voit d'après l'équation différentielle que le facteur de qualité est sans dimension,
   \eqbox{
   \left[ \frac{1}{R}\sqrt{\frac{L}{C}} \right] = 1.
   }
   \item Pour le démontrer sans recourir à l'équation différentielle, utilisons les définitions de $R$ et $C$. 
   \item La résistance est définie par la loi d'Ohm $U=RI$ et a donc pour dimension 
    $$\left[R\right] = \SI{}{U.I^{-1}}.$$
    \item Une capacité électrique est définie par la caractéristique du condensateur $Q=CU$, et la charge est reliée à l'intensité électrique par $I=\dd{Q}{t}$.
    Une capacité électrique a donc pour dimension $$\left[C\right] = \SI{}{Q.U^{-1}} = \SI{}{I.T.U^{-1}}.$$
    \item L'auto-inductance est définie par la caractéristique de la bobine à induction $U=L\dd{I}{t}$ et a donc pour dimension $$\left[L\right] = \SI{}{I^{-1}.U.T}.$$ Ainsi, on a 
   \eqbox{
     \left[ \frac{1}{R}\sqrt{\frac{L}{C}} \right] = \SI{}{U^{-1}.I}\cdot  \sqrt{\SI{}{U^2.I^{-2}}} = 1.
     }
\end{enumerate}
  }
  \item $\mu_0 \frac{N}{L} I$ avec $\mu_0$ perméabilité magnétique du vide, $N$ nombre, $L$ longueur, et $I$ intensité du courant.
  \sol{
  Il s'agit du champ magnétique créé par un solénoïde constitué de $n=\frac{N}{L}$ spires par unité de longueur,
  $$B = \mu_0 n I.$$
Un champ magnétique se mesure en tesla, de symbole $\rm T$. Pour trouver l'unité SI du tesla, penser à l'expression de la force de Lorentz magnétique 
$$\Vec{F} = q \Vec{v}\pv \Vec{B},$$ avec $q$ la charge et $\Vec{v}$ la vitesse. On sait qu'une force a pour dimension $[F] = \SI{}{M.L.T^{-2}}$. On a donc 
$$
[B] = \frac{[F]}{[qv]} = \frac{\SI{}{M.L.T^{-2}}}{\SI{}{Q.L.T^{-1}}} = \frac{\SI{}{M}}{\SI{}{I.T^2}},
$$
soit 
\eqbox{
\left[ \mu_0 \frac{N}{L} I\right] =  \SI{}{M.I^{-1}.T^{-2}}.
}

\udl{Remarque} : Attention à ne pas confondre l'unité tesla $\SI{}{T}$ avec la dimension du temps $\rm T$.
  }
  \item $\omega B S$ avec $B$ champ magnétique, $\omega$ pulsation, et $S$ surface.
  \sol{

  % \udl{Remarque} : une pulsation a pour dimension $[\omega]=\SI{}{T^{-1}}$. Néanmoins, une pulsation intervient souvent dans une fonction sinusoïdale de la forme $\sin(\omega t + \vphi)$, dont l'argument représente un angle. On utilise donc souvent l'unité des angles, le rad, pour la grandeur sans dimension $\omega t$, et donc $\omega$ se mesure souvent en $\SI{}{rad.s^{-1}}$. On ne vous en voudra pas si vous ne prenez pas compte l'unité rad dans une analyse dimensionnelle.
En utilisant les résultats des questions précédentes, on trouve
$$
  [\omega B S] = [\omega][B][S] = \SI{}{T^{-1}}\cdot\SI{}{M.I^{-1}.T^{-2}}\cdot\SI{}{L^2} = \SI{}{M.L^2.I^{-1}.T^{-3}}.
$$
  \eqbox{[\omega B S]=\SI{}{M.L^2.I^{-1}.T^{-3}}.}

    \udl{Remarque}  La grandeur $\omega B S$ a la dimension de la dérivée temporelle d'un flux magnétique $\Phi=BS$. On peut donc interpréter cette grandeur comme une tension électromotrice $e$, qui a la même dimension qu'une tension. On rappelle la loi de Faraday de l'induction
  $$
e= - \dd{\Phi}{t}.
  $$
  La dimension trouvée précédemment est donc aussi celle d'une tension $U$ ou d'un potentiel, soit 
  \eqbox{
  [U] = \SI{}{M.L^2.I^{-1}.T^{-3}}.
  }
  }

  % \item
  \item $\sqrt{\frac{\gamma R T}{M}}$, avec $R$ constante des gaz parfaits, $T$ température, et $M$ masse molaire, $\gamma$ sans dimension.
  \sol{
    Deux méthodes sont possibles.
    \begin{enumerate}
      \item 
      On reconnaît la célérité du son dans un gaz parfait,
      $$
      c = \sqrt{\frac{\gamma R T}{M}}.
      $$
      On a donc
      \eqbox{\left[\sqrt{\frac{\gamma R T}{M}}\right]=\SI{}{L.T^{-1}}.}
      \item 
      On peut retrouver cette expression a partir des dimensions de chaque grandeur.
      \begin{liste}
        \item le facteur adiabatique $\gamma$ est sans dimension, soit 
        $$
        [\gamma] = 1
        $$
        \item la constante des gaz parfaits $R$ intervient dans l'équation d'état des gaz parfaits,
        $$
        PV = nRT,
        $$
        avec $P$ la pression, $V$ le volume, $n$ le nombre de moles, et $T$ la température.
        On a donc 
        $$
        [RT] = \frac{[P][V]}{[n]} = \frac{\SI{}{M.L^{-1}.T^{-2}}\cdot\SI{}{L^3}}{\SI{}{N}} = \SI{}{M.L^2.T^{-2}.N^{-1}}.
        $$
        \item la masse molaire $M$ est la masse d'une mole de matière, soit
        $$
        [M] = \SI{}{M.N^{-1}}.
        $$
      \end{liste}
      On a donc bien 
      $$
      \left[ \sqrt{\frac{\gamma R T}{M}} \right] = \sqrt{\frac{[\gamma][R][T]}{[M]}} = \sqrt{\frac{1\cdot\SI{}{M.L^2.T^{-2}.N^{-1}}}{\SI{}{M.N^{-1}}}} = \sqrt{\SI{}{L^2.T^{-2}}} = \SI{}{L.T^{-1}}.
      $$
    \end{enumerate}
  }
  \item $\frac{1}{2}\rho v^2$, avec $\rho$ masse volumique, et $v$ vitesse.
  \sol{
  \eqbox{\left[\frac{1}{2}\rho v^2\right]=\SI{}{M.L^{-1}.T^{-2}}.}
  }
  % \item $\rho g z$, avec $\rho$ masse volumique, $g$ intensité de la pesanteur, et $z$ altitude.
  \item $\rho v S$, avec $\rho$ masse volumique, $v$ vitesse, et $S$ surface.
  \sol{
  \eqbox{\left[\rho v S\right] = \SI{}{M.T^{-1}}.}
  }
    % \item $\frac{L}{\omega}$ avec $L$ moment cinétique, et $\omega$ vitesse angulaire.
  \item $2m\Omega v$ avec $m$ masse, $\Omega$ vitesse angulaire, et $v$ vitesse.
  \sol{
  \eqbox{\left[2m\Omega v\right]=\SI{}{M.L.T^{-2}}.}
  }
  % \item $m\Omega^2 r$ avec $m$ masse, $\Omega$ vitesse angulaire, et $r$ distance.
  % \item $\frac{\lambda}{\rho c}$ avec $\lambda$ en $\SI{}{W.m^{-1}.K^{-1}}$, $\rho$ masse volumique, $c$ capacité calorifique massique.
  % \sol{
  % La grandeur $\lambda$ est appelée conductivité thermique. La puissance est une énergie par unité de temps. On rappelle que la dimension d'une énergie s'obtient par exemple à partir de l'énergie cinétique

  % \eqbox{\left[\frac{\lambda}{\rho c}\right] = \SI{}{L^2.T^{-1}}.}
  % }
  \item $\sqrt{\frac{2\hbar}{m\omega}}$, avec $\hbar$ en $\SI{}{J.s}$, $m$ masse, $\omega$ pulsation.
  \sol{
  La grandeur $\hbar = \SI{1.05e-34}{J.s}$ est une constante fondamentale appelée constante de Planck réduite. Elle intervient dans les phénomènes quantiques. Sa dimension est
  $$
[\hbar] = [E][T] = \SI{}{M.L^2.T^{-2}.T} = \SI{}{M.L^2.T^{-1}}.
  $$
  On en déduit
  \eqbox{\left[\sqrt{\frac{2\hbar}{m\omega}}\right] = \SI{}{L}.}
  \udl{Remarque :} nous verrons que la grandeur $a= \sqrt{\frac{2\hbar}{m\omega}}$ représente l'extension spatiale de la fonction d'onde dans un puits de potentiel harmonique de pulsation propre $\omega$.
  }
  \item $\frac{\hbar^2 k^2}{2m}$, avec $\hbar$ en $\SI{}{J.s}$, $m$ masse, $k$ vecteur d’onde.
  % \item $\sqrt{\frac{2}{\mu_0 \omega \sigma}}$ avec $\mu_0$ perméabilité magnétique du vide, $\omega$ pulsation, et $\sigma$ en $\SI{}{A^2.s^3.m^{-3}.kg^{-1}}$.
  \sol{
    On trouve
  \eqbox{\left[\frac{\hbar^2 k^2}{2m}\right] = \SI{}{E}=\SI{}{M.L.T^{-2}}.}
  La grandeur $E = \frac{\hbar^2 k^2}{2m}$ représente l'énergie d'une particule quantique de masse $m$ et de vecteur d'onde $k$.
  }
  \item $\frac{2\pi^5 \kB^4}{15 h^3 c^2}$, avec $k_B$ en $\SI{}{J.K^{-1}}$, $h$ en $\SI{}{J.s}$ et $c$ célérité de la lumière.
  \sol{
  La grandeur $h = 2\pi \hbar = \SI{6.63e-34}{J.s}$ est une constante fondamentale appelée constante de Planck. Elle intervient dans les phénomènes quantiques. La grandeur $k_{\rm B} = \SI{1.38e-23}{J.K^{-1}}$ est une constante fondamentale appelée constante de Boltzmann. Elle intervient en physique statistique. On trouve

  \eqbox{\left[\frac{2\pi^5 \kB^4}{15 h^3 c^2}\right] = \SI{}{M.T^{-3}.\Theta^{-4}}}
  La grandeur 
  $$
  \sigma = \frac{2\pi^5 \kB^4}{15 h^3 c^2}
  $$
  représente en fait la constante de Stefan, qui intervient dans la loi de Stefan donnant le flux surfacique radiatif d'une corps noir,
  $$\vphi_0 = \sigma T^4.$$
  }
\end{quest}

\begin{exo}[PC][1][TD]{Dérivées partielles}
  Sans se soucier de la signification physique des symboles ci-dessous, calculer les dérivées partielles suivantes.
  \begin{quest}
    \item  $\dr{B}{t}$ et $\dr{B}{x}$ avec $B(x,t) = B_0 \cos(\omega t - kx)$, avec $B_0, \omega$ et $k$ des constantes.
    \sol{
    Une dérivée partielle d'ordre $1$ fonctionne comme une dérivée classique en fixant par la pensée toutes les autres variables à des constantes. Par exemple, pour la première dérivée, $t$ est considérée comme une variable et $x$ comme une constante, soit
    $$
\dr{B}{t} = \dr{}{t}( B_0 \cos(\omega t - kx)) = \dd{}{t}(B_0 \cos(\omega t + \cte )) = -B_0 \omega \sin(\omega t + \cte ),
    $$
    soit \eqbox{\dr{B}{t}= - B_0 \omega \sin (\omega t - k x).}
    De même pour la seconde dérivée partielle, \eqbox{\dr{B}{x}= B_0 k \sin (\omega t - k x).}
    }
    \item $\dr{p}{z}$ avec $p(x,z) = p_0 + A x + \rho g z$, avec $p_0$ et $A$ des constantes.
    \sol{
    \eqbox{
    \dr{p}{z} = \rho g.
    }
    }
    \item $\dr{B}{r}$ et $\dr{B}{z}$ avec $B(r,z) = B_0 \frac{r}{R^2+z^2}$, avec $B_0$ et $R$ des constantes.
    \sol{
    
      \eqbox{
        \dr{B}{r} = B_0 \frac{1}{R^2+z^2}.
        }
        \eqbox{
          \dr{B}{z} = - 2B_0 \frac{rz}{(R^2+z^2)^2}.
          }
    }
    \item  $\dr{E}{t}$ et  $\dr{E}{z}$ avec $E(x,t) = E_0 e^{-z/\delta} \cos(\omega t - z/\delta)$, avec $E_0, \omega$ et $\delta$ des constantes.
    \sol{
    \eqbox{
    \dr{E}{t} = -\omega E_0 e^{-z/\delta} \sin(\omega t - z/\delta)
    }
    \eqbox{
    \dr{E}{z} = - \frac{1}{\delta} E_0 e^{-z/\delta} (\sin(\omega t - z/\delta) - \cos(\omega t - z/\delta))
    }
    }
    \item $\dr{\psi}{t}$ avec $\psi(x,t) = Ae^{-x^2/a^2}e^{-iEt/\hbar}$, avec $A,a,\hbar,m,E$ des constantes.
    \sol{
      \eqbox{
      \dr{\psi}{t} = -\frac{iE}{\hbar} \psi.
      }
    }
    \item $\dr{n}{x}$ et $\dr{n}{t}$ avec $n(x,t) = \frac{A}{\sqrt{t}} e^{-x^2/4Dt}$, avec $A$ et $D$ des constantes.
    \sol{
      \eqbox{
      \dr{n}{x} = -\frac{x}{2Dt} \times \frac{A}{\sqrt{t}}  e^{-x^2/4Dt}
      }
      On a aussi 
      $$
    \dr{n}{t} = -\frac{A}{2t^{3/2}} e^{-x^2/4Dt} + \frac{A}{\sqrt{t}} \frac{x^2}{4Dt^2} e^{-x^2/4Dt},
      $$
      ce qu 'on peut factoriser comme
      \eqbox{
      \dr{n}{t} = -\frac{A}{2t^{3/2}} e^{-x^2/4Dt} \left( 1- \frac{x^2}{2Dt} \right).
      }
    }
  \end{quest}
Pour une fonction scalaire de plusieurs variables $f(x,y,...)$, on définit les dérivées partielles d’ordre 2 par
\begin{equation}
  \ddr{f}{x} = \dr{}{x}\left(\dr{f}{x}\right), \quad  \ddr{f}{y} = \dr{}{y}\left(\dr{f}{y}\right), \quad \text{etc.}
\end{equation}
Calculer les dérivées partielles d’ordre deux suivantes.
\begin{quest}
    \item  $\ddr{p}{x}$ et $\ddr{p}{t}$ avec $p(x,t) = p_0 \cos(\omega t - kx + \vphi)$, avec $p_0,\omega,k$ et $\vphi$ des constantes.
    \sol{
      On calcule la dérivée première,
      $$
      \dr{p}{x} = k p_0 \sin(\omega t - kx + \vphi),
      $$
      puis la dérivée seconde,
      $$
      \ddr{p}{x} = \dr{}{x} \left( k p_0 \sin(\omega t - kx + \vphi) \right) = -k^2 p_0 \cos(\omega t - kx + \vphi),
      $$
      soit 
      \eqbox{
      \ddr{p}{x} = -k^2 p.
      }
      On précède de même pour la dérivée temporelle, soit 
      \eqbox{
      \ddr{p}{t} = -\omega^2 p.
      }
    }
  \item  $\ddr{\psi}{x}$ avec $\psi(x,t) = Ae^{-x^2/a^2}e^{-iEt/\hbar}$, avec $A,a,\hbar,m,E$ des constantes.
  \sol{
    La dérivée première donne 
    $$
    \dr{\psi}{x} = -\frac{2x}{a^2} \psi.
    $$
    On peut ensuite calculer la dérivée seconde,
    $$
    \ddr{\psi}{x} = \dr{}{x} \left( -\frac{2x}{a^2} \psi \right) = -\frac{2}{a^2} \psi - \frac{2x}{a^2} \dr{\psi}{x} = -\frac{2}{a^2} \psi + \frac{4x^2}{a^4} \psi.
    $$
    On en déduit finalement
    \eqbox{
    \ddr{\psi}{x} = -\frac{2}{a^2}\left( 1 - \frac{2x^2}{a^2} \right)  \psi .
    }
  }
\end{quest}
\end{exo}

\begin{exo}[PC][1][TD]{Développements limités}
Effectuer les développements limités des expressions suivantes à l’ordre $2$ en le petit paramètre $x$.
\begin{quest}
\item  $\ln\left(1+\frac{e}{R}\right)$ en fonction de $x=\frac{e}{R}$.
\sol{
On se ramène aux formules connues de développement limité. Ici on suppose que $e \ll R$, ce qui implique que $x=\frac{e}{R}$ est un « petit » paramètre. Un petit paramètre est toujours adimensionné. On peut alors utiliser le développement limité de $\ln(1+x)$, ce qui donne $\ln(1+x) \simeq x - \frac{x^2}{2}$, soit en remplaçant $x$ par son expression,
\eqbox{
\ln\left(1+\frac{e}{R}\right) \simeq \frac{e}{R} - \frac{e^2}{2R^2}.
}
\udl{Remarque} : mathématiquement, le développement limité d'une fonction $f(x)$ revient à approximer cette fonction à son développement de Taylor tronqué à un certain ordre. Le développement de Taylor s'écrit avec un \guill{petit o} ou un \guill{grand o}, e.g.
\eqbox{
\ln\left(1+x\right) = x - \frac{x^2}{2} + O\left(x^3\right) = x - \frac{x^2}{2} + o(x^2).
}
}
  \item  $\left(1-\frac{d}{r} \cos \theta\right)^{-1/2}$ en fonction de $x=\frac{d}{r} \cos \theta$.
  \sol{
  \eqbox{\left(1-\frac{d}{r} \cos \theta\right)^{-1/2} = 1+ \frac{1}{2}\frac{d}{r} \cos \theta + \frac{3}{8}\frac{d^2}{r^2} \cos^2 \theta}
  }
  \item  $\left(r-d\cos \theta\right)^{-1/2}$ en fonction de $x=\frac{d}{r} \cos \theta$.
  \sol{
  On ne peut pas appliquer directement les formules de développement limité sur cette expression. Il faut d'abord factoriser par $r$ dans la parenthèse pour faire apparaître le petit paramètre $\frac{d}{r} \cos \theta$. On trouve alors
  $$
  \left(r-d\cos \theta\right)^{-1/2} = \left(r \left(1- \frac{d \cos \theta}{r} \right) \right)^{-1/2} = \left(r \left(1-x\right) \right)^{-2} = r^{-1/2} \left(1-x\right)^{-1/2}.
  $$
  On peut ensuite appliquer la formule du développement limité de $\left(1-x\right)^{-1/2}$, soit
  $$
  r^{-1/2} \left(1-x\right)^{-1/2} \simeq r^{-1/2} \left(1 +\frac{x}{2} + \frac{3}{8}x^2\right),
  $$
  et donc
  \eqbox{\left(r-d\cos \theta\right)^{-1/2} \simeq r^{-1/2} \left(1 +\frac{d \cos \theta}{2 r} + \frac{3}{8}\frac{d^2 \cos^2\theta}{r^2}\right).}
  }
\item  $\left(1+\frac{e}{R}\right)^{-1}$ en fonction de $x=\frac{e}{R}$.
\sol{
\eqbox{
\left(1+\frac{e}{R}\right)^{-1} \simeq 1-\frac{e}{R} + \frac{e^2}{R^2}.
}
}
\item  $\frac{R^2}{R+e}$ en fonction de $x=\frac{e}{R}$.
\sol{
On a $\frac{R^2}{R+e} = R \frac{1}{1+e/R} = R(1-\frac{e}{R} + \frac{e^2}{R^2})$, soit
\eqbox{
\frac{R^2}{R+e} \simeq R-e + \frac{e^2}{R}.
}
}
  \item  $\left(1-\frac{d}{r} + \frac{d^2}{r^2}\right)^{-1/2}$ en fonction de $x=\frac{d}{r}$.
  \sol{
  On pose $y = -\frac{d}{r} + \frac{d^2}{r^2}$. Alors
$\left(1-\frac{d}{r} + \frac{d^2}{r^2}\right)^{-1/2} = \left(1+y\right)^{-1/2} \simeq 1-\frac{y}{2}+\frac{3}{8}y^2 = 1+\frac{d}{r}-\frac{d^2}{r^2} + \frac{3}{8}\frac{d^2}{r^2}$, soit
\eqbox{
\left(1-\frac{d}{r} + \frac{d^2}{r^2}\right)^{-1/2} \simeq 1+\frac{d}{r}-\frac{5}{8}\frac{d^2}{r^2}.
}
  }
\end{quest}
\end{exo}
