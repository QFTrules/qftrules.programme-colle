\section{Points de vue des bilans macroscopiques}
Il existe deux méthodes équivalentes pour réaliser des bilans macroscopiques en mécanique des fluides: considérer un système ouvert mais fixe (indéformable), ou considérer un système fermé mais mobile (déformable). Nous avons vus pour les systèmes thermodynamiques ouverts qu’il est préférable de se ramener à un système fermé pour appliquer les théorèmes généraux de la thermodynamique. Il en va de même pour les théorèmes de la mécanique.

\subsection{Bilan de masse}
\parag{Système ouvert fixe}
Ce point de vue correspond à un point de vue eulérien: dans le référentiel du laboratoire, on prend pour système \imp{ouvert}  un volume fixe $\So$. Dans la suite, on considérera toujours des écoulements \imp{unidirectionnels}, et on supposera que toutes les grandeurs sont uniformes sur une section donnée de l'écoulement. Pour illustrer ce point de vue, effectuons un bilan de masse sur $\So$.
%%%%%%%%%%%%%%%%%%%%%%%
\begin{figure}[h!]
	\centering
	\includegraphics[width=0.45\textwidth]{syst_ouvert.pdf}
	\caption{Le système $\So$ est un système ouvert et fixe (indéformable). Durant $\d t$, il y rentre la quantité de fluide présente dans le bloc bleu, de masse $\delta m_{\rm{e}}$, et il en sort la quantité de fluide présente dans le bloc rouge, de masse $\delta m_{\rm{s}}$.}
	\label{fig:système_ouvert}
\end{figure}
%%%%%%%%%%%%%%%%

Durant $\d t$, il entre une masse $\delta m_{\rm{e}}$ et il sort une masse $\delta m_{\rm{s}}$ (Fig.~\ref{fig:système_ouvert}).
En notant $v_{\rm{e,s}}$ la vitesse du fluide entrant ou sortant, $\rho_{\rm{e,s}}$ sa masse volumique et $S_{\rm{e,s}}$ la section d'entrée ou de sortie, on a donc
\begin{equation}
\begin{split}
\d m_{\So} =
m_{\So}(t+\mathrm{d}t)-m_{\So}(t) = \delta m_{\rm{e}} - \delta m_{\rm{s}} = \rho_{\rm{e}}\times S_{\rm{e}}v_{\rm{e}}\d t-\rho_{\rm{s}}\times S_{\rm{s}}v_{\rm{s}}\d t.
\end{split}
\end{equation}
On reconnaît les débits massiques d’entrée $\Dm^{\rm{(e)}} = \rho_{\rm{e}} S_{\rm{e}}v_{\rm{e}} $ et de sortie $\Dm^{\rm{(s)}} = \rho_{\rm{s}} S_{\rm{s}}v_{\rm{s}}$. On trouve alors
\begin{equation}\label{eq:bilan_mo}
\frac{\d m_{\So}}{\d t}=\Dm^{\rm{(e)}}-\Dm^{\rm{s}},
\end{equation}
en accord avec la définition du débit massique.
% Retrouvons cette équation en effectuant un bilan sur un système fermé.

%En régime stationnaire, $\frac{\d m_{\So}}{\d t}=0$ et donc le débit massique est conservé ($\Dm^{\rm{(e)}}=\Dm^{\rm{(s)}}$).

\parag{Système fermé mobile}
Ce point de vue correspond à un point de vue lagrangien: dans le référentiel du laboratoire, on prend comme système \imp{fermé} mais mobile (déformable) le volume de contrôle $\Sigma_0$, plus la masse entrante ou sortante.
%L’ensemble est noté $\Sf$.
Pour le système ouvert $\So$ précédent étudié entre deux instants $t$ et $t+\d t$, nous définissons le système fermé $\Sf$ de la façon suivante.
\begin{definition}[Système fermé mobile]\label{def:sigmaf}
	Soit $\So$ un système ouvert (volume fixe). On définit le système fermé associé $\Sf$ par :
	\begin{liste}
		\item $\Sf(t) = \So(t) \cup \delta m_{\rm e}$ (masse entrante dans $\So$ entre $t$ et $t+\d t$) ;% (en tireté rouge sur la \Fig{ouvert});
\item $\Sf(t+\d t) = \So(t+\d t) \cup \delta m_{\rm s}$ (masse sortante dans $\So$ entre $t$ et $t+\d t$).%(en tireté jaune sur la \Fig{ouvert}).
\end{liste}
\end{definition}

%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{figure}[h!]
	\centering
	\includegraphics[width=0.45\textwidth]{syst_fermé.pdf}
	\caption{Le système $\Sf$ est un système fermé et mobile (déformable) qui suit l'écoulement. À l'instant $t$, il est constitué du bloc central qui correspondait à $\So$ et du volume entrant de fluide de masse $\delta m_{\rm{e}}$ (bleu). À l'instant $t+\d t$, il est constitué du bloc central qui correspondait à $\So$ et du volume sortant de fluide de masse $\delta m_{\rm{s}}$ (en rouge).}
	\label{fig:système_fermé}
\end{figure}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%Le volume de ce système dépend donc du temps, mais rien n'y rentre et rien n'en sort au cours du temps, c'est simplement que le système se déforme.

\begin{methode}[Bilan de masse]
	\item Définir soigneusement les systèmes ouvert $\So$ et fermé $\Sf$.
	\item Exprimer $m_{\Sf}$ en fonction de $m_{\So}$ et des masses entrante $\delta m_e$ et sortante $\delta m_s$.
	\item Utiliser le caractère fermé de $\Sf$ pour justifier que $m_{\Sf}(t)= m_{\Sf}(t+\d t)$.
	\item Utiliser le caractère stationnaire de l'écoulement pour justifier que $m_{\So}(t)= m_{\So}(t+\d t)$.
\end{methode}
\begin{enumerate}
	\item Les systèmes $\So$ et $\Sf$ ont déjà été défini.
	\item Pour réaliser le bilan de masse sur $\Sf$, on détermine séparément la masse aux instants $t$ et $t+\d t$ : %Réécrivons le bilan de masse précédent :
	\begin{equation}
	\begin{cases}
	m_{\Sf}(t)=m_{\So}(t)+\delta m_{\rm{e}}=m_{\So}(t)+\rho_{\rm{e}}S_{\rm{e}}v_{\rm{e}}\d t, \\
	m_{\Sf}(t+\d t)=m_{\So}(t+\d t)+\delta m_{\rm{s}}=m_{\So}(t+\d t)+\rho_{\rm{s}}S_{\rm{s}}v_{\rm{s}}\d t.
	\end{cases}
	\end{equation}
	\item Or $\Sf$ est fermé par définition, donc sa masse est constante
\begin{equation}
	m_{\Sf}(t)= m_{\Sf}(t+\d t).
\end{equation}
On trouve alors
	$
m_{\So}(t)+\rho_{\rm{e}}S_{\rm{e}}v_{\rm{e}}\d t = m_{\So}(t+\d t)+\rho_{\rm{s}}S_{\rm{s}}v_{\rm{s}}\d t
$ puis $m_{\So}(t+\d t)-m_{\So}(t) = (\Dm^{\rm{(e)}}-\Dm^{\rm{s}})\d t$, soit
\begin{equation}
\frac{\d m_{\So}}{\d t}=\Dm^{\rm{(e)}}-\Dm^{\rm{s}}.
\end{equation}
On retrouve bien l'\Eq{bilan_mo}. Les deux points de vue sont donc bien équivalents.
\item Lorsque l'écoulement est stationnaire, on a de plus
\begin{equation}
	m_{\So}(t) = m_{\So}(t+ \d t) \Rar \frac{\d m_{\So}}{\d t}=0 \Rar \Dm^{\rm{(e)}}=\Dm^{\rm{s}}.
\end{equation}
On retrouve que le débit massique est conservé le long d'un tube de courant (ici le volume fixe $\So$).
\end{enumerate}

\subsection{Bilans de quantité de mouvement et d'énergie}
L’atout principal des systèmes fermés est qu'ils permettent d’appliquer directement les théorèmes fondamentaux de la mécanique.
Contrairement à la masse, qui est une grandeur \imp{conservative}, les grandeurs cinétiques ne sont pas conservées en général.
% \begin{methode}[Bilan d'une grandeur non conservée]
% 	\item Définir soigneusement les systèmes ouvert $\So$ et fermé $\Sf$.
% 	\item Calculer la grandeur pour $\Sf$ aux instants $t$ et $t+\d t$.
% 	\item Appliquer le théorème fondamental de la mécanique.
% \end{methode}

\begin{liste}
\item Bilan de quantité de mouvement : on calcule $\vv{p}_{\Sf}(t+\d t)$ et $\vv{p}_{\Sf}(t)$, puis on applique le PFD :
\begin{equation}
\frac{\d\vv{p}_{\Sf}}{\d t}=\frac{\vv{p}_{\Sf}(t+\d t)-\vv{p}_{\Sf}(t)}{\d t}=\vv{F}_{\rm ext},
\end{equation}
où $\vv{F}_{\rm ext}$ désigne la résultante des forces extérieures que subit le fluide (cela exclut donc les forces de contact internes aux fluide).
\item Bilan d’énergie : on calcule par exemple les énergies cinétiques $E_{\rm c,\Sf}(t+\d t)$ et $E_{\rm c,\Sf}(t)$, puis on applique le TEC :
\begin{equation}
\frac{\d E_{\rm{c},\Sf}}{\d t}=\frac{E_{\rm{c},\Sf}(t+\d t)-E_{\rm{c},\Sf}(t)}{\d t}=P_{\rm{ext}}+P_{\rm{int}},
\end{equation}
où $P_{\rm{ext}}$ est la puissance fournie par les forces extérieures, et $P_{\rm{int}}$ est la puissance fournie par les forces intérieures (en générale, la dissipation visqueuse).
On admettra la propriété suivante.
\begin{prop}\label{prop:}
Dans un écoulement \imp{parfait et incompressible}, la puissance des forces intérieures $P_{\rm{int}}$ est nulle.
\end{prop}


\end{liste}

\begin{remarque}
	On peut également effectuer un bilan de moment cinétique, mais ce n'est pas au programme. On calcule les moments cinétiques $\vv{L}_{O,\Sf}(t+\d t)$ et $\vv{L}_{\Sf}(t)$ par rapport à un point $O$ fixe, puis on applique le TMC :
	\begin{equation}
	\frac{\d\vv{L}_{O,\Sf}}{\d t}=\frac{\vv{L}_{O,\Sf}(t+\d t)-\vv{L}_{O,\Sf}(t)}{\d t} = \vv{\mathcal{M}}_{O,\rm ext},
	\end{equation}
	où $\vv{\mathcal{M}}_{O,\rm ext}$ désigne la résultante des moments des forces extérieures que subit le fluide.
\end{remarque}

\section{Exemples de bilans macroscopiques}
\subsection{Le tuyau coudé}
On considère une canalisation coudée de section $S=\cte$ (\Fig{bilans_tuyau}). On suppose le fluide homogène de masse volumique $\rho$ et l'écoulement parfait et stationnaire. On néglige l'influence de la pesanteur.
\begin{figure}[H]
	\centering
	\includegraphics[width=0.4\textwidth]{tuyau.pdf}
	\caption{Bilan de quantité de mouvement sur une canalisation coudé.}
	\label{fig:bilans_tuyau}
\end{figure}

\parag{Propriétés de l'écoulement}
Caractérisons tout d'abord les variables de sortie en fonction de celles d'entrée. On applique pour cela la méthode d'analyse des écoulements parfait.
\begin{enumerate}
	\item Le fluide est homogène donc l'écoulement est incompressible et le débit volumique $\DV = v S$ est conservé, et puisque $S=\cte$ on a
\begin{equation}
	v_{\rm{s}}=v_{\rm{e}}=v.
\end{equation}
\item L'écoulement est aussi stationnaire, donc toutes les hypothèses du théorème de Bernoulli sont réunies. On peut l'appliquer le long de n'importe quelle ligne de courant dans la canalisation, et il vient donc
\begin{equation}
	p_{\rm{s}}=p_{\rm{e}}=p.
\end{equation}
\end{enumerate}

\parag{Bilan de quantité de mouvement}
 On souhaite calculer la force que le fluide, de l'eau par exemple, exerce sur la canalisation.
 Pour cela, réalisons un bilan de quantité de mouvement dans le référentiel du laboratoire supposé galiléen.
 \begin{methode}[Bilan de quantité de mouvement]
 	\item Définir soigneusement le référentiel $\Rcal$, le système ouvert $\So$ et le système fermé $\Sf$.
 	\item Exprimer $\vv{p}_{\Sf}$ aux instants $t$ et $t+\d t$ en fonction de $\vv{p}_{\So}$ et des grandeurs entrante et sortante.
	\item Utiliser la stationnarité de l'écoulement pour justifier que $\vv{p}_{\So}(t)=\vv{p}_{\So}(t+\d t)$.
 	\item Appliquer le PFD au système fermé $\Sf$, soit
	$$
\frac{\d \vv{p}_{\Sf}(t)}{\d t}=\vv{F}_{\rm{ext}}.
	$$
 \end{methode}
 \begin{enumerate}
 	\item On définit $\So$ comme le partie coudée fixe de la canalisation. Le système fermé $\Sf$ mobile, est constitué à l'instant $t$ de la partie bleue et de $\So$, et à l'instant $t+\d t$ de la partie rouge et de $\So$. Le référentiel $\Rcal$ est celui du laboratoire supposé galiléen.
	\item On calcule
	\begin{equation}
	\begin{cases}
	\vv{p}_{\Sf}(t)=\vv{p}_{\So}(t)+\delta\vv{p}_{\rm{e}},\\
	\vv{p}_{\Sf}(t+\d t)=\vv{p}_{\So}(t+\d t)+\delta\vv{p}_{\rm{s}}.
	\end{cases}
	\end{equation}
	Or  $\delta \Vec{p_{\rm{e}}} = \delta m_e v_e \ex = \rho S v^2 \d t \ex$, et de même $\delta\vv{p}_{\rm{s}} = \delta m_s v_s \ex = \rho S v^2 \d t \ey$.
	\item En régime stationnaire, on a $\vv{p}_{\So}(t)=\vv{p}_{\So}(t+\d t)$ et donc
	\begin{equation}\label{eq:p_stat}
		\vv{p}_{\Sf}(t+\d t) - \vv{p}_{\Sf}(t) = \delta\vv{p}_{\rm{s}} - \delta\vv{p}_{\rm{e}} = \rho S v^2 \d t (\ey - \ex) \Rar  \frac{\d \vv{p}_{\Sf}(t)}{\d t}=\rho S v^2\left(\vv{e_y}-\vv{e_x}\right).
	\end{equation}

%  et donc on obtient par différence entre les deux équations $\d \vv{p}_{\Sf}(t)=\delta\vv{p}_{\rm{s}}-\delta\vv{p}_{\rm{e}}$.
% Or d'après l'étude préliminaire, $\delta p_{\rm{s}}=\delta p_{\rm{e}}=(\rho S v\d t)\times v$ (attention ici ce sont les normes des quantités de mouvement, pas les pressions !) et avec les vecteurs
% \begin{equation}
%
% \end{equation}

% \paragraph{Étape 2: utilisation d'un théorème fondamental.}
\item On applique le PFD au système fermé $\Sf$,
\begin{equation}\label{eq:PFD}
% \begin{split}
\frac{\d \vv{p}_{\Sf}(t)}{\d t}=\vv{F}_{\rm{ext}}=\vv{F}_{\rm{can\rightarrow eau}}+\Vec{F}_p,
% \end{split}
\end{equation}
où $\Vec{F}_p$ désigne la résultante des forces de pression. Elle comprend la force de pression exercée sur le fluide entrant $p_e S \ex $ et celle exercée sur le fluide sortant $-p_s S \ey$, soit $\Vec{F}_p = pS(\ex-\ey)$.
\end{enumerate}
% car d'après l'étude préliminaire $P_{\rm{e}}=P_{\rm{s}}=P$.

% \begin{remarque} C'est bien la force exercée par la canalisation sur l'eau qui intervient dans le PFD puisque le système étudié c'est justement le fluide en écoulement. On obtiendra la force recherchée exercée par l'eau sur la canalisation avec le principe des actions réciproques.
% \end{remarque}
On combine les \Eq{p_stat} et \Eq{PFD} pour trouver
\begin{equation}
	\vv{F}_{\rm{can\rightarrow eau}} = \frac{\d \vv{p}_{\Sf}(t)}{\d t} - \Vec{F}_p = \left(p+\rho v^{2}\right)S\left(\vv{e_y}-\vv{e_x}\right)
\end{equation}
On conclut en utilisant le principe des actions réciproques,
\begin{equation}
\vv{F}_{\rm{eau}\rightarrow \rm{can}}=-\vv{F}_{\rm{can\rightarrow eau}} = \left(p+\rho v^{2}\right)S\left(\vv{e_x}-\vv{e_y}\right).
\end{equation}

\subsection{Le moulin à eau}
On étudie le principe d'un moulin à eau, schématisé en Fig.~\ref{fig:bilans_moulin}. On souhaite déterminer la vitesse de rotation $\omega$ en fonction de la vitesse d'entrée et du couple $\Gamma$ que le fluide exerce sur le moulin. On applique pour cela un bilan d'énergie cinétique.

\begin{figure}[h!]
	\centering
	\includegraphics[width=0.5\textwidth]{moulin2.pdf}
	\caption{Principe du moulin à eau.}
	\label{fig:bilans_moulin}
\end{figure}

\begin{methode}[Bilan d'énergie cinétique]
 \item Définir soigneusement le référentiel $\Rcal$, le système ouvert $\So$ et le système fermé $\Sf$.
 \item Exprimer $E_{c,\Sf}$ aux instants $t$ et $t+\d t$ en fonction de $E_{c,\So}$ et des grandeurs entrante et sortante.
 \item Utiliser la stationnarité de l'écoulement pour justifier que $E_{c,\So}(t)=E_{c,\So}(t+\d t)$.
 \item Appliquer le TEC au système fermé $\Sf$, soit
 $$
\frac{\d E_{c,\Sf}(t)}{\d t}=\mc{P}_{\rm{ext}} + \mc{P}_{\rm{int}},
 $$
 avec $\mc{P}_{\rm{int}}=0$ si l'écoulement est parfait et incompressible.
\end{methode}

\begin{enumerate}

		\item On a
		\begin{equation}
		\begin{cases}
		E_{\rm{c},\Sf}(t)=E_{\rm{c},\So}(t)+\delta E_{\rm{c},\rm{e}},\\
		E_{\rm{c},\Sf}(t+\d t)=E_{\rm{c},\So}(t+\d t)+\delta E_{\rm{c},\rm{s}},
		\end{cases}
		\end{equation}
		avec $\delta E_{\rm{c},\rm{e}} = \frac{1}{2} \delta m_{\rm e} v_{\rm e}^2$ et  $\delta E_{\rm{c},\rm{s}} = \frac{1}{2} \delta m_{\rm s} v_{\rm s}^2$.
		\item Or en régime stationnaire, $E_{\rm{c},\So}(t)=E_{\rm{c},\So}(t+\d t)$ et donc on obtient par différence entre les deux équations $\d E_{\rm{c},\Sf}(t)=\delta E_{\rm{c},\rm{s}}-\delta E_{\rm{c},\rm{e}} = \frac{1}{2}\Dm\d t v_{\rm{s}}^2 - \frac{1}{2}\Dm\d t v_{\rm{e}}^2$,
		et donc
		\begin{equation}
		\frac{\d E_{\rm{c},\Sf}(t)}{\d t}=\frac{1}{2}\Dm\left(v_{\rm{s}}^2-v_{\rm{e}}^2\right).
		\end{equation}
		\item On applique le TEC en supposant l'écoulement parfait et incompressible, ce qui donne
		\begin{equation}
			\begin{split}
				\frac{\d E_{\rm{c},\Sf}(t)}{\d t}=\mc{P}_{\rm ext}=-\Gamma\omega.
			\end{split}
		\end{equation}
\end{enumerate}


On admet que l'application du TMC permet de trouver la relation suivante entre les vitesses d'entrée et de sortie,
\begin{equation}
v_{\rm{s}}=v_{\rm{e}}-\frac{\Gamma}{D_{m}R}.
\end{equation}
En combinant les résultats précédents, on en déduit la vitesse angulaire
\begin{equation}
\begin{split}
\omega=\dfrac{v_{\rm{e}}}{R}-\frac{\Gamma}{2R^{2}\Dm}.
\end{split}
\end{equation}
Le moulin tourne d'autant plus vite que la vitesse d'entrée du fluide est élevée, que le rayon de la roue est petit, et que le couple résistant de la roue est faible.
